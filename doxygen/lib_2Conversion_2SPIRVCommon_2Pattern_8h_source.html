19 /// Converts elementwise unary, binary and ternary standard operations to SPIR-V
20 /// operations.
21 template <typename Op, typename SPIRVOp>
22 struct ElementwiseOpPattern : public OpConversionPattern<Op> {
23  using OpConversionPattern<Op>::OpConversionPattern;
24 
25  LogicalResult
26  matchAndRewrite(Op op, typename Op::Adaptor adaptor,
27  ConversionPatternRewriter &rewriter) const override {
28  assert(adaptor.getOperands().size() <= 3);
29  Type dstType = this->getTypeConverter()->convertType(op.getType());
30  if (!dstType) {
31  return rewriter.notifyMatchFailure(
32  op->getLoc(),
33  llvm::formatv("failed to convert type {0} for SPIR-V", op.getType()));
34  }
35 
36  if (SPIRVOp::template hasTrait<OpTrait::spirv::UnsignedOp>() &&
37  !op.getType().isIndex() && dstType != op.getType()) {
38  return op.emitError(
39  "bitwidth emulation is not implemented yet on unsigned op");
40  }
41  rewriter.template replaceOpWithNewOp<SPIRVOp>(op, dstType,
42  adaptor.getOperands());
43  return success();
44  }
45 };
