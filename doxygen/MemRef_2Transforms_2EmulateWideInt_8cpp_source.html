100 struct EmulateWideIntPass final
101  : memref::impl::MemRefEmulateWideIntBase<EmulateWideIntPass> {
102  using MemRefEmulateWideIntBase::MemRefEmulateWideIntBase;
103 
104  void runOnOperation() override {
105  if (!llvm::isPowerOf2_32(widestIntSupported) || widestIntSupported < 2) {
106  signalPassFailure();
107  return;
108  }
109 
110  Operation *op = getOperation();
111  MLIRContext *ctx = op->getContext();
112 
113  arith::WideIntEmulationConverter typeConverter(widestIntSupported);
114  memref::populateMemRefWideIntEmulationConversions(typeConverter);
115  ConversionTarget target(*ctx);
116  target.addDynamicallyLegalDialect<
117  arith::ArithDialect, memref::MemRefDialect, vector::VectorDialect>(
118  [&typeConverter](Operation *op) { return typeConverter.isLegal(op); });
119 
120  RewritePatternSet patterns(ctx);
121  // Add common pattenrs to support contants, functions, etc.
122  arith::populateArithWideIntEmulationPatterns(typeConverter, patterns);
123 
124  memref::populateMemRefWideIntEmulationPatterns(typeConverter, patterns);
125 
126  if (failed(applyPartialConversion(op, target, std::move(patterns))))
127  signalPassFailure();
128  }
129 };

145 void memref::populateMemRefWideIntEmulationConversions(
146  arith::WideIntEmulationConverter &typeConverter) {
147  typeConverter.addConversion(
148  [&typeConverter](MemRefType ty) -> Optional<Type> {
149  auto intTy = ty.getElementType().dyn_cast<IntegerType>();
150  if (!intTy)
151  return ty;
152 
153  if (intTy.getIntOrFloatBitWidth() <=
154  typeConverter.getMaxTargetIntBitWidth())
155  return ty;
156 
157  Type newElemTy = typeConverter.convertType(intTy);
158  if (!newElemTy)
159  return None;
160 
161  return ty.cloneWith(None, newElemTy);
162  });
163 }
