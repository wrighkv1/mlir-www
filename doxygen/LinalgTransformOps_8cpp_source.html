61 /// Attempts to apply the pattern specified as template argument to the given
62 /// operation. The pattern is expected to have a `returningMatchAndRewrite`
63 /// function that returns the "main" result or failure. Returns failure if the
64 /// pattern failed to apply. Extra arguments are forwarded to the pattern
65 /// constructor.
66 template <typename PatternTy, typename... Args>
67 static FailureOr<LinalgOp> tryApply(Operation *operation, Args &&...args) {
68  // Check if the given operation has the type expected by the pattern.
69  using OpTy = typename llvm::function_traits<
70  decltype(&PatternTy::returningMatchAndRewrite)>::template arg_t<0>;
71  auto op = dyn_cast<OpTy>(operation);
72  if (!op)
73  return failure();
74 
75  // Apply the pattern directly to the op.
76  PatternTy pattern(operation->getContext(), std::forward<Args>(args)...);
77  SimpleRewriter rewriter(operation->getContext());
78  rewriter.setInsertionPoint(operation);
79  auto result = pattern.returningMatchAndRewrite(op, rewriter);
80  if (failed(result))
81  return failure();
82  return cast<LinalgOp>(result->getOperation());
83 }

120 /// Apply a tiling transformation to all payload ops and store both the
121 /// tiled operation as well as the created tile loops.
122 static LogicalResult applyTilingToAll(
123  Operation *transformOp, ArrayRef<Operation *> payloadOps, unsigned numLoops,
124  transform::TransformResults &transformResults,
125  function_ref<FailureOr<scf::SCFTileAndFuseResult>(TilingInterface)>
126  applyFn) {
127  SmallVector<Operation *> tiledLinalgOps;
128  SmallVector<SmallVector<Operation *>> loopOps(numLoops);
129  for (unsigned int i = 0; i < numLoops; ++i)
130  loopOps[i].reserve(payloadOps.size());
131 
132  for (Operation *target : payloadOps) {
133  auto tilingInterfaceOp = dyn_cast<TilingInterface>(target);
134  if (!tilingInterfaceOp)
135  return transformOp->emitError("only TilingInterface ops are supported");
136 
137  SimpleRewriter rewriter(target->getContext());
138  rewriter.setInsertionPoint(target);
139  FailureOr<scf::SCFTileAndFuseResult> tiledResults =
140  applyFn(tilingInterfaceOp);
141  if (failed(tiledResults))
142  return failure();
143 
144  // Perform the replacement of tiled and fused values.
145  SmallVector<Operation *> opsToReplace{target};
146  llvm::append_range(opsToReplace, tiledResults->fusedProducers);
147  for (Operation *toReplace : opsToReplace) {
148  SmallVector<Value> replacements;
149  replacements.reserve(toReplace->getNumResults());
150  for (OpResult res : toReplace->getResults()) {
151  auto it = tiledResults->replacements.find(res);
152  if (it == tiledResults->replacements.end())
153  replacements.push_back(res);
154  else
155  replacements.push_back(it->getSecond());
156  }
157  rewriter.replaceOp(toReplace, replacements);
158  }
159 
160  // Report back the relevant handles to the transform op.
161  tiledLinalgOps.push_back(tiledResults->tiledAndFusedOps.front());
162  assert(tiledResults->loops.size() == numLoops &&
163  "Mismatched number of loops, tile and fuse transform should have "
164  "failed");
165  for (unsigned int i = 0; i < numLoops; ++i)
166  loopOps[i].push_back(tiledResults->loops[i]);
167  }
168 
169  transformResults.set(transformOp->getOpResult(0), tiledLinalgOps);
170  for (unsigned int i = 0; i < numLoops; ++i)
171  transformResults.set(transformOp->getOpResult(i + 1), loopOps[i]);
172 
173  return success();
174 }

176 /// Parse a tiling-like operation that returns the tiled op as well as the
177 /// created tile loops. The function counts the non-zero tile sizes to compute
178 /// the number of results.
179 static ParseResult parseTileLikeOp(OpAsmParser &parser, OperationState &result,
180  StringRef sizesAttrName) {
181  OpAsmParser::UnresolvedOperand targetOperand;
182  SMLoc opLoc = parser.getCurrentLocation();
183  if (parser.parseOperand(targetOperand) ||
184  parser.parseOptionalAttrDict(result.attributes))
185  return failure();
186  Attribute sizesAttr = result.attributes.get(sizesAttrName);
187  if (!sizesAttr)
188  return parser.emitError(opLoc)
189  << "expected '" << sizesAttrName << "' attribute";
190  auto sizesArrayAttr = sizesAttr.dyn_cast<ArrayAttr>();
191  if (!sizesArrayAttr)
192  return parser.emitError(opLoc)
193  << "'" << sizesAttrName << "' attribute must be an array";
194  Type pdlOpType = parser.getBuilder().getType<pdl::OperationType>();
195  size_t numExpectedLoops =
196  sizesArrayAttr.size() -
197  llvm::count(extractFromI64ArrayAttr(sizesArrayAttr), 0);
198  result.addTypes(SmallVector<Type>(numExpectedLoops + 1, pdlOpType));
199  if (parser.resolveOperand(targetOperand, pdlOpType, result.operands))
200  return failure();
201  return success();
202 }

265 /// Find the first "extract" user of `producerOp` and tile it right before its
266 /// use. The tiled op is fused under the `containingOp`.
267 /// Return this fused op on success or nullptr if anything fails.
268 static Operation *tileAndFuseFirstExtractUse(RewriterBase &rewriter,
269  Diagnostic &diag,
270  Operation *producerOp,
271  Operation *containingOp) {
272  LLVM_DEBUG(llvm::dbgs() << "Try to fuse a direct extract use\n");
273  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
274  if (!tileableProducer) {
275  diag.attachNote(producerOp->getLoc())
276  << "producer is not a TileableInterface: " << *producerOp;
277  return nullptr;
278  }
279 
280  // Search the producer slices accessed within the containing operation.
281  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
282  // evolve into an interface.
283  auto it = llvm::find_if(tileableProducer->getUsers(), [&](Operation *user) {
284  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
285  return sliceOp && containingOp->isProperAncestor(sliceOp);
286  });
287 
288  // Find a fusion opportunity.
289  if (it == tileableProducer->getUsers().end()) {
290  diag.attachNote(tileableProducer->getLoc())
291  << "could not find fusion opportunity for: " << *tileableProducer;
292  return nullptr;
293  }
294  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*it);
295 
296  // Try to fuse the producer in-place.
297  OpBuilder::InsertionGuard guard(rewriter);
298  rewriter.setInsertionPoint(sliceOpToTile);
299 
300  // Tile the producer.
301  int64_t resultNumber =
302  sliceOpToTile.getSource().cast<OpResult>().getResultNumber();
303  LLVM_DEBUG(llvm::dbgs() << "resultNumber: " << resultNumber << "\n");
304 
305  FailureOr<Value> tiledProducer = tileableProducer.generateResultTileValue(
306  rewriter, resultNumber, sliceOpToTile.getMixedOffsets(),
307  sliceOpToTile.getMixedSizes());
308  if (failed(tiledProducer)) {
309  diag.attachNote(tileableProducer->getLoc())
310  << "failed to tile producer op: " << *tileableProducer;
311  return nullptr;
312  }
313  LLVM_DEBUG(llvm::dbgs() << "tiledProducer: " << *tiledProducer << "\n");
314 
315  // Replace the extract op.
316  Operation *fusedOp = tiledProducer->getDefiningOp();
317  rewriter.replaceOp(sliceOpToTile, fusedOp->getResult(resultNumber));
318  return fusedOp;
319 }

321 /// First, find the first "scf::ForeachThreadOp" user of `producerOp` and ensure
322 /// it is exactly the `containingOp`, otherwise bail.
323 /// Then, find the first "extract" user of the tied block argument and tile it
324 /// right before its "extract" use. The tiled op is fused under the
325 /// `containingOp`.
326 /// Return this fused op on success or nullptr if anything fails.
327 static Operation *tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
328  RewriterBase &rewriter, Diagnostic &diag, Operation *producerOp,
329  Operation *containingOp) {
330  LLVM_DEBUG(
331  llvm::dbgs() << "Try to fuse an extract use through block argument\n");
332 
333  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
334  if (!tileableProducer) {
335  diag.attachNote(producerOp->getLoc())
336  << "producer is not a TileableInterface: " << *producerOp;
337  return nullptr;
338  }
339 
340  // Search the first use by a "scf::ForeachThreadOp" user.
341  scf::ForeachThreadOp foreachThreadOp;
342  auto itProducerUses =
343  llvm::find_if(tileableProducer->getUses(), [&](OpOperand &use) {
344  foreachThreadOp = dyn_cast<scf::ForeachThreadOp>(use.getOwner());
345  return foreachThreadOp;
346  });
347  // If it's not from the containing op, return.
348  if (!foreachThreadOp || foreachThreadOp != containingOp) {
349  diag.attachNote(tileableProducer->getLoc())
350  << "could not find a use by the containing op: " << *tileableProducer;
351  return nullptr;
352  }
353 
354  // Search the producer slices accessed within the containing
355  // operation.
356  // TODO: Generalize to more extract/insert/parallel_insert triples.
357  // Maybe evolve into an interface.
358  OpOperand *pUse = &(*itProducerUses);
359  BlockArgument bbArg = foreachThreadOp.getTiedBlockArgument(pUse);
360 
361  // Search the producer slices accessed within the containing operation.
362  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
363  // evolve into an interface.
364  auto itBBArgUsers = llvm::find_if(bbArg.getUsers(), [&](Operation *user) {
365  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
366  return sliceOp && containingOp->isProperAncestor(sliceOp);
367  });
368 
369  // Find a fusion opportunity.
370  if (itBBArgUsers == bbArg.getUsers().end()) {
371  diag.attachNote(containingOp->getLoc())
372  << "could not find fusion opportunity for bbArg: " << bbArg;
373  return nullptr;
374  }
375  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*itBBArgUsers);
376 
377  // Try to fuse the producer in-place.
378  OpBuilder::InsertionGuard guard(rewriter);
379  rewriter.setInsertionPoint(sliceOpToTile);
380 
381  // Replace the use in the tileableProducer before tiling: clone, replace and
382  // then tile.
383  int64_t resultNumber = pUse->get().cast<OpResult>().getResultNumber();
384  LLVM_DEBUG(llvm::dbgs() << "resultNumber: " << resultNumber << "\n");
385 
386  // Gather destination tensors.
387  SmallVector<Value> destinationTensors;
388  if (failed(tensor::getOrCreateDestinations(
389  rewriter, tileableProducer->getLoc(), tileableProducer,
390  destinationTensors))) {
391  diag.attachNote(tileableProducer->getLoc())
392  << "failed to get destination tensors for: " << *tileableProducer;
393  return nullptr;
394  }
395 
396  BlockAndValueMapping bvm;
397  bvm.map(destinationTensors[resultNumber], bbArg);
398  auto tileableProducerClone =
399  cast<TilingInterface>(rewriter.clone(*tileableProducer, bvm));
400  auto scopeGuard =
401  llvm::make_scope_exit([&]() { rewriter.eraseOp(tileableProducerClone); });
402 
403  // Tile the producer.
404  FailureOr<Value> tiledProducer =
405  tileableProducerClone.generateResultTileValue(
406  rewriter, resultNumber, sliceOpToTile.getMixedOffsets(),
407  sliceOpToTile.getMixedSizes());
408  if (failed(tiledProducer)) {
409  diag.attachNote(tileableProducer->getLoc())
410  << "failed to tile producer op: " << *tileableProducer;
411  return nullptr;
412  }
413  LLVM_DEBUG(llvm::dbgs() << "tiledProducer: " << *tiledProducer << "\n");
414 
415  // Replace the extract op.
416  Operation *fusedOp = tiledProducer->getDefiningOp();
417  rewriter.replaceOp(sliceOpToTile, fusedOp->getResult(resultNumber));
418 
419  // Replace the use in containingOp.
420  rewriter.updateRootInPlace(containingOp, [&]() {
421  containingOp->setOperand(pUse->getOperandNumber(),
422  destinationTensors.front());
423  });
424 
425  return fusedOp;
426 }

428 static Operation *cloneAndFuseFirstUse(RewriterBase &rewriter, Diagnostic &diag,
429  Operation *producerOp,
430  Operation *containingOp) {
431  LLVM_DEBUG(llvm::dbgs() << "Try to fuse an use by cloning\n");
432 
433  // Gather all uses inside the containing op.
434  SmallVector<OpOperand *> uses;
435  for (OpResult result : producerOp->getOpResults()) {
436  for (OpOperand &use : result.getUses()) {
437  if (containingOp->isProperAncestor(use.getOwner())) {
438  uses.push_back(&use);
439  continue;
440  }
441  // Cannot clone and fuse if the use is by the containing op itself: fail
442  // immediately.
443  if (containingOp == use.getOwner()) {
444  diag.attachNote(producerOp->getLoc())
445  << "producer op use by containing op cannot be fused by cloning";
446  return nullptr;
447  }
448  }
449  }
450 
451  // Check for a non-empty list of fusion opportunities.
452  if (uses.empty()) {
453  diag.attachNote(producerOp->getLoc()) << "no fusion opportunity by cloning";
454  return nullptr;
455  }
456 
457  // Clone and fuse inside the containing op.
458  Operation *fusedOp = nullptr;
459  OpOperand *use = uses.front();
460  // Parallel insert slice is not a valid clone destination.
461  // TODO: Generalize to other type of ops.
462  assert(!isa<tensor::ParallelInsertSliceOp>(use->getOwner()) &&
463  "Parallel insert slice is not a valid clone destination");
464  unsigned resultNumber = use->get().cast<OpResult>().getResultNumber();
465  LLVM_DEBUG(llvm::dbgs() << "resultNumber: " << resultNumber << "\n");
466 
467  OpBuilder::InsertionGuard guard(rewriter);
468  rewriter.setInsertionPoint(use->getOwner());
469  fusedOp = rewriter.clone(*producerOp);
470  rewriter.updateRootInPlace(
471  use->getOwner(), [&] { use->set(fusedOp->getOpResult(resultNumber)); });
472 
473  return fusedOp;
474 }

476 DiagnosedSilenceableFailure
477 transform::FuseIntoContainingOp::apply(transform::TransformResults &results,
478  transform::TransformState &state) {
479  SmallVector<Operation *> fusedOps;
480  ArrayRef<Operation *> producerOps = state.getPayloadOps(getProducerOp());
481  // If nothing to fuse, propagate success.
482  if (producerOps.empty()) {
483  results.set(getFusedOp().cast<OpResult>(),
484  SmallVector<mlir::Operation *>{});
485  return DiagnosedSilenceableFailure::success();
486  }
487  ArrayRef<Operation *> containingOps = state.getPayloadOps(getContainingOp());
488  if (containingOps.size() != 1) {
489  return emitDefiniteFailure()
490  << "requires exactly one containing_op handle (got "
491  << containingOps.size() << ")";
492  }
493  Operation *containingOp = containingOps.front();
494 
495  // Helper function to find the next producer that should be fused. Take any
496  // producer that has a use inside the containing op.
497  SmallVector<Operation *> remainingProducers(producerOps.begin(),
498  producerOps.end());
499  auto getNextProducer = [&]() -> FailureOr<Operation *> {
500  for (const auto &it : enumerate(remainingProducers)) {
501  Operation *producerOp = it.value();
502  // The containing op may be a user of producerOp: use isAncestor.
503  int64_t numUsesInContainingOp =
504  llvm::count_if(producerOp->getUsers(), [&](Operation *op) {
505  return containingOp->isAncestor(op);
506  });
507  // TODO: When resolving the TODO below (no duplicate ops), take an op
508  // that has no use among the remaining producers. This is a topological
509  // sorting.
510  if (numUsesInContainingOp > 0) {
511  if (numUsesInContainingOp == 1)
512  remainingProducers.erase(remainingProducers.begin() + it.index());
513  return producerOp;
514  }
515  }
516  return failure();
517  };
518 
519  IRRewriter rewriter(getContext());
520  while (!remainingProducers.empty()) {
521  auto nextProducer = getNextProducer();
522  if (failed(nextProducer)) {
523  results.set(getFusedOp().cast<OpResult>(), ArrayRef<Operation *>());
524  Diagnostic diag(containingOp->getLoc(), DiagnosticSeverity::Remark);
525  diag << "could not find next producer to fuse into container";
526  return DiagnosedSilenceableFailure::silenceableFailure(std::move(diag));
527  }
528 
529  Operation *producerOp = *nextProducer;
530 
531  // Default diagnostic, to be complemented with more failure information.
532  Diagnostic diag(producerOp->getLoc(), DiagnosticSeverity::Remark);
533  diag << "could not fuse " << *producerOp << " into " << *containingOp;
534 
535  // TODO: If there are multiple uses of the producer in the containing op,
536  // we currently tile/clone the op multiple times (once per use). In some
537  // cases, we can tile/clone once and reuse the value for each use.
538  // Futhermore, producers should then be traversed according to a
539  // topological sorting.
540  Operation *tiled =
541  tileAndFuseFirstExtractUse(rewriter, diag, producerOp, containingOp);
542  if (tiled) {
543  LLVM_DEBUG(llvm::dbgs() << "\nFused a direct extract use\n"
544  << *containingOp);
545  fusedOps.push_back(tiled);
546  continue;
547  }
548 
549  Operation *tiledContainingOpOperand =
550  tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
551  rewriter, diag, producerOp, containingOp);
552  if (tiledContainingOpOperand) {
553  LLVM_DEBUG(llvm::dbgs()
554  << "\nFused an extract use through block argument\n"
555  << *containingOp);
556  fusedOps.push_back(tiledContainingOpOperand);
557  continue;
558  }
559 
560  Operation *cloned =
561  cloneAndFuseFirstUse(rewriter, diag, producerOp, containingOp);
562  if (cloned) {
563  LLVM_DEBUG(llvm::dbgs() << "\nFused an use by cloning\n"
564  << *containingOp);
565  fusedOps.push_back(cloned);
566  continue;
567  }
568  results.set(getFusedOp().cast<OpResult>(), ArrayRef<Operation *>());
569  return DiagnosedSilenceableFailure::silenceableFailure(std::move(diag));
570  }
571 
572  results.set(getFusedOp().cast<OpResult>(), fusedOps);
573  return DiagnosedSilenceableFailure::success();
574 }

647 DiagnosedSilenceableFailure
648 transform::MatchOp::apply(transform::TransformResults &results,
649  transform::TransformState &state) {
650  llvm::StringSet<> strs;
651  if (getOps().has_value())
652  strs.insert(getOps()->getAsValueRange<StringAttr>().begin(),
653  getOps()->getAsValueRange<StringAttr>().end());
654 
655  ArrayRef<Operation *> payloadOps = state.getPayloadOps(getTarget());
656  if (payloadOps.size() != 1) {
657  results.set(getResult().cast<OpResult>(), {});
658  return DiagnosedSilenceableFailure(
659  this->emitOpError("requires exactly one target handle"));
660  }
661 
662  SmallVector<Operation *> res;
663  auto matchFun = [&](Operation *op) {
664  if (getOps().has_value() && !strs.contains(op->getName().getStringRef()))
665  return;
666 
667  // Interfaces cannot be matched by name, just by ID.
668  // So we specifically encode the interfaces we care about for this op.
669  if (getInterface().has_value()) {
670  auto iface = getInterface().value();
671  if (iface == transform::MatchInterfaceEnum::LinalgOp &&
672  !isa<linalg::LinalgOp>(op))
673  return;
674  if (iface == transform::MatchInterfaceEnum::TilingInterface &&
675  isa<TilingInterface>(op))
676  return;
677  }
678 
679  // Check if all specified attributes match.
680  if (getOpAttrs().has_value()) {
681  DictionaryAttr opAttrs = getOpAttrs().value();
682  for (NamedAttribute attr : opAttrs) {
683  if (attr.getName() == getInterfaceAttrName() ||
684  attr.getName() == getOpsAttrName())
685  continue;
686  if (!op->hasAttr(attr.getName()))
687  return;
688  if (op->getAttr(attr.getName()) != attr.getValue())
689  return;
690  }
691  }
692 
693  if (getFilterResultType().has_value()) {
694  Type t = getFilterResultType().value();
695  if (op->getNumResults() != 1 || op->getResultTypes().front() != t)
696  return;
697  }
698 
699  // All constraints are satisfied.
700  res.push_back(op);
701  return;
702  };
703 
704  payloadOps.front()->walk(matchFun);
705  results.set(getResult().cast<OpResult>(), res);
706  return DiagnosedSilenceableFailure(success());
707 }

815  return emitDefaultSilenceableFailure(target);

942 DiagnosedSilenceableFailure SplitOp::apply(TransformResults &results,
943  TransformState &state) {
944  // Collect the dynamic split points if provided.
945  ArrayRef<Operation *> payload = state.getPayloadOps(getTarget());
946  SimpleRewriter rewriter(getContext());
947  SmallVector<OpFoldResult> splitPoints;
948  splitPoints.reserve(payload.size());
949  if (getDynamicSplitPoint()) {
950  auto diag = DiagnosedSilenceableFailure::success();
951  splitPoints = llvm::to_vector(llvm::map_range(
952  state.getPayloadOps(getDynamicSplitPoint()), [&](Operation *op) {
953  if (op->getNumResults() != 1 ||
954  !op->getResult(0).getType().isIndex()) {
955  diag = emitSilenceableError()
956  << "expected dynamic split point handle to point to a "
957  "single-result index-typed op";
958  diag.attachNote(op->getLoc()) << "dynamic split point";
959  }
960  return OpFoldResult(op->getResult(0));
961  }));
962  if (diag.isSilenceableFailure()) {
963  results.set(getFirst().cast<OpResult>(), {});
964  results.set(getSecond().cast<OpResult>(), {});
965  return diag;
966  }
967 
968  if (splitPoints.size() != payload.size()) {
969  return emitDefiniteFailure()
970  << "expected the dynamic split point handle to point to as "
971  "many operations ("
972  << splitPoints.size() << ") as the target handle ("
973  << payload.size() << ")";
974  }
975  } else {
976  splitPoints.resize(payload.size(),
977  rewriter.getIndexAttr(getStaticSplitPoint()));
978  }
979 
980  // Split each target operation.
981  SmallVector<Operation *> first, second;
982  for (const auto &pair : llvm::zip(payload, splitPoints)) {
983  Operation *target = std::get<0>(pair);
984  auto linalgOp = dyn_cast<LinalgOp>(target);
985  if (!linalgOp) {
986  auto diag = emitSilenceableError() << "only applies to structured ops";
987  diag.attachNote(target->getLoc()) << "target op";
988  results.set(getFirst().cast<OpResult>(), {});
989  results.set(getSecond().cast<OpResult>(), {});
990  return diag;
991  }
992 
993  if (getDimension() >= linalgOp.getNumLoops()) {
994  auto diag = emitSilenceableError() << "dimension " << getDimension()
995  << " does not exist in target op";
996  diag.attachNote(target->getLoc()) << "target op";
997  results.set(getFirst().cast<OpResult>(), {});
998  results.set(getSecond().cast<OpResult>(), {});
999  return diag;
1000  }
1001 
1002  rewriter.setInsertionPoint(linalgOp);
1003  std::tie(first.emplace_back(), second.emplace_back()) = linalg::splitOp(
1004  rewriter, cast<TilingInterface>(linalgOp.getOperation()),
1005  getDimension(), std::get<1>(pair));
1006  }
1007 
1008  results.set(getFirst().cast<OpResult>(), first);
1009  results.set(getSecond().cast<OpResult>(), second);
1010  return DiagnosedSilenceableFailure::success();
1011 }
