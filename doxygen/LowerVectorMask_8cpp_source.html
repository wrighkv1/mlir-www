34 /// The `MaskOpRewritePattern` implements a pattern that follows a two-fold
35 /// matching:
36 /// 1. It matches a `vector.mask` operation.
37 /// 2. It invokes `matchAndRewriteMaskableOp` on `MaskableOpInterface` nested
38 /// in the matched `vector.mask` operation.
39 ///
40 /// It is required that the replacement op in the pattern replaces the
41 /// `vector.mask` operation and not the nested `MaskableOpInterface`. This
42 /// approach allows having patterns that "stop" at every `vector.mask` operation
43 /// and actually match the traits of its the nested `MaskableOpInterface`.
44 template <class SourceOp>
45 struct MaskOpRewritePattern : OpRewritePattern<MaskOp> {
46  using OpRewritePattern<MaskOp>::OpRewritePattern;
47 
48 private:
49  LogicalResult matchAndRewrite(MaskOp maskOp,
50  PatternRewriter &rewriter) const final {
51  MaskableOpInterface maskableOp = maskOp.getMaskableOp();
52  SourceOp sourceOp = dyn_cast<SourceOp>(maskableOp.getOperation());
53  if (!sourceOp)
54  return failure();
55 
56  return matchAndRewriteMaskableOp(sourceOp, maskOp, rewriter);
57  }
58 
59 protected:
60  virtual LogicalResult
61  matchAndRewriteMaskableOp(SourceOp sourceOp, MaskingOpInterface maskingOp,
62  PatternRewriter &rewriter) const = 0;
63 };

65 /// Lowers a masked `vector.transfer_read` operation.
66 struct MaskedTransferReadOpPattern
67  : public MaskOpRewritePattern<TransferReadOp> {
68 public:
69  using MaskOpRewritePattern<TransferReadOp>::MaskOpRewritePattern;
70 
71  LogicalResult
72  matchAndRewriteMaskableOp(TransferReadOp readOp, MaskingOpInterface maskingOp,
73  PatternRewriter &rewriter) const override {
74  // TODO: The 'vector.mask' passthru is a vector and 'vector.transfer_read'
75  // expects a scalar. We could only lower one to the other for cases where
76  // the passthru is a broadcast of a scalar.
77  if (maskingOp.hasPassthru())
78  return rewriter.notifyMatchFailure(
79  maskingOp, "Can't lower passthru to vector.transfer_read");
80 
81  // Replace the `vector.mask` operation.
82  rewriter.replaceOpWithNewOp<TransferReadOp>(
83  maskingOp.getOperation(), readOp.getVectorType(), readOp.getSource(),
84  readOp.getIndices(), readOp.getPermutationMap(), readOp.getPadding(),
85  maskingOp.getMask(), readOp.getInBounds().value_or(ArrayAttr()));
86  return success();
87  }
88 };

97  matchAndRewriteMaskableOp(TransferWriteOp writeOp,
98  MaskingOpInterface maskingOp,
99  PatternRewriter &rewriter) const override {
100  Type resultType =
101  writeOp.getResult() ? writeOp.getResult().getType() : Type();
102 
103  // Replace the `vector.mask` operation.
104  rewriter.replaceOpWithNewOp<TransferWriteOp>(
105  maskingOp.getOperation(), resultType, writeOp.getVector(),
106  writeOp.getSource(), writeOp.getIndices(), writeOp.getPermutationMap(),
107  maskingOp.getMask(), writeOp.getInBounds().value_or(ArrayAttr()));
108  return success();
109  }
