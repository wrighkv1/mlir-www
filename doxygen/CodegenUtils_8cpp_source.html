24 /// Generates a pointer/index load from the sparse storage scheme. Narrower
25 /// data types need to be zero extended before casting the value into the
26 /// index type used for looping and indexing.
27 static Value genIndexLoad(OpBuilder &builder, Location loc, Value ptr,
28  Value s) {
29  // For the scalar case, we simply zero extend narrower indices into 64-bit
30  // values before casting to index without a performance penalty. Here too,
31  // however, indices that already are 64-bit, in theory, cannot express the
32  // full range as explained above.
33  Value load = builder.create<memref::LoadOp>(loc, ptr, s);
34  if (!load.getType().isa<IndexType>()) {
35  if (load.getType().getIntOrFloatBitWidth() < 64)
36  load = builder.create<arith::ExtUIOp>(loc, builder.getI64Type(), load);
37  load =
38  builder.create<arith::IndexCastOp>(loc, builder.getIndexType(), load);
39  }
40  return load;
41 }

62  Value indices, Value values,

97 SparseTensorLoopEmitter::SparseTensorLoopEmitter(ValueRange tensors,
98  StringAttr loopTag,
99  bool hasOutput,
100  bool isSparseOut)
101  : loopTag(loopTag), hasOutput(hasOutput), isSparseOut(isSparseOut),
102  tensors(tensors.begin(), tensors.end()), dimTypes(tensors.size()),
103  pidxs(tensors.size()), coord(tensors.size()), highs(tensors.size()),
104  ptrBuffer(tensors.size()), idxBuffer(tensors.size()),
105  valBuffer(tensors.size()), loopStack() {
106  for (size_t tid = 0, e = tensors.size(); tid < e; tid++) {
107  auto t = tensors[tid];
108  // a scalar or 0-dimension tensors
109  if (isZeroRankedTensorOrScalar(t.getType()))
110  continue;
111  auto rtp = t.getType().cast<RankedTensorType>();
112  auto rank = static_cast<size_t>(rtp.getRank());
113  auto enc = getSparseTensorEncoding(rtp);
114  // We always treat sparse output tensor as dense so that we always iterate
115  // it based on dim size.
116  if (enc && !(isOutputTensor(tid) && isSparseOut))
117  for (auto dimTp : enc.getDimLevelType())
118  dimTypes[tid].push_back(dimTp);
119  else
120  dimTypes[tid].assign(rank, DimLevelType::Dense);
121 
122  // Initialize using empty value.
123  pidxs[tid].assign(rank, Value());
124  coord[tid].assign(rank, Value());
125  highs[tid].assign(rank, Value());
126  ptrBuffer[tid].assign(rank, Value());
127  idxBuffer[tid].assign(rank, Value());
128  }
129 }

131 void SparseTensorLoopEmitter::initializeLoopEmit(
132  OpBuilder &builder, Location loc,
133  SparseTensorLoopEmitter::OutputUpdater updater) {
134  // For every tensor, find lower and upper bound on dimensions, set the
135  // same bounds on loop indices, and obtain dense or sparse buffer(s).
136  for (size_t t = 0, e = tensors.size(); t < e; t++) {
137  auto tensor = tensors[t];
138  auto rtp = tensor.getType().dyn_cast<RankedTensorType>();
139  if (!rtp)
140  // Skips only scalar, zero ranked tensor still need to be bufferized and
141  // (probably) filled with zeros by users.
142  continue;
143  auto rank = rtp.getRank();
144  auto shape = rtp.getShape();
145  auto enc = getSparseTensorEncoding(rtp);
146  auto dynShape = {ShapedType::kDynamicSize};
147  // Scan all dimensions of current tensor.
148  for (int64_t d = 0; d < rank; d++) {
149  // This should be called only once at beginning.
150  assert(!ptrBuffer[t][d] && !idxBuffer[t][d] && !highs[t][d]);
151  // Handle sparse storage schemes.
152  if (isCompressedDLT(dimTypes[t][d])) {
153  auto ptrTp =
154  MemRefType::get(dynShape, getPointerOverheadType(builder, enc));
155  auto indTp =
156  MemRefType::get(dynShape, getIndexOverheadType(builder, enc));
157  auto dim = builder.getIndexAttr(d);
158  // Generate sparse primitives to obtains pointer and indices.
159  ptrBuffer[t][d] = builder.create<ToPointersOp>(loc, ptrTp, tensor, dim);
160  idxBuffer[t][d] = builder.create<ToIndicesOp>(loc, indTp, tensor, dim);
161  } else if (isSingletonDLT(dimTypes[t][d])) {
162  // Singleton dimension, fetch indices.
163  auto indTp =
164  MemRefType::get(dynShape, getIndexOverheadType(builder, enc));
165  auto dim = builder.getIndexAttr(d);
166  idxBuffer[t][d] = builder.create<ToIndicesOp>(loc, indTp, tensor, dim);
167  } else {
168  // Dense dimension, nothing to fetch.
169  assert(isDenseDLT(dimTypes[t][d]));
170  }
171 
172  // Find upper bound in current dimension.
173  unsigned p = toOrigDim(enc, d);
174  Value up = mlir::linalg::createOrFoldDimOp(builder, loc, tensor, p);
175  highs[t][d] = up;
176  }
177 
178  // Perform the required bufferization. Dense inputs materialize
179  // from the input tensors. Sparse inputs use sparse primitives to obtain the
180  // values.
181  // Delegates extra output initialization to clients.
182  bool isOutput = isOutputTensor(t);
183  Type elementType = rtp.getElementType();
184  if (!enc) {
185  // Non-annotated dense tensors.
186  auto denseTp = MemRefType::get(shape, elementType);
187  Value denseVal =
188  builder.create<bufferization::ToMemrefOp>(loc, denseTp, tensor);
189  // Dense outputs need special handling.
190  if (isOutput && updater)
191  denseVal = updater(builder, loc, denseVal, tensor);
192 
193  valBuffer[t] = denseVal;
194  } else {
195  // Annotated sparse tensors.
196  // We also need the value buffer for annotated all dense `sparse` tensor.
197  auto dynShape = {ShapedType::kDynamicSize};
198  auto sparseTp = MemRefType::get(dynShape, elementType);
199  valBuffer[t] = builder.create<ToValuesOp>(loc, sparseTp, tensor);
200  }
201  // NOTE: we can also prepares for 0 dim here in advance, this will hosit
202  // some loop preparation from tensor iteration, but will also (undesirably)
203  // hosit the code ouside if conditions.
204  }
205 }

219 Operation *SparseTensorLoopEmitter::enterLoopOverTensorAtDim(
220  OpBuilder &builder, Location loc, size_t tid, size_t dim,
221  MutableArrayRef<Value> reduc, bool isParallel, ArrayRef<size_t> extraTids,
222  ArrayRef<size_t> extraDims) {
223 
224  assert(dimTypes[tid].size() > dim);
225  // We can not re-enter the same level.
226  assert(!coord[tid][dim]);
227  // TODO: support multiple return on parallel for?
228  assert(!isParallel || reduc.size() <= 1);
229 
230  Value step = constantIndex(builder, loc, 1);
231  auto dimType = dimTypes[tid][dim];
232  bool isSparseInput = isCompressedDLT(dimType) || isSingletonDLT(dimType);
233  assert(isDenseDLT(dimType) || isCompressedDLT(dimType) ||
234  isSingletonDLT(dimType));
235 
236  Value lo = isSparseInput ? pidxs[tid][dim] // current offset
237  : loopSeqStack.back(); // univeral tid
238  Value hi = highs[tid][dim];
239  Operation *loop = nullptr;
240  Value iv;
241  if (isParallel) {
242  scf::ParallelOp parOp =
243  builder.create<scf::ParallelOp>(loc, lo, hi, step, reduc);
244  builder.setInsertionPointToStart(parOp.getBody());
245  assert(parOp.getNumReductions() == reduc.size());
246  iv = parOp.getInductionVars()[0];
247 
248  // In-place update on the reduction variable vector.
249  // Note that the init vals is not the actual reduction variables but instead
250  // used as a `special handle` to (temporarily) represent them. The
251  // expression on init vals will be moved into scf.reduce and replaced with
252  // the block arguments when exiting the loop (see exitForLoop). This is
253  // needed as we can not build the actual reduction block and get the actual
254  // reduction varaible before users fill parallel loop body.
255  for (int i = 0, e = reduc.size(); i < e; i++)
256  reduc[i] = parOp.getInitVals()[i];
257  loop = parOp;
258  } else {
259  scf::ForOp forOp = builder.create<scf::ForOp>(loc, lo, hi, step, reduc);
260  builder.setInsertionPointToStart(forOp.getBody());
261  iv = forOp.getInductionVar();
262 
263  // In-place update on the reduction variable vector.
264  assert(forOp.getNumRegionIterArgs() == reduc.size());
265  for (int i = 0, e = reduc.size(); i < e; i++)
266  reduc[i] = forOp.getRegionIterArg(i);
267  loop = forOp;
268  }
269  assert(loop && iv);
270 
271  if (isSparseInput) {
272  pidxs[tid][dim] = iv;
273  // Generating a load on the indices array yields the coordinate.
274  Value ptr = idxBuffer[tid][dim];
275  coord[tid][dim] = genIndexLoad(builder, loc, ptr, iv);
276  } else {
277  // Dense tensor, the coordinates is the inducation variable.
278  coord[tid][dim] = iv;
279  // generate pidx for dense dim (pidx = i * sz + j)
280  auto enc = getSparseTensorEncoding(tensors[tid].getType());
281  if (enc && !isSparseOutput(tid))
282  pidxs[tid][dim] = genAddress(builder, loc, tid, dim, iv);
283  }
284 
285  // NOTE: we can also prepares for next dim here in advance
286  // Push the loop into stack
287  loopStack.emplace_back(ArrayRef<size_t>(tid), ArrayRef<size_t>(dim), loop,
288  coord[tid][dim], loopTag);
289  // Emit extra locals.
290  emitExtraLocalsForTensorsAtDenseDims(builder, loc, extraTids, extraDims);
291 
292  return loop;
293 }

295 Operation *SparseTensorLoopEmitter::enterCoIterationOverTensorsAtDims(
296  OpBuilder &builder, Location loc, ArrayRef<size_t> tids,
297  ArrayRef<size_t> dims, bool needsUniv, MutableArrayRef<Value> reduc,
298  ArrayRef<size_t> extraTids, ArrayRef<size_t> extraDims) {
299  assert(tids.size() == dims.size());
300  SmallVector<Type> types;
301  SmallVector<Value> operands;
302  // Construct the while-loop with a parameter for each index.
303  Type indexType = builder.getIndexType();
304  for (auto [tid, dim] : llvm::zip(tids, dims)) {
305  if (isCompressedDLT(dimTypes[tid][dim]) ||
306  isSingletonDLT(dimTypes[tid][dim])) {
307  assert(pidxs[tid][dim]);
308  types.push_back(indexType);
309  operands.push_back(pidxs[tid][dim]);
310  }
311  }
312  // The position where user-supplied reduction variable starts.
313  for (Value rec : reduc) {
314  types.push_back(rec.getType());
315  operands.push_back(rec);
316  }
317  if (needsUniv) {
318  types.push_back(indexType);
319  // Update universal index.
320  operands.push_back(loopSeqStack.back());
321  }
322  assert(types.size() == operands.size());
323  scf::WhileOp whileOp = builder.create<scf::WhileOp>(loc, types, operands);
324 
325  SmallVector<Location> locs(types.size(), loc);
326  Block *before = builder.createBlock(&whileOp.getBefore(), {}, types, locs);
327  Block *after = builder.createBlock(&whileOp.getAfter(), {}, types, locs);
328 
329  // Build the "before" region, which effectively consists
330  // of a conjunction of "i < upper" tests on all induction.
331  builder.setInsertionPointToStart(&whileOp.getBefore().front());
332  Value cond;
333  unsigned o = 0;
334  for (auto [tid, dim] : llvm::zip(tids, dims)) {
335  if (isCompressedDLT(dimTypes[tid][dim]) ||
336  isSingletonDLT(dimTypes[tid][dim])) {
337  Value op1 = before->getArgument(o);
338  Value op2 = highs[tid][dim];
339  Value opc = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::ult,
340  op1, op2);
341  cond = cond ? builder.create<arith::AndIOp>(loc, cond, opc) : opc;
342  // Update
343  pidxs[tid][dim] = after->getArgument(o++);
344  }
345  }
346  builder.create<scf::ConditionOp>(loc, cond, before->getArguments());
347 
348  // Generates while body.
349  builder.setInsertionPointToStart(&whileOp.getAfter().front());
350  Value min;
351  for (auto [tid, dim] : llvm::zip(tids, dims)) {
352  // Prepares for next level.
353  if (isCompressedDLT(dimTypes[tid][dim]) ||
354  isSingletonDLT(dimTypes[tid][dim])) {
355  Value ptr = idxBuffer[tid][dim];
356  Value s = pidxs[tid][dim];
357  Value load = genIndexLoad(builder, loc, ptr, s);
358  coord[tid][dim] = load;
359  if (!needsUniv) {
360  if (min) {
361  Value cmp = builder.create<arith::CmpIOp>(
362  loc, arith::CmpIPredicate::ult, load, min);
363  min = builder.create<arith::SelectOp>(loc, cmp, load, min);
364  } else {
365  min = load;
366  }
367  }
368  }
369  }
370 
371  if (needsUniv) {
372  assert(!min);
373  // Otherwise, universal index is the minimal pidx.
374  min = after->getArguments().back();
375  }
376 
377  for (auto [tid, dim] : llvm::zip(tids, dims)) {
378  // All dense dim (as well as sparse output tensor) shared the same pidx in
379  // the while loop.
380  if (isDenseDLT(dimTypes[tid][dim])) {
381  pidxs[tid][dim] = min;
382  // generate pidx for dense dim (pidx = i * sz + j)
383  auto enc = getSparseTensorEncoding(tensors[tid].getType());
384  if (enc && !isSparseOutput(tid))
385  pidxs[tid][dim] = genAddress(builder, loc, tid, dim, min);
386  }
387  // NOTE: we can also prepares for next dim here in advance
388  }
389  // Sets up the loop stack.
390  loopStack.emplace_back(tids, dims, whileOp, min, loopTag);
391  assert(loopStack.size() == loopSeqStack.size());
392 
393  // Emits extra locals
394  emitExtraLocalsForTensorsAtDenseDims(builder, loc, extraTids, extraDims);
395 
396  // Updates reduction variables
397  assert(after->getNumArguments() == o + reduc.size() + (needsUniv ? 1 : 0));
398  // In-place update on reduction variable.
399  for (unsigned i = 0, e = reduc.size(); i < e; i++)
400  reduc[i] = after->getArgument(o + i);
401 
402  return whileOp;
403 }

405 void SparseTensorLoopEmitter::prepareLoopOverTensorAtDim(OpBuilder &builder,
406  Location loc,
407  size_t tid,
408  size_t dim) {
409  assert(dimTypes[tid].size() > dim);
410  auto dimType = dimTypes[tid][dim];
411 
412  if (isDenseDLT(dimType))
413  return;
414 
415  // Either the first dimension, or the previous dimension has been set.
416  assert(dim == 0 || pidxs[tid][dim - 1]);
417  Value c0 = constantIndex(builder, loc, 0);
418  Value c1 = constantIndex(builder, loc, 1);
419  if (isCompressedDLT(dimType)) {
420  Value ptr = ptrBuffer[tid][dim];
421 
422  Value pLo = dim == 0 ? c0 : pidxs[tid][dim - 1];
423  pidxs[tid][dim] = genIndexLoad(builder, loc, ptr, pLo);
424 
425  Value pHi = builder.create<arith::AddIOp>(loc, pLo, c1);
426  highs[tid][dim] = genIndexLoad(builder, loc, ptr, pHi);
427  return;
428  }
429  if (isSingletonDLT(dimType)) {
430  Value pLo = dim == 0 ? c0 : pidxs[tid][dim - 1];
431  Value pHi = builder.create<arith::AddIOp>(loc, pLo, c1);
432 
433  pidxs[tid][dim] = pLo;
434  highs[tid][dim] = pHi;
435  return;
436  }

438  llvm_unreachable("Unrecognizable dimesion type!");
439 }

443  ArrayRef<size_t> dims) {

446  // but may be needed for linearized codegen.

454  // unconditionally required by the sparsification.

456  continue;
457  }

464 void SparseTensorLoopEmitter::exitForLoop(RewriterBase &rewriter, Location loc,
465  MutableArrayRef<Value> reduc) {
466  LoopLevelInfo &loopInfo = loopStack.back();
467  auto &dims = loopStack.back().dims;
468  auto &tids = loopStack.back().tids;
469  auto forOp = llvm::dyn_cast<scf::ForOp>(loopInfo.loop);
470  if (forOp) {
471  if (!reduc.empty()) {
472  assert(reduc.size() == forOp.getNumResults());
473  rewriter.setInsertionPointToEnd(forOp.getBody());
474  rewriter.create<scf::YieldOp>(loc, reduc);
475  }
476  // Exit the loop.
477  rewriter.setInsertionPointAfter(forOp);
478  // In-place update reduction variables.
479  for (unsigned i = 0, e = forOp.getResults().size(); i < e; i++)
480  reduc[i] = forOp.getResult(i);
481  } else {
482  auto parOp = llvm::cast<scf::ParallelOp>(loopInfo.loop);
483  if (!reduc.empty()) {
484  assert(reduc.size() == parOp.getInitVals().size() && reduc.size() == 1);
485  Operation *redExp = reduc.front().getDefiningOp();
486  // Reduction expression should have no use.
487  assert(redExp->getUses().empty());
488  // This must be a binary operation.
489  // NOTE: This is users' responsibilty to ensure the operation are
490  // commutative.
491  assert(redExp->getNumOperands() == 2 && redExp->getNumResults() == 1);
492 
493  Value redVal = parOp.getInitVals().front();
494  Value curVal;
495  if (redExp->getOperand(0) == redVal)
496  curVal = redExp->getOperand(1);
497  else if (redExp->getOperand(1) == redVal)
498  curVal = redExp->getOperand(0);
499  // One of the operands must be the init value (which is also the
500  // previous reduction value).
501  assert(curVal);
502  // The reduction expression should be the only user of the reduction val
503  // inside the parallel for.
504  unsigned numUsers = 0;
505  for (Operation *op : redVal.getUsers()) {
506  if (op->getParentOp() == parOp)
507  numUsers++;
508  }
509  assert(numUsers == 1);
510  (void)numUsers; // to silence unused variable warning in release build
511 
512  rewriter.setInsertionPointAfter(redExp);
513  auto redOp = rewriter.create<scf::ReduceOp>(loc, curVal);
514  // Attach to the reduction op.
515  Block *redBlock = &redOp.getRegion().getBlocks().front();
516  rewriter.setInsertionPointToEnd(redBlock);
517  Operation *newRed = rewriter.clone(*redExp);
518  // Replaces arguments of the reduction expression by using the block
519  // arguments from scf.reduce.
520  rewriter.updateRootInPlace(
521  newRed, [&]() { newRed->setOperands(redBlock->getArguments()); });
522  // Erases the out-dated reduction expression.
523  rewriter.eraseOp(redExp);
524  rewriter.setInsertionPointToEnd(redBlock);
525  rewriter.create<scf::ReduceReturnOp>(loc, newRed->getResult(0));
526  }
527  rewriter.setInsertionPointAfter(parOp);
528  // In-place update reduction variables.
529  for (unsigned i = 0, e = parOp.getResults().size(); i < e; i++)
530  reduc[i] = parOp.getResult(i);
531  }
532 
533  // Finished iterating a tensor, clean up
534  // We only do the clean up on for loop as while loops do not necessarily
535  // finish the iteration on a sparse tensor
536  for (auto [tid, dim] : llvm::zip(tids, dims)) {
537  // Reset to null.
538  coord[tid][dim] = Value();
539  pidxs[tid][dim] = Value();
540  // Dense dimension, high is fixed.
541  if (!isDenseDLT(dimTypes[tid][dim]))
542  highs[tid][dim] = Value();
543  }
544 }

546 void SparseTensorLoopEmitter::exitCoIterationLoop(
547  OpBuilder &builder, Location loc, MutableArrayRef<Value> reduc) {
548  auto whileOp = llvm::cast<scf::WhileOp>(loopStack.back().loop);
549  auto &dims = loopStack.back().dims;
550  auto &tids = loopStack.back().tids;
551  Value iv = loopStack.back().iv;
552  // Generation while loop induction at the end.
553  builder.setInsertionPointToEnd(&whileOp.getAfter().front());
554  // Finalize the induction. Note that the induction could be performed
555  // in the individual if-branches to avoid re-evaluating the conditions.
556  // However, that would result in a rather elaborate forest of yield
557  // instructions during code generation. Moreover, performing the induction
558  // after the if-statements more closely resembles code generated by TACO.
559  unsigned o = 0;
560  SmallVector<Value> operands;
561  Value one = constantIndex(builder, loc, 1);
562  for (auto [tid, dim] : llvm::zip(tids, dims)) {
563  if (isCompressedDLT(dimTypes[tid][dim]) ||
564  isSingletonDLT(dimTypes[tid][dim])) {
565  Value op1 = coord[tid][dim];
566  Value op3 = pidxs[tid][dim];
567  Value cmp =
568  builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq, op1, iv);
569  Value add = builder.create<arith::AddIOp>(loc, op3, one);
570  operands.push_back(builder.create<arith::SelectOp>(loc, cmp, add, op3));
571  // Following loops continue iteration from the break point of the
572  // current while loop.
573  pidxs[tid][dim] = whileOp->getResult(o++);
574  // The coordinates are invalid now.
575  coord[tid][dim] = nullptr;
576  // highs remains unchanged.
577  }
578  }
579 
580  // Reduction value from users.
581  for (auto &i : reduc) {
582  operands.push_back(i);
583  // In place update reduction variable.
584  i = whileOp->getResult(o++);
585  }
586 
587  // An (optional) universal index.
588  if (operands.size() < whileOp.getNumResults()) {
589  assert(operands.size() + 1 == whileOp.getNumResults());
590  // The last one is the universial index.
591  operands.push_back(builder.create<arith::AddIOp>(loc, iv, one));
592  // update the loop starting point of current loop sequence
593  loopSeqStack.back() = whileOp->getResult(o++);
594  }
595 
596  assert(o == operands.size());
597  builder.create<scf::YieldOp>(loc, operands);
598  builder.setInsertionPointAfter(whileOp);
599 }

780 void mlir::sparse_tensor::genReshapeDstShape(
781  Location loc, PatternRewriter &rewriter, SmallVectorImpl<Value> &dstShape,
782  ArrayRef<Value> srcShape, ArrayRef<int64_t> staticDstShape,
783  ArrayRef<ReassociationIndices> reassociation) {
784  // Collapse shape.
785  if (reassociation.size() < srcShape.size()) {
786  unsigned start = 0;
787  for (const auto &map : llvm::enumerate(reassociation)) {
788  auto dstDim = constantIndex(rewriter, loc, 1);
789  for (unsigned i = start; i < start + map.value().size(); i++) {
790  dstDim = rewriter.create<arith::MulIOp>(loc, dstDim, srcShape[i]);
791  }
792  dstShape.push_back(dstDim);
793  start = start + map.value().size();
794  }
795  assert(start == srcShape.size());
796  return;
797  }
798 
799  // Expand shape.
800  assert(reassociation.size() == srcShape.size());
801  unsigned start = 0;
802  // Expand the i-th dimension in srcShape.
803  for (unsigned i = 0, size = srcShape.size(); i < size; i++) {
804  const auto &map = reassociation[i];
805  auto srcDim = srcShape[i];
806  // Iterate through dimensions expanded from the i-th dimension.
807  for (unsigned j = start; j < start + map.size(); j++) {
808  // There can be only one dynamic sized dimension among dimensions
809  // expanded from the i-th dimension in srcShape.
810  // For example, if srcDim = 8, then the expanded shape could be <2x?x2>,
811  // but not <2x?x?>.
812  if (staticDstShape[j] == ShapedType::kDynamicSize) {
813  // The expanded dimension has dynamic size. We compute the dimension
814  // by dividing srcDim by the product of the static dimensions.
815  int64_t product = 1;
816  for (unsigned k = start; k < start + map.size(); k++) {
817  if (staticDstShape[k] != ShapedType::kDynamicSize) {
818  product *= staticDstShape[k];
819  }
820  }
821  // Compute the dynamic dimension size.
822  Value productVal = constantIndex(rewriter, loc, product);
823  Value dynamicSize =
824  rewriter.create<arith::DivUIOp>(loc, srcDim, productVal);
825  dstShape.push_back(dynamicSize);
826  } else {
827  // The expanded dimension is statically known.
828  dstShape.push_back(constantIndex(rewriter, loc, staticDstShape[j]));
829  }
830  }
831  start = start + map.size();
832  }
833  assert(start == staticDstShape.size());
834 }

836 void mlir::sparse_tensor::translateIndicesArray(
837  OpBuilder &builder, Location loc,
838  ArrayRef<ReassociationIndices> reassociation, ValueRange srcIndices,
839  ArrayRef<Value> srcShape, ArrayRef<Value> dstShape,
840  SmallVectorImpl<Value> &dstIndices) {
841  unsigned i = 0;
842  unsigned start = 0;
843  unsigned dstRank = dstShape.size();
844  unsigned srcRank = srcShape.size();
845  assert(srcRank == srcIndices.size());
846  bool isCollapse = srcRank > dstRank;
847  ArrayRef<Value> shape = isCollapse ? srcShape : dstShape;
848  // Iterate over reassociation map.
849  for (const auto &map : llvm::enumerate(reassociation)) {
850  // Prepare strides information in dimension slice.
851  Value linear = constantIndex(builder, loc, 1);
852  for (unsigned j = start, end = start + map.value().size(); j < end; j++) {
853  linear = builder.create<arith::MulIOp>(loc, linear, shape[j]);
854  }
855  // Start expansion.
856  Value val;
857  if (!isCollapse)
858  val = srcIndices[i];
859  // Iterate over dimension slice.
860  for (unsigned j = start, end = start + map.value().size(); j < end; j++) {
861  linear = builder.create<arith::DivUIOp>(loc, linear, shape[j]);
862  if (isCollapse) {
863  Value old = srcIndices[j];
864  Value mul = builder.create<arith::MulIOp>(loc, old, linear);
865  val = val ? builder.create<arith::AddIOp>(loc, val, mul) : mul;
866  } else {
867  Value old = val;
868  val = builder.create<arith::DivUIOp>(loc, val, linear);
869  assert(dstIndices.size() == j);
870  dstIndices.push_back(val);
871  val = builder.create<arith::RemUIOp>(loc, old, linear);
872  }
873  }
874  // Finalize collapse.
875  if (isCollapse) {
876  assert(dstIndices.size() == i);
877  dstIndices.push_back(val);
878  }
879  start += map.value().size();
880  i++;
881  }
882  assert(dstIndices.size() == dstRank);
883 }
