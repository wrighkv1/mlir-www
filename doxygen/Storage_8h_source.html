71 /// Abstract base class for `SparseTensorStorage<P,I,V>`. This class
72 /// takes responsibility for all the `<P,I,V>`-independent aspects
73 /// of the tensor (e.g., shape, sparsity, permutation). In addition,
74 /// we use function overloading to implement "partial" method
75 /// specialization, which the C-API relies on to catch type errors
76 /// arising from our use of opaque pointers.
77 ///
78 /// Because this class forms a bridge between the denotational semantics
79 /// of "tensors" and the operational semantics of how we store and
80 /// compute with them, it also distinguishes between two different
81 /// coordinate spaces (and their associated rank, shape, sizes, etc).
82 /// Denotationally, we have the *dimensions* of the tensor represented
83 /// by this object. Operationally, we have the *levels* of the storage
84 /// representation itself. We use this "dimension" vs "level" terminology
85 /// throughout, since alternative terminology like "tensor-dimension",
86 /// "original-dimension", "storage-dimension", etc, is both more verbose
87 /// and prone to introduce confusion whenever the qualifiers are dropped.
88 /// Where necessary, we use "axis" as the generic term.
89 ///
90 /// The *size* of an axis is the cardinality of possible coordinate/index
91 /// values along that axis (regardless of which coordinates have stored
92 /// element values). As such, each size must be non-zero since if any
93 /// axis has size-zero then the whole tensor would have trivial storage
94 /// (since there are no possible coordinates). Thus we use the plural
95 /// term *sizes* for a collection of non-zero cardinalities, and use
96 /// this term whenever referring to run-time cardinalities. Whereas we
97 /// use the term *shape* for a collection of compile-time cardinalities,
98 /// where zero is used to indicate cardinalities which are dynamic (i.e.,
99 /// unknown/unspecified at compile-time). At run-time, these dynamic
100 /// cardinalities will be inferred from or checked against sizes otherwise
101 /// specified. Thus, dynamic cardinalities always have an "immutable but
102 /// unknown" value; so the term "dynamic" should not be taken to indicate
103 /// run-time mutability.
104 //
105 // TODO: we'd like to factor out a class akin to `PermutationRef` for
106 // capturing known-valid sizes to avoid redundant validity assertions.
107 // But calling that class "SizesRef" would be a terrible name (and
108 // "ValidSizesRef" isn't much better). Whereas, calling it "ShapeRef"
109 // would be a lot nicer, but then that conflicts with the terminology
110 // introduced above. So we need to come up with some new terminology
111 // for distinguishing things, which allows a reasonable class name too.

422  /// Partially specialize lexicographical insertions based on template types.
423  void lexInsert(const uint64_t *lvlInd, V val) final {
424  assert(lvlInd && "Received nullptr for level-indices");
425  // First, wrap up pending insertion path.
426  uint64_t diffLvl = 0;
427  uint64_t topIdx = 0;
428  if (!values.empty()) {
429  diffLvl = lexDiff(lvlInd);
430  endPath(diffLvl + 1);
431  topIdx = lvlCursor[diffLvl] + 1;
432  }
433  // Then continue with insertion path.
434  insPath(lvlInd, diffLvl, topIdx, val);
435  }

437  /// Partially specialize expanded insertions based on template types.
438  void expInsert(uint64_t *lvlInd, V *values, bool *filled, uint64_t *added,
439  uint64_t count) final {
440  assert((lvlInd && values && filled && added) && "Received nullptr");
441  if (count == 0)
442  return;
443  // Sort.
444  std::sort(added, added + count);
445  // Restore insertion path for first insert.
446  const uint64_t lastLvl = getLvlRank() - 1;
447  uint64_t index = added[0];
448  assert(filled[index] && "added index is not filled");
449  lvlInd[lastLvl] = index;
450  lexInsert(lvlInd, values[index]);
451  values[index] = 0;
452  filled[index] = false;
453  // Subsequent insertions are quick.
454  for (uint64_t i = 1; i < count; ++i) {
455  assert(index < added[i] && "non-lexicographic insertion");
456  index = added[i];
457  assert(filled[index] && "added index is not filled");
458  lvlInd[lastLvl] = index;
459  insPath(lvlInd, lastLvl, added[i - 1] + 1, values[index]);
460  values[index] = 0;
461  filled[index] = false;
462  }
463  }

465  /// Finalizes lexicographic insertions.
466  void endInsert() final {
467  if (values.empty())
468  finalizeSegment(0);
469  else
470  endPath(0);

507  /// checks that `pos` is representable in the `P` type; however, it
508  /// does not check that `pos` is semantically valid (i.e., larger than
509  /// the previous position and smaller than `indices[l].capacity()`).
510  void appendPointer(uint64_t l, uint64_t pos, uint64_t count = 1) {
511  ASSERT_COMPRESSED_LVL(l);
512  // TODO: we'd like to recover the nicer error message:
513  // "Pointer value is too large for the P-type"
514  pointers[l].insert(pointers[l].end(), count,
515  detail::checkOverflowCast<P>(pos));
516  }
517 
518  /// Appends index `i` to level `l`, in the semantically general sense.
519  /// For non-dense levels, that means appending to the `indices[l]` array,
520  /// checking that `i` is representable in the `I` type; however, we do
521  /// not verify other semantic requirements (e.g., that `i` is in bounds
522  /// for `lvlSizes[l]`, and not previously occurring in the same segment).
523  /// For dense levels, this method instead appends the appropriate number
524  /// of zeros to the `values` array, where `full` is the number of "entries"
525  /// already written to `values` for this segment (aka one after the highest
526  /// index previously appended).
527  void appendIndex(uint64_t l, uint64_t full, uint64_t i) {
528  const auto dlt = getLvlType(l); // Avoid redundant bounds checking.
529  if (isCompressedDLT(dlt) || isSingletonDLT(dlt)) {
530  // TODO: we'd like to recover the nicer error message:
531  // "Index value is too large for the I-type"
532  indices[l].push_back(detail::checkOverflowCast<I>(i));
533  } else { // Dense dimension.
534  ASSERT_DENSE_DLT(dlt);
535  assert(i >= full && "Index was already filled");
536  if (i == full)
537  return; // Short-circuit, since it'll be a nop.
538  if (l + 1 == getLvlRank())
539  values.insert(values.end(), i - full, 0);
540  else
541  finalizeSegment(l + 1, 0, i - full);
542  }
543  }
544 
545  /// Writes the given coordinate to `indices[l][pos]`. This method
546  /// checks that `i` is representable in the `I` type; however, it
547  /// does not check that `i` is semantically valid (i.e., in bounds
548  /// for `dimSizes[l]` and not elsewhere occurring in the same segment).
549  void writeIndex(uint64_t l, uint64_t pos, uint64_t i) {
550  ASSERT_COMPRESSED_OR_SINGLETON_LVL(l);
551  // Subscript assignment to `std::vector` requires that the `pos`-th
552  // entry has been initialized; thus we must be sure to check `size()`
553  // here, instead of `capacity()` as would be ideal.
554  assert(pos < indices[l].size() && "Index position is out of bounds");
555  // TODO: we'd like to recover the nicer error message:
556  // "Index value is too large for the I-type"
557  indices[l][pos] = detail::checkOverflowCast<I>(i);
558  }
559 
560  /// Computes the assembled-size associated with the `l`-th level,
561  /// given the assembled-size associated with the `(l-1)`-th level.
562  /// "Assembled-sizes" correspond to the (nominal) sizes of overhead
563  /// storage, as opposed to "level-sizes" which are the cardinality
564  /// of possible coordinates for that level.
565  ///
566  /// Precondition: the `pointers[l]` array must be fully initialized
567  /// before calling this method.
568  uint64_t assembledSize(uint64_t parentSz, uint64_t l) const {
569  const auto dlt = getLvlType(l); // Avoid redundant bounds checking.
570  if (isCompressedDLT(dlt))
571  return pointers[l][parentSz];
572  if (isSingletonDLT(dlt))
573  return parentSz; // New size is same as the parent.
574  if (isDenseDLT(dlt))
575  return parentSz * getLvlSizes()[l];
576  MLIR_SPARSETENSOR_FATAL("unsupported level type: %d\n",
577  static_cast<uint8_t>(dlt));
578  }

580  /// Initializes sparse tensor storage scheme from a memory-resident sparse
581  /// tensor in coordinate scheme. This method prepares the pointers and
582  /// indices arrays under the given per-dimension dense/sparse annotations.
583  ///
584  /// Preconditions:
585  /// * the `lvlElements` must be lexicographically sorted.
586  /// * the indices of every element are valid for `getLvlSizes()`
587  /// (i.e., equal rank and pointwise less-than).
588  void fromCOO(const std::vector<Element<V>> &lvlElements, uint64_t lo,
589  uint64_t hi, uint64_t l) {
590  const uint64_t lvlRank = getLvlRank();
591  assert(l <= lvlRank && hi <= lvlElements.size());
592  // Once dimensions are exhausted, insert the numerical values.
593  if (l == lvlRank) {
594  assert(lo < hi);
595  values.push_back(lvlElements[lo].value);
596  return;
597  }
598  // Visit all elements in this interval.
599  uint64_t full = 0;
600  while (lo < hi) { // If `hi` is unchanged, then `lo < lvlElements.size()`.
601  // Find segment in interval with same index elements in this level.
602  const uint64_t i = lvlElements[lo].indices[l];
603  uint64_t seg = lo + 1;
604  if (isUniqueLvl(l))
605  while (seg < hi && lvlElements[seg].indices[l] == i)
606  ++seg;
607  // Handle segment in interval for sparse or dense level.
608  appendIndex(l, full, i);
609  full = i + 1;
610  fromCOO(lvlElements, lo, seg, l + 1);
611  // And move on to next segment in interval.
612  lo = seg;
613  }
614  // Finalize the sparse pointer structure at this level.
615  finalizeSegment(l, full);
616  }
617 
618  /// Finalizes the sparse pointer structure at this level.
619  void finalizeSegment(uint64_t l, uint64_t full = 0, uint64_t count = 1) {
620  if (count == 0)
621  return; // Short-circuit, since it'll be a nop.
622  const auto dlt = getLvlType(l); // Avoid redundant bounds checking.
623  if (isCompressedDLT(dlt)) {
624  appendPointer(l, indices[l].size(), count);
625  } else if (isSingletonDLT(dlt)) {
626  return; // Nothing to finalize.
627  } else { // Dense dimension.
628  ASSERT_DENSE_DLT(dlt);
629  const uint64_t sz = getLvlSizes()[l];
630  assert(sz >= full && "Segment is overfull");
631  count = detail::checkedMul(count, sz - full);
632  // For dense storage we must enumerate all the remaining coordinates
633  // in this level (i.e., coordinates after the last non-zero
634  // element), and either fill in their zero values or else recurse
635  // to finalize some deeper level.
636  if (l + 1 == getLvlRank())
637  values.insert(values.end(), count, 0);
638  else
639  finalizeSegment(l + 1, 0, count);
640  }
641  }
642 
643  /// Wraps up a single insertion path, inner to outer.
644  void endPath(uint64_t diffLvl) {
645  const uint64_t lvlRank = getLvlRank();
646  const uint64_t lastLvl = lvlRank - 1;
647  assert(diffLvl <= lvlRank && "Level-diff is out of bounds");
648  const uint64_t stop = lvlRank - diffLvl;
649  for (uint64_t i = 0; i < stop; ++i) {
650  const uint64_t l = lastLvl - i;
651  finalizeSegment(l, lvlCursor[l] + 1);
652  }
653  }
654 
655  /// Continues a single insertion path, outer to inner. The first
656  /// argument is the storage-level indices for the value being inserted.
657  void insPath(const uint64_t *lvlInd, uint64_t diffLvl, uint64_t topIdx,
658  V val) {
659  const uint64_t lvlRank = getLvlRank();
660  assert(diffLvl <= lvlRank && "Level-diff is out of bounds");
661  for (uint64_t l = diffLvl; l < lvlRank; ++l) {
662  const uint64_t i = lvlInd[l];
663  appendIndex(l, topIdx, i);
664  topIdx = 0;
665  lvlCursor[l] = i;
666  }
667  values.push_back(val);
668  }
669 
670  /// Finds the lexicographically first level where the level-indices
671  /// in the argument differ from those in the current cursor.
672  uint64_t lexDiff(const uint64_t *lvlInd) const {
673  const uint64_t lvlRank = getLvlRank();
674  for (uint64_t l = 0; l < lvlRank; ++l)
675  if (lvlInd[l] > lvlCursor[l])
676  return l;
677  else
678  assert(lvlInd[l] == lvlCursor[l] && "non-lexicographic insertion");
679  assert(0 && "duplicate insertion");
680  return -1u;
681  }

685  // making them public to other client code.

687 
688  std::vector<std::vector<P>> pointers;
689  std::vector<std::vector<I>> indices;
690  std::vector<V> values;
691  std::vector<uint64_t> lvlCursor; // cursor for lexicographic insertion.

699 /// A (higher-order) function object for enumerating the elements of some
700 /// `SparseTensorStorage` under a permutation. That is, the `forallElements`
701 /// method encapsulates the loop-nest for enumerating the elements of
702 /// the source tensor (in whatever order is best for the source tensor),
703 /// and applies a permutation to the coordinates/indices before handing
704 /// each element to the callback. A single enumerator object can be
705 /// freely reused for several calls to `forallElements`, just so long
706 /// as each call is sequential with respect to one another.
707 ///
708 /// N.B., this class stores a reference to the `SparseTensorStorageBase`
709 /// passed to the constructor; thus, objects of this class must not
710 /// outlive the sparse tensor they depend on.
711 ///
712 /// Design Note: The reason we define this class instead of simply using
713 /// `SparseTensorEnumerator<P,I,V>` is because we need to hide/generalize
714 /// the `<P,I>` template parameters from MLIR client code (to simplify the
715 /// type parameters used for direct sparse-to-sparse conversion). And the
716 /// reason we define the `SparseTensorEnumerator<P,I,V>` subclasses rather
717 /// than simply using this class, is to avoid the cost of virtual-method
718 /// dispatch within the loop-nest.
719 template <typename V>
720 class MLIR_SPARSETENSOR_GSL_POINTER [[nodiscard]] SparseTensorEnumeratorBase {
721 public:
722  /// Constructs an enumerator which automatically applies the given
723  /// mapping from the source tensor's dimensions to the desired
724  /// target tensor dimensions.
725  ///
726  /// Preconditions:
727  /// * the `src` must have the same `V` value type.
728  /// * `trgSizes` must be valid for `trgRank`.
729  /// * `src2trg` must be valid for `srcRank`, and must map indices
730  /// valid for `src.getDimSizes()` to indices valid for `trgSizes`.
731  ///
732  /// Asserts:
733  /// * `trgSizes` must be nonnull and must contain only nonzero sizes.
734  /// * `srcRank == src.getDimRank()`.
735  /// * `src2trg` must be nonnull.
736  SparseTensorEnumeratorBase(const SparseTensorStorageBase &src,
737  uint64_t trgRank, const uint64_t *trgSizes,
738  uint64_t srcRank, const uint64_t *src2trg)
739  : src(src), trgSizes(trgSizes, trgSizes + trgRank),
740  lvl2trg(src.getLvlRank()), trgCursor(trgRank) {
741  assert(trgSizes && "Received nullptr for target-sizes");
742  assert(src2trg && "Received nullptr for source-to-target mapping");
743  assert(srcRank == src.getDimRank() && "Source-rank mismatch");
744  for (uint64_t t = 0; t < trgRank; ++t)
745  assert(trgSizes[t] > 0 && "Target-size zero has trivial storage");
746  const auto &lvl2src = src.getLvl2Dim();
747  for (uint64_t lvlRank = src.getLvlRank(), l = 0; l < lvlRank; ++l)
748  lvl2trg[l] = src2trg[lvl2src[l]];
749  }
750 
751  virtual ~SparseTensorEnumeratorBase() = default;
752 
753  // We disallow copying to help avoid leaking the `src` reference.
754  // (In addition to avoiding the problem of slicing.)
755  SparseTensorEnumeratorBase(const SparseTensorEnumeratorBase &) = delete;
756  SparseTensorEnumeratorBase &
757  operator=(const SparseTensorEnumeratorBase &) = delete;
758 
759  /// Gets the source's dimension-rank.
760  uint64_t getSrcDimRank() const { return src.getDimRank(); }
761 
762  /// Gets the target's dimension-/level-rank. (This is usually
763  /// "dimension-rank", though that may coincide with "level-rank"
764  /// depending on usage.)
765  uint64_t getTrgRank() const { return trgSizes.size(); }
766 
767  /// Gets the target's dimension-/level-sizes. (These are usually
768  /// "dimensions", though that may coincide with "level-rank" depending
769  /// on usage.)
770  const std::vector<uint64_t> &getTrgSizes() const { return trgSizes; }
771 
772  /// Enumerates all elements of the source tensor, permutes their
773  /// indices, and passes the permuted element to the callback.
774  /// The callback must not store the cursor reference directly,
775  /// since this function reuses the storage. Instead, the callback
776  /// must copy it if they want to keep it.
777  virtual void forallElements(ElementConsumer<V> yield) = 0;
778 
779 protected:
780  const SparseTensorStorageBase &src;
781  std::vector<uint64_t> trgSizes; // in target order.
782  std::vector<uint64_t> lvl2trg; // source-levels -> target-dims/lvls.
783  std::vector<uint64_t> trgCursor; // in target order.
784 };

787 template <typename P, typename I, typename V>
788 class MLIR_SPARSETENSOR_GSL_POINTER [[nodiscard]] SparseTensorEnumerator final
789  : public SparseTensorEnumeratorBase<V> {
790  using Base = SparseTensorEnumeratorBase<V>;
791  using StorageImpl = SparseTensorStorage<P, I, V>;
792 
793 public:
794  /// Constructs an enumerator which automatically applies the given
795  /// mapping from the source tensor's dimensions to the desired
796  /// target tensor dimensions.
797  ///
798  /// Preconditions/assertions are as per the `SparseTensorEnumeratorBase` ctor.
799  SparseTensorEnumerator(const StorageImpl &src, uint64_t trgRank,
800  const uint64_t *trgSizes, uint64_t srcRank,
801  const uint64_t *src2trg)
802  : Base(src, trgRank, trgSizes, srcRank, src2trg) {}
803 
804  ~SparseTensorEnumerator() final = default;
805 
806  void forallElements(ElementConsumer<V> yield) final {
807  forallElements(yield, 0, 0);
808  }
809 
810 private:
811  // TODO: Once we functionalize the mappings, then we'll no longer
812  // be able to use the current approach of constructing `lvl2trg` in the
813  // ctor and using it to incrementally fill the `trgCursor` cursor as we
814  // recurse through `forallElements`. Instead we'll want to incrementally
815  // fill a `lvlCursor` as we recurse, and then use `src.getLvl2Dim()`
816  // and `src2trg` to convert it just before yielding to the callback.
817  // It's probably most efficient to just store the `srcCursor` and
818  // `trgCursor` buffers in this object, but we may want to benchmark
819  // that against using `std::calloc` to stack-allocate them instead.
820  //
821  /// The recursive component of the public `forallElements`.
822  void forallElements(ElementConsumer<V> yield, uint64_t parentPos,
823  uint64_t l) {
824  // Recover the `<P,I,V>` type parameters of `src`.
825  const auto &src = static_cast<const StorageImpl &>(this->src);
826  if (l == src.getLvlRank()) {
827  assert(parentPos < src.values.size() &&
828  "Value position is out of bounds");
829  // TODO: <https://github.com/llvm/llvm-project/issues/54179>
830  yield(this->trgCursor, src.values[parentPos]);
831  return;
832  }
833  uint64_t &cursorL = this->trgCursor[this->lvl2trg[l]];
834  const auto dlt = src.getLvlType(l); // Avoid redundant bounds checking.
835  if (isCompressedDLT(dlt)) {
836  // Look up the bounds of the `l`-level segment determined by the
837  // `(l - 1)`-level position `parentPos`.
838  const std::vector<P> &pointersL = src.pointers[l];
839  assert(parentPos + 1 < pointersL.size() &&
840  "Parent pointer position is out of bounds");
841  const uint64_t pstart = static_cast<uint64_t>(pointersL[parentPos]);
842  const uint64_t pstop = static_cast<uint64_t>(pointersL[parentPos + 1]);
843  // Loop-invariant code for looking up the `l`-level coordinates/indices.
844  const std::vector<I> &indicesL = src.indices[l];
845  assert(pstop <= indicesL.size() && "Index position is out of bounds");
846  for (uint64_t pos = pstart; pos < pstop; ++pos) {
847  cursorL = static_cast<uint64_t>(indicesL[pos]);
848  forallElements(yield, pos, l + 1);
849  }
850  } else if (isSingletonDLT(dlt)) {
851  cursorL = src.getIndex(l, parentPos);
852  forallElements(yield, parentPos, l + 1);
853  } else { // Dense dimension.
854  ASSERT_DENSE_DLT(dlt);
855  const uint64_t sz = src.getLvlSizes()[l];
856  const uint64_t pstart = parentPos * sz;
857  for (uint64_t i = 0; i < sz; ++i) {
858  cursorL = i;
859  forallElements(yield, pstart + i, l + 1);
860  }
861  }
862  }
863 };

867 /// a source tensor, for direct sparse=>sparse conversion a la
868 /// <https://arxiv.org/abs/2001.02609>.

870 /// N.B., this class stores references to the parameters passed to

872 /// those parameters.

874 /// This class does not have the "dimension" vs "level" distinction, but
875 /// since it is used for initializing the levels of a `SparseTensorStorage`

878 public:
879  /// Allocates the statistics structure for the desired target-tensor
880  /// level structure (i.e., sizes and types). This constructor does not
881  /// actually populate the statistics, however; for that see `initialize`.

891

894 
895  /// Enumerates the source tensor to fill in the statistics.
896  /// The enumerator should already incorporate the mapping from
897  /// the source tensor-dimensions to the target storage-levels.
898  ///
899  /// Asserts:
900  /// * `enumerator.getTrgRank() == getLvlRank()`.
901  /// * `enumerator.getTrgSizes() == lvlSizes`.

905  assert(enumerator.getTrgSizes() == lvlSizes && "Tensor size mismatch");

909

913  /// Lexicographically enumerates all indicies for levels strictly
914  /// less than `stopLvl`, and passes their nnz statistic to the callback.
915  /// Since our use-case only requires the statistic not the coordinates
916  /// themselves, we do not bother to construct those coordinates.

918

921  /// a method rather than inlining into the lambda in `initialize`,
922  /// to avoid spurious templating over `V`. And this method is private
923  /// to avoid needing to re-assert validity of `lvlInd` (which is

929  uint64_t l) const;
930 
931  // All of these are in the target storage-order.
932  const std::vector<uint64_t> &lvlSizes;
933  const std::vector<DimLevelType> &lvlTypes;
934  std::vector<std::vector<uint64_t>> nnz;
935 };

994  : SparseTensorStorage(dimRank, dimSizes, lvlRank, lvlSizes, lvlTypes,

996  // Provide hints on capacity of pointers and indices.
997  // TODO: needs much fine-tuning based on actual sparsity; currently
998  // we reserve pointer/index space based on all previous dense
999  // dimensions, which works well up to first sparse dim; but
1000  // we should really use nnz and dense/sparse distribution.

1007  // `sz` by that before reserving. (For now we just use 1.)
1008  pointers[l].reserve(sz + 1);
1009  pointers[l].push_back(0);
1010  indices[l].reserve(sz);
1011  sz = 1;
1012  allDense = false;
1013  } else if (isSingletonDLT(dlt)) {
1014  indices[l].reserve(sz);
1015  sz = 1;
1016  allDense = false;
1017  } else { // Dense dimension.
1018  ASSERT_DENSE_DLT(dlt);
1019  sz = detail::checkedMul(sz, lvlSizes[l]);

1021  }

1045 template <typename P, typename I, typename V>
1046 SparseTensorStorage<P, I, V>::SparseTensorStorage(
1047  uint64_t dimRank, const uint64_t *dimSizes, uint64_t lvlRank,
1048  const DimLevelType *lvlTypes, const uint64_t *lvl2dim,
1049  SparseTensorEnumeratorBase<V> &lvlEnumerator)
1050  : SparseTensorStorage(dimRank, dimSizes, lvlRank,
1051  lvlEnumerator.getTrgSizes().data(), lvlTypes,
1052  lvl2dim) {
1053  assert(lvlRank == lvlEnumerator.getTrgRank() && "Level-rank mismatch");
1054  {
1055  // Initialize the statistics structure.
1056  SparseTensorNNZ nnz(getLvlSizes(), getLvlTypes());
1057  nnz.initialize(lvlEnumerator);
1058  // Initialize "pointers" overhead (and allocate "indices", "values").
1059  uint64_t parentSz = 1; // assembled-size of the `(l - 1)`-level.
1060  for (uint64_t l = 0; l < lvlRank; ++l) {
1061  const auto dlt = lvlTypes[l]; // Avoid redundant bounds checking.
1062  if (isCompressedDLT(dlt)) {
1063  pointers[l].reserve(parentSz + 1);
1064  pointers[l].push_back(0);
1065  uint64_t currentPos = 0;
1066  nnz.forallIndices(l, [this, &currentPos, l](uint64_t n) {
1067  currentPos += n;
1068  appendPointer(l, currentPos);
1069  });
1070  assert(pointers[l].size() == parentSz + 1 &&
1071  "Final pointers size doesn't match allocated size");
1072  // That assertion entails `assembledSize(parentSz, l)`
1073  // is now in a valid state. That is, `pointers[l][parentSz]`
1074  // equals the present value of `currentPos`, which is the
1075  // correct assembled-size for `indices[l]`.
1076  }
1077  // Update assembled-size for the next iteration.
1078  parentSz = assembledSize(parentSz, l);
1079  // Ideally we need only `indices[l].reserve(parentSz)`, however
1080  // the `std::vector` implementation forces us to initialize it too.
1081  // That is, in the yieldPos loop we need random-access assignment
1082  // to `indices[l]`; however, `std::vector`'s subscript-assignment
1083  // only allows assigning to already-initialized positions.
1084  if (isCompressedDLT(dlt) || isSingletonDLT(dlt))
1085  indices[l].resize(parentSz, 0);
1086  else
1087  ASSERT_DENSE_DLT(dlt); // Future-proofing.
1088  }
1089  values.resize(parentSz, 0); // Both allocate and zero-initialize.
1090  }
1091  // The yieldPos loop
1092  lvlEnumerator.forallElements([this](const auto &lvlInd, V val) {
1093  uint64_t parentSz = 1, parentPos = 0;
1094  for (uint64_t lvlRank = getLvlRank(), l = 0; l < lvlRank; ++l) {
1095  const auto dlt = getLvlTypes()[l]; // Avoid redundant bounds checking.
1096  if (isCompressedDLT(dlt)) {
1097  // If `parentPos == parentSz` then it's valid as an array-lookup;
1098  // however, it's semantically invalid here since that entry
1099  // does not represent a segment of `indices[l]`. Moreover, that
1100  // entry must be immutable for `assembledSize` to remain valid.
1101  assert(parentPos < parentSz && "Pointers position is out of bounds");
1102  const uint64_t currentPos = pointers[l][parentPos];
1103  // This increment won't overflow the `P` type, since it can't
1104  // exceed the original value of `pointers[l][parentPos+1]`
1105  // which was already verified to be within bounds for `P`
1106  // when it was written to the array.
1107  pointers[l][parentPos]++;
1108  writeIndex(l, currentPos, lvlInd[l]);
1109  parentPos = currentPos;
1110  } else if (isSingletonDLT(dlt)) {
1111  writeIndex(l, parentPos, lvlInd[l]);
1112  // the new parentPos equals the old parentPos.
1113  } else { // Dense dimension.
1114  ASSERT_DENSE_DLT(dlt);
1115  parentPos = parentPos * getLvlSizes()[l] + lvlInd[l];
1116  }
1117  parentSz = assembledSize(parentSz, l);
1118  }
1119  assert(parentPos < values.size() && "Value position is out of bounds");
1120  values[parentPos] = val;
1121  });
1122  // The finalizeYieldPos loop
1123  for (uint64_t parentSz = 1, l = 0; l < lvlRank; ++l) {
1124  const auto dlt = lvlTypes[l]; // Avoid redundant bounds checking.
1125  if (isCompressedDLT(dlt)) {
1126  assert(parentSz == pointers[l].size() - 1 &&
1127  "Actual pointers size doesn't match the expected size");
1128  // Can't check all of them, but at least we can check the last one.
1129  assert(pointers[l][parentSz - 1] == pointers[l][parentSz] &&
1130  "Pointers got corrupted");
1131  // TODO: optimize this by using `memmove` or similar.
1132  for (uint64_t n = 0; n < parentSz; ++n) {
1133  const uint64_t parentPos = parentSz - n;
1134  pointers[l][parentPos] = pointers[l][parentPos - 1];
1135  }
1136  pointers[l][0] = 0;
1137  } else {
1138  // Both dense and singleton are no-ops for the finalizeYieldPos loop.
1139  // This assertion is for future-proofing.
1140  assert((isDenseDLT(dlt) || isSingletonDLT(dlt)) &&
1141  "Level is neither dense nor singleton");
1142  }
1143  parentSz = assembledSize(parentSz, l);
1144  }
1145 }
