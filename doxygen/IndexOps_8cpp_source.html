29 Operation *IndexDialect::materializeConstant(OpBuilder &b, Attribute value,
30  Type type, Location loc) {
31  // Materialize bool constants as `i1`.
32  if (auto boolValue = dyn_cast<BoolAttr>(value)) {
33  if (!type.isSignlessInteger(1))
34  return nullptr;
35  return b.create<BoolConstantOp>(loc, type, boolValue);
36  }
37 
38  // Materialize integer attributes as `index`.
39  if (auto indexValue = dyn_cast<IntegerAttr>(value)) {
40  if (!indexValue.getType().isa<IndexType>() || !type.isa<IndexType>())
41  return nullptr;
42  assert(indexValue.getValue().getBitWidth() ==
43  IndexType::kInternalStorageBitWidth);
44  return b.create<ConstantOp>(loc, indexValue);
45  }
46 
47  return nullptr;
48 }

54 /// Fold an index operation irrespective of the target bitwidth. The
55 /// operation must satisfy the property:
56 ///
57 /// ```
58 /// trunc(f(a, b)) = f(trunc(a), trunc(b))
59 /// ```
60 ///
61 /// For all values of `a` and `b`. The function accepts a lambda that computes
62 /// the integer result, which in turn must satisfy the above property.
63 static OpFoldResult foldBinaryOpUnchecked(
64  ArrayRef<Attribute> operands,
65  function_ref<Optional<APInt>(const APInt &, const APInt &)> calculate) {
66  assert(operands.size() == 2 && "binary operation expected 2 operands");
67  auto lhs = dyn_cast_if_present<IntegerAttr>(operands[0]);
68  auto rhs = dyn_cast_if_present<IntegerAttr>(operands[1]);
69  if (!lhs || !rhs)
70  return {};
71 
72  Optional<APInt> result = calculate(lhs.getValue(), rhs.getValue());
73  if (!result)
74  return {};
75  assert(result->trunc(32) ==
76  calculate(lhs.getValue().trunc(32), rhs.getValue().trunc(32)));
77  return IntegerAttr::get(IndexType::get(lhs.getContext()), *result);
78 }
79 
80 /// Fold an index operation only if the truncated 64-bit result matches the
81 /// 32-bit result for operations that don't satisfy the above property. These
82 /// are operations where the upper bits of the operands can affect the lower
83 /// bits of the results.
84 ///
85 /// The function accepts a lambda that computes the integer result in both
86 /// 64-bit and 32-bit. If either call returns `None`, the operation is not
87 /// folded.
88 static OpFoldResult foldBinaryOpChecked(
89  ArrayRef<Attribute> operands,
90  function_ref<Optional<APInt>(const APInt &, const APInt &lhs)> calculate) {
91  assert(operands.size() == 2 && "binary operation expected 2 operands");
92  auto lhs = dyn_cast_if_present<IntegerAttr>(operands[0]);
93  auto rhs = dyn_cast_if_present<IntegerAttr>(operands[1]);
94  // Only fold index operands.
95  if (!lhs || !rhs)
96  return {};
97 
98  // Compute the 64-bit result and the 32-bit result.
99  Optional<APInt> result64 = calculate(lhs.getValue(), rhs.getValue());
100  if (!result64)
101  return {};
102  Optional<APInt> result32 =
103  calculate(lhs.getValue().trunc(32), rhs.getValue().trunc(32));
104  if (!result32)
105  return {};
106  // Compare the truncated 64-bit result to the 32-bit result.
107  if (result64->trunc(32) != *result32)
108  return {};
109  // The operation can be folded for these particular operands.
110  return IntegerAttr::get(IndexType::get(lhs.getContext()), *result64);
111 }

172 /// Compute `ceildivs(n, m)` as `x = m > 0 ? -1 : 1` and then
173 /// `n*m > 0 ? (n+x)/m + 1 : -(-n/m)`.
174 static Optional<APInt> calculateCeilDivS(const APInt &n, const APInt &m) {
175  // Don't fold division by zero.
176  if (m.isZero())
177  return None;
178  // Short-circuit the zero case.
179  if (n.isZero())
180  return n;
181 
182  bool mGtZ = m.sgt(0);
183  if (n.sgt(0) != mGtZ) {
184  // If the operands have different signs, compute the negative result. Signed
185  // division overflow is not possible, since if `m == -1`, `n` can be at most
186  // `INT_MAX`, and `-INT_MAX != INT_MIN` in two's complement.
187  return -(-n).sdiv(m);
188  }
189  // Otherwise, compute the positive result. Signed division overflow is not
190  // possible since if `m == -1`, `x` will be `1`.
191  int64_t x = mGtZ ? -1 : 1;
192  return (n + x).sdiv(m) + 1;
193 }

204  // Compute `ceildivu(n, m)` as `n == 0 ? 0 : (n-1)/m + 1`.
205  return foldBinaryOpChecked(

211  if (n.isZero())
212  return n;
213 
214  return (n - 1).udiv(m) + 1;
215  });

222 /// Compute `floordivs(n, m)` as `x = m < 0 ? 1 : -1` and then
223 /// `n*m < 0 ? -1 - (x-n)/m : n/m`.
224 static Optional<APInt> calculateFloorDivS(const APInt &n, const APInt &m) {
225  // Don't fold division by zero.
226  if (m.isZero())
227  return None;
228  // Short-circuit the zero case.
229  if (n.isZero())
230  return n;
231 
232  bool mLtZ = m.slt(0);
233  if (n.slt(0) == mLtZ) {
234  // If the operands have the same sign, compute the positive result.
235  return n.sdiv(m);
236  }
237  // If the operands have different signs, compute the negative result. Signed
238  // division overflow is not possible since if `m == -1`, `x` will be 1 and
239  // `n` can be at most `INT_MAX`.
240  int64_t x = mLtZ ? 1 : -1;
241  return -1 - (x - n).sdiv(m);
242 }

292 OpFoldResult ShlOp::fold(ArrayRef<Attribute> operands) {
293  return foldBinaryOpUnchecked(
294  operands, [](const APInt &lhs, const APInt &rhs) -> Optional<APInt> {
295  // We cannot fold if the RHS is greater than or equal to 32 because
296  // this would be UB in 32-bit systems but not on 64-bit systems. RHS is
297  // already treated as unsigned.
298  if (rhs.uge(32))
299  return {};
300  return lhs << rhs;
301  });
302 }

352 /// Compare two integers according to the comparison predicate.
353 bool compareIndices(const APInt &lhs, const APInt &rhs,
354  IndexCmpPredicate pred) {
355  switch (pred) {
356  case IndexCmpPredicate::EQ:
357  return lhs.eq(rhs);
358  case IndexCmpPredicate::NE:
359  return lhs.ne(rhs);
360  case IndexCmpPredicate::SGE:
361  return lhs.sge(rhs);
362  case IndexCmpPredicate::SGT:
363  return lhs.sgt(rhs);
364  case IndexCmpPredicate::SLE:
365  return lhs.sle(rhs);
366  case IndexCmpPredicate::SLT:
367  return lhs.slt(rhs);
368  case IndexCmpPredicate::UGE:
369  return lhs.uge(rhs);
370  case IndexCmpPredicate::UGT:
371  return lhs.ugt(rhs);
372  case IndexCmpPredicate::ULE:
373  return lhs.ule(rhs);
374  case IndexCmpPredicate::ULT:
375  return lhs.ult(rhs);
376  }
377  llvm_unreachable("unhandled IndexCmpPredicate predicate");
378 }

380 OpFoldResult CmpOp::fold(ArrayRef<Attribute> operands) {
381  assert(operands.size() == 2 && "compare expected 2 operands");
382  auto lhs = dyn_cast_if_present<IntegerAttr>(operands[0]);
383  auto rhs = dyn_cast_if_present<IntegerAttr>(operands[1]);
384  if (!lhs || !rhs)
385  return {};
386 
387  // Perform the comparison in 64-bit and 32-bit.
388  bool result64 = compareIndices(lhs.getValue(), rhs.getValue(), getPred());
389  bool result32 = compareIndices(lhs.getValue().trunc(32),
390  rhs.getValue().trunc(32), getPred());
391  if (result64 != result32)
392  return {};
393  return BoolAttr::get(getContext(), result64);
394 }
