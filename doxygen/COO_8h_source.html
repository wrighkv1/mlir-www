29 /// An element of a sparse tensor in coordinate-scheme representation
30 /// (i.e., a pair of indices and value). For example, a rank-1 vector
31 /// element would look like
32 /// ({i}, a[i])
33 /// and a rank-5 tensor element would look like
34 /// ({i,j,k,l,m}, a[i,j,k,l,m])
35 ///
36 /// The indices are represented as a (non-owning) pointer into a shared
37 /// pool of indices, rather than being stored directly in this object.
38 /// This significantly improves performance because it: (1) reduces
39 /// the per-element memory footprint, and (2) centralizes the memory
40 /// management for indices. The only downside is that the indices
41 /// themselves cannot be retrieved without knowing the rank of the
42 /// tensor to which this element belongs (and that rank is not stored
43 /// in this object).
44 template <typename V>
45 struct Element final {
46  Element(const uint64_t *ind, V val) : indices(ind), value(val){};
47  const uint64_t *indices; // pointer into shared index pool
48  V value;
49 };
50 
51 /// Closure object for `operator<` on `Element` with a given rank.
52 template <typename V>
53 struct ElementLT final {
54  ElementLT(uint64_t rank) : rank(rank) {}
55 
56  /// Compares two elements a la `operator<`.
57  ///
58  /// Precondition: the elements must both be valid for `rank`.
59  bool operator()(const Element<V> &e1, const Element<V> &e2) const {
60  for (uint64_t d = 0; d < rank; ++d) {
61  if (e1.indices[d] == e2.indices[d])
62  continue;
63  return e1.indices[d] < e2.indices[d];
64  }
65  return false;
66  }
67 
68  const uint64_t rank;
69 };

78 /// A memory-resident sparse tensor in coordinate-scheme representation
79 /// (a collection of `Element`s). This data structure is used as
80 /// an intermediate representation; e.g., for reading sparse tensors
81 /// from external formats into memory, or for certain conversions between
82 /// different `SparseTensorStorage` formats.
83 ///
84 /// This class provides all the typedefs required by the "Container"
85 /// concept (<https://en.cppreference.com/w/cpp/named_req/Container>);
86 /// however, beware that it cannot fully implement that concept since
87 /// it cannot have a default ctor (because the `dimSizes` field is const).
88 /// Thus these typedefs are provided for familiarity reasons, rather
89 /// than as a proper implementation of the concept.
90 template <typename V>
91 class SparseTensorCOO final {
92 public:
93  using value_type = const Element<V>;
94  using reference = value_type &;
95  using const_reference = reference;
96  // The types associated with `std::vector` differ significantly between
97  // C++11/17 vs C++20; so we explicitly defer to whatever `std::vector`
98  // says the types should be.
99  using vector_type = std::vector<Element<V>>;
100  using iterator = typename vector_type::const_iterator;
101  using const_iterator = iterator;
102  using difference_type = typename vector_type::difference_type;
103  using size_type = typename vector_type::size_type;
104 
105  /// Constructs a new coordinate-scheme sparse tensor with the given
106  /// sizes and initial storage capacity.
107  ///
108  /// Asserts:
109  /// * `dimSizes` has nonzero size.
110  /// * the elements of `dimSizes` are nonzero.
111  explicit SparseTensorCOO(const std::vector<uint64_t> &dimSizes,
112  uint64_t capacity = 0)
113  : SparseTensorCOO(dimSizes.size(), dimSizes.data(), capacity) {}
114 
115  // TODO: make a class for capturing known-valid sizes (a la PermutationRef),
116  // so that `SparseTensorStorage::toCOO` can avoid redoing these assertions.
117  // Also so that we can enforce the asserts *before* copying into `dimSizes`.
118  //
119  /// Constructs a new coordinate-scheme sparse tensor with the given
120  /// sizes and initial storage capacity.
121  ///
122  /// Precondition: `dimSizes` must be valid for `dimRank`.
123  ///
124  /// Asserts:
125  /// * `dimRank` is nonzero.
126  /// * the elements of `dimSizes` are nonzero.
127  explicit SparseTensorCOO(uint64_t dimRank, const uint64_t *dimSizes,
128  uint64_t capacity = 0)
129  : dimSizes(dimSizes, dimSizes + dimRank), isSorted(true) {
130  assert(dimRank > 0 && "Trivial shape is not supported");
131  for (uint64_t d = 0; d < dimRank; ++d)
132  assert(dimSizes[d] > 0 && "Dimension size zero has trivial storage");
133  if (capacity) {
134  elements.reserve(capacity);
135  indices.reserve(capacity * dimRank);
136  }
137  }
138 
139  /// Gets the rank of the tensor.
140  uint64_t getRank() const { return dimSizes.size(); }
141 
142  /// Gets the dimension-sizes array.
143  const std::vector<uint64_t> &getDimSizes() const { return dimSizes; }
144 
145  /// Gets the elements array.
146  const std::vector<Element<V>> &getElements() const { return elements; }
147 
148  /// Returns the `operator<` closure object for the COO's element type.
149  ElementLT<V> getElementLT() const { return ElementLT<V>(getRank()); }
150 
151  /// Adds an element to the tensor. This method does not check whether
152  /// `ind` is already associated with a value, it adds it regardless.
153  /// Resolving such conflicts is left up to clients of the iterator
154  /// interface.
155  ///
156  /// This method invalidates all iterators.
157  ///
158  /// Asserts:
159  /// * the `ind` is valid for `rank`
160  /// * the elements of `ind` are valid for `dimSizes`.
161  void add(const std::vector<uint64_t> &ind, V val) {
162  const uint64_t *base = indices.data();
163  uint64_t size = indices.size();
164  uint64_t rank = getRank();
165  assert(ind.size() == rank && "Element rank mismatch");
166  for (uint64_t r = 0; r < rank; ++r) {
167  assert(ind[r] < dimSizes[r] && "Index is too large for the dimension");
168  indices.push_back(ind[r]);
169  }
170  // This base only changes if indices were reallocated. In that case, we
171  // need to correct all previous pointers into the vector. Note that this
172  // only happens if we did not set the initial capacity right, and then only
173  // for every internal vector reallocation (which with the doubling rule
174  // should only incur an amortized linear overhead).
175  const uint64_t *newBase = indices.data();
176  if (newBase != base) {
177  for (uint64_t i = 0, n = elements.size(); i < n; ++i)
178  elements[i].indices = newBase + (elements[i].indices - base);
179  base = newBase;
180  }
181  // Add the new element and update the sorted bit.
182  Element<V> addedElem(base + size, val);
183  if (!elements.empty() && isSorted)
184  isSorted = getElementLT()(elements.back(), addedElem);
185  elements.push_back(addedElem);
186  }
187 
188  const_iterator begin() const { return elements.cbegin(); }
189  const_iterator end() const { return elements.cend(); }
190 
191  /// Sorts elements lexicographically by index. If an index is mapped to
192  /// multiple values, then the relative order of those values is unspecified.
193  ///
194  /// This method invalidates all iterators.
195  void sort() {
196  if (isSorted)
197  return;
198  std::sort(elements.begin(), elements.end(), getElementLT());
199  isSorted = true;
200  }
201 
202 private:
203  const std::vector<uint64_t> dimSizes; // per-dimension sizes
204  std::vector<Element<V>> elements; // all COO elements
205  std::vector<uint64_t> indices; // shared index pool
206  bool isSorted;
207 };
