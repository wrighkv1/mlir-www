107 /// Helper method to construct a permuted dimension ordering
108 /// that adheres to the given topological sort.
109 static AffineMap permute(MLIRContext *context, AffineMap m,
110  std::vector<unsigned> &topSort) {
111  unsigned sz = topSort.size();
112  assert(m.getNumResults() == sz && "TopoSort/AffineMap size mismatch");
113  // Construct the inverse of `m`; to avoid the asymptotic complexity
114  // of calling `m.getPermutedPosition` repeatedly.
115  SmallVector<unsigned> inv(sz);
116  for (unsigned i = 0; i < sz; i++)
117  inv[i] = m.getDimPosition(i);
118  // Construct the permutation.
119  SmallVector<unsigned> perm(sz);
120  for (unsigned i = 0; i < sz; i++)
121  perm[i] = inv[topSort[i]];
122  return AffineMap::getPermutationMap(perm, context);
123 }

125 /// Helper method to inspect affine expressions. Rejects cases where the
126 /// same index is used more than once. Also rejects compound affine
127 /// expressions in sparse dimensions.
128 static bool findAffine(Merger &merger, unsigned tensor, unsigned dim,
129  AffineExpr a, DimLevelType dlt,
130  bool setLvlFormat = true) {
131  switch (a.getKind()) {
132  case AffineExprKind::DimId: {
133  unsigned idx = a.cast<AffineDimExpr>().getPosition();
134  if (!isUndefDLT(merger.getDimLevelType(tensor, idx)))
135  return false; // used more than once
136 
137  if (setLvlFormat)
138  merger.setDimAndDimLevelType(tensor, idx, dim, dlt);
139  return true;
140  }
141  case AffineExprKind::Add:
142  case AffineExprKind::Mul: {
143  if (!isDenseDLT(dlt))
144  return false; // compound only in dense dim
145  auto binOp = a.cast<AffineBinaryOpExpr>();
146  // We do not set dim level format for affine expresssion like d0 + d1 on
147  // both loop index at d0 and d1,
148  return findAffine(merger, tensor, dim, binOp.getLHS(), dlt, false) &&
149  findAffine(merger, tensor, dim, binOp.getRHS(), dlt, false);
150  }
151  case AffineExprKind::Constant:
152  return isDenseDLT(dlt); // const only in dense dim
153  default:
154  return false;
155  }
156 }

158 /// Helper method to inspect sparse encodings in the tensor types.
159 /// Fills the per-dimension sparsity information for all tensors.
160 /// Returns true if the sparse annotations and affine subscript
161 /// expressions of all tensors are admissible. Returns false if
162 /// no annotations are found or inadmissible constructs occur.
163 static bool findSparseAnnotations(Merger &merger, linalg::GenericOp op) {
164  bool annotated = false;
165  for (OpOperand &t : op->getOpOperands()) {
166  auto map = op.getMatchingIndexingMap(&t);
167  auto enc = getSparseTensorEncoding(t.get().getType());
168  if (enc)
169  annotated = true;
170  assert(map.getNumResults() == op.getRank(&t));
171  for (unsigned d = 0, rank = map.getNumResults(); d < rank; d++) {
172  unsigned tensor = t.getOperandNumber();
173  AffineExpr a = map.getResult(toOrigDim(enc, d));
174  if (!findAffine(merger, tensor, d, a, getDimLevelType(enc, d)))
175  return false; // inadmissible affine expression
176  }
177  }
178  return annotated;
179 }

181 /// A helper to compute a topological sort. O(n^2) time complexity
182 /// as we use adj matrix for the graph.
183 /// The sorted result will put the first Reduction iterator to the
184 /// latest possible index.
185 static bool topSortOptimal(unsigned n,
186  ArrayRef<utils::IteratorType> iteratorTypes,
187  std::vector<unsigned> &topSort,
188  std::vector<unsigned> &inDegree,
189  std::vector<std::vector<bool>> &adjM) {
190  std::vector<unsigned> redIt; // reduce iterator with 0 degree
191  std::vector<unsigned> parIt; // parallel iterator with 0 degree
192  for (unsigned i = 0; i < n; i++) {
193  if (inDegree[i] == 0) {
194  if (linalg::isReductionIterator(iteratorTypes[i]))
195  redIt.push_back(i);
196  else
197  parIt.push_back(i);
198  }
199  }
200 
201  while (!redIt.empty() || !parIt.empty()) {
202  // We always choose parallel iterator if there is any.
203  auto &it = !parIt.empty() ? parIt : redIt;
204  auto src = it.back();
205  topSort.push_back(src);
206  it.pop_back();
207  // Update in-degree, and push 0-degree node into worklist.
208  for (unsigned dst = 0; dst < n; dst++)
209  if (adjM[src][dst] && --inDegree[dst] == 0) {
210  if (linalg::isReductionIterator(iteratorTypes[dst]))
211  redIt.push_back(dst);
212  else
213  parIt.push_back(dst);
214  }
215  }
216  return topSort.size() == n;
217 }

219 /// Helper method to add all constraints from the indices in one affine
220 /// expression before all indices in the other affine expression. For
221 /// example i0+i1 < i2+i3+1 yields i0<i2, i0<i3, i1<i2, and i1<i3.
222 static void addAffineOrderings(std::vector<std::vector<bool>> &adjM,
223  std::vector<unsigned> &inDegree, AffineExpr a,
224  AffineExpr b, unsigned fidx) {
225  switch (a.getKind()) {
226  case AffineExprKind::DimId: {
227  unsigned idx = a.cast<AffineDimExpr>().getPosition();
228  if (b)
229  addAffineOrderings(adjM, inDegree, b, AffineExpr(), idx);
230  else if (!adjM[fidx][idx]) {
231  adjM[fidx][idx] = true;
232  inDegree[idx]++;
233  }
234  break;
235  }
236  case AffineExprKind::Add:
237  case AffineExprKind::Mul: {
238  auto binOp = a.cast<AffineBinaryOpExpr>();
239  addAffineOrderings(adjM, inDegree, binOp.getLHS(), b, fidx);
240  addAffineOrderings(adjM, inDegree, binOp.getRHS(), b, fidx);
241  break;
242  }
243  default:
244  break;
245  }
246 }

248 /// Computes a topologically sorted iteration graph for the linalg operation.
249 /// Ensures all tensors are visited in natural index order. This is essential
250 /// for sparse storage formats since these only support access along fixed
251 /// dimensions. Even for dense storage formats, however, the natural index
252 /// order yields innermost unit-stride access with better spatial locality.
253 static bool computeIterationGraph(Merger &merger, linalg::GenericOp op,
254  std::vector<unsigned> &topSort, unsigned mask,
255  OpOperand *skip = nullptr) {
256  // Set up an n x n from/to adjacency matrix of the iteration graph
257  // for the implicit loop indices i_0 .. i_n-1.
258  unsigned n = op.getNumLoops();
259  std::vector<std::vector<bool>> adjM(n, std::vector<bool>(n, false));
260  std::vector<unsigned> inDegree(n, 0); // in-degree of each node.
261  auto iteratorTypes = op.getIteratorTypesArray();
262  // Iterate over the indexing maps of every tensor in the tensor expression.
263  for (OpOperand &t : op->getOpOperands()) {
264  // Skip tensor during cycle resolution.
265  if (&t == skip)
266  continue;
267  // Get map and encoding.
268  auto map = op.getMatchingIndexingMap(&t);
269  auto enc = getSparseTensorEncoding(t.get().getType());
270  assert(map.getNumDims() == n);
271  // Skip dense tensor constraints when not requested.
272  if (!(mask & SortMask::kIncludeDense) && !enc)
273  continue;
274  // Each tensor expression and optional dimension ordering (row-major
275  // by default) puts an ordering constraint on the loop indices. For
276  // example, the tensor expresion A_ijk forces the ordering i < j < k
277  // on the loop indices if no explicit dimension ordering is given.
278  for (unsigned d = 1, rank = map.getNumResults(); d < rank; d++) {
279  AffineExpr f = map.getResult(toOrigDim(enc, d - 1));
280  AffineExpr t = map.getResult(toOrigDim(enc, d));
281  addAffineOrderings(adjM, inDegree, f, t, 0);
282  }
283  // Push unrelated loops into sparse iteration space, so these
284  // will be skipped more often.
285  if (mask & SortMask::kIncludeUndef) {
286  unsigned tensor = t.getOperandNumber();
287  for (unsigned i = 0; i < n; i++)
288  if (isCompressedDLT(merger.getDimLevelType(tensor, i)) ||
289  isSingletonDLT(merger.getDimLevelType(tensor, i))) {
290  for (unsigned j = 0; j < n; j++)
291  if (isUndefDLT(merger.getDimLevelType(tensor, j))) {
292  adjM[i][j] = true;
293  inDegree[j]++;
294  }
295  } else {
296  assert(isDenseDLT(merger.getDimLevelType(tensor, i)) ||
297  isUndefDLT(merger.getDimLevelType(tensor, i)));
298  }
299  }
300  }
301  // Topologically sort the iteration graph to determine loop order.
302  // Report failure for a cyclic iteration graph.
303  topSort.clear();
304  topSort.reserve(n);
305  return topSortOptimal(n, iteratorTypes, topSort, inDegree, adjM);
306 }

314 /// Returns true when the tensor expression is admissible for codegen.
315 /// Since all sparse input tensors are admissible, we just need to check
316 /// whether the out tensor in the tensor expression codegen is admissible.
317 /// Sets `sparseOut` to the tensor and `outerParNest` to the outer injective
318 /// nesting depth when a "truly dynamic" sparse tensor output occurs.
319 static bool isAdmissibleTensorExp(Merger &merger, linalg::GenericOp op,
320  std::vector<unsigned> &topSort, unsigned exp,
321  OpOperand **sparseOut,
322  unsigned &outerParNest) {
323  OpOperand *lhs = op.getDpsInitOperand(0);
324  unsigned tensor = lhs->getOperandNumber();
325  auto enc = getSparseTensorEncoding(lhs->get().getType());
326  // An non-annotated output tensor is assumed dense, and becomes a random
327  // access n-dim memref. Admissible since insertions cannot occur.
328  if (!enc)
329  return true;
330  // An all-dense annotated "sparse" output tensor becomes a linearized random
331  // access 1-dim memref. Also admissible since insertions cannot occur.
332  bool allDense = true;
333  auto iteratorTypes = op.getIteratorTypesArray();
334  unsigned numLoops = iteratorTypes.size();
335  for (unsigned i = 0; i < numLoops; i++)
336  if (isCompressedDLT(merger.getDimLevelType(tensor, i)) ||
337  isSingletonDLT(merger.getDimLevelType(tensor, i))) {
338  allDense = false;
339  break;
340  } else {
341  assert(isDenseDLT(merger.getDimLevelType(tensor, i)) ||
342  isUndefDLT(merger.getDimLevelType(tensor, i)));
343  }
344  if (allDense)
345  return true;
346  // A tensor expression with a sparse output tensor that changes its values
347  // but not its nonzero structure, an operation called "simply dynamic" in
348  // [Bik96,Ch9], is also admissible without special codegen.
349  if (merger.isSingleCondition(tensor, exp))
350  return true;
351  // Accept "truly dynamic" if the output tensor materializes uninitialized
352  // into the computation and insertions occur in lexicographic index order.
353  if (isMaterializing(lhs->get())) {
354  unsigned nest = 0;
355  for (unsigned i = 0; i < numLoops; i++) {
356  if (linalg::isReductionIterator(iteratorTypes[topSort[i]]))
357  break; // terminate at first reduction
358  nest++;
359  }
360  // Determine admissible dynamic insertion situations:
361  // (1) fully injective, since there are no reductions,
362  // (2) admissible 1-d expansion in innermost dimension.
363  if (nest >= op.getRank(lhs) - 1) {
364  *sparseOut = lhs;
365  outerParNest = nest;
366  return true;
367  }
368  }
369  return false;
370 }

378  switch (kind) {

418 /// Generates loop boundary statements (entering/exiting loops). The function
419 /// passes and updates the reduction value.
420 static Optional<Operation *> genLoopBoundary(
421  CodeGen &codegen, Merger &merger,
422  function_ref<Optional<Operation *>(MutableArrayRef<Value> reduc)>
423  callback) {
424  SmallVector<Value> reduc;
425  if (codegen.redVal)
426  reduc.push_back(codegen.redVal);
427  if (codegen.expValues)
428  reduc.push_back(codegen.expCount);
429  if (codegen.insChain)
430  reduc.push_back(codegen.insChain);
431 
432  auto r = callback(reduc);
433 
434  // Callback should do in-place update on reduction value vector.
435  unsigned i = 0;
436  if (codegen.redVal)
437  updateReduc(merger, codegen, reduc[i++]);
438  if (codegen.expValues)
439  codegen.expCount = reduc[i++];
440  if (codegen.insChain)
441  codegen.insChain = reduc[i];
442 
443  return r;
444 }

446 /// Local bufferization of all dense and sparse data structures.
447 static void genBuffers(Merger &merger, CodeGen &codegen, OpBuilder &builder,
448  linalg::GenericOp op) {
449  Location loc = op.getLoc();
450  assert(op.getNumOperands() == op.getNumDpsInputs() + 1);
451 
452  codegen.loopEmitter.initializeLoopEmit(
453  builder, loc,
454  /// Generates buffer for the output tensor.
455  /// Note that all sparse kernels assume that when all elements are written
456  /// to (viz. x(i) = y(i) * z(i)), the output buffer is already initialized
457  /// to all zeroes and only nonzeroes values are computed and written out.
458  /// For updates (viz. x(i) += y(i) * z(i)), only nonzeroes values are used
459  /// for the updates and no assumption on the original contents of the
460  /// output buffer is necessary.
461  [&op](OpBuilder &builder, Location loc, Value memref,
462  Value tensor) -> Value {
463  // Must not be a sparse tensor.
464  assert(!getSparseTensorEncoding(tensor.getType()));
465  OpOperand *lhs = op.getDpsInitOperand(0);
466  // Two output tensors references should pointed to the same object.
467  assert(lhs->get() == tensor);
468  bool isInit = op.isInitTensor(lhs);
469  // An output tensor can simply materialize from the buffer of the tensor
470  // that appears in the outs() clause. For updates, this has the
471  // advantage that only the nonzero value are involved in the
472  // computation, keeping the operation O(nnz). In all other cases, we are
473  // forced to zero out the buffer to enforce the assumption above, which
474  // may negatively impact running complexity (viz. O(n^2 + nnz) vs.
475  // O(nnz) for matrices).
476  // TODO: use better analysis to avoid zeroing out the buffer?
477  Value init = memref;
478  if (!isInit) {
479  Value zero = constantZero(builder, loc,
480  getElementTypeOrSelf(tensor.getType()));
481  builder.create<linalg::FillOp>(loc, ValueRange{zero},
482  ValueRange{init});
483  }
484  return init;
485  });
486 }

488 /// Generates an affine expression.
489 //
490 // TODO: generalize for sparse tensor subscripts
491 //
492 static Value genAffine(CodeGen &codegen, OpBuilder &builder, AffineExpr a,
493  Location loc) {
494  switch (a.getKind()) {
495  case AffineExprKind::DimId: {
496  unsigned idx = a.cast<AffineDimExpr>().getPosition();
497  return codegen.getLoopIdxValue(idx); // universal dense index
498  }
499  case AffineExprKind::Add: {
500  auto binOp = a.cast<AffineBinaryOpExpr>();
501  return builder.create<arith::AddIOp>(
502  loc, genAffine(codegen, builder, binOp.getLHS(), loc),
503  genAffine(codegen, builder, binOp.getRHS(), loc));
504  }
505  case AffineExprKind::Mul: {
506  auto binOp = a.cast<AffineBinaryOpExpr>();
507  return builder.create<arith::MulIOp>(
508  loc, genAffine(codegen, builder, binOp.getLHS(), loc),
509  genAffine(codegen, builder, binOp.getRHS(), loc));
510  }
511  case AffineExprKind::Constant: {
512  int64_t c = a.cast<AffineConstantExpr>().getValue();
513  return constantIndex(builder, loc, c);
514  }
515  default:
516  llvm_unreachable("unexpected affine subscript");
517  }

530 /// Generates subscript for load/store on a dense or sparse tensor.
531 static Value genSubscript(CodeGen &codegen, OpBuilder &builder,
532  linalg::GenericOp op, OpOperand *t,
533  SmallVectorImpl<Value> &args) {
534  unsigned tensor = t->getOperandNumber();
535  auto map = op.getMatchingIndexingMap(t);
536  auto enc = getSparseTensorEncoding(t->get().getType());
537  unsigned rank = map.getNumResults();
538  if (enc) {
539  // Note that currently, all sparse subscripts are simple.
540  // TODO: accept affine too?
541  assert(map.getResult(toOrigDim(enc, rank - 1)).getKind() ==
542  AffineExprKind::DimId);
543  Value pidx = codegen.loopEmitter.getPidxs()[tensor].back();
544  assert(pidx);
545  args.push_back(pidx); // position index
546  } else {
547  for (unsigned d = 0; d < rank; d++) {
548  AffineExpr a = map.getResult(d);
549  args.push_back(genAffine(codegen, builder, a, op.getLoc()));
550  }
551  }
552  return codegen.loopEmitter.getValBuffer()[tensor];

569 /// Generates insertion code to implement dynamic tensor load for reduction.
570 static Value genInsertionLoadReduce(Merger &merger, CodeGen &codegen,
571  OpBuilder &builder, linalg::GenericOp op,
572  OpOperand *t) {
573  Location loc = op.getLoc();
574  Value identity = getCustomRedId(merger.exp(codegen.redCustom).op);
575  // Direct lexicographic index order, tensor loads as identity.
576  if (!codegen.expValues) {
577  return identity;
578  }
579  // Load from expanded access pattern if filled, identity otherwise.
580  Value index = genIndex(codegen, op, t);
581  Value isFilled =
582  builder.create<memref::LoadOp>(loc, codegen.expFilled, index);
583  Value valAtIndex =
584  builder.create<memref::LoadOp>(loc, codegen.expValues, index);
585  return builder.create<arith::SelectOp>(loc, isFilled, valAtIndex, identity);

588 /// Generates insertion code to implement dynamic tensor store.
589 static void genInsertionStore(CodeGen &codegen, OpBuilder &builder,
590  linalg::GenericOp op, OpOperand *t, Value rhs) {
591  Location loc = op.getLoc();
592  // Direct insertion in lexicographic index order.
593  if (!codegen.expValues) {
594  unsigned rank = op.getRank(t);
595  SmallVector<Value> indices;
596  for (unsigned i = 0; i < rank; i++) {
597  assert(codegen.loopEmitter.getLoopIV(i));
598  indices.push_back(codegen.loopEmitter.getLoopIV(i));
599  }
600  codegen.insChain =
601  builder.create<InsertOp>(loc, rhs, codegen.insChain, indices);
602  return;
603  }
604  // Generates insertion code along expanded access pattern.
605  // if (!expFilled[i]) then
606  // expFilled[i] = true
607  // expAdded[inserts++] = i
608  // endif
609  // values[i] = rhs
610  Value index = genIndex(codegen, op, t);
611  Value fval = constantI1(builder, loc, false);
612  Value tval = constantI1(builder, loc, true);
613  // If statement.
614  Value filled = builder.create<memref::LoadOp>(loc, codegen.expFilled, index);
615  Value cond = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq,
616  filled, fval);
617  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, builder.getIndexType(), cond,
618  /*else=*/true);
619  // True branch.
620  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
621  builder.create<memref::StoreOp>(loc, tval, codegen.expFilled, index);
622  builder.create<memref::StoreOp>(loc, index, codegen.expAdded,
623  codegen.expCount);
624  Value one = constantIndex(builder, loc, 1);
625  Value add = builder.create<arith::AddIOp>(loc, codegen.expCount, one);
626  builder.create<scf::YieldOp>(loc, add);
627  // False branch.
628  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
629  builder.create<scf::YieldOp>(loc, codegen.expCount);
630  builder.setInsertionPointAfter(ifOp);
631  // Value assignment.
632  codegen.expCount = ifOp.getResult(0);
633  builder.create<memref::StoreOp>(loc, rhs, codegen.expValues, index);
634 }

636 /// Generates a load on a dense or sparse tensor.
637 static Value genTensorLoad(Merger &merger, CodeGen &codegen, OpBuilder &builder,
638  linalg::GenericOp op, unsigned exp) {
639  // Test if the load was hoisted to a higher loop nest.
640  Value val = merger.exp(exp).val;
641  if (val)
642  return val;
643 
644  // Load during insertion.
645  OpOperand &t = op->getOpOperand(merger.exp(exp).tensor);
646  if (&t == codegen.sparseOut) {
647  if (codegen.redCustom != -1u)
648  return genInsertionLoadReduce(merger, codegen, builder, op, &t);
649  return genInsertionLoad(codegen, builder, op, &t);
650  }
651  // Actual load.
652  SmallVector<Value> args;
653  Value ptr = genSubscript(codegen, builder, op, &t, args);
654  return builder.create<memref::LoadOp>(op.getLoc(), ptr, args);

657 /// Generates a store on a dense or sparse tensor.
658 static void genTensorStore(Merger &merger, CodeGen &codegen, OpBuilder &builder,
659  linalg::GenericOp op, unsigned exp, Value rhs) {
660  Location loc = op.getLoc();
661  // Test if this is a scalarized reduction.
662  if (codegen.redVal) {
663  updateReduc(merger, codegen, rhs);
664  return;
665  }
666  // Store during insertion.
667  OpOperand *t = op.getDpsInitOperand(0);
668  if (t == codegen.sparseOut) {
669  if (!rhs) {
670  // Only unary and binary are allowed to return uninitialized rhs
671  // to indicate missing output.
672  assert(merger.exp(exp).kind == kUnary || merger.exp(exp).kind == kBinary);
673  } else if (merger.exp(exp).kind == kSelect) {
674  // Select operation insertion.
675  Value insChain = codegen.insChain;
676  assert(insChain);
677  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, insChain.getType(), rhs,
678  /*else=*/true);
679  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
680  // Existing value was preserved to be used here.
681  assert(merger.exp(exp).val);
682  Value v0 = merger.exp(exp).val;
683  genInsertionStore(codegen, builder, op, t, v0);
684  merger.exp(exp).val = Value();
685  // Yield modified insertion chain along true branch.
686  builder.create<scf::YieldOp>(op.getLoc(), codegen.insChain);
687  // Yield original insertion chain along false branch.
688  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
689  builder.create<scf::YieldOp>(loc, insChain);
690  // Done with if statement.
691  codegen.insChain = ifOp->getResult(0);
692  builder.setInsertionPointAfter(ifOp);
693  } else {
694  genInsertionStore(codegen, builder, op, t, rhs);
695  }
696  return;
697  }
698  // Actual store.
699  SmallVector<Value> args;
700  Value ptr = genSubscript(codegen, builder, op, t, args);
701  builder.create<memref::StoreOp>(loc, rhs, ptr, args);
702 }

716 /// Semi-ring branches are simply inlined by the sparse compiler. Prior
717 /// analysis has verified that all computations are "local" to the inlined
718 /// branch or otherwise invariantly defined outside the loop nest, with the
719 /// exception of index computations, which need to be relinked to actual
720 /// inlined cloned code.
721 static Value relinkBranch(CodeGen &codegen, RewriterBase &rewriter,
722  Block *block, Value e, unsigned ldx) {
723  if (Operation *def = e.getDefiningOp()) {
724  if (auto indexOp = dyn_cast<linalg::IndexOp>(def))
725  return genIndexValue(codegen, rewriter, indexOp.getDim());
726  if (def->getBlock() == block) {
727  for (unsigned i = 0, n = def->getNumOperands(); i < n; i++)
728  def->setOperand(
729  i, relinkBranch(codegen, rewriter, block, def->getOperand(i), ldx));
730  }
731  }
732  return e;
733 }

735 /// Recursively generates tensor expression.
736 static Value genExp(Merger &merger, CodeGen &codegen, RewriterBase &rewriter,
737  linalg::GenericOp op, unsigned exp, unsigned ldx) {
738  Location loc = op.getLoc();
739  if (exp == -1u)
740  return Value();
741  if (merger.exp(exp).kind == Kind::kTensor)
742  return genTensorLoad(merger, codegen, rewriter, op, exp);
743  if (merger.exp(exp).kind == Kind::kInvariant)
744  return genInvariantValue(merger, codegen, rewriter, exp);
745  if (merger.exp(exp).kind == Kind::kIndex)
746  return genIndexValue(codegen, rewriter, merger.exp(exp).index);
747 
748  if (merger.exp(exp).kind == Kind::kReduce) {
749  // Make custom reduction identity accessible for expanded access pattern.
750  assert(codegen.redCustom == -1u);
751  codegen.redCustom = exp;
752  }
753 
754  Value v0 =
755  genExp(merger, codegen, rewriter, op, merger.exp(exp).children.e0, ldx);
756  Value v1 =
757  genExp(merger, codegen, rewriter, op, merger.exp(exp).children.e1, ldx);
758  Value ee = merger.buildExp(rewriter, loc, exp, v0, v1);
759  if (ee && (merger.exp(exp).kind == Kind::kUnary ||
760  merger.exp(exp).kind == Kind::kBinary ||
761  merger.exp(exp).kind == Kind::kBinaryBranch ||
762  merger.exp(exp).kind == Kind::kReduce ||
763  merger.exp(exp).kind == Kind::kSelect))
764  ee = relinkBranch(codegen, rewriter, ee.getParentBlock(), ee, ldx);
765 
766  if (merger.exp(exp).kind == kSelect) {
767  assert(!merger.exp(exp).val);
768  merger.exp(exp).val = v0; // Preserve value for later use.
769  }
770 
771  if (merger.exp(exp).kind == Kind::kReduce) {
772  assert(codegen.redCustom != -1u);
773  codegen.redCustom = -1u;
774  }
775 
776  return ee;
777 }

779 /// Determines if affine expression is invariant.
780 static bool isInvariantAffine(const CodeGen &codegen, AffineExpr a,
781  unsigned ldx, bool &atLevel) {
782  switch (a.getKind()) {
783  case AffineExprKind::DimId: {
784  unsigned idx = a.cast<AffineDimExpr>().getPosition();
785  if (idx == ldx)
786  atLevel = true;
787  return codegen.getLoopIdxValue(idx) != nullptr; // no longer in play?
788  }
789  case AffineExprKind::Add:
790  case AffineExprKind::Mul: {
791  auto binOp = a.cast<AffineBinaryOpExpr>();
792  return isInvariantAffine(codegen, binOp.getLHS(), ldx, atLevel) &&
793  isInvariantAffine(codegen, binOp.getRHS(), ldx, atLevel);
794  }
795  default:
796  return true;
797  }
798 }

800 /// Hoists loop invariant tensor loads for which indices have been exhausted.
801 static void genInvariants(Merger &merger, CodeGen &codegen, OpBuilder &builder,
802  linalg::GenericOp op, unsigned exp, unsigned ldx,
803  bool atStart, unsigned last = -1u) {
804  if (exp == -1u)
805  return;
806  if (merger.exp(exp).kind == Kind::kTensor) {
807  // Inspect tensor indices.
808  bool atLevel = ldx == -1u;
809  OpOperand &t = op->getOpOperand(merger.exp(exp).tensor);
810  auto map = op.getMatchingIndexingMap(&t);
811  auto enc = getSparseTensorEncoding(t.get().getType());
812  for (unsigned d = 0, rank = map.getNumResults(); d < rank; d++) {
813  AffineExpr a = map.getResult(toOrigDim(enc, d));
814  if (!isInvariantAffine(codegen, a, ldx, atLevel))
815  return; // still in play
816  }
817  // All exhausted at this level (atLevel denotes exactly at this level).
818  if (!atLevel)
819  return;
820  OpOperand *lhs = op.getDpsInitOperand(0);
821  if (lhs == &t) {
822  // Start or end a scalarized reduction
823  if (atStart) {
824  Kind kind = merger.exp(last).kind;
825  Value load = kind == Kind::kReduce
826  ? getCustomRedId(merger.exp(last).op)
827  : genTensorLoad(merger, codegen, builder, op, exp);
828  codegen.redKind = getReduction(kind);
829  codegen.redExp = exp;
830  updateReduc(merger, codegen, load);
831  } else {
832  Value redVal = codegen.redVal;
833  updateReduc(merger, codegen, Value());
834  codegen.redExp = -1u;
835  codegen.redKind = kNoReduc;
836  genTensorStore(merger, codegen, builder, op, exp, redVal);
837  }
838  } else {
839  // Start or end loop invariant hoisting of a tensor load.
840  merger.exp(exp).val =
841  atStart ? genTensorLoad(merger, codegen, builder, op, exp) : Value();
842  }
843  } else if (merger.exp(exp).kind != Kind::kInvariant &&
844  merger.exp(exp).kind != Kind::kIndex) {
845  // Traverse into the binary operations. Note that we only hoist
846  // tensor loads, since subsequent MLIR/LLVM passes know how to
847  // deal with all other kinds of derived loop invariants.
848  unsigned e0 = merger.exp(exp).children.e0;
849  unsigned e1 = merger.exp(exp).children.e1;
850  genInvariants(merger, codegen, builder, op, e0, ldx, atStart, exp);
851  genInvariants(merger, codegen, builder, op, e1, ldx, atStart, exp);
852  }
853 }

855 /// Generates an expanded access pattern in innermost dimension.
856 static void genExpansion(Merger &merger, CodeGen &codegen, OpBuilder &builder,
857  linalg::GenericOp op, unsigned at, bool atStart) {
858  OpOperand *lhs = codegen.sparseOut;
859  if (!lhs || codegen.outerParNest != op.getRank(lhs) - 1 ||
860  at != codegen.outerParNest)
861  return; // not needed at this level
862  assert(codegen.redVal == nullptr);
863  // Generate start or end of an expanded access pattern. Note that because
864  // an expension does not rely on the ongoing contents of the sparse storage
865  // scheme, we can use the original tensor as incoming SSA value (which
866  // simplifies codegen a bit). If expansion on the actual contents is ever
867  // needed, we will need to use the SSA value in the insertion chain instead.
868  Value tensor = lhs->get();
869  Location loc = op.getLoc();
870  if (atStart) {
871  auto dynShape = {ShapedType::kDynamicSize};
872  Type etp = tensor.getType().cast<ShapedType>().getElementType();
873  Type t1 = MemRefType::get(dynShape, etp);
874  Type t2 = MemRefType::get(dynShape, builder.getI1Type());
875  Type t3 = MemRefType::get(dynShape, builder.getIndexType());
876  Type t4 = builder.getIndexType();
877  auto res =
878  builder.create<ExpandOp>(loc, TypeRange({t1, t2, t3, t4}), tensor);
879  assert(res.getNumResults() == 4);
880  assert(!codegen.expValues);
881  codegen.expValues = res.getResult(0);
882  codegen.expFilled = res.getResult(1);
883  codegen.expAdded = res.getResult(2);
884  codegen.expCount = res.getResult(3);
885  } else {
886  assert(codegen.expValues);
887  SmallVector<Value> indices;
888  for (unsigned i = 0; i < at; i++) {
889  assert(codegen.loopEmitter.getLoopIV(i));
890  indices.push_back(codegen.loopEmitter.getLoopIV(i));
891  }
892  codegen.insChain = builder.create<CompressOp>(
893  loc, codegen.expValues, codegen.expFilled, codegen.expAdded,
894  codegen.expCount, codegen.insChain, indices);
895  codegen.expValues = codegen.expFilled = codegen.expAdded =
896  codegen.expCount = Value();
897  }
898 }

900 /// Returns parallelization strategy. Any implicit loop in the Linalg
901 /// operation that is marked "parallel" is a candidate. Whether it is actually
902 /// converted to a parallel operation depends on the requested strategy.
903 static bool isParallelFor(CodeGen &codegen, bool isOuter, bool isSparse) {
904  // Reject parallelization of sparse output.
905  if (codegen.sparseOut)
906  return false;
907  // Parallel loops on tensor expansion can cause data races.
908  if (codegen.expCount)
909  return false;
910  // Inspect strategy.
911  switch (codegen.options.parallelizationStrategy) {
912  case SparseParallelizationStrategy::kNone:
913  return false;
914  case SparseParallelizationStrategy::kDenseOuterLoop:
915  return isOuter && !isSparse;
916  case SparseParallelizationStrategy::kAnyStorageOuterLoop:
917  return isOuter;
918  case SparseParallelizationStrategy::kDenseAnyLoop:
919  return !isSparse;
920  case SparseParallelizationStrategy::kAnyStorageAnyLoop:
921  return true;
922  }
923  llvm_unreachable("unexpected parallelization strategy");
924 }

1018 static scf::IfOp genIf(Merger &merger, CodeGen &codegen, OpBuilder &builder,
1019  linalg::GenericOp op, unsigned idx,

1021  Location loc = op.getLoc();

1081 /// Starts a loop sequence at given level. Returns true if
1082 /// the universal loop index must be maintained at this level.
1083 static bool startLoopSeq(Merger &merger, CodeGen &codegen, OpBuilder &builder,
1084  linalg::GenericOp op, unsigned exp, unsigned at,
1085  unsigned idx, unsigned ldx, unsigned lts) {
1086  assert(!codegen.getLoopIdxValue(idx));
1087  // Emit invariants at this loop sequence level.
1088  genInvariants(merger, codegen, builder, op, exp, ldx, /*atStart=*/true);
1089  // Emit access pattern expansion for sparse tensor output.
1090  genExpansion(merger, codegen, builder, op, at, /*atStart=*/true);
1091  // Emit further intitialization at this loop sequence level.
1092  unsigned l0 = merger.set(lts)[0];
1093  bool needsUniv = false;
1094 
1095  SmallVector<size_t> tids;
1096  SmallVector<size_t> dims;
1097  merger.foreachTidDimPairInBits(
1098  merger.lat(l0).bits,
1099  [&](unsigned b, unsigned tid, Optional<unsigned> dim, DimLevelType dlt) {
1100  assert(merger.index(b) == idx);
1101  if (isDenseDLT(dlt) || isUndefDLT(dlt)) {
1102  needsUniv = true;
1103  } else {
1104  // sparse/singleton dim levels.
1105  tids.push_back(tid);
1106  dims.push_back(dim.value());
1107  }
1108  });
1109 
1110  codegen.loopEmitter.enterNewLoopSeq(builder, op.getLoc(), tids, dims);
1111 
1112  // Maintain the universal index only if it is actually

1315 /// Sparse rewriting rule for generic Lingalg operation.
1316 struct GenericOpSparsifier : public OpRewritePattern<linalg::GenericOp> {
1317 public:
1318  GenericOpSparsifier(MLIRContext *context, SparsificationOptions o)
1319  : OpRewritePattern<linalg::GenericOp>(context), options(o) {}
1320 
1321  LogicalResult matchAndRewrite(linalg::GenericOp op,
1322  PatternRewriter &rewriter) const override {
1323  // Detects sparse annotations and translate the per-dimension sparsity
1324  // information for all tensors to loop indices in the kernel.
1325  if (op.getNumDpsInits() != 1)
1326  return failure();
1327  unsigned numTensors = op->getNumOperands();
1328  unsigned numLoops = op.getNumLoops();
1329  Merger merger(numTensors, numLoops);
1330  if (!findSparseAnnotations(merger, op))
1331  return failure();
1332 
1333  // Builds the tensor expression for the Linalg operation in SSA form.
1334  Optional<unsigned> optExp = merger.buildTensorExpFromLinalg(op);
1335  if (!optExp.has_value())
1336  return failure();
1337 
1338  unsigned exp = optExp.value();
1339  OpOperand *sparseOut = nullptr;
1340  unsigned outerParNest = 0;
1341  // Computes a topologically sorted iteration graph to ensure tensors
1342  // are visited in natural index order. Gradually relaxes the considered
1343  // constraints until an acyclic iteration graph results, such that sparse
1344  // code generation can proceed. As a last resort, an attempt is made
1345  // to resolve cycles by inserting a conversion.
1346  std::vector<unsigned> topSort;
1347  // Whether the current GenericOp is admissible.
1348  bool isAdmissible = false;
1349  bool hasCycle = true;
1350  // An const list of all masks that we used for interation graph
1351  // computation. Must be ordered from strict -> loose.
1352  const auto allMask = {SortMask::kIncludeAll, SortMask::kIncludeUndef,
1353  SortMask::kIncludeDense, SortMask::kSparseOnly};
1354  for (auto mask : allMask)
1355  if (computeIterationGraph(merger, op, topSort, mask)) {
1356  hasCycle = false;
1357  if (isAdmissibleTensorExp(merger, op, topSort, exp, &sparseOut,
1358  outerParNest)) {
1359  isAdmissible = true;
1360  break;
1361  }
1362  // else try a set of less strict constraints.
1363  }
1364 
1365  if (hasCycle)
1366  // Give it one last shot to resolve the cycle.
1367  return resolveCycle(merger, rewriter, op);
1368  if (!isAdmissible)
1369  // Inadmissible expression, reject.
1370  return failure();
1371 
1372  merger.setHasSparseOut(sparseOut != nullptr);
1373 
1374  SmallVector<Value> tensors;
1375  for (OpOperand &t : op->getOpOperands())
1376  tensors.push_back(t.get());
1377 
1378  // Recursively generates code if admissible.
1379  CodeGen codegen(options, op.getContext(), tensors, numTensors, numLoops,
1380  sparseOut, outerParNest, topSort);
1381  genBuffers(merger, codegen, rewriter, op);
1382  genStmt(merger, codegen, rewriter, op, exp, 0);
1383  genResult(merger, codegen, rewriter, op);
1384  return success();
1385  }
1386 
1387 private:
1388  // Last resort cycle resolution.
1389  LogicalResult resolveCycle(Merger &merger, PatternRewriter &rewriter,
1390  linalg::GenericOp op) const {
1391  // Compute topological sort while leaving out every
1392  // sparse input tensor in succession until an acylic
1393  // iteration graph results.
1394  std::vector<unsigned> topSort;
1395  for (OpOperand *t : op.getDpsInputOperands()) {
1396  unsigned tensor = t->getOperandNumber();
1397  Value tval = t->get();
1398  auto srcEnc = getSparseTensorEncoding(tval.getType());
1399  if (!srcEnc ||
1400  !computeIterationGraph(merger, op, topSort, SortMask::kSparseOnly, t))
1401  continue;
1402  // Found an input tensor that resolves the cycle by inserting a
1403  // conversion into a sparse tensor that adheres to the iteration
1404  // graph order. Also releases the temporary sparse tensor.
1405  //
1406  // TODO: investigate fusing the conversion with computation,
1407  // especially if it is a direct yield!
1408  //
1409  auto srcTp = tval.getType().cast<RankedTensorType>();
1410  auto dstEnc = SparseTensorEncodingAttr::get(
1411  op->getContext(), srcEnc.getDimLevelType(),
1412  permute(getContext(), op.getMatchingIndexingMap(t),
1413  topSort), // new order
1414  srcEnc.getHigherOrdering(), srcEnc.getPointerBitWidth(),
1415  srcEnc.getIndexBitWidth());

1417  srcTp.getElementType(), dstEnc);
1418  auto convert = rewriter.create<ConvertOp>(tval.getLoc(), dstTp, tval);
1419  op->setOperand(tensor, convert);
1420  rewriter.setInsertionPointAfter(op);
1421  rewriter.create<bufferization::DeallocTensorOp>(tval.getLoc(), convert);
1422  return success();
1423  }
1424  // Cannot be resolved with a single conversion.
1425  // TODO: convert more than one?
1426  return failure();
1427  }
