21 /// A utility walker that interrupts if the operation has valid debug
22 /// information.
23 static WalkResult interruptIfValidLocation(Operation *op) {
24  return op->getLoc().isa<UnknownLoc>() ? WalkResult::advance()
25  : WalkResult::interrupt();

28 DebugTranslation::DebugTranslation(Operation *module, llvm::Module &llvmModule)
29  : debugEmissionIsEnabled(false), llvmModule(llvmModule),
30  llvmCtx(llvmModule.getContext()) {
31  // If the module has no location information, there is nothing to do.
32  if (!module->walk(interruptIfValidLocation).wasInterrupted())
33  return;
34  debugEmissionIsEnabled = true;
35 
36  // TODO: The version information should be encoded on the LLVM module itself,
37  // not implicitly set here.
38 
39  // Mark this module as having debug information.
40  StringRef debugVersionKey = "Debug Info Version";
41  if (!llvmModule.getModuleFlag(debugVersionKey))
42  llvmModule.addModuleFlag(llvm::Module::Warning, debugVersionKey,
43  llvm::DEBUG_METADATA_VERSION);
44 
45  if (auto targetTripleAttr =
46  module->getAttr(LLVM::LLVMDialect::getTargetTripleAttrName())) {
47  auto targetTriple =
48  llvm::Triple(targetTripleAttr.cast<StringAttr>().getValue());
49  if (targetTriple.isKnownWindowsMSVCEnvironment()) {
50  // Dwarf debugging files will be generated by default, unless "CodeView"
51  // is set explicitly. Windows/MSVC should use CodeView instead.
52  llvmModule.addModuleFlag(llvm::Module::Warning, "CodeView", 1);
53  }
54  }
55 }

60 /// Translate the debug information for the given function.
61 void DebugTranslation::translate(LLVMFuncOp func, llvm::Function &llvmFunc) {
62  if (!debugEmissionIsEnabled)
63  return;
64 
65  // If we are to create debug info for the function, we need to ensure that all
66  // inlinable calls in it are with debug info, otherwise the LLVM verifier will
67  // complain. For now, be more restricted and treat all calls as inlinable.
68  const bool hasCallWithoutDebugInfo =
69  func.walk([&](LLVM::CallOp call) {
70  return call.getLoc()->walk([](Location l) {
71  return l.isa<UnknownLoc>() ? WalkResult::interrupt()
72  : WalkResult::advance();
73  });
74  })
75  .wasInterrupted();
76  if (hasCallWithoutDebugInfo)
77  return;
78 
79  // Look for a sub program attached to the function.
80  auto spLoc =
81  func.getLoc()->findInstanceOf<FusedLocWith<LLVM::DISubprogramAttr>>();
82  if (!spLoc)
83  return;
84  llvmFunc.setSubprogram(translate(spLoc.getMetadata()));
85 }

158 /// Return a new subprogram that is either distinct or not, depending on

160 template <class... Ts>
161 static llvm::DISubprogram *getSubprogram(bool isDistinct, Ts &&...args) {
162  if (isDistinct)
163  return llvm::DISubprogram::getDistinct(std::forward<Ts>(args)...);
164  return llvm::DISubprogram::get(std::forward<Ts>(args)...);

208 llvm::DINode *DebugTranslation::translate(DINodeAttr attr) {
209  if (!attr)
210  return nullptr;
211  // Check for a cached instance.
212  if (llvm::DINode *node = attrToNode.lookup(attr))
213  return node;
214 
215  llvm::DINode *node =
216  TypeSwitch<DINodeAttr, llvm::DINode *>(attr)
217  .Case<DIBasicTypeAttr, DICompileUnitAttr, DICompositeTypeAttr,
218  DIDerivedTypeAttr, DIFileAttr, DILexicalBlockAttr,
219  DILexicalBlockFileAttr, DILocalVariableAttr, DISubprogramAttr,
220  DISubrangeAttr, DISubroutineTypeAttr>(
221  [&](auto attr) { return translateImpl(attr); });
222  attrToNode.insert({attr, node});
223  return node;
224 }

238 /// Translate the given location to an llvm DebugLoc.
239 const llvm::DILocation *
240 DebugTranslation::translateLoc(Location loc, llvm::DILocalScope *scope,
241  const llvm::DILocation *inlinedAt) {
242  // LLVM doesn't have a representation for unknown.
243  if (!scope || loc.isa<UnknownLoc>())
244  return nullptr;
245 
246  // Check for a cached instance.
247  auto existingIt = locationToLoc.find(std::make_tuple(loc, scope, inlinedAt));
248  if (existingIt != locationToLoc.end())
249  return existingIt->second;
250 
251  const llvm::DILocation *llvmLoc = nullptr;
252  if (auto callLoc = loc.dyn_cast<CallSiteLoc>()) {
253  // For callsites, the caller is fed as the inlinedAt for the callee.
254  const auto *callerLoc = translateLoc(callLoc.getCaller(), scope, inlinedAt);
255  llvmLoc = translateLoc(callLoc.getCallee(), scope, callerLoc);
256 
257  } else if (auto fileLoc = loc.dyn_cast<FileLineColLoc>()) {
258  auto *file = translateFile(fileLoc.getFilename());
259  auto *fileScope = llvm::DILexicalBlockFile::get(llvmCtx, scope, file,
260  /*Discriminator=*/0);
261  llvmLoc = llvm::DILocation::get(llvmCtx, fileLoc.getLine(),
262  fileLoc.getColumn(), fileScope,
263  const_cast<llvm::DILocation *>(inlinedAt));
264 
265  } else if (auto fusedLoc = loc.dyn_cast<FusedLoc>()) {
266  ArrayRef<Location> locations = fusedLoc.getLocations();
267 
268  // Check for a scope encoded with the location.
269  if (auto scopedAttr =
270  fusedLoc.getMetadata().dyn_cast_or_null<LLVM::DIScopeAttr>())
271  scope = cast<llvm::DILocalScope>(translate(scopedAttr));
272 
273  // For fused locations, merge each of the nodes.
274  llvmLoc = translateLoc(locations.front(), scope, inlinedAt);
275  for (Location locIt : locations.drop_front()) {
276  llvmLoc = llvm::DILocation::getMergedLocation(
277  llvmLoc, translateLoc(locIt, scope, inlinedAt));
278  }
279 
280  } else if (auto nameLoc = loc.dyn_cast<NameLoc>()) {
281  llvmLoc = translateLoc(loc.cast<NameLoc>().getChildLoc(), scope, inlinedAt);
282 
283  } else if (auto opaqueLoc = loc.dyn_cast<OpaqueLoc>()) {
284  llvmLoc = translateLoc(loc.cast<OpaqueLoc>().getFallbackLocation(), scope,
285  inlinedAt);
286  } else {
287  llvm_unreachable("unknown location kind");
288  }
289 
290  locationToLoc.try_emplace(std::make_tuple(loc, scope, inlinedAt), llvmLoc);
291  return llvmLoc;
292 }

294 /// Create an llvm debug file for the given file path.
295 llvm::DIFile *DebugTranslation::translateFile(StringRef fileName) {
296  auto *&file = fileMap[fileName];
297  if (file)
298  return file;
299 
300  // Make sure the current working directory is up-to-date.
301  if (currentWorkingDir.empty())
302  llvm::sys::fs::current_path(currentWorkingDir);
303 
304  StringRef directory = currentWorkingDir;
305  SmallString<128> dirBuf;
306  SmallString<128> fileBuf;
307  if (llvm::sys::path::is_absolute(fileName)) {
308  // Strip the common prefix (if it is more than just "/") from current
309  // directory and FileName for a more space-efficient encoding.
310  auto fileIt = llvm::sys::path::begin(fileName);
311  auto fileE = llvm::sys::path::end(fileName);
312  auto curDirIt = llvm::sys::path::begin(directory);
313  auto curDirE = llvm::sys::path::end(directory);
314  for (; curDirIt != curDirE && *curDirIt == *fileIt; ++curDirIt, ++fileIt)
315  llvm::sys::path::append(dirBuf, *curDirIt);
316  if (std::distance(llvm::sys::path::begin(directory), curDirIt) == 1) {
317  // Don't strip the common prefix if it is only the root "/" since that
318  // would make LLVM diagnostic locations confusing.
319  directory = StringRef();
320  } else {
321  for (; fileIt != fileE; ++fileIt)
322  llvm::sys::path::append(fileBuf, *fileIt);
323  directory = dirBuf;
324  fileName = fileBuf;
325  }
326  }
327  return (file = llvm::DIFile::get(llvmCtx, fileName, directory));
328 }
