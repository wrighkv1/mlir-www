52 void TosaValidation::runOnOperation() {
53  profile_type = symbolizeEnum<TosaProfileEnum>(profileName);
54 
55  getOperation().walk([&](Operation *op) {
56  for (Value operand : op->getOperands()) {
57  if ((profile_type == TosaProfileEnum::BaseInference) &&
58  getElementTypeOrSelf(operand).isa<FloatType>()) {
