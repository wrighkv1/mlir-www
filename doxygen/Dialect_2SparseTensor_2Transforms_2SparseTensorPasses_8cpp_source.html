101 struct SparseTensorConversionPass
102  : public impl::SparseTensorConversionPassBase<SparseTensorConversionPass> {
103 
104  SparseTensorConversionPass() = default;
105  SparseTensorConversionPass(const SparseTensorConversionPass &pass) = default;
106  SparseTensorConversionPass(const SparseTensorConversionOptions &options) {
107  sparseToSparse = static_cast<int32_t>(options.sparseToSparseStrategy);
108  }
109 
110  void runOnOperation() override {
111  auto *ctx = &getContext();
112  RewritePatternSet patterns(ctx);
113  SparseTensorTypeToPtrConverter converter;
114  ConversionTarget target(*ctx);
115  // Everything in the sparse dialect must go!
116  target.addIllegalDialect<SparseTensorDialect>();
117  // All dynamic rules below accept new function, call, return, and various
118  // tensor and bufferization operations as legal output of the rewriting
119  // provided that all sparse tensor types have been fully rewritten.
120  target.addDynamicallyLegalOp<func::FuncOp>([&](func::FuncOp op) {
121  return converter.isSignatureLegal(op.getFunctionType());
122  });
123  target.addDynamicallyLegalOp<func::CallOp>([&](func::CallOp op) {
124  return converter.isSignatureLegal(op.getCalleeType());
125  });
126  target.addDynamicallyLegalOp<func::ReturnOp>([&](func::ReturnOp op) {
127  return converter.isLegal(op.getOperandTypes());
128  });
129  target.addDynamicallyLegalOp<tensor::DimOp>([&](tensor::DimOp op) {
130  return converter.isLegal(op.getOperandTypes());
131  });
132  target.addDynamicallyLegalOp<tensor::CastOp>([&](tensor::CastOp op) {
133  return converter.isLegal(op.getSource().getType()) &&
134  converter.isLegal(op.getDest().getType());
135  });
136  target.addDynamicallyLegalOp<tensor::ExpandShapeOp>(
137  [&](tensor::ExpandShapeOp op) {
138  return converter.isLegal(op.getSrc().getType()) &&
139  converter.isLegal(op.getResult().getType());
140  });
141  target.addDynamicallyLegalOp<tensor::CollapseShapeOp>(
142  [&](tensor::CollapseShapeOp op) {
143  return converter.isLegal(op.getSrc().getType()) &&
144  converter.isLegal(op.getResult().getType());
145  });
146  target.addDynamicallyLegalOp<bufferization::AllocTensorOp>(
147  [&](bufferization::AllocTensorOp op) {
148  return converter.isLegal(op.getType());
149  });
150  target.addDynamicallyLegalOp<bufferization::DeallocTensorOp>(
151  [&](bufferization::DeallocTensorOp op) {
152  return converter.isLegal(op.getTensor().getType());
153  });
154  // The following operations and dialects may be introduced by the
155  // rewriting rules, and are therefore marked as legal.
156  target.addLegalOp<complex::ConstantOp, complex::NotEqualOp, linalg::FillOp,
157  linalg::YieldOp, tensor::ExtractOp>();
158  target.addLegalDialect<
159  arith::ArithDialect, bufferization::BufferizationDialect,
160  LLVM::LLVMDialect, memref::MemRefDialect, scf::SCFDialect>();
161  // Translate strategy flags to strategy options.
162  SparseTensorConversionOptions options(
163  sparseToSparseConversionStrategy(sparseToSparse));
164  // Populate with rules and apply rewriting rules.
165  populateFunctionOpInterfaceTypeConversionPattern<func::FuncOp>(patterns,
166  converter);
167  populateCallOpTypeConversionPattern(patterns, converter);
168  scf::populateSCFStructuralTypeConversionsAndLegality(converter, patterns,
169  target);
170  populateSparseTensorConversionPatterns(converter, patterns, options);
171  if (failed(applyPartialConversion(getOperation(), target,
172  std::move(patterns))))
173  signalPassFailure();
174  }
175 };

259 SparseToSparseConversionStrategy
260 mlir::sparseToSparseConversionStrategy(int32_t flag) {
261  switch (flag) {
262  default:
263  return SparseToSparseConversionStrategy::kAuto;
264  case 1:
265  return SparseToSparseConversionStrategy::kViaCOO;
