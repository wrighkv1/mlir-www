41 /// Return true if the last dimension of the MemRefType has unit stride. Also
42 /// return true for memrefs with no strides.
43 static bool isLastMemrefDimUnitStride(MemRefType type) {
44  int64_t offset;
45  SmallVector<int64_t> strides;
46  if (failed(getStridesAndOffset(type, strides, offset))) {
47  return false;
48  }
49  return strides.back() == 1;
50 }

52 LogicalResult DeviceAsyncCopyOp::verify() {
53  auto srcMemref = getSrc().getType().cast<MemRefType>();
54  auto dstMemref = getDst().getType().cast<MemRefType>();
55  unsigned workgroupAddressSpace = gpu::GPUDialect::getWorkgroupAddressSpace();
56  if (!isLastMemrefDimUnitStride(srcMemref))
57  return emitError("source memref most minor dim must have unit stride");
58  if (!isLastMemrefDimUnitStride(dstMemref))
59  return emitError("destination memref most minor dim must have unit stride");
60  if (dstMemref.getMemorySpaceAsInt() != workgroupAddressSpace)
61  return emitError("destination memref must have memory space ")
62  << workgroupAddressSpace;
63  if (dstMemref.getElementType() != srcMemref.getElementType())
64  return emitError("source and destination must have the same element type");
65  if (size_t(srcMemref.getRank()) != getSrcIndices().size())
66  return emitOpError() << "expected " << srcMemref.getRank()
67  << " source indices, got " << getSrcIndices().size();
68  if (size_t(dstMemref.getRank()) != getDstIndices().size())
69  return emitOpError() << "expected " << dstMemref.getRank()
70  << " destination indices, got "
71  << getDstIndices().size();
72  return success();
73 }

85 /// Performs verification for MmaSyncOp and MmaSparseSyncOp.
86 static LogicalResult verifyMmaSyncOp(Operation *op,
87  TypedValue<VectorType> matrixA,
88  TypedValue<VectorType> matrixB,
89  TypedValue<VectorType> matrixC,
90  const std::array<int64_t, 3> &mmaShape,
91  bool tf32Enabled, bool sparse = false) {
92 
93  // The verification for mma.sync covering various shapes and data types is
94  // based on the fundamental tensor core shape.
95 
96  // "Fundamental" tensor core shapes:
97  // - For F32 (TF32), F16, S8, and S4 data
98  // types the fundamental tensor core operation is of shape 8-by-8-by-128b.
99  // - F64 is an exception and is of shape 8-by-8-by-256b.
100  constexpr int kThreads = 32; // 32 threads per warp
101  int64_t shapeM = 8;
102  int64_t shapeN = 8;
103  int64_t shapeK; // set based on data type (128b for all data types except F64)
104 
105  // Number of elements A, B, and C per thread per fundamental tensor core tile
106  int64_t numElementA; // set based on data type (32b except F64)
107  int64_t numElementB; // set based on data type (32b except F64)
108  int64_t numElementC{2}; // two accumulator elements per fundamental tile
109 
110  // nvgpu.mma.sync vector operands (per thread)
111  auto aVector = matrixA.getType();
112  auto bVector = matrixB.getType();
113  auto cVector = matrixC.getType();
114 
115  // vector shapes
116  ArrayRef<int64_t> aShape = aVector.getShape();
117  ArrayRef<int64_t> bShape = bVector.getShape();
118  ArrayRef<int64_t> cShape = cVector.getShape();
119 
120  // vector element type
121  Type aType = aVector.getElementType();
122 
123  // Certain data types are not allowed in sparse mode.
124  if (sparse && aType.isF64())
125  return op->emitError() << "f64 is not supported for sparse mode";
126 
127  if (aType.isF64()) {
128  // exception to 8-by-8-128b fundamental tensor core tile size
129  shapeK = 4;
130  numElementA = 1;
131  numElementB = 1;
132  } else if (aType.isF32() || aType.isBF16() || aType.isF16() ||
133  aType.isInteger(8) || aType.isInteger(4)) {
134  // 8-by-8-128b fundamental tensor core tile size
135  int operandBitwidth = aType.getIntOrFloatBitWidth();
136  shapeK = 128 / operandBitwidth; // 128b wide shapeK
137 
138  numElementA = 32 / operandBitwidth; // 32b wide operand A
139  numElementB = 32 / operandBitwidth; // 32b wide operand B
140  } else {
141  return op->emitError()
142  << "expected input data type (i4,i8,f16,bf16,tf32,f64) "
143  "supported by "
144  << op->getName();
145  }
146 
147  //
148  // Basic verification
149  //
150 
151  auto [m, n, k] = mmaShape;
152 
153  // verify warp-wide size for vector a
154  int64_t sparseFactor = sparse ? 2 : 1;
155  if (aShape[0] * aShape[1] * kThreads != m * k / sparseFactor)
156  return op->emitOpError()
157  << "expected " << m * k << " warp-wide matrix A elements";
158 
159  // verify warp-wide size for vector b
160  if (bShape[0] * bShape[1] * kThreads != k * n)
161  return op->emitOpError()
162  << "expected " << k * n << " warp-wide matrix B elements";
163 
164  // verify warp-wide size for vector c
165  if (cShape[0] * cShape[1] * kThreads != m * n)
166  return op->emitOpError()
167  << "expected " << m * n << " warp-wide matrix C elements";
168 
169  // verify tf32 tensor cores are enabled for only F32 datatype
170  if (tf32Enabled && !(aType.isF32()))
171  return op->emitOpError()
172  << "expected tf32 tensor cores only for F32 operands";
173 
174  //
175  // Extended verification
176  //
177 
178  // tiles of fundamental tensor core operations
179  int64_t mTile = m / shapeM;
180  int64_t nTile = n / shapeN;
181  int64_t kTile = k / shapeK;
182 
183  // verify shape of aVector
184  if ((aShape[0] != mTile * kTile / (sparse ? 2 : 1)) ||
185  (aShape[1] != numElementA))
186  return op->emitOpError() << "expected matrix A to be shaped ("
187  << mTile * kTile << " x " << numElementA << ")";
188 
189  // verify shape of bVector
190  if ((bShape[0] != kTile * nTile) || (bShape[1] != numElementB))
191  return op->emitOpError() << "expected matrix B to be shaped ("
192  << kTile * nTile << " x " << numElementB << ")";
193 
194  // verify shape of cVector
195  if ((cShape[0] != mTile * nTile) || (cShape[1] != numElementC))
196  return op->emitOpError() << "expected matrix C to be shaped ("
197  << mTile * nTile << " x " << numElementC << ")";
198 
199  return success();
200 }

227 // NVGPU_LdMatrixOp
228 //===----------------------------------------------------------------------===//
229 LogicalResult LdMatrixOp::verify() {
230 
231  // ldmatrix reads data from source in shared memory
232  auto srcMemref = getSrcMemref().getType().cast<MemRefType>();
233 
234  // ldmatrix writes data to result/destination in vector registers
235  auto resVector = getRes().getType().cast<VectorType>();
236 
237  // vector register shape, element type, and bitwidth
238  ArrayRef<int64_t> resShape = resVector.getShape();
239  Type resType = resVector.getElementType();
240  int64_t elementBitWidth = resType.getIntOrFloatBitWidth();
241 
242  // ldmatrix loads 32 bits into vector registers per 8-by-8 tile per thread
243  int64_t numElementsPer32b = 32 / elementBitWidth;
244 
245  // number of 8-by-8 tiles
246  int64_t numTiles = getNumTiles();
247 
248  // transpose elements in vector registers at 16b granularity when true
249  bool isTranspose = getTranspose();
250 
251  // address space id for shared memory
252  unsigned smemAddressSpace = gpu::GPUDialect::getWorkgroupAddressSpace();
253 
254  //
255  // verification
256  //
257 
258  if (!(srcMemref.getMemorySpaceAsInt() == smemAddressSpace))
259  return emitError()
260  << "expected nvgpu.ldmatrix srcMemref must have memory space "
261  << smemAddressSpace;
262  if (elementBitWidth > 32)
263  return emitError() << "nvgpu.ldmatrix works for 32b or lower";
264  if (isTranspose && !(elementBitWidth == 16))
265  return emitError()
266  << "nvgpu.ldmatrix transpose works only at 16b granularity";
267  if (!(resShape[1] == numElementsPer32b))
268  return emitError() << "expected vector register shape[1] = "
269  << numElementsPer32b;
270  if (!(resShape[0] == numTiles))
271  return emitError()
272  << "expected vector register shape[0] and numTiles to match";
273 
274  return success();
275 }
