179 template <typename OpTy, typename ExtCastTy>
180 LogicalResult IndexCastOpLowering<OpTy, ExtCastTy>::matchAndRewrite(
181  OpTy op, typename OpTy::Adaptor adaptor,
182  ConversionPatternRewriter &rewriter) const {
183  Type resultType = op.getResult().getType();
184  Type targetElementType =
185  this->typeConverter->convertType(getElementTypeOrSelf(resultType));
186  Type sourceElementType =
187  this->typeConverter->convertType(getElementTypeOrSelf(op.getIn()));
188  unsigned targetBits = targetElementType.getIntOrFloatBitWidth();
189  unsigned sourceBits = sourceElementType.getIntOrFloatBitWidth();
190 
191  if (targetBits == sourceBits) {
192  rewriter.replaceOp(op, adaptor.getIn());
193  return success();
194  }
195 
196  // Handle the scalar and 1D vector cases.
197  Type operandType = adaptor.getIn().getType();
198  if (!operandType.isa<LLVM::LLVMArrayType>()) {
199  Type targetType = this->typeConverter->convertType(resultType);
200  if (targetBits < sourceBits)
201  rewriter.replaceOpWithNewOp<LLVM::TruncOp>(op, targetType,
202  adaptor.getIn());
203  else
204  rewriter.replaceOpWithNewOp<ExtCastTy>(op, targetType, adaptor.getIn());
205  return success();
206  }
207 
208  if (!resultType.isa<VectorType>())
209  return rewriter.notifyMatchFailure(op, "expected vector result type");
210 
211  return LLVM::detail::handleMultidimensionalVectors(
212  op.getOperation(), adaptor.getOperands(), *(this->getTypeConverter()),
213  [&](Type llvm1DVectorTy, ValueRange operands) -> Value {
214  typename OpTy::Adaptor adaptor(operands);
215  if (targetBits < sourceBits) {
216  return rewriter.create<LLVM::TruncOp>(op.getLoc(), llvm1DVectorTy,
217  adaptor.getIn());
218  }
219  return rewriter.create<ExtCastTy>(op.getLoc(), llvm1DVectorTy,
220  adaptor.getIn());
221  },
222  rewriter);
223 }

229 LogicalResult AddUICarryOpLowering::matchAndRewrite(
230  arith::AddUICarryOp op, OpAdaptor adaptor,
231  ConversionPatternRewriter &rewriter) const {
232  Type operandType = adaptor.getLhs().getType();
233  Type sumResultType = op.getSum().getType();
234  Type carryResultType = op.getCarry().getType();
235 
236  if (!LLVM::isCompatibleType(operandType))
237  return failure();
238 
239  MLIRContext *ctx = rewriter.getContext();
240  Location loc = op.getLoc();
241 
242  // Handle the scalar and 1D vector cases.
243  if (!operandType.isa<LLVM::LLVMArrayType>()) {
244  Type newCarryType = typeConverter->convertType(carryResultType);
245  Type structType =
246  LLVM::LLVMStructType::getLiteral(ctx, {sumResultType, newCarryType});
247  Value addOverflow = rewriter.create<LLVM::UAddWithOverflowOp>(
248  loc, structType, adaptor.getLhs(), adaptor.getRhs());
249  Value sumExtracted =
250  rewriter.create<LLVM::ExtractValueOp>(loc, addOverflow, 0);
251  Value carryExtracted =
252  rewriter.create<LLVM::ExtractValueOp>(loc, addOverflow, 1);
253  rewriter.replaceOp(op, {sumExtracted, carryExtracted});
254  return success();
255  }
256 
257  if (!sumResultType.isa<VectorType>())
258  return rewriter.notifyMatchFailure(loc, "expected vector result types");
259 
260  return rewriter.notifyMatchFailure(loc,
261  "ND vector types are not supported yet");
262 }

268 // Convert arith.cmp predicate into the LLVM dialect CmpPredicate. The two enums
269 // share numerical values so just cast.
270 template <typename LLVMPredType, typename PredType>
271 static LLVMPredType convertCmpPredicate(PredType pred) {
272  return static_cast<LLVMPredType>(pred);
273 }

275 LogicalResult
276 CmpIOpLowering::matchAndRewrite(arith::CmpIOp op, OpAdaptor adaptor,
277  ConversionPatternRewriter &rewriter) const {
278  Type operandType = adaptor.getLhs().getType();
279  Type resultType = op.getResult().getType();
280 
281  // Handle the scalar and 1D vector cases.
282  if (!operandType.isa<LLVM::LLVMArrayType>()) {
283  rewriter.replaceOpWithNewOp<LLVM::ICmpOp>(
284  op, typeConverter->convertType(resultType),
285  convertCmpPredicate<LLVM::ICmpPredicate>(op.getPredicate()),
286  adaptor.getLhs(), adaptor.getRhs());
287  return success();
288  }
289 
290  if (!resultType.isa<VectorType>())
291  return rewriter.notifyMatchFailure(op, "expected vector result type");
292 
293  return LLVM::detail::handleMultidimensionalVectors(
294  op.getOperation(), adaptor.getOperands(), *getTypeConverter(),
295  [&](Type llvm1DVectorTy, ValueRange operands) {
296  OpAdaptor adaptor(operands);
297  return rewriter.create<LLVM::ICmpOp>(
298  op.getLoc(), llvm1DVectorTy,
299  convertCmpPredicate<LLVM::ICmpPredicate>(op.getPredicate()),
300  adaptor.getLhs(), adaptor.getRhs());
301  },
302  rewriter);
303 }
