9 // A pass that converts sparse tensor primitives into calls into a runtime
10 // support library. Sparse tensor types are converted into opaque pointers
11 // to the underlying sparse storage schemes. The use of opaque pointers
12 // together with runtime support library keeps the conversion relatively
13 // simple, but at the expense of IR opacity, which obscures opportunities
14 // for subsequent optimization of the IR. An alternative is provided by
15 // the SparseTensorCodegen pass.

112 /// Populates the given sizes array for concatenation from type (for static
113 /// sizes) and from an already-converted opaque pointer source (for dynamic
114 /// sizes).
115 static void concatSizesFromInputs(OpBuilder &builder,
116  SmallVectorImpl<Value> &sizes, Location loc,
117  ShapedType dstTp, ValueRange srcs,
118  unsigned dim) {
119  auto dstShape = dstTp.getShape();
120 
121  auto srcTp = srcs[0].getType().cast<ShapedType>();
122  auto srcEnc = getSparseTensorEncoding(srcTp);
123  // We first fills the sizes from an input tensor, and then
124  // compute the size of the concatenation dimension if necessary.
125  if (srcEnc)
126  // Reuses sizes from an arbitrary input tensor is fine.
127  sizesFromPtr(builder, sizes, loc, srcEnc, srcTp, srcs[0]);
128  else
129  sizesFromSrc(builder, sizes, loc, srcs[0]);
130 
131  // Sum up on the `dim` if the dimension is dynamic.
132  if (dstShape[dim] != ShapedType::kDynamicSize) {
133  // Faithfully take the static size.
134  sizes[dim] = constantIndex(builder, loc, dstShape[dim]);
135  } else {
136  // Else, compute the shape dynamically.
137  for (size_t i = 1, sz = srcs.size(); i < sz; i++) {
138  auto srcTp = srcs[i].getType().cast<ShapedType>();
139  auto encSrc = getSparseTensorEncoding(srcTp);
140  Value srcSz =
141  encSrc ? sizeFromPtrAtDim(builder, loc, encSrc, srcTp, srcs[i], dim)
142  : linalg::createOrFoldDimOp(builder, loc, srcs[i], dim);
143  // Sum up all the sizes.
144  sizes[dim] = builder.create<arith::AddIOp>(loc, sizes[dim], srcSz);
145  }
146  }
147 }

257 // TODO: see the note at `_mlir_ciface_newSparseTensor` about how
258 // the meaning of the various arguments (e.g., "sizes" vs "shapes")
259 // is inconsistent between the different actions.
260 NewCallParams &NewCallParams::genBuffers(SparseTensorEncodingAttr enc,
261  ValueRange dimSizes, ShapedType stp) {
262  const unsigned lvlRank = enc.getDimLevelType().size();
263  const unsigned dimRank = stp.getRank();
264  // Sparsity annotations.
265  SmallVector<Value> lvlTypes;
266  for (auto dlt : enc.getDimLevelType())
267  lvlTypes.push_back(constantDimLevelTypeEncoding(builder, loc, dlt));
268  assert(lvlTypes.size() == lvlRank && "Level-rank mismatch");
269  params[kParamLvlTypes] = genBuffer(builder, loc, lvlTypes);
270  // Dimension-sizes array of the enveloping tensor. Useful for either
271  // verification of external data, or for construction of internal data.
272  assert(dimSizes.size() == dimRank && "Dimension-rank mismatch");
273  params[kParamDimSizes] = genBuffer(builder, loc, dimSizes);
274  // The level-sizes array must be passed as well, since for arbitrary
275  // dim2lvl mappings it cannot be trivially reconstructed at runtime.
276  // For now however, since we're still assuming permutations, we will
277  // initialize this parameter alongside the `dim2lvl` and `lvl2dim`
278  // parameters below. We preinitialize `lvlSizes` for code symmetry.
279  SmallVector<Value> lvlSizes(lvlRank);
280  // The dimension-to-level mapping and its inverse. We must preinitialize
281  // `dim2lvl` so that the true branch below can perform random-access
282  // `operator[]` assignment. We preinitialize `lvl2dim` for code symmetry.
283  SmallVector<Value> dim2lvl(dimRank);
284  SmallVector<Value> lvl2dim(lvlRank);
285  auto dimOrder = enc.getDimOrdering();
286  if (dimOrder) {
287  assert(dimOrder.isPermutation());
288  for (unsigned l = 0; l < lvlRank; l++) {
289  // The `d`th source variable occurs in the `l`th result position.
290  uint64_t d = dimOrder.getDimPosition(l);
291  dim2lvl[d] = constantIndex(builder, loc, l);
292  lvl2dim[l] = constantIndex(builder, loc, d);
293  lvlSizes[l] = dimSizes[d];
294  }
295  } else {
296  assert(dimRank == lvlRank && "Rank mismatch");
297  for (unsigned i = 0; i < lvlRank; i++) {
298  dim2lvl[i] = lvl2dim[i] = constantIndex(builder, loc, i);
299  lvlSizes[i] = dimSizes[i];
300  }
301  }
302  params[kParamLvlSizes] = genBuffer(builder, loc, lvlSizes);
303  params[kParamLvl2Dim] = genBuffer(builder, loc, lvl2dim);
304  params[kParamDim2Lvl] =
305  dimOrder ? genBuffer(builder, loc, dim2lvl) : params[kParamLvl2Dim];
306  // Secondary and primary types encoding.
307  setTemplateTypes(enc, stp);
308  // Finally, make note that initialization is complete.
309  assert(isInitialized() && "Initialization failed");
310  // And return `this` for method chaining.
311  return *this;
312 }

456 /// Generate code for a general sparse to sparse reshaping operation.
457 /// Note that unlike dense reshaping (which can be done with a "cheap"
458 /// change of view), sparse reshaping is currently done with actual
459 /// data shuffling.
460 ///
461 /// TODO: proportional to nnz, but still a lot of data movement
462 /// https://github.com/llvm/llvm-project/issues/56477
463 ///
464 /// iter = src->toCOO();
465 /// coo = newSparseCOO()
466 /// while (elem = iter->getNext()) {
467 /// coo->add(reshape(elem.indices), elem.value)
468 /// }
469 /// s = newSparseTensor(coo)
470 template <typename ReshapeOp>
471 static LogicalResult
472 genSparse2SparseReshape(ReshapeOp op, typename ReshapeOp::Adaptor adaptor,
473  ConversionPatternRewriter &rewriter) {
474  Location loc = op.getLoc();
475  auto srcTp = op.getSrc().getType().template cast<RankedTensorType>();
476  auto dstTp = op.getResult().getType().template cast<RankedTensorType>();
477  auto encSrc = getSparseTensorEncoding(srcTp);
478  auto encDst = getSparseTensorEncoding(dstTp);
479  if (!encDst || !encSrc)
480  return failure();
481 
482  unsigned srcRank = srcTp.getRank();
483  unsigned dstRank = dstTp.getRank();
484  Type elemTp = srcTp.getElementType();
485  assert(elemTp == dstTp.getElementType() &&
486  "reshape should not change element type");
487  // Start an iterator over the source tensor (in original index order).
488  auto noPerm = SparseTensorEncodingAttr::get(
489  op->getContext(), encSrc.getDimLevelType(), AffineMap(), AffineMap(),
490  encSrc.getPointerBitWidth(), encSrc.getIndexBitWidth());
491  SmallVector<Value> srcSizes;
492  sizesFromPtr(rewriter, srcSizes, loc, encSrc, srcTp, adaptor.getSrc());
493  NewCallParams params(rewriter, loc);
494  Value iter = params.genBuffers(noPerm, srcSizes, srcTp)
495  .genNewCall(Action::kToIterator, adaptor.getSrc());
496  // Start a new COO for the destination tensor.
497  SmallVector<Value> dstSizes;
498  if (dstTp.hasStaticShape()) {
499  sizesFromType(rewriter, dstSizes, loc, dstTp);
500  } else {
501  ArrayRef<int64_t> dstShape = dstTp.getShape();
502  genReshapeDstShape(loc, rewriter, dstSizes, srcSizes, dstShape,
503  op.getReassociationIndices());
504  }
505  Value coo =
506  params.genBuffers(encDst, dstSizes, dstTp).genNewCall(Action::kEmptyCOO);
507  Value dstPerm = params.getDim2LvlMap();
508  // Construct a while loop over the iterator.
509  Value srcIdx = genAlloca(rewriter, loc, srcRank, rewriter.getIndexType());
510  Value dstIdx = genAlloca(rewriter, loc, dstRank, rewriter.getIndexType());
511  Value elemPtr = genAllocaScalar(rewriter, loc, elemTp);
512  SmallVector<Value> noArgs;
513  SmallVector<Type> noTypes;
514  auto whileOp = rewriter.create<scf::WhileOp>(loc, noTypes, noArgs);
515  Block *before = rewriter.createBlock(&whileOp.getBefore(), {}, noTypes);
516  rewriter.setInsertionPointToEnd(before);
517  Value cond = genGetNextCall(rewriter, loc, iter, srcIdx, elemPtr);
518  rewriter.create<scf::ConditionOp>(loc, cond, before->getArguments());
519  // Translate indices from source to target and insert. Note that we do
520  // not need to store the value in elemPtr, as the value is still there.
521  Block *after = rewriter.createBlock(&whileOp.getAfter(), {}, noTypes);
522  rewriter.setInsertionPointToStart(after);
523  translateIndices(loc, rewriter, op.getReassociationIndices(), dstTp, srcTp,
524  dstIdx, srcIdx, dstSizes, srcSizes);
525  genAddEltCall(rewriter, loc, elemTp, coo, elemPtr, dstIdx, dstPerm);
526  rewriter.create<scf::YieldOp>(loc);
527  // Final call to construct sparse tensor storage and free temporary resources.
528  rewriter.setInsertionPointAfter(whileOp);
529  Value dst = params.genNewCall(Action::kFromCOO, coo);
530  genDelCOOCall(rewriter, loc, elemTp, coo);

536 // Generates a while loop that iterates over the COO list extracted
537 // from `t`, using `bodyBuilder` to build the loop body.
538 // while (elem = coo->getNext()) {
539 // bodyBuilder
540 // }
541 // TODO: It can be used by other operators (ReshapeOp, ConvertOP) conversion to
542 // reduce code repetition!
543 // TODO: rename to `genSparseIterationLoop`?
544 static void genSparseCOOIterationLoop(
545  ConversionPatternRewriter &rewriter, Location loc, Value t,
546  RankedTensorType tensorTp,
547  function_ref<void(OpBuilder &, Location, Value, Value)> bodyBuilder) {
548  auto enc = getSparseTensorEncoding(tensorTp);
549  assert(enc && "Generating Sparse Tensor COO Loop on a Dense Tensor!");
550 
551  unsigned rank = tensorTp.getRank();
552  Type elemTp = tensorTp.getElementType();
553 
554  // Start an iterator over the tensor (in original index order).
555  auto noPerm = SparseTensorEncodingAttr::get(
556  rewriter.getContext(), enc.getDimLevelType(), AffineMap(), AffineMap(),
557  enc.getPointerBitWidth(), enc.getIndexBitWidth());
558  SmallVector<Value> sizes;
559  sizesFromPtr(rewriter, sizes, loc, noPerm, tensorTp, t);
560  Value iter = NewCallParams(rewriter, loc)
561  .genBuffers(noPerm, sizes, tensorTp)
562  .genNewCall(Action::kToIterator, t);
563 
564  // Construct a while loop over the iterator.
565  Value srcIdx = genAlloca(rewriter, loc, rank, rewriter.getIndexType());
566  Value elemPtr = genAllocaScalar(rewriter, loc, elemTp);
567  SmallVector<Value> noArgs;
568  SmallVector<Type> noTypes;
569  auto whileOp = rewriter.create<scf::WhileOp>(loc, noTypes, noArgs);
570  Block *before = rewriter.createBlock(&whileOp.getBefore(), {}, noTypes);
571  rewriter.setInsertionPointToEnd(before);
572  Value cond = genGetNextCall(rewriter, loc, iter, srcIdx, elemPtr);
573  rewriter.create<scf::ConditionOp>(loc, cond, before->getArguments());
574  Block *after = rewriter.createBlock(&whileOp.getAfter(), {}, noTypes);
575  rewriter.setInsertionPointToStart(after);
576 
577  bool hasDenseDim = llvm::any_of(
578  enc.getDimLevelType(), [](DimLevelType dlt) { return isDenseDLT(dlt); });
579  if (hasDenseDim) {
580  Value elemV = rewriter.create<memref::LoadOp>(loc, elemPtr);
581  Value isZero = genIsNonzero(rewriter, loc, elemV);
582  scf::IfOp ifOp = rewriter.create<scf::IfOp>(loc, isZero, /*else*/ false);
583  rewriter.setInsertionPointToStart(&ifOp.getThenRegion().front());
584  }
585  // Callback here to build loop body.
586  bodyBuilder(rewriter, loc, srcIdx, elemPtr);
587 
588  // Exit the scope from the IfOp.
589  if (hasDenseDim)
590  rewriter.setInsertionPointToEnd(after);
591 
592  rewriter.create<scf::YieldOp>(loc);
593  // Finish generating loop.
594  rewriter.setInsertionPointAfter(whileOp);
595 
596  // Free memory for iterator.
597  genDelIteratorCall(rewriter, loc, elemTp, iter);
598 }
599 
600 // Generate loop that iterates over a dense tensor.
601 // for i1 in dim1
602 // ..
603 // for ik in dimk
604 // val = a[i1,..,ik]
605 // if val != 0
606 // bodyBuilder(v, [i1, ..., ik])
607 // TODO: It can be used by other operators (ReshapeOp, ConvertOP) conversion to
608 // reduce code repetition!
609 static void genDenseTensorIterationLoop(
610  ConversionPatternRewriter &rewriter, Location loc, Value t,
611  RankedTensorType tensorTp,
612  function_ref<void(OpBuilder &, Location, ValueRange)> bodyBuilder) {
613  assert(!getSparseTensorEncoding(tensorTp) &&
614  "Generating Dense Tensor Loop on a Sparse Tensor!");
615 
616  unsigned rank = tensorTp.getRank();
617  Value zero = constantIndex(rewriter, loc, 0);
618  Value one = constantIndex(rewriter, loc, 1);
619 
620  SmallVector<Value> lo;
621  SmallVector<Value> hi;
622  SmallVector<Value> st;
623 
624  // Fill out loop iteration information.
625  for (unsigned i = 0; i < rank; i++) {
626  lo.push_back(zero);
627  hi.push_back(linalg::createOrFoldDimOp(rewriter, loc, t, i));
628  st.push_back(one);
629  }
630 
631  scf::buildLoopNest(rewriter, loc, lo, hi, st, {},
632  [&](OpBuilder &builder, Location loc, ValueRange ivs,
633  ValueRange args) -> scf::ValueVector {
634  // Invoke callback to build the body of the loop.
635  bodyBuilder(builder, loc, ivs);
636  return {};
637  });
638 }

772 /// Sparse conversion rule for the convert operator.
773 class SparseTensorConvertConverter : public OpConversionPattern<ConvertOp> {
774 public:
775  using OpConversionPattern::OpConversionPattern;
776  SparseTensorConvertConverter(MLIRContext *context,
777  SparseTensorConversionOptions o)
778  : OpConversionPattern<ConvertOp>(context), options(o) {}
779  SparseTensorConvertConverter(TypeConverter &typeConv, MLIRContext *context,
780  SparseTensorConversionOptions o)
781  : OpConversionPattern<ConvertOp>(typeConv, context), options(o) {}
782 
783  LogicalResult
784  matchAndRewrite(ConvertOp op, OpAdaptor adaptor,
785  ConversionPatternRewriter &rewriter) const override {
786  Location loc = op->getLoc();
787  Type resType = op.getType();
788  Type srcType = op.getSource().getType();
789  auto encDst = getSparseTensorEncoding(resType);
790  auto encSrc = getSparseTensorEncoding(srcType);
791  Value src = adaptor.getOperands()[0];
792  if (encDst && encSrc) {
793  // This is a sparse => sparse conversion, which is handled as follows:
794  // t = src->toCOO(); ; src to COO in dst order
795  // dst = newSparseTensor(t)
796  // Using the coordinate scheme as an intermediate does not always
797  // yield the fastest conversion but avoids the need for a full
798  // O(N^2) conversion matrix.
799  if (encDst == encSrc) {
800  rewriter.replaceOp(op, adaptor.getOperands()); // hidden nop cast
801  return success();
802  }
803  SmallVector<Value> sizes;
804  NewCallParams params(rewriter, loc);
805  ShapedType stp = srcType.cast<ShapedType>();
806  sizesFromPtr(rewriter, sizes, loc, encSrc, stp, src);
807  bool useDirectConversion;
808  switch (options.sparseToSparseStrategy) {
809  case SparseToSparseConversionStrategy::kViaCOO:
810  useDirectConversion = false;
811  break;
812  case SparseToSparseConversionStrategy::kDirect:
813  useDirectConversion = true;
814  assert(canUseDirectConversion(encDst.getDimLevelType()) &&
815  "Unsupported target for direct sparse-to-sparse conversion");
816  break;
817  case SparseToSparseConversionStrategy::kAuto:
818  useDirectConversion = canUseDirectConversion(encDst.getDimLevelType());
819  break;
820  }
821  if (useDirectConversion) {
822  rewriter.replaceOp(op, params.genBuffers(encDst, sizes, stp)
823  .genNewCall(Action::kSparseToSparse, src));
824  } else { // use via-COO conversion.
825  // Set up encoding with right mix of src and dst so that the two
826  // method calls can share most parameters, while still providing
827  // the correct sparsity information to either of them.
828  auto enc = SparseTensorEncodingAttr::get(
829  op->getContext(), encDst.getDimLevelType(), encDst.getDimOrdering(),
830  encDst.getHigherOrdering(), encSrc.getPointerBitWidth(),
831  encSrc.getIndexBitWidth());
832  // TODO: This is the only place where `kToCOO` (or `kToIterator`)
833  // is called with a non-identity permutation. Is there any clean
834  // way to push the permutation over to the `kFromCOO` side instead?
835  Value coo =
836  params.genBuffers(enc, sizes, stp).genNewCall(Action::kToCOO, src);
837  Value dst = params.setTemplateTypes(encDst, stp)
838  .genNewCall(Action::kFromCOO, coo);
839  genDelCOOCall(rewriter, loc, stp.getElementType(), coo);
840  rewriter.replaceOp(op, dst);
841  }
842  return success();
843  }
844  if (!encDst && encSrc) {
845  // This is sparse => dense conversion, which is handled as follows:
846  // dst = new Tensor(0);
847  // iter = new SparseTensorIterator(src);
848  // while (elem = iter->getNext()) {
849  // dst[elem.indices] = elem.value;
850  // }
851  // delete iter;
852  RankedTensorType dstTensorTp = resType.cast<RankedTensorType>();
853  RankedTensorType srcTensorTp = srcType.cast<RankedTensorType>();
854  unsigned rank = dstTensorTp.getRank();
855  Type elemTp = dstTensorTp.getElementType();
856  // Fabricate a no-permutation encoding for NewCallParams
857  // The pointer/index types must be those of `src`.
858  // The dimLevelTypes aren't actually used by Action::kToIterator.
859  encDst = SparseTensorEncodingAttr::get(
860  op->getContext(),
861  SmallVector<DimLevelType>(rank, DimLevelType::Dense), AffineMap(),
862  AffineMap(), encSrc.getPointerBitWidth(), encSrc.getIndexBitWidth());
863  SmallVector<Value> sizes;
864  sizesFromPtr(rewriter, sizes, loc, encSrc, srcTensorTp, src);
865  Value iter = NewCallParams(rewriter, loc)
866  .genBuffers(encDst, sizes, dstTensorTp)
867  .genNewCall(Action::kToIterator, src);
868  Value ind = genAlloca(rewriter, loc, rank, rewriter.getIndexType());
869  Value elemPtr = genAllocaScalar(rewriter, loc, elemTp);
870  Block *insertionBlock = rewriter.getInsertionBlock();
871  // TODO: Dense buffers should be allocated/deallocated via the callback
872  // in BufferizationOptions.
873  Value dst = allocDenseTensor(rewriter, loc, dstTensorTp, sizes);
874  SmallVector<Value> noArgs;
875  SmallVector<Type> noTypes;
876  auto whileOp = rewriter.create<scf::WhileOp>(loc, noTypes, noArgs);
877  Block *before = rewriter.createBlock(&whileOp.getBefore(), {}, noTypes);
878  rewriter.setInsertionPointToEnd(before);
879  Value cond = genGetNextCall(rewriter, loc, iter, ind, elemPtr);
880  rewriter.create<scf::ConditionOp>(loc, cond, before->getArguments());
881  Block *after = rewriter.createBlock(&whileOp.getAfter(), {}, noTypes);
882  rewriter.setInsertionPointToStart(after);
883  SmallVector<Value> ivs = loadIndices(rewriter, loc, rank, ind);
884  insertScalarIntoDenseTensor(rewriter, loc, elemPtr, dst, ivs);
885  rewriter.create<scf::YieldOp>(loc);
886  rewriter.setInsertionPointAfter(whileOp);
887  genDelIteratorCall(rewriter, loc, elemTp, iter);
888  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, resType, dst);
889  // Deallocate the buffer.
890  if (bufferization::allocationDoesNotEscape(op->getOpResult(0))) {
891  rewriter.setInsertionPoint(insertionBlock->getTerminator());
892  deallocDenseTensor(rewriter, loc, dst);
893  }
894  return success();
895  }
896  if (!encDst && !encSrc) {
897  // dense => dense
898  return failure();
899  }
900  // This is a dense => sparse conversion or a sparse constant in COO =>
901  // sparse conversion, which is handled as follows:
902  // t = newSparseCOO()
903  // ...code to fill the COO tensor t...
904  // s = newSparseTensor(t)
905  //
906  // To fill the COO tensor from a dense tensor:
907  // for i1 in dim1
908  // ..
909  // for ik in dimk
910  // val = a[i1,..,ik]
911  // if val != 0
912  // t->add(val, [i1,..,ik], [p1,..,pk])
913  //
914  // To fill the COO tensor from a sparse constant in COO format:
915  // for i in range(NNZ)
916  // val = values[i]
917  // [i1,..,ik] = indices[i]
918  // t->add(val, [i1,..,ik], [p1,..,pk])
919  //
920  // Note that the dense tensor traversal code is actually implemented
921  // using MLIR IR to avoid having to expose too much low-level
922  // memref traversal details to the runtime support library.
923  // Also note that the code below only generates the "new" ops and
924  // the loop-nest per se; whereas the entire body of the innermost
925  // loop is generated by genAddElt().
926  ShapedType stp = resType.cast<ShapedType>();
927  unsigned rank = stp.getRank();
928  SmallVector<Value> sizes;
929  sizesFromSrc(rewriter, sizes, loc, src);
930  NewCallParams params(rewriter, loc);
931  Value coo =
932  params.genBuffers(encDst, sizes, stp).genNewCall(Action::kEmptyCOO);
933  Value ind = genAlloca(rewriter, loc, rank, rewriter.getIndexType());
934  Value perm = params.getDim2LvlMap();
935  Type eltType = stp.getElementType();
936  Value elemPtr = genAllocaScalar(rewriter, loc, eltType);
937  genDenseTensorOrSparseConstantIterLoop(
938  rewriter, loc, src, rank,
939  [&](OpBuilder &builder, Location loc, Value val, ValueRange indices) {
940  for (unsigned i = 0; i < rank; i++) {
941  Value idx = constantIndex(builder, loc, i);
942  builder.create<memref::StoreOp>(loc, indices[i], ind, idx);
943  }
944  builder.create<memref::StoreOp>(loc, val, elemPtr);
945  genAddEltCall(builder, loc, eltType, coo, elemPtr, ind, perm);
946  });
947  // Final call to construct sparse tensor storage.
948  Value dst = params.genNewCall(Action::kFromCOO, coo);
949  genDelCOOCall(rewriter, loc, eltType, coo);
950  rewriter.replaceOp(op, dst);
951  return success();
952  }
953 
954 private:
955  /// Options to control sparse code generation.
956  SparseTensorConversionOptions options;
957 };

1096 public:

1100  ConversionPatternRewriter &rewriter) const override {
1101  Location loc = op->getLoc();
1102  RankedTensorType srcType =
1103  op.getTensor().getType().cast<RankedTensorType>();
1104  Type eltType = srcType.getElementType();
1105  Type boolType = rewriter.getIntegerType(1);
1106  Type idxType = rewriter.getIndexType();
1107  // All initialization should be done on entry of the loop nest.
1108  rewriter.setInsertionPointAfter(op.getTensor().getDefiningOp());
1109  // Determine the size for access expansion (always the innermost stored
1110  // dimension size, translated back to original dimension).
1111  auto enc = getSparseTensorEncoding(srcType);
1112  unsigned innerDim = toOrigDim(srcType, srcType.getRank() - 1);
1113  auto sz = sizeFromPtrAtDim(rewriter, loc, enc, srcType, adaptor.getTensor(),
1114  innerDim);

1116  // We do not use stack buffers for this, since the expanded size may
1117  // be rather large (as it envelops a single expanded dense dimension).
1118  Value values = genAlloc(rewriter, loc, sz, eltType);
1119  Value filled = genAlloc(rewriter, loc, sz, boolType);
1120  Value indices = genAlloc(rewriter, loc, sz, idxType);
1121  Value zero = constantZero(rewriter, loc, idxType);
1122  // Reset the values/filled-switch to all-zero/false. Note that this
1123  // introduces an O(N) operation into the computation, but this reset
1124  // operation is amortized over the innermost loops for the access
1125  // pattern expansion. As noted in the operation doc, we would like
1126  // to amortize this setup cost even between kernels.
1127  rewriter.create<linalg::FillOp>(
1128  loc, ValueRange{constantZero(rewriter, loc, eltType)},
1129  ValueRange{values});
1130  rewriter.create<linalg::FillOp>(
1131  loc, ValueRange{constantZero(rewriter, loc, boolType)},
1132  ValueRange{filled});
1133  // Replace expansion op with these buffers and initial index.
1134  assert(op.getNumResults() == 4);
1135  rewriter.replaceOp(op, {values, filled, indices, zero});

1179 /// Sparse conversion rule for the concatenate operator.
1180 class SparseTensorConcatConverter : public OpConversionPattern<ConcatenateOp> {
1181 public:
1182  using OpConversionPattern::OpConversionPattern;
1183  LogicalResult
1184  matchAndRewrite(ConcatenateOp op, OpAdaptor adaptor,
1185  ConversionPatternRewriter &rewriter) const override {
1186  // The conversion works as follow:
1187  // (1). When output is sparse, and mix of inputs:
1188  // a_sparse = concat (b_dense, c_sparse, ....)
1189  // =>
1190  // coo_for_a = newSparseCOO(shapeOf(a))
1191  // for i, j, k // dense input
1192  // coo->add(adjustForOffset(i,j,k), b[i,j,k])
1193  //
1194  // for elem in sparse_input
1195  // coo->add(adjustForOffset(elem.indices), elem.value)
1196  // ...
1197  // a = newSparseTensor(coo_for_a)
1198  // return a
1199  //
1200  // (2). When output is dense, and mix of inputs:
1201  // a_dense = concat (b_dense, c_sparse, ....)
1202  // =>
1203  // a = malloc(shapeOf(a))
1204  // for i, j, k // dense input
1205  // a[ adjustForOffset(i,j,k) ] = b[i,j,k]
1206  //
1207  // for elem in sparse_input
1208  // a[ adjustForOffset(elem.indices) ] = elem.value
1209  // return a
1210  Location loc = op.getLoc();
1211  auto dstTp = op.getType().cast<RankedTensorType>();
1212  auto encDst = getSparseTensorEncoding(dstTp);
1213  Type elemTp = dstTp.getElementType();
1214  uint64_t concatDim = op.getDimension().getZExtValue();
1215  unsigned rank = dstTp.getRank();
1216 
1217  Value dst; // destination tensor
1218  Value dstPerm; // destination tensor permutation (if sparse out)
1219  // A pointer to the value being inserted (if dense => sparse)
1220  Value elemPtr;
1221  // Memory that holds the COO for destination tensor (if sparse out)
1222  Value dstIdx;
1223  // The offset applied to the dimenstion to be concated (starting from 0)
1224  Value offset = constantIndex(rewriter, loc, 0);
1225 
1226  SmallVector<Value> sizes;
1227  NewCallParams params(rewriter, loc);
1228  concatSizesFromInputs(rewriter, sizes, loc, dstTp, op.getInputs(),
1229  concatDim);
1230 
1231  if (encDst) {
1232  // Start a new COO for the destination tensor.
1233  dst =
1234  params.genBuffers(encDst, sizes, dstTp).genNewCall(Action::kEmptyCOO);
1235  dstPerm = params.getDim2LvlMap();
1236  elemPtr = genAllocaScalar(rewriter, loc, elemTp);
1237  dstIdx = genAlloca(rewriter, loc, rank, rewriter.getIndexType());
1238  } else {
1239  // TODO: Dense buffers should be allocated/deallocated via the callback
1240  // in BufferizationOptions.
1241  dst = allocDenseTensor(rewriter, loc, dstTp, sizes);
1242  }
1243  for (auto it : llvm::zip(op.getInputs(), adaptor.getInputs())) {
1244  Value orignalOp = std::get<0>(it); // Input (with encoding) from Op
1245  Value adaptedOp = std::get<1>(it); // Input (type converted) from adaptor
1246  RankedTensorType srcTp = orignalOp.getType().cast<RankedTensorType>();
1247  auto encSrc = getSparseTensorEncoding(srcTp);
1248  if (encSrc) {
1249  genSparseCOOIterationLoop(
1250  rewriter, loc, adaptedOp, srcTp,
1251  [&](OpBuilder &builder, Location loc, Value idx,
1252  Value elemPtr) -> void {
1253  auto indVec =
1254  loadIndices(builder, loc, rank, idx, concatDim, offset);
1255  if (encDst) {
1256  // Case: sparse => sparse
1257  storeIndices(builder, loc, rank, dstIdx, indVec);
1258  genAddEltCall(builder, loc, elemTp, dst, elemPtr, dstIdx,
1259  dstPerm);
1260  } else {
1261  // Case: sparse => dense
1262  insertScalarIntoDenseTensor(builder, loc, elemPtr, dst, indVec);
1263  }
1264  });
1265  } else {
1266  genDenseTensorIterationLoop(
1267  rewriter, loc, adaptedOp, srcTp,
1268  [&](OpBuilder &builder, Location loc, ValueRange idx) -> void {
1269  if (encDst) {
1270  // Case: dense => sparse
1271  storeIndices(builder, loc, rank, dstIdx, idx, concatDim,
1272  offset);
1273  Value val = genValueForDense(builder, loc, adaptedOp, idx);
1274  builder.create<memref::StoreOp>(loc, val, elemPtr);
1275  genAddEltCall(builder, loc, elemTp, dst, elemPtr, dstIdx,
1276  dstPerm);
1277  } else {
1278  // Case: dense => dense
1279  Value val = genValueForDense(builder, loc, adaptedOp, idx);
1280  SmallVector<Value> indVec(idx);
1281  // Apply offset.
1282  indVec[concatDim] = builder.create<arith::AddIOp>(
1283  loc, indVec[concatDim], offset);
1284  builder.create<memref::StoreOp>(loc, val, dst, indVec);
1285  }
1286  });
1287  }
1288  // Accumulate offset.
1289  // TODO: avoid calling sparseDimSize multiple times by caching the result!
1290  Value curDim = encSrc ? sizeFromPtrAtDim(rewriter, loc, encSrc, srcTp,
1291  adaptedOp, concatDim)
1292  : linalg::createOrFoldDimOp(rewriter, loc,
1293  adaptedOp, concatDim);
1294 
1295  offset = rewriter.create<arith::AddIOp>(loc, offset, curDim);
1296  }
1297  if (encDst) {
1298  // In sparse output case, the destination holds the COO.
1299  Value coo = dst;
1300  dst = params.genNewCall(Action::kFromCOO, coo);
1301  // Release resources.
1302  genDelCOOCall(rewriter, loc, elemTp, coo);
1303  rewriter.replaceOp(op, dst);
1304  } else {
1305  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, dstTp, dst);
1306  }
1307  return success();
1308  }
1309 };
