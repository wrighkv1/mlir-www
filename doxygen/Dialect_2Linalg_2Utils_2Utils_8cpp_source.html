45 static bool isZero(OpFoldResult v) {
46  if (!v)
47  return false;
48  if (auto attr = v.dyn_cast<Attribute>()) {
49  IntegerAttr intAttr = attr.dyn_cast<IntegerAttr>();
50  return intAttr && intAttr.getValue().isZero();
51  }
52  if (auto cst = v.get<Value>().getDefiningOp<arith::ConstantIndexOp>())
53  return cst.value() == 0;
54  return false;

59 // Helper visitor to determine whether an AffineExpr is tiled.
60 // This is achieved by traversing every AffineDimExpr with position `pos` and
61 // checking whether the corresponding `tileSizes[pos]` is non-zero.
62 // This also enforces only positive coefficients occur in multiplications.
63 //
64 // Example:
65 // `d0 + 2 * d1 + d3` is tiled by [0, 0, 0, 2] but not by [0, 0, 2, 0]
66 //
67 struct TileCheck : public AffineExprVisitor<TileCheck> {
68  TileCheck(ArrayRef<OpFoldResult> tileSizes) : tileSizes(tileSizes) {}
69 
70  void visitDimExpr(AffineDimExpr expr) {
71  isTiled |= !isZero(tileSizes[expr.getPosition()]);
72  }
73  void visitAffineBinaryOpExpr(AffineBinaryOpExpr expr) {
74  visit(expr.getLHS());
75  visit(expr.getRHS());
76  if (expr.getKind() == mlir::AffineExprKind::Mul)
77  assert(expr.getRHS().cast<AffineConstantExpr>().getValue() > 0 &&
78  "nonpositive multiplying coefficient");
79  }
80  bool isTiled = false;
81  ArrayRef<OpFoldResult> tileSizes;
82 };

106  auto &region = op.getRegion();
107  if (!llvm::hasSingleElement(region))
108  return llvm::None;
109 
110  Block &block = region.front();
111  if (block.getNumArguments() != 2 ||
112  !block.getArgument(0).getType().isSignlessIntOrFloat() ||
113  !block.getArgument(1).getType().isSignlessIntOrFloat())

227 void getUpperBoundForIndex(Value value, AffineMap &boundMap,
228  SmallVectorImpl<Value> &boundOperands,
229  bool constantRequired) {
230  // Initialize `boundMap` and `boundOperands` to the identity returning
231  // `value`. This combination is the default result of the method if no
232  // simplification is possible.
233  assert(value.getType().isIndex() && "expect value to have index type");
234  boundMap = AffineMap::getMultiDimIdentityMap(1, value.getContext());
235  boundOperands.assign({value});
236  canonicalizeMapAndOperands(&boundMap, &boundOperands);
237 
238  // Continue only if there is an affine index computation to simplify.
239  Operation *definingOp = value.getDefiningOp();
240  if (!definingOp || !isa<AffineApplyOp, AffineMinOp>(definingOp))
241  return;
242 
243  // Get the backward slice containing the affine index computation.
244  SetVector<Operation *> backwardSlice;
245  getBackwardSlice(definingOp, &backwardSlice, [](Operation *op) {
246  return isa<AffineApplyOp, AffineMinOp>(op);
247  });
248  backwardSlice.insert(definingOp);
249 
250  // Setup a system of affine constraints that describe the index computation.
251  FlatAffineValueConstraints constraints;
252 
253  // Helper to find or create an identifier for the given value.
254  auto findOrCreateId = [&](Value value) {
255  if (!constraints.containsVar(value)) {
256  constraints.appendDimVar(value);
257  return true;
258  }
259  unsigned pos;
260  constraints.findVar(value, &pos);
261  return pos < constraints.getNumDimVars();
262  };
263  // Helper to get the position for the given value.
264  auto getPosition = [&](Value value) {
265  unsigned pos;
266  bool exists = constraints.findVar(value, &pos);
267  (void)exists;
268  assert(exists && "expect to find the identifier");
269  return pos;
270  };
271 
272  // Add the affine operations in `backwardSlice` to the constraints.
273  for (Operation *op : llvm::reverse(backwardSlice)) {
274  // Add an identifier for all op results and operands.
275  if (!(llvm::all_of(op->getResults(), findOrCreateId) &&
276  llvm::all_of(op->getOperands(), findOrCreateId)))
277  return;
278 
279  // Add AffineApplyOps to the constraints.
280  if (auto applyOp = dyn_cast<AffineApplyOp>(op)) {
281  AffineMap map = constraints.computeAlignedMap(applyOp.getAffineMap(),
282  applyOp.getOperands());
283  if (failed(constraints.addBound(IntegerPolyhedron::EQ,
284  getPosition(applyOp.getResult()), map)))
285  return;
286  continue;
287  }
288  // Add AffineMinOps to the constraints.
289  auto minOp = cast<AffineMinOp>(op);
290  AffineMap map = constraints.computeAlignedMap(minOp.getAffineMap(),
291  minOp.getOperands());
292  if (failed(constraints.addBound(IntegerPolyhedron::UB,
293  getPosition(minOp.getResult()), map,
294  /*isClosedBound=*/true)))
295  return;
296  }
297 
298  // Obtain an upper bound for the affine index computation by projecting out
299  // all temporary results and expressing the upper bound for `value` in terms
300  // of the terminals of the index computation.
301  unsigned pos = getPosition(value);
302  if (constantRequired) {
303  auto ubConst = constraints.getConstantBound64(
304  FlatAffineValueConstraints::BoundType::UB, pos);
305  if (!ubConst)
306  return;
307 
308  boundMap = AffineMap::getConstantMap(*ubConst, value.getContext());
309  return;
310  }
311 
312  SmallVector<AffineMap> lowerBounds(1), upperBounds(1);
313  constraints.getSliceBounds(pos, 1, value.getContext(), &lowerBounds,
314  &upperBounds,
315  /*getClosedUB=*/true);
316  // Verify `upperBounds[0]` is valid and has at least one result.
317  if (!upperBounds[0] || upperBounds[0].getNumResults() == 0)
318  return;
319 
320  // Set `boundMap` and `boundOperands` to the computed upper bound.
321  boundMap = upperBounds[0];
322  constraints.getAllValues(&boundOperands);
323  erase_value(boundOperands, value);
324  canonicalizeMapAndOperands(&boundMap, &boundOperands);
325 }

327 FailureOr<int64_t> getConstantUpperBoundForIndex(Value value) {
328  // Compute an upper bound for `value`.
329  AffineMap boundMap;
330  SmallVector<Value> boundOperands;
331  getUpperBoundForIndex(value, boundMap, boundOperands,
332  /*constantRequired=*/true);
333 
334  // Search the results of `boundMap` for constant upper bounds.
335  SmallVector<int64_t> constantBounds;
336  for (AffineExpr result : boundMap.getResults())
337  if (auto constExpr = result.dyn_cast<AffineConstantExpr>())
338  constantBounds.push_back(constExpr.getValue());
339 
340  // Return the minimal upper bound or failure if none is found.
341  if (constantBounds.empty())
342  return failure();
343  return *std::min_element(constantBounds.begin(), constantBounds.end());
344 }

346 Value makeComposedPadHighOp(OpBuilder &b, Location loc, RankedTensorType type,
347  Value source, Value pad, bool nofold) {
348  // Exit if `source` is not defined by an ExtractSliceOp.
349  auto sliceOp = source.getDefiningOp<tensor::ExtractSliceOp>();
350  if (!sliceOp)
351  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
352 
353  // Search the `source` use-def chain for padded LinalgOps.
354  Value current = sliceOp.getSource();
355  while (current) {
356  auto linalgOp = current.getDefiningOp<LinalgOp>();
357  if (!linalgOp)
358  break;
359  OpResult opResult = current.cast<OpResult>();
360  current = linalgOp.getDpsInitOperand(opResult.getResultNumber())->get();
361  }
362  auto padOp = current ? current.getDefiningOp<tensor::PadOp>() : nullptr;
363 
364  // Exit if the search fails to match a tensor::PadOp at the end of the matched
365  // LinalgOp sequence.
366  if (!padOp)
367  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
368 
369  // Exit if the padded result type does not match.
370  if (sliceOp.getSource().getType() != type)
371  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
372 
373  // Exit if the LinalgOps are not high padded.
374  if (llvm::any_of(padOp.getMixedLowPad(), [](OpFoldResult ofr) {
375  return getConstantIntValue(ofr) != static_cast<int64_t>(0);
376  }))
377  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
378 
379  // Exit if `padOpSliceOp`, which defines the slice used by
380  // `padOp`, is rank-reducing.
381  auto padOpSliceOp = padOp.getSource().getDefiningOp<tensor::ExtractSliceOp>();
382  if (!padOpSliceOp ||
383  sliceOp.getMixedSizes().size() != padOpSliceOp.getMixedSizes().size())
384  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
385 
386  // Exit if the sizes of the dynamic sizes of `sliceOp` do not match the size
387  // of the slice padded by `padOp`.
388  if (llvm::any_of(
389  llvm::zip(sliceOp.getMixedSizes(), padOpSliceOp.getMixedSizes()),
390  [](std::tuple<OpFoldResult, OpFoldResult> it) {
391  return !isEqualConstantIntOrValue(std::get<0>(it), std::get<1>(it));
392  }))
393  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
394 
395  // Exit if the padding values do not match.
396  Attribute padOpPadAttr, padAttr;
397  Value padOpPad = padOp.getConstantPaddingValue();
398  if (!padOpPad || !matchPattern(padOpPad, m_Constant(&padOpPadAttr)) ||
399  !matchPattern(pad, m_Constant(&padAttr)) || padOpPadAttr != padAttr)
400  return tensor::createPadHighOp(type, source, pad, nofold, loc, b);
401 
402  // Return the padded result if the padding values and sizes match.
403  return sliceOp.getSource();
404 }

406 GenericOp makeTransposeOp(OpBuilder &b, Location loc, Value inputTensor,
407  Value outputTensor,
408  ArrayRef<int64_t> transposeVector) {
409  auto resultTensorType = outputTensor.getType().cast<RankedTensorType>();
410  Type elementType = resultTensorType.getElementType();
411 
412  assert(isPermutation(transposeVector) &&
413  "expect transpose vector to be a permutation");
414  assert(transposeVector.size() ==
415  static_cast<size_t>(resultTensorType.getRank()) &&
416  "expect transpose vector size to match result tensor rank");
417 
418  // Compute the transpose and the indentity indexing maps.
419  SmallVector<AffineMap> indexingMaps = {
420  inversePermutation(AffineMap::getPermutationMap(
421  SmallVector<unsigned>(transposeVector.begin(), transposeVector.end()),
422  b.getContext())),
423  AffineMap::getMultiDimIdentityMap(transposeVector.size(),
424  b.getContext())};
425  SmallVector<utils::IteratorType> iteratorTypes(transposeVector.size(),
426  utils::IteratorType::parallel);
427 
428  // Create a GenericOp to transpose `inputTensor` into `outputTensor`.
429  auto transposeOp =
430  b.create<GenericOp>(loc, resultTensorType, inputTensor, outputTensor,
431  indexingMaps, iteratorTypes);
432  Region &body = transposeOp.getRegion();
433  body.push_back(new Block());
434  body.front().addArguments({elementType, elementType}, {loc, loc});
435 
436  // Create the body of the transpose operation.
437  OpBuilder::InsertionGuard g(b);
438  b.setInsertionPointToEnd(&body.front());
439  b.create<YieldOp>(loc, transposeOp.getRegion().front().getArgument(0));
440  return transposeOp;
441 }

564  OpBuilder &b, Location loc, ValueRange lbs, ValueRange ubs,
565  ValueRange steps, ArrayRef<utils::IteratorType> iteratorTypes,
566  ArrayRef<linalg::ProcInfo> procInfo,
567  function_ref<void(OpBuilder &, Location, ValueRange)> bodyBuilderFn,
568  SmallVectorImpl<Value> &ivStorage) {
569  assert(lbs.size() == ubs.size());
570  assert(lbs.size() == steps.size());
571  assert(lbs.size() == iteratorTypes.size());
572  assert(procInfo.empty() || (lbs.size() == procInfo.size()));
573 
574  // If there are no (more) loops to be generated, generate the body and be
575  // done with it.
576  if (iteratorTypes.empty()) {
577  bodyBuilderFn(b, loc, ivStorage);
578  return;
579  }
580 
581  // If there are no outer parallel loops, generate one sequential loop and
582  // recurse.
583  if (!isParallelIterator(iteratorTypes.front())) {
584  LoopNest singleLoop = buildLoopNest(
585  b, loc, lbs.take_front(), ubs.take_front(), steps.take_front(),
586  [&](OpBuilder &b, Location loc, ValueRange ivs) {
587  ivStorage.append(ivs.begin(), ivs.end());
588  generateParallelLoopNest(
589  b, loc, lbs.drop_front(), ubs.drop_front(), steps.drop_front(),
590  iteratorTypes.drop_front(),
591  procInfo.empty() ? procInfo : procInfo.drop_front(),
592  bodyBuilderFn, ivStorage);
593  });
594  return;
595  }
596 
597  unsigned nLoops = iteratorTypes.size();
598  unsigned numProcessed = 0;
599  DistributionMethod distributionMethod = DistributionMethod::None;
600  if (procInfo.empty()) {
601  numProcessed = nLoops - iteratorTypes.drop_while(isParallelIterator).size();
602  } else {
603  distributionMethod = procInfo.front().distributionMethod;
604  numProcessed =
605  nLoops - procInfo
606  .drop_while([&](linalg::ProcInfo p) {
607  return p.distributionMethod == distributionMethod;
608  })
609  .size();
610  }
611 
612  auto remainderProcInfo =
613  procInfo.empty() ? procInfo : procInfo.drop_front(numProcessed);
614  switch (distributionMethod) {
615  case DistributionMethod::None: {
616  // Generate a single parallel loop-nest operation for all outermost
617  // parallel loops and recurse.
618  b.create<scf::ParallelOp>(
619  loc, lbs.take_front(numProcessed), ubs.take_front(numProcessed),
620  steps.take_front(numProcessed),
621  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
622  ivStorage.append(localIvs.begin(), localIvs.end());
623  generateParallelLoopNest(
624  nestedBuilder, nestedLoc, lbs.drop_front(numProcessed),
625  ubs.drop_front(numProcessed), steps.drop_front(numProcessed),
626  iteratorTypes.drop_front(numProcessed), remainderProcInfo,
627  bodyBuilderFn, ivStorage);
628  });
629  return;
630  }
631  case DistributionMethod::Cyclic: {
632  // Generate a single parallel loop-nest operation for all outermost
633  // parallel loops and recurse.
634  b.create<scf::ParallelOp>(
635  loc, lbs.take_front(numProcessed), ubs.take_front(numProcessed),
636  steps.take_front(numProcessed),
637  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
638  ivStorage.append(localIvs.begin(), localIvs.end());
639  generateParallelLoopNest(
640  nestedBuilder, nestedLoc, lbs.drop_front(numProcessed),
641  ubs.drop_front(numProcessed), steps.drop_front(numProcessed),
642  iteratorTypes.drop_front(numProcessed), remainderProcInfo,
643  bodyBuilderFn, ivStorage);
644  });
645  return;
646  }
647  case DistributionMethod::CyclicNumProcsGeNumIters: {
648  // Check (for the processed loops) that the iteration is in-bounds.
649  ArithBuilder ab(b, loc);
650  Value cond = ab.slt(lbs[0], ubs[0]);
651  for (unsigned i = 1; i < numProcessed; ++i)
652  cond = ab._and(cond, ab.slt(lbs[i], ubs[i]));
653  ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
654  b.create<scf::IfOp>(loc, cond, [&](OpBuilder &b, Location loc) {
655  generateParallelLoopNest(b, loc, lbs.drop_front(numProcessed),
656  ubs.drop_front(numProcessed),
657  steps.drop_front(numProcessed),
658  iteratorTypes.drop_front(numProcessed),
659  remainderProcInfo, bodyBuilderFn, ivStorage);
660  b.create<scf::YieldOp>(loc, ValueRange{});
661  });
662  return;
663  }
664  case DistributionMethod::CyclicNumProcsEqNumIters:
665  // No check/loops needed here. Set the `%iv` to be the `%lb` and proceed
666  // with inner loop generation.
667  ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
668  generateParallelLoopNest(
669  b, loc, lbs.drop_front(numProcessed), ubs.drop_front(numProcessed),
670  steps.drop_front(numProcessed), iteratorTypes.drop_front(numProcessed),
671  remainderProcInfo, bodyBuilderFn, ivStorage);
672  return;
673  }
674 }

676 /// Specialization for generating a mix of parallel and sequential scf loops.
677 template <>
678 void GenerateLoopNest<scf::ParallelOp>::doit(
679  OpBuilder &b, Location loc, ArrayRef<Range> loopRanges, LinalgOp linalgOp,
680  ArrayRef<utils::IteratorType> iteratorTypes,
681  function_ref<scf::ValueVector(OpBuilder &, Location, ValueRange,
682  ValueRange)>
683  bodyBuilderFn,
684  ArrayRef<linalg::ProcInfo> procInfo) {
685  SmallVector<Value> iterArgInitValues = linalgOp.hasBufferSemantics()
686  ? SmallVector<Value>{}
687  : linalgOp.getDpsInitOperands();
688  assert(iterArgInitValues.empty() && "unexpected ParallelOp init values");
689  // This function may be passed more iterator types than ranges.
690  assert(iteratorTypes.size() >= loopRanges.size() &&
691  "expected iterator type for all ranges");
692  assert((procInfo.empty() || (procInfo.size() == loopRanges.size())) &&
693  "expected proc information for all loops when present");
694  iteratorTypes = iteratorTypes.take_front(loopRanges.size());
695  SmallVector<Value, 8> lbsStorage, ubsStorage, stepsStorage, ivs;
696  unsigned numLoops = iteratorTypes.size();
697  ivs.reserve(numLoops);
698  lbsStorage.reserve(numLoops);
699  ubsStorage.reserve(numLoops);
700  stepsStorage.reserve(numLoops);
701 
702  // Get the loop lb, ub, and step.
703  unpackRanges(b, loc, loopRanges, lbsStorage, ubsStorage, stepsStorage);
704 
705  // Modify the lb, ub, and step based on the distribution options.
706  for (const auto &it : llvm::enumerate(procInfo)) {
707  if (it.value().distributionMethod != linalg::DistributionMethod::None) {
708  updateBoundsForCyclicDistribution(
709  b, loc, it.value().procId, it.value().nprocs, lbsStorage[it.index()],
710  ubsStorage[it.index()], stepsStorage[it.index()]);
711  }
712  }
713  ValueRange lbs(lbsStorage), ubs(ubsStorage), steps(stepsStorage);
714  generateParallelLoopNest(
715  b, loc, lbs, ubs, steps, iteratorTypes, procInfo,
716  [&](OpBuilder &b, Location loc, ValueRange ivs) {
717  bodyBuilderFn(b, loc, ivs, linalgOp->getOperands());
718  },
719  ivs);
720 
721  assert(ivs.size() == iteratorTypes.size() && "did not generate enough loops");
722 }

725  Value valueToTile,
726  const SliceParameters &sliceParams) {

756 SliceParameters
757 computeSliceParameters(OpBuilder &builder, Location loc, Value valueToTile,
758  ArrayRef<OpFoldResult> tileSizes, AffineMap map,
759  ArrayRef<OpFoldResult> lbs, ArrayRef<OpFoldResult> ubs,
760  ArrayRef<OpFoldResult> subShapeSizes,
761  bool omitPartialTileCheck) {
762  auto shapedType = valueToTile.getType().dyn_cast<ShapedType>();
763  assert(shapedType && "only shaped types can be tiled");
764  ArrayRef<int64_t> shape = shapedType.getShape();
765  int64_t rank = shapedType.getRank();
766 
767  // Compute offsets/sizes/strides for the tile.
768  SliceParameters sliceParams;
769  sliceParams.offsets.reserve(rank);
770  sliceParams.sizes.reserve(rank);
771  sliceParams.strides.reserve(rank);
772  for (unsigned r = 0; r < rank; ++r) {
773  LLVM_DEBUG(llvm::dbgs() << "computeSliceParameters: for dim#" << r);
774  if (!isTiled(map.getSubMap({r}), tileSizes)) {
775  sliceParams.offsets.push_back(builder.getIndexAttr(0));
776  OpFoldResult dim = createFoldedDimOp(builder, loc, valueToTile, r);
777  sliceParams.sizes.push_back(dim);
778  sliceParams.strides.push_back(builder.getIndexAttr(1));
779  LLVM_DEBUG(llvm::dbgs() << ": not tiled: use size: " << dim << "\n");
780  continue;
781  }
782  LLVM_DEBUG(llvm::dbgs() << ": tiled: figure out subsize...\n");
783 
784  // Tiling creates a new slice at the proper index, the slice step is 1
785  // (i.e. the op does not subsample, stepping occurs in the loop).
786  auto m = map.getSubMap({r});
787  LLVM_DEBUG(llvm::dbgs() << "computeSliceParameters: submap: " << m << "\n");
788  IRRewriter rewriter(builder);
789  OpFoldResult offset = makeComposedFoldedAffineApply(rewriter, loc, m, lbs);
790  sliceParams.offsets.push_back(offset);
791  OpFoldResult closedIntSize =
792  makeComposedFoldedAffineApply(rewriter, loc, m, subShapeSizes);
793  // Resulting size needs to be made half open interval again.
794  AffineExpr s0 = getAffineSymbolExpr(0, builder.getContext());
795  OpFoldResult size =
796  makeComposedFoldedAffineApply(rewriter, loc, s0 + 1, closedIntSize);
797  LLVM_DEBUG(llvm::dbgs()
798  << "computeSliceParameters: raw size: " << size << "\n");
799  LLVM_DEBUG(llvm::dbgs()
800  << "computeSliceParameters: new offset: " << offset << "\n");
801  sliceParams.strides.push_back(builder.getIndexAttr(1));
802 
803  if (omitPartialTileCheck) {
804  // We statically know that the partial/boundary tile condition is
805  // unnecessary.
806  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: new size: " << size << "\n");
807  sliceParams.sizes.push_back(size);
808  continue;
809  }
810 
811  // The size of the subview / extract_slice should be trimmed to avoid
812  // out-of-bounds accesses, unless:
813  // a. We statically know the subshape size divides the shape size evenly.
814  // b. The subshape size is 1. According to the way the loops are set up,
815  // tensors with "0" dimensions would never be constructed.
816  int64_t shapeSize = shape[r];
817  Optional<int64_t> sizeCst = getConstantIntValue(size);
818  auto hasTileSizeOne = sizeCst && *sizeCst == 1;
819  auto dividesEvenly = sizeCst && !ShapedType::isDynamic(shapeSize) &&
820  ((shapeSize % *sizeCst) == 0);
821  if (!hasTileSizeOne && !dividesEvenly) {
822  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: shapeSize=" << shapeSize
823  << ", size: " << size
824  << ": make sure in bound with affine.min\n");
825 
826  AffineExpr dim0, dim1, dim2;
827  bindDims(builder.getContext(), dim0, dim1, dim2);
828 
829  // Get the dimension size for this dimension. We need to first calculate
830  // the max index and then plus one. This is important because for
831  // convolution ops, we have its input window dimension's affine map of the
832  // form `(d0 * s0 + d1)`, where `d0`/`d1 is an output/filter window
833  // dimension and `s0` is stride. Directly use the dimension size of
834  // output/filer window dimensions will cause incorrect calculation.
835  AffineMap minusOneMap =
836  AffineMap::inferFromExprList({ArrayRef<AffineExpr>{dim0 - 1}})
837  .front();
838  AffineMap plusOneMap =
839  AffineMap::inferFromExprList({ArrayRef<AffineExpr>{dim0 + 1}})
840  .front();
841  SmallVector<OpFoldResult> maxIndices =
842  llvm::to_vector(llvm::map_range(ubs, [&](OpFoldResult ub) {
843  return makeComposedFoldedAffineApply(rewriter, loc, minusOneMap,
844  {ub});
845  }));
846  OpFoldResult maxIndex =
847  makeComposedFoldedAffineApply(rewriter, loc, m, maxIndices);
848  OpFoldResult d =
849  makeComposedFoldedAffineApply(rewriter, loc, plusOneMap, {maxIndex});
850 
851  // Compute min(dim - offset, size) to avoid out-of-bounds accesses.
852  AffineMap minMap = AffineMap::inferFromExprList(
853  {ArrayRef<AffineExpr>{dim1 - dim2, dim0}})
854  .front();
855  size =
856  makeComposedFoldedAffineMin(rewriter, loc, minMap, {size, d, offset});
857  }
858  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: new size: " << size << "\n");
859  sliceParams.sizes.push_back(size);
860  }
861  return sliceParams;
862 }

865  ArrayRef<OpFoldResult> ivs,
866  ArrayRef<OpFoldResult> tileSizes) {

878 SmallVector<OpFoldResult> computeTileSizes(OpBuilder &b, Location loc,
879  ArrayRef<OpFoldResult> tileSizes,
880  ArrayRef<OpFoldResult> sizeBounds) {
881  SmallVector<OpFoldResult> sizes;
882  for (unsigned idx = 0, e = tileSizes.size(); idx < e; ++idx) {
883  bool isTiled = !isZero(tileSizes[idx]);
884  // Before composing, we need to make range a closed interval.
885  OpFoldResult size = isTiled ? tileSizes[idx] : sizeBounds[idx];

887  IRRewriter rewriter(b);
888  sizes.push_back(makeComposedFoldedAffineApply(rewriter, loc, d0 - 1, size));
889  LLVM_DEBUG(llvm::dbgs() << "computeTileSizes: " << sizes.back() << "\n");
890  }
891  return sizes;

931 SmallVector<Optional<SliceParameters>>
932 computeAllSliceParameters(OpBuilder &builder, Location loc, LinalgOp linalgOp,
933  ValueRange valuesToTile, ArrayRef<OpFoldResult> ivs,
934  ArrayRef<OpFoldResult> tileSizes,
935  ArrayRef<OpFoldResult> sizeBounds,
936  bool omitPartialTileCheck) {
937  assert(ivs.size() == static_cast<size_t>(llvm::count_if(
938  llvm::make_range(tileSizes.begin(), tileSizes.end()),
939  [](OpFoldResult v) { return !isZero(v); })) &&
940  "expected as many ivs as non-zero sizes");
941 
942  // Construct (potentially temporary) mins and maxes on which to apply maps
943  // that define tile subshapes.
944  SmallVector<OpFoldResult> lbs =
945  computeTileOffsets(builder, loc, ivs, tileSizes);
946  SmallVector<OpFoldResult> subShapeSizes =
947  computeTileSizes(builder, loc, tileSizes, sizeBounds);
948 
949  assert(static_cast<int64_t>(valuesToTile.size()) <=
950  linalgOp->getNumOperands() &&
951  "more value to tile than operands.");
952  SmallVector<Optional<SliceParameters>> allSliceParams;
953  allSliceParams.reserve(valuesToTile.size());
954  for (auto [opOperand, val] :
955  llvm::zip(linalgOp->getOpOperands(), valuesToTile)) {
956  Value shapedOp = val;
957  LLVM_DEBUG(llvm::dbgs() << "makeTiledShapes: for operand " << shapedOp);
958  AffineMap map = linalgOp.getMatchingIndexingMap(&opOperand);
959  // Use `opOperand` as is if it is not tiled and not an output tensor. Having
960  // an extract/insert slice pair for all output tensors simplifies follow up
961  // transformations such as padding and bufferization since the
962  // extract/insert slice pairs make the accessed iteration argument
963  // subdomains explicit.
964 
965  Type operandType = opOperand.get().getType();
966  if (!isTiled(map, tileSizes) && !(operandType.isa<RankedTensorType>() &&
967  linalgOp.isDpsInit(&opOperand))) {
968  allSliceParams.push_back(llvm::None);
969  LLVM_DEBUG(llvm::dbgs()
970  << ": not tiled: use shape: " << operandType << "\n");
971  continue;
972  }
973  LLVM_DEBUG(llvm::dbgs() << ": tiled: figure out subshape...\n");
974 
975  allSliceParams.push_back(computeSliceParameters(
976  builder, loc, shapedOp, tileSizes, map, lbs, sizeBounds, subShapeSizes,
977  omitPartialTileCheck));
978  }
979 
980  return allSliceParams;
981 }

1033 /// Get the reassociation maps to fold the result of a extract_slice (or source
1034 /// of a insert_slice) operation with given offsets, and sizes to its
1035 /// rank-reduced version. This is only done for the cases where the size is 1
1036 /// and offset is 0. Strictly speaking the offset 0 is not required in general,
1037 /// but non-zero offsets are not handled by SPIR-V backend at this point (and
1038 /// potentially cannot be handled).
1039 Optional<SmallVector<ReassociationIndices>>
1040 getReassociationMapForFoldingUnitDims(ArrayRef<OpFoldResult> mixedSizes) {
1041  SmallVector<ReassociationIndices> reassociation;
1042  ReassociationIndices curr;
1043  for (const auto &it : llvm::enumerate(mixedSizes)) {
1044  auto dim = it.index();
1045  auto size = it.value();
1046  curr.push_back(dim);
1047  auto attr = size.dyn_cast<Attribute>();
1048  if (attr && attr.cast<IntegerAttr>().getInt() == 1)
1049  continue;
1050  reassociation.emplace_back(ReassociationIndices{});
1051  std::swap(reassociation.back(), curr);
1052  }
1053  // When the reassociations are not empty, then fold the remaining
1054  // unit-dimensions into the last dimension. If the reassociations so far is
1055  // empty, then leave it emtpy. This will fold everything to a rank-0 tensor.
1056  if (!curr.empty() && !reassociation.empty())
1057  reassociation.back().append(curr.begin(), curr.end());
1058  return reassociation;
1059 }
