44 Attribute SparseTensorEncodingAttr::parse(AsmParser &parser, Type type) {
45  if (failed(parser.parseLess()))
46  return {};
47  // Parse the data as a dictionary.
48  DictionaryAttr dict;
49  if (failed(parser.parseAttribute(dict)))
50  return {};
51  if (failed(parser.parseGreater()))
52  return {};
53  // Process the data from the parsed dictionary value into struct-like data.
54  SmallVector<DimLevelType> dlt;
55  AffineMap dimOrd = {};
56  AffineMap higherOrd = {};
57  unsigned ptr = 0;
58  unsigned ind = 0;
59  for (const NamedAttribute &attr : dict) {
60  if (attr.getName() == "dimLevelType") {
61  auto arrayAttr = attr.getValue().dyn_cast<ArrayAttr>();
62  if (!arrayAttr) {
63  parser.emitError(parser.getNameLoc(),
64  "expected an array for dimension level types");
65  return {};
66  }
67  for (auto i : arrayAttr) {
68  auto strAttr = i.dyn_cast<StringAttr>();
69  if (!strAttr) {
70  parser.emitError(parser.getNameLoc(),
71  "expected a string value in dimension level types");
72  return {};
73  }
74  auto strVal = strAttr.getValue();
75  if (strVal == "dense") {
76  dlt.push_back(DimLevelType::Dense);
77  } else if (strVal == "compressed") {
78  dlt.push_back(DimLevelType::Compressed);
79  } else if (strVal == "compressed-nu") {
80  dlt.push_back(DimLevelType::CompressedNu);
81  } else if (strVal == "compressed-no") {
82  dlt.push_back(DimLevelType::CompressedNo);
83  } else if (strVal == "compressed-nu-no") {
84  dlt.push_back(DimLevelType::CompressedNuNo);
85  } else if (strVal == "singleton") {
86  dlt.push_back(DimLevelType::Singleton);
87  } else if (strVal == "singleton-nu") {
88  dlt.push_back(DimLevelType::SingletonNu);
89  } else if (strVal == "singleton-no") {
90  dlt.push_back(DimLevelType::SingletonNo);
91  } else if (strVal == "singleton-nu-no") {
92  dlt.push_back(DimLevelType::SingletonNuNo);
93  } else {
94  parser.emitError(parser.getNameLoc(),
95  "unexpected dimension level type: ")
96  << strVal;
97  return {};
98  }
99  }
100  } else if (attr.getName() == "dimOrdering") {
101  auto affineAttr = attr.getValue().dyn_cast<AffineMapAttr>();
102  if (!affineAttr) {
103  parser.emitError(parser.getNameLoc(),
104  "expected an affine map for dimension ordering");
105  return {};
106  }
107  dimOrd = affineAttr.getValue();
108  } else if (attr.getName() == "higherOrdering") {
109  auto affineAttr = attr.getValue().dyn_cast<AffineMapAttr>();
110  if (!affineAttr) {
111  parser.emitError(parser.getNameLoc(),
112  "expected an affine map for higher ordering");
113  return {};
114  }
115  higherOrd = affineAttr.getValue();
116  } else if (attr.getName() == "pointerBitWidth") {
117  auto intAttr = attr.getValue().dyn_cast<IntegerAttr>();
118  if (!intAttr) {
119  parser.emitError(parser.getNameLoc(),
120  "expected an integral pointer bitwidth");
121  return {};
122  }
123  ptr = intAttr.getInt();
124  } else if (attr.getName() == "indexBitWidth") {
125  auto intAttr = attr.getValue().dyn_cast<IntegerAttr>();
126  if (!intAttr) {
127  parser.emitError(parser.getNameLoc(),
128  "expected an integral index bitwidth");
129  return {};
130  }
131  ind = intAttr.getInt();
132  } else {
133  parser.emitError(parser.getNameLoc(), "unexpected key: ")
134  << attr.getName().strref();
135  return {};
136  }
137  }
138  // Construct struct-like storage for attribute.
139  return parser.getChecked<SparseTensorEncodingAttr>(
140  parser.getContext(), dlt, dimOrd, higherOrd, ptr, ind);
141 }

196 LogicalResult SparseTensorEncodingAttr::verify(
197  function_ref<InFlightDiagnostic()> emitError,
198  ArrayRef<DimLevelType> dimLevelType, AffineMap dimOrdering,
199  AffineMap higherOrdering, unsigned pointerBitWidth,
200  unsigned indexBitWidth) {
201  if (!acceptBitWidth(pointerBitWidth))
202  return emitError() << "unexpected pointer bitwidth: " << pointerBitWidth;
203  if (!acceptBitWidth(indexBitWidth))
204  return emitError() << "unexpected index bitwidth: " << indexBitWidth;
205  if (dimOrdering) {
206  if (!dimOrdering.isPermutation())
207  return emitError()
208  << "expected a permutation affine map for dimension ordering";
209  if (dimOrdering.getNumResults() != dimLevelType.size())
210  return emitError() << "unexpected mismatch in ordering and dimension "
211  "level types size";
212  }
213  if (higherOrdering) {
214  if (higherOrdering.getNumDims() >= higherOrdering.getNumResults())
215  return emitError() << "unexpected higher ordering mapping from "
216  << higherOrdering.getNumDims() << " to "
217  << higherOrdering.getNumResults();
218  if (higherOrdering.getNumResults() != dimLevelType.size())
219  return emitError() << "unexpected mismatch in higher ordering and "
220  "dimension level types size";
221  }
222  return success();
223 }

225 LogicalResult SparseTensorEncodingAttr::verifyEncoding(
226  ArrayRef<int64_t> shape, Type elementType,
227  function_ref<InFlightDiagnostic()> emitError) const {
228  // Check structural integrity.
229  if (failed(verify(emitError, getDimLevelType(), getDimOrdering(),
230  getHigherOrdering(), getPointerBitWidth(),
231  getIndexBitWidth())))
232  return failure();
233  // Check integrity with tensor type specifics. Dimension ordering is optional,
234  // but we always should have dimension level types for the full rank.
235  unsigned size = shape.size();
236  if (size == 0)
237  return emitError() << "expected non-scalar sparse tensor";
238  if (getHigherOrdering()) {
239  if (getHigherOrdering().getNumDims() != size)
240  return emitError() << "expected an affine map of size " << size
241  << " for higher ordering";
242 
243  // TODO: verification of higher ordering contents
244 
245  size = getHigherOrdering().getNumResults(); // higher-order size!
246  }
247  if (getDimOrdering() && getDimOrdering().getNumResults() != size)
248  return emitError() << "expected an affine map of size " << size
249  << " for dimension ordering";
250  if (getDimLevelType().size() != size)
251  return emitError() << "expected an array of size " << size
252  << " for dimension level types";
253  return success();
254 }

337 LogicalResult ConvertOp::verify() {
338  if (auto tp1 = getSource().getType().dyn_cast<RankedTensorType>()) {
339  if (auto tp2 = getDest().getType().dyn_cast<RankedTensorType>()) {
340  if (tp1.getRank() != tp2.getRank())
341  return emitError("unexpected conversion mismatch in rank");
342  auto shape1 = tp1.getShape();
343  auto shape2 = tp2.getShape();
344  // Accept size matches between the source and the destination type
345  // (e.g. 10 vs. 10, 10 vs. ?, or ? vs. ?), but reject direct mismatches or
346  // matches that would need a runtime assert (e.g. 10 vs. 20 or ? vs. 10).
347  for (unsigned d = 0, rank = tp1.getRank(); d < rank; d++)
348  if (shape1[d] != shape2[d] && shape2[d] != ShapedType::kDynamicSize)
349  return emitError("unexpected conversion mismatch in dimension ") << d;
350  return success();
351  }
352  }
353  return emitError("unexpected type in convert");
354 }

397 template <class T>
398 static LogicalResult verifyNumBlockArgs(T *op, Region &region,
399  const char *regionName,
400  TypeRange inputTypes, Type outputType) {
401  unsigned numArgs = region.getNumArguments();
402  unsigned expectedNum = inputTypes.size();
403  if (numArgs != expectedNum)
404  return op->emitError() << regionName << " region must have exactly "
405  << expectedNum << " arguments";
406 
407  for (unsigned i = 0; i < numArgs; i++) {
408  Type typ = region.getArgument(i).getType();
409  if (typ != inputTypes[i])
410  return op->emitError() << regionName << " region argument " << (i + 1)
411  << " type mismatch";
412  }
413  Operation *term = region.front().getTerminator();
414  YieldOp yield = dyn_cast<YieldOp>(term);
415  if (!yield)
416  return op->emitError() << regionName
417  << " region must end with sparse_tensor.yield";
418  if (!yield.getResult() || yield.getResult().getType() != outputType)
419  return op->emitError() << regionName << " region yield type mismatch";
420 
421  return success();
422 }

424 LogicalResult BinaryOp::verify() {
425  NamedAttrList attrs = (*this)->getAttrs();
426  Type leftType = getX().getType();
427  Type rightType = getY().getType();
428  Type outputType = getOutput().getType();
429  Region &overlap = getOverlapRegion();
430  Region &left = getLeftRegion();
431  Region &right = getRightRegion();
432 
433  // Check correct number of block arguments and return type for each
434  // non-empty region.
435  LogicalResult regionResult = success();
436  if (!overlap.empty()) {
437  regionResult = verifyNumBlockArgs(
438  this, overlap, "overlap", TypeRange{leftType, rightType}, outputType);
439  if (failed(regionResult))
440  return regionResult;
441  }
442  if (!left.empty()) {
443  regionResult =
444  verifyNumBlockArgs(this, left, "left", TypeRange{leftType}, outputType);
445  if (failed(regionResult))
446  return regionResult;
447  } else if (getLeftIdentity()) {
448  if (leftType != outputType)
449  return emitError("left=identity requires first argument to have the same "
450  "type as the output");
451  }
452  if (!right.empty()) {
453  regionResult = verifyNumBlockArgs(this, right, "right",
454  TypeRange{rightType}, outputType);
455  if (failed(regionResult))
456  return regionResult;
457  } else if (getRightIdentity()) {
458  if (rightType != outputType)
459  return emitError("right=identity requires second argument to have the "
460  "same type as the output");
461  }
462 
463  return success();
464 }

491 LogicalResult ConcatenateOp::verify() {
492  auto dstTp = getType().cast<RankedTensorType>();
493  uint64_t concatDim = getDimension().getZExtValue();
494  unsigned rank = dstTp.getRank();
495 
496  if (getInputs().size() <= 1)
497  return emitError("Need at least two tensors to concatenate.");
498 
499  for (auto type : getInputs().getTypes()) {
500  auto shape = type.cast<RankedTensorType>().getShape();
501  for (auto dim : shape) {
502  if (dim == ShapedType::kDynamicSize)
503  return emitError("Only statically-sized input tensors are supported.");
504  }
505  }
506 
507  if (concatDim >= rank)
508  return emitError(llvm::formatv(
509  "Failed to concatentate tensors with rank={0} on dimension={1}.", rank,
510  concatDim));
511 
512  for (size_t i = 0, e = getInputs().size(); i < e; i++) {
513  Value input = getInputs()[i];
514  auto inputRank = input.getType().cast<RankedTensorType>().getRank();
515  if (inputRank != rank)
516  return emitError(
517  llvm::formatv("The input tensor ${0} has a different rank (rank={1}) "
518  "from the output tensor (rank={2}).",
519  i, inputRank, rank));
520  }
521 
522  for (unsigned i = 0; i < rank; i++) {
523  auto dstDim = dstTp.getShape()[i];
524  if (i == concatDim) {
525  if (dstDim != ShapedType::kDynamicSize) {
526  unsigned sumDim = 0;
527  for (auto src : getInputs()) {
528  // If we reach here, all inputs should have static shapes.
529  auto d = src.getType().cast<RankedTensorType>().getShape()[i];
530  sumDim += d;
531  }
532  // If all dimension are statically known, the sum of all the input
533  // dimensions should be equal to the output dimension.
534  if (sumDim != dstDim)
535  return emitError(
536  "The concatenation dimension of the output tensor should be the "
537  "sum of all the concatenation dimensions of the input tensors.");
538  }
539  } else {
540  int64_t prev = dstDim;
541  for (auto src : getInputs()) {
542  auto d = src.getType().cast<RankedTensorType>().getShape()[i];
543  if (prev != ShapedType::kDynamicSize && d != prev)
544  return emitError("All dimensions (expect for the concatenating one) "
545  "should be equal.");
546  prev = d;
547  }
548  }
549  }
550 
551  return success();
552 }

592 void ForeachOp::build(
593  OpBuilder &builder, OperationState &result, Value tensor,
594  ValueRange initArgs,
595  function_ref<void(OpBuilder &, Location, ValueRange, Value, ValueRange)>
596  bodyBuilder) {
597  build(builder, result, initArgs.getTypes(), tensor, initArgs);
598  // Builds foreach body.
599  if (!bodyBuilder)
600  return;
601  auto rtp = tensor.getType().cast<RankedTensorType>();
602  int64_t rank = rtp.getRank();
603 
604  SmallVector<Type> blockArgTypes;
605  // Starts with n index.
606  std::fill_n(std::back_inserter(blockArgTypes), rank, builder.getIndexType());
607  // Followed by one value.
608  blockArgTypes.push_back(rtp.getElementType());
609  // Followed by reduction variable.
610  blockArgTypes.append(initArgs.getTypes().begin(), initArgs.getTypes().end());
611 
612  SmallVector<Location> blockArgLocs;
613  std::fill_n(std::back_inserter(blockArgLocs), blockArgTypes.size(),
614  tensor.getLoc());
615 
616  OpBuilder::InsertionGuard guard(builder);
617  auto &region = *result.regions.front();
618  Block *bodyBlock =
619  builder.createBlock(&region, region.end(), blockArgTypes, blockArgLocs);
620  bodyBuilder(builder, result.location,
621  bodyBlock->getArguments().slice(0, rank),
622  bodyBlock->getArguments()[rank],
623  bodyBlock->getArguments().drop_front(rank + 1));
624 }

626 LogicalResult ForeachOp::verify() {
627  auto t = getTensor().getType().cast<RankedTensorType>();
628  auto args = getBody()->getArguments();
629 
630  if (static_cast<size_t>(t.getRank()) + 1 + getInitArgs().size() !=
631  args.size())
632  return emitError("Unmatched number of arguments in the block");
633 
634  if (getNumResults() != getInitArgs().size())
635  return emitError("Mismatch in number of init arguments and results");
636 
637  if (getResultTypes() != getInitArgs().getTypes())
638  return emitError("Mismatch in types of init arguments and results");
639

690 LogicalResult SortOp::verify() {
691  if (getXs().empty())
692  return emitError("need at least one xs buffer.");
693 
694  auto n = getN().getDefiningOp<arith::ConstantIndexOp>();
695 
696  Type xtp = getXs().front().getType().cast<MemRefType>().getElementType();
697  auto checkTypes = [&](ValueRange operands,
698  bool checkEleType = true) -> LogicalResult {
699  for (Value opnd : operands) {
700  MemRefType mtp = opnd.getType().cast<MemRefType>();
701  int64_t dim = mtp.getShape()[0];
702  // We can't check the size of dynamic dimension at compile-time, but all
703  // xs and ys should have a dimension not less than n at runtime.
704  if (n && dim != ShapedType::kDynamicSize && dim < n.value())
705  return emitError(llvm::formatv("xs and ys need to have a dimension >= n"
706  ": {0} < {1}",
707  dim, n.value()));
708 
709  if (checkEleType && xtp != mtp.getElementType())
710  return emitError("mismatch xs element types");
711  }
712  return success();
713  };
714 
715  LogicalResult result = checkTypes(getXs());
716  if (failed(result))
717  return result;
718 
719  if (n)
720  return checkTypes(getYs(), false);
721 
722  return success();
723 }

727  // We can't check the size of the buffers when n or buffer dimensions aren't
728  // compile-time constants.

741  ny = nyAttr.getInt();
742  }

745  MemRefType tp = v.getType().cast<MemRefType>();

749  }
