16 /// Return `true` if the `result` of an operation `genericOp` is dead.
17 static bool isResultValueDead(linalg::GenericOp genericOp, OpResult result) {
18  if (!result.use_empty())
19  return false;
20  // If out operand not used in payload, we can drop it.
21  OpOperand *outputOpOperand =
22  genericOp.getDpsInitOperand(result.getResultNumber());
23  if (!genericOp.payloadUsesValueFromOperand(outputOpOperand))
24  return true;
25 
26  // The out operand that is part of a payload can be dropped if
27  // these conditions are met:
28  // - Result from out operand is dead.
29  // - User of arg is yield.
30  // - outArg data is not being used by other outArgs.
31 
32  // Check block arg and cycle from out operand has a single use.
33  BlockArgument outputArg =
34  genericOp.getRegionOutputArgs()[result.getResultNumber()];
35  if (!outputArg.hasOneUse())
36  return false;
37  Operation *argUserOp = *outputArg.user_begin();
38 
39  // Check argUser has no other use.
40  if (!argUserOp->use_empty())
41  return false;
42 
43  // Check that argUser is a yield.
44  auto yieldOp = dyn_cast<linalg::YieldOp>(argUserOp);
45  if (!yieldOp)
46  return false;
47 
48  // Check outArg data is not being used by other outArgs.
49  if (yieldOp.getOperand(result.getResultNumber()) != outputArg)
50  return false;
51 
52  return true;
53 }

57 struct DeduplicateAndRemoveDeadOperandsAndResults
58  : public OpRewritePattern<GenericOp> {
59  using OpRewritePattern<GenericOp>::OpRewritePattern;
60 
61  LogicalResult matchAndRewrite(GenericOp genericOp,
62  PatternRewriter &rewriter) const override {
63  // Create a map from argument position in the original op to the argument
64  // position in the new op. If the argument is dropped it wont have an entry.
65  SmallVector<OpOperand *> droppedOpOperands;
66 
67  // Information needed to build the new op.
68  SmallVector<Value> newInputOperands, newOutputOperands;
69  SmallVector<AffineMap> newIndexingMaps;
70 
71  // Gather information about duplicate input operands.
72  llvm::SmallDenseMap<unsigned, unsigned> origInsToNewInsPos =
73  deduplicateInputOperands(genericOp, droppedOpOperands, newInputOperands,
74  newIndexingMaps);
75 
76  // Gather information about the dropped outputs.
77  llvm::SmallDenseMap<unsigned, unsigned> origOutsToNewOutsPos =
78  deduplicateOutputOperands(genericOp, droppedOpOperands,
79  newOutputOperands, newIndexingMaps);
80 
81  // Check if there is any change to operands.
82  if (newInputOperands.size() + newOutputOperands.size() ==
83  genericOp->getNumOperands())
84  return failure();
85 
86  // Create the new op with the body being empty.
87  Location loc = genericOp.getLoc();
88  SmallVector<Type> newResultTypes;
89  for (Value v : newOutputOperands)
90  if (v.getType().isa<TensorType>())
91  newResultTypes.push_back(v.getType());
92  auto newOp = rewriter.create<GenericOp>(
93  loc, newResultTypes, newInputOperands, newOutputOperands,
94  rewriter.getAffineMapArrayAttr(newIndexingMaps),
95  genericOp.getIteratorTypes(), genericOp.getDocAttr(),
96  genericOp.getLibraryCallAttr(),
97  [](OpBuilder & /*builder*/, Location /*loc*/, ValueRange /*args*/) {
98  return;
99  });
100  // Copy over unknown attributes. They might be load bearing for some flow.
101  ArrayRef<StringRef> odsAttrs = genericOp.getAttributeNames();
102  for (NamedAttribute kv : genericOp->getAttrs())
103  if (!llvm::is_contained(odsAttrs, kv.getName().getValue()))
104  newOp->setAttr(kv.getName(), kv.getValue());
105 
106  // Fix up the payload of the canonicalized operation.
107  populateOpPayload(genericOp, newOp, origInsToNewInsPos,
108  origOutsToNewOutsPos, rewriter);
109 
110  // Replace all live uses of the op.
111  SmallVector<Value> replacementsVals(genericOp->getNumResults(), nullptr);
112  for (const auto &result : llvm::enumerate(genericOp.getResults())) {
113  auto it = origOutsToNewOutsPos.find(result.index());
114  if (it == origOutsToNewOutsPos.end())
115  continue;
116  replacementsVals[result.index()] = newOp.getResult(it->second);
117  }
118  rewriter.replaceOp(genericOp, replacementsVals);
119  return success();
120  }
121 
122 private:
123  // Deduplicate input operands, and return the
124  // - Mapping from operand position in the original op, to operand position in
125  // the canonicalized op.
126  // - The preserved input operands list (by reference).
127  llvm::SmallDenseMap<unsigned, unsigned>
128  deduplicateInputOperands(GenericOp genericOp,
129  SmallVector<OpOperand *> &droppedOpOperands,
130  SmallVector<Value> &newInputOperands,
131  SmallVector<AffineMap> &newIndexingMaps) const {
132  llvm::SmallDenseMap<unsigned, unsigned> origToNewPos;
133  llvm::SmallDenseMap<std::pair<Value, AffineMap>, unsigned> dedupedInputs;
134  for (const auto &en : llvm::enumerate(genericOp.getDpsInputOperands())) {
135  OpOperand *inputOpOperand = en.value();
136  // Check if operand is dead and if dropping the indexing map makes the
137  // loops to shape computation invalid.
138  if (!genericOp.payloadUsesValueFromOperand(inputOpOperand)) {
139  // Add the current operands to the list of potentially droppable
140  // operands. If it cannot be dropped, this needs to be popped back.
141  droppedOpOperands.push_back(inputOpOperand);
142  if (genericOp.canOpOperandsBeDropped(droppedOpOperands))
143  continue;
144  droppedOpOperands.pop_back();
145  }
146 
147  // Check if this operand is a duplicate.
148  AffineMap indexingMap = genericOp.getMatchingIndexingMap(inputOpOperand);
149  auto it = dedupedInputs.find(
150  std::make_pair(inputOpOperand->get(), indexingMap));
151  if (it != dedupedInputs.end()) {
152  origToNewPos[en.index()] = it->second;
153  droppedOpOperands.push_back(inputOpOperand);
154  continue;
155  }
156 
157  // This is a preserved argument.
158  origToNewPos[en.index()] = newInputOperands.size();
159  dedupedInputs[{inputOpOperand->get(), indexingMap}] =
160  newInputOperands.size();
161  newInputOperands.push_back(inputOpOperand->get());
162  newIndexingMaps.push_back(indexingMap);
163  }
164  return origToNewPos;
165  }
166 
167  // Deduplicate output operands, and return the
168  // - Mapping from operand position in the original op, to operand position in
169  // the canonicalized op.
170  // - The preserved output operands list (by reference).
171  llvm::SmallDenseMap<unsigned, unsigned>
172  deduplicateOutputOperands(GenericOp genericOp,
173  SmallVector<OpOperand *> &droppedOpOperands,
174  SmallVector<Value> &newOutputOperands,
175  SmallVector<AffineMap> &newIndexingMaps) const {
176  llvm::SmallDenseMap<unsigned, unsigned> origToNewPos;
177  llvm::SmallDenseMap<std::tuple<Value, AffineMap, Value>, unsigned>
178  dedupedOutpts;
179  // If the op doesnt have tensor semantics, keep all the outputs as
180  // preserved.
181  if (!genericOp.hasTensorSemantics()) {
182  for (const auto &en : llvm::enumerate(genericOp.getDpsInitOperands())) {
183  origToNewPos[en.index()] = newOutputOperands.size();
184  newOutputOperands.push_back(en.value()->get());
185  newIndexingMaps.push_back(genericOp.getMatchingIndexingMap(en.value()));
186  }
187  return origToNewPos;
188  }
189  // Output argument can be dropped if the result has
190  // - no users, and
191  // - it is not used in the payload, and
192  // - the corresponding indexing maps are not needed for loop bound
193  // computation.
194  auto yieldOp = cast<YieldOp>(genericOp.getBody()->getTerminator());
195  for (const auto &outputOpOperand :
196  llvm::enumerate(genericOp.getDpsInitOperands())) {
197  OpResult result = genericOp.getTiedOpResult(outputOpOperand.value());
198  AffineMap indexingMap =
199  genericOp.getMatchingIndexingMap(outputOpOperand.value());
200  auto key = std::make_tuple(outputOpOperand.value()->get(), indexingMap,
201  yieldOp->getOperand(outputOpOperand.index()));
202  if (isResultValueDead(genericOp, result)) {
203  // Check if the opoperand can be dropped without affecting loop
204  // bound computation. Add the operand to the list of dropped op
205  // operand for checking. If it cannot be dropped, need to pop the
206  // value back.
207  droppedOpOperands.push_back(outputOpOperand.value());
208  if (genericOp.canOpOperandsBeDropped(droppedOpOperands)) {
209  continue;
210  }
211  droppedOpOperands.pop_back();
212  }
213 
214  if (!genericOp.payloadUsesValueFromOperand(outputOpOperand.value())) {
215  // The out operand can also be dropped if it is computed redundantly
216  // by another result, the conditions for that are
217  // - The same operand is used as the out operand
218  // - The same indexing map is used
219  // - The same yield value is used.
220  auto it = dedupedOutpts.find(key);
221  if (it != dedupedOutpts.end()) {
222  origToNewPos[outputOpOperand.index()] = it->second;
223  droppedOpOperands.push_back(outputOpOperand.value());
224  continue;
225  }
226  }
227 
228  origToNewPos[outputOpOperand.index()] = newOutputOperands.size();
229  dedupedOutpts[key] = newOutputOperands.size();
230  newOutputOperands.push_back(outputOpOperand.value()->get());
231  newIndexingMaps.push_back(
232  genericOp.getMatchingIndexingMap(outputOpOperand.value()));
233  }
234  return origToNewPos;
235  }

237  // Populate the body of the canonicalized operation.
238  void populateOpPayload(
239  GenericOp genericOp, GenericOp newOp,
240  const llvm::SmallDenseMap<unsigned, unsigned> &origInsToNewInsPos,
241  const llvm::SmallDenseMap<unsigned, unsigned> &origOutsToNewOutsPos,
242  PatternRewriter &rewriter) const {
243  // Merge the body of the original op with the new op.
244  Block *newOpBlock = &newOp.getRegion().front();
245  assert(newOpBlock->empty() && "expected new op to have an empty payload");
246  Block *origOpBlock = &genericOp.getRegion().front();
247  SmallVector<Value> replacements(origOpBlock->getNumArguments(), nullptr);
248 
249  // Replace all arguments in the original op, with arguments from the
250  // canonicalized op.
251  auto updateReplacements =
252  [&](OpOperandVector &origOperands, OpOperandVector &newOperands,
253  const llvm::SmallDenseMap<unsigned, unsigned> &map) {
254  for (const auto &origOperand : llvm::enumerate(origOperands)) {
255  auto it = map.find(origOperand.index());
256  if (it == map.end())
257  continue;
258  OpOperand *newOperand = newOperands[it->second];
259  replacements[origOperand.value()->getOperandNumber()] =
260  newOpBlock->getArgument(newOperand->getOperandNumber());
261  }
262  };
263 
264  OpOperandVector origInputOperands = genericOp.getDpsInputOperands();
265  OpOperandVector newInputOperands = newOp.getDpsInputOperands();
266  updateReplacements(origInputOperands, newInputOperands, origInsToNewInsPos);
267 
268  OpOperandVector origOutputOperands = genericOp.getDpsInitOperands();
269  OpOperandVector newOutputOperands = newOp.getDpsInitOperands();
270  updateReplacements(origOutputOperands, newOutputOperands,
271  origOutsToNewOutsPos);
272 
273  // Drop the unused yield args.
274  if (newOp.getNumDpsInits() != genericOp.getNumDpsInits()) {
275  OpBuilder::InsertionGuard g(rewriter);
276  YieldOp origYieldOp = cast<YieldOp>(origOpBlock->getTerminator());
277  rewriter.setInsertionPoint(origYieldOp);
278 
279  SmallVector<Value> newYieldVals(newOp.getNumDpsInits(), nullptr);
280  for (const auto &yieldOpOperands :
281  llvm::enumerate(origYieldOp.getValues())) {
282  auto it = origOutsToNewOutsPos.find(yieldOpOperands.index());
283  if (it == origOutsToNewOutsPos.end())
284  continue;
285  newYieldVals[it->second] = yieldOpOperands.value();
286  }
287  rewriter.replaceOpWithNewOp<YieldOp>(origYieldOp, newYieldVals);
288  }
289 
290  rewriter.mergeBlocks(origOpBlock, newOpBlock, replacements);
291  }

294 /// Remove unused cycles.
295 /// We can remove unused cycle within a payload of generic region
296 /// if these conditions are met:
297 /// - Result from out operand is dead.
298 /// - Block arg from out operand has a single use in the %cycle
299 /// instruction.
300 /// - Cycle has a single use and it is in yield.
301 struct RemoveUnusedCycleInGenericOp : public OpRewritePattern<GenericOp> {
302  using OpRewritePattern<GenericOp>::OpRewritePattern;
303 
304  LogicalResult matchAndRewrite(GenericOp genericOp,
305  PatternRewriter &rewriter) const override {
306 
307  // If the op doesnt have tensor semantics, preserve the outputs as is.
308  if (!genericOp.hasTensorSemantics())
309  return failure();
310 
311  bool hasRemovedCycles = false;
312  // Iterate over output operands and remove any unused cycles.
313  for (const auto &outputOpOperand :
314  llvm::enumerate(genericOp.getDpsInitOperands())) {
315 
316  // Check that result from out operand is dead.
317  Value result = genericOp.getResult(outputOpOperand.index());
318  if (!result.use_empty())
319  continue;
320 
321  // Check that outputArg has one use in cycle.
322  BlockArgument outputArg =
323  genericOp.getRegionOutputArgs()[outputOpOperand.index()];
324  if (!outputArg.hasOneUse())
325  continue;
326 
327  // Check cycle has at most one use.
328  Operation *cycleOp = *outputArg.user_begin();
329  if (!cycleOp->hasOneUse())
330  continue;
331 
332  // Check that the cycleUser is a yield.
333  Operation *cycleUserOp = *cycleOp->user_begin();
334  if (!isa<linalg::YieldOp>(cycleUserOp))
335  continue;
336 
337  // Check that argIndex matches yieldIndex, else data is being used.
338  if (cycleUserOp->getOperand(outputOpOperand.index()) !=
339  cycleOp->getResult(0))
340  continue;
341 
342  // Directly replace the cycle with the blockArg such that
343  // Deduplicate pattern can eliminate it along with unused yield.
344  rewriter.replaceOp(cycleOp, outputArg);
345  rewriter.updateRootInPlace(genericOp, [] {});
346  hasRemovedCycles = true;
347  }
348 
349  if (hasRemovedCycles) {
350  return success();
351  }
352 
353  return failure();
354  }
355 };
