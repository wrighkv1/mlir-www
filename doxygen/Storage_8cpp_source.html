27 SparseTensorStorageBase::SparseTensorStorageBase( // NOLINT
28  uint64_t dimRank, const uint64_t *dimSizes, uint64_t lvlRank,
29  const uint64_t *lvlSizes, const DimLevelType *lvlTypes,
30  const uint64_t *lvl2dim)
31  : dimSizes(dimSizes, dimSizes + dimRank),
32  lvlSizes(lvlSizes, lvlSizes + lvlRank),
33  lvlTypes(lvlTypes, lvlTypes + lvlRank),
34  lvl2dim(lvl2dim, lvl2dim + lvlRank) {
35  // TODO: If we do get any nullptrs, I'm pretty sure these assertions
36  // will run too late (i.e., after copying things into vectors above).
37  // But since those fields are const I'm not sure there's any clean way
38  // to assert things before copying...
39  assert(dimSizes && "Got nullptr for dimension sizes");
40  assert(lvlSizes && "Got nullptr for level sizes");
41  assert(lvlTypes && "Got nullptr for level types");
42  assert(lvl2dim && "Got nullptr for level-to-dimension mapping");
43  // Validate dim-indexed parameters.
44  assert(dimRank > 0 && "Trivial shape is unsupported");
45  for (uint64_t d = 0; d < dimRank; ++d)
46  assert(dimSizes[d] > 0 && "Dimension size zero has trivial storage");
47  // Validate level-indexed parameters.
48  assert(lvlRank > 0 && "Trivial shape is unsupported");
49  for (uint64_t l = 0; l < lvlRank; ++l) {
50  assert(lvlSizes[l] > 0 && "Level size zero has trivial storage");
51  const auto dlt = lvlTypes[l]; // Avoid redundant bounds checking.
52  // We use `MLIR_SPARSETENSOR_FATAL` here instead of `assert` so that
53  // when this ctor is successful then all the methods can rely on the
54  // fact that each level-type satisfies one of these options (even
55  // when `NDEBUG` is true), thereby reducing the need to re-assert things.
56  if (!(isDenseDLT(dlt) || isCompressedDLT(dlt) || isSingletonDLT(dlt)))
57  MLIR_SPARSETENSOR_FATAL("unsupported level type: %d\n",
58  static_cast<uint8_t>(dlt));
59  }
60 }
