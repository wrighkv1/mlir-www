45 /// Reads an element of a non-complex type for the current indices in
46 /// coordinate scheme.
47 template <typename V>
48 inline std::enable_if_t<!is_complex<V>::value, V>
49 readCOOValue(char **linePtr, bool is_pattern) {
50  // The external formats always store these numerical values with the type
51  // double, but we cast these values to the sparse tensor object type.
52  // For a pattern tensor, we arbitrarily pick the value 1 for all entries.
53  return is_pattern ? 1.0 : strtod(*linePtr, linePtr);
54 }

56 /// Reads an element of a complex type for the current indices in
57 /// coordinate scheme.
58 template <typename V>
59 inline std::enable_if_t<is_complex<V>::value, V> readCOOValue(char **linePtr,
60  bool is_pattern) {
61  // Read two values to make a complex. The external formats always store
62  // numerical values with the type double, but we cast these values to the
63  // sparse tensor object type. For a pattern tensor, we arbitrarily pick the
64  // value 1 for all entries.
65  double re = is_pattern ? 1.0 : strtod(*linePtr, linePtr);
66  double im = is_pattern ? 1.0 : strtod(*linePtr, linePtr);
67  // Avoiding brace-notation since that forbids narrowing to `float`.
68  return V(re, im);
69 }

212 /// Reads a sparse tensor with the given filename into a memory-resident
213 /// sparse tensor.
214 ///
215 /// Preconditions:
216 /// * `dimShape` and `dim2lvl` must be valid for `dimRank`.
217 /// * `lvlTypes` and `lvl2dim` must be valid for `lvlRank`.
218 /// * `dim2lvl` is the inverse of `lvl2dim`.
219 ///
220 /// Asserts:
221 /// * the file's actual value type can be read as `valTp`.
222 /// * the file's actual dimension-sizes match the expected `dimShape`.
223 /// * `dim2lvl` is a permutation, and therefore also `dimRank == lvlRank`.
224 //
225 // TODO: As currently written, this function uses `dim2lvl` in two
226 // places: first, to construct the level-sizes from the file's actual
227 // dimension-sizes; and second, to map the file's dimension-indices into
228 // level-indices. The latter can easily generalize to arbitrary mappings,
229 // however the former cannot. Thus, once we functionalize the mappings,
230 // this function will need both the sizes-to-sizes and indices-to-indices
231 // variants of the `dim2lvl` mapping. For the `lvl2dim` direction we only
232 // need the indices-to-indices variant, for handing off to `newFromCOO`.
233 template <typename P, typename I, typename V>
234 inline SparseTensorStorage<P, I, V> *
235 openSparseTensor(uint64_t dimRank, const uint64_t *dimShape, uint64_t lvlRank,
236  const DimLevelType *lvlTypes, const uint64_t *lvl2dim,
237  const uint64_t *dim2lvl, const char *filename,
238  PrimaryType valTp) {
239  // Read the file's header and check the file's actual element type and
240  // dimension-sizes against the expected element type and dimension-shape.
241  SparseTensorReader stfile(filename);
242  stfile.openFile();
243  stfile.readHeader();
244  if (!stfile.canReadAs(valTp))
245  MLIR_SPARSETENSOR_FATAL(
246  "Tensor element type %d not compatible with values in file %s\n",
247  static_cast<int>(valTp), filename);
248  stfile.assertMatchesShape(dimRank, dimShape);
249  const uint64_t *dimSizes = stfile.getDimSizes();
250  // Construct the level-sizes from the file's dimension-sizes
251  // TODO: This doesn't generalize to arbitrary mappings. (See above.)
252  assert(dimRank == lvlRank && "Rank mismatch");
253  detail::PermutationRef d2l(dimRank, dim2lvl);
254  std::vector<uint64_t> lvlSizes = d2l.pushforward(dimRank, dimSizes);
255  // Prepare a COO object with the number of nonzeros as initial capacity.
256  uint64_t nnz = stfile.getNNZ();
257  auto *lvlCOO = new SparseTensorCOO<V>(lvlSizes, nnz);
258  // Read all nonzero elements.
259  std::vector<uint64_t> dimInd(dimRank);
260  std::vector<uint64_t> lvlInd(lvlRank);
261  for (uint64_t k = 0; k < nnz; ++k) {
262  const V value = stfile.readCOOElement<V>(dimRank, dimInd.data());
263  d2l.pushforward(dimRank, dimInd.data(), lvlInd.data());
264  // TODO: <https://github.com/llvm/llvm-project/issues/54179>
265  lvlCOO->add(lvlInd, value);
266  // We currently chose to deal with symmetric matrices by fully
267  // constructing them. In the future, we may want to make symmetry
268  // implicit for storage reasons.
269  if (stfile.isSymmetric() && lvlInd[0] != lvlInd[1])
270  lvlCOO->add({lvlInd[1], lvlInd[0]}, value);
271  }
272  // Close the file, convert the COO to SparseTensorStorage, and return.
273  stfile.closeFile();
274  auto *tensor = SparseTensorStorage<P, I, V>::newFromCOO(
275  dimRank, dimSizes, lvlRank, lvlTypes, lvl2dim, *lvlCOO);
276  delete lvlCOO;
277  return tensor;
278 }

280 /// Writes the sparse tensor to `filename` in extended FROSTT format.
281 template <typename V>
282 inline void writeExtFROSTT(const SparseTensorCOO<V> &coo,
283  const char *filename) {
284  assert(filename && "Got nullptr for filename");
285  auto &dimSizes = coo.getDimSizes();
286  auto &elements = coo.getElements();
287  const uint64_t rank = coo.getRank();
288  const uint64_t nnz = elements.size();
289  std::fstream file;
290  file.open(filename, std::ios_base::out | std::ios_base::trunc);
291  assert(file.is_open());
292  file << "; extended FROSTT format\n" << rank << " " << nnz << std::endl;
293  for (uint64_t r = 0; r < rank - 1; ++r)
294  file << dimSizes[r] << " ";
295  file << dimSizes[rank - 1] << std::endl;
296  for (uint64_t i = 0; i < nnz; ++i) {
297  auto &idx = elements[i].indices;
298  for (uint64_t r = 0; r < rank; ++r)
299  file << (idx[r] + 1) << " ";
300  file << elements[i].value << std::endl;
301  }
302  file.flush();
303  file.close();
304  assert(file.good());
305 }
