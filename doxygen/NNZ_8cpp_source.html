24 SparseTensorNNZ::SparseTensorNNZ(const std::vector<uint64_t> &lvlSizes,
25  const std::vector<DimLevelType> &lvlTypes)
26  : lvlSizes(lvlSizes), lvlTypes(lvlTypes), nnz(getLvlRank()) {
27  assert(lvlSizes.size() == lvlTypes.size() && "Rank mismatch");
28  bool alreadyCompressed = false;
29  (void)alreadyCompressed;
30  uint64_t sz = 1; // the product of all `lvlSizes` strictly less than `l`.
31  for (uint64_t l = 0, lvlrank = getLvlRank(); l < lvlrank; ++l) {
32  const DimLevelType dlt = lvlTypes[l];
33  if (isCompressedDLT(dlt)) {
34  if (alreadyCompressed)
35  MLIR_SPARSETENSOR_FATAL(
36  "Multiple compressed levels not currently supported");
37  alreadyCompressed = true;
38  nnz[l].resize(sz, 0); // Both allocate and zero-initialize.
39  } else if (isDenseDLT(dlt)) {
40  if (alreadyCompressed)
41  MLIR_SPARSETENSOR_FATAL(
42  "Dense after compressed not currently supported");
43  } else if (isSingletonDLT(dlt)) {
44  // Singleton after Compressed causes no problems for allocating
45  // `nnz` nor for the yieldPos loop. This remains true even
46  // when adding support for multiple compressed dimensions or
47  // for dense-after-compressed.
48  } else {
49  MLIR_SPARSETENSOR_FATAL("unsupported level type: %d\n",
50  static_cast<uint8_t>(dlt));
51  }
52  sz = detail::checkedMul(sz, lvlSizes[l]);
53  }
54 }

56 void SparseTensorNNZ::forallIndices(uint64_t stopLvl,
57  SparseTensorNNZ::NNZConsumer yield) const {
58  assert(stopLvl < getLvlRank() && "Level out of bounds");
59  assert(isCompressedDLT(lvlTypes[stopLvl]) &&
60  "Cannot look up non-compressed levels");
61  forallIndices(yield, stopLvl, 0, 0);
62 }

64 void SparseTensorNNZ::add(const std::vector<uint64_t> &lvlInd) {
65  uint64_t parentPos = 0;
66  for (uint64_t l = 0, lvlrank = getLvlRank(); l < lvlrank; ++l) {
67  if (isCompressedDLT(lvlTypes[l]))
68  nnz[l][parentPos]++;
69  parentPos = parentPos * lvlSizes[l] + lvlInd[l];
70  }
71 }

73 void SparseTensorNNZ::forallIndices(SparseTensorNNZ::NNZConsumer yield,
74  uint64_t stopLvl, uint64_t parentPos,
75  uint64_t l) const {
76  assert(l <= stopLvl);
77  if (l == stopLvl) {
78  assert(parentPos < nnz[l].size() && "Cursor is out of range");
79  yield(nnz[l][parentPos]);
80  } else {
81  const uint64_t sz = lvlSizes[l];
82  const uint64_t pstart = parentPos * sz;
83  for (uint64_t i = 0; i < sz; ++i)
84  forallIndices(yield, stopLvl, pstart + i, l + 1);
85  }
