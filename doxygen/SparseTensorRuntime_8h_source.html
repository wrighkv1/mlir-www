43 /// The @newSparseTensor function for constructing a new sparse tensor.
44 /// This is the "swiss army knife" method for materializing sparse
45 /// tensors into the computation. The types of the `ptr` argument and
46 /// the result depend on the action, as explained in the following table
47 /// (where "STS" means a sparse-tensor-storage object, "COO" means
48 /// a coordinate-scheme object, and "Iterator" means an iterator object).
49 ///
50 /// Action: `ptr`: Returns:
51 /// kEmpty unused STS, empty
52 /// kEmptyCOO unused COO, empty
53 /// kFromFile char* filename STS, read from the file
54 /// kFromCOO COO STS, copied from the COO source
55 /// kToCOO STS COO, copied from the STS source
56 /// kSparseToSparse STS STS, copied from the STS source
57 /// kToIterator STS Iterator, call @getNext to use and
58 /// @delSparseTensorIterator to free.
59 MLIR_CRUNNERUTILS_EXPORT void *_mlir_ciface_newSparseTensor( // NOLINT
60  StridedMemRefType<index_type, 1> *dimSizesRef,
61  StridedMemRefType<index_type, 1> *lvlSizesRef,
62  StridedMemRefType<DimLevelType, 1> *lvlTypesRef,
63  StridedMemRefType<index_type, 1> *lvl2dimRef,
64  StridedMemRefType<index_type, 1> *dim2lvlRef, OverheadType ptrTp,
65  OverheadType indTp, PrimaryType valTp, Action action, void *ptr);

174 /// Initializes sparse tensor from a COO-flavored format expressed using
175 /// C-style data structures. The expected parameters are:
176 ///
177 /// rank: rank of tensor
178 /// nse: number of specified elements (usually the nonzeros)
179 /// shape: array with dimension size for each rank
180 /// values: a "nse" array with values for all specified elements
181 /// indices: a flat "nse * rank" array with indices for all specified elements
182 /// perm: the permutation of the dimensions in the storage
183 /// sparse: the sparsity for the dimensions
184 ///
185 /// For example, the sparse matrix
186 /// | 1.0 0.0 0.0 |
187 /// | 0.0 5.0 3.0 |
188 /// can be passed as
189 /// rank = 2
190 /// nse = 3
191 /// shape = [2, 3]
192 /// values = [1.0, 5.0, 3.0]
193 /// indices = [ 0, 0, 1, 1, 1, 2]
194 #define DECL_CONVERTTOMLIRSPARSETENSOR(VNAME, V) \
195  MLIR_CRUNNERUTILS_EXPORT void *convertToMLIRSparseTensor##VNAME( \
196  uint64_t rank, uint64_t nse, uint64_t *shape, V *values, \
197  uint64_t *indices, uint64_t *perm, uint8_t *sparse);
198 MLIR_SPARSETENSOR_FOREVERY_V(DECL_CONVERTTOMLIRSPARSETENSOR)
199 #undef DECL_CONVERTTOMLIRSPARSETENSOR
