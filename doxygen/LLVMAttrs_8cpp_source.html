77 LoopOptionsAttrBuilder &LoopOptionsAttrBuilder::setOption(LoopOptionCase tag,
78  Optional<T> value) {
79  auto option = llvm::find_if(
80  options, [tag](auto option) { return option.first == tag; });
81  if (option != options.end()) {
82  if (value)
83  option->second = *value;
84  else
85  options.erase(option);
86  } else {
87  options.push_back(LoopOptionsAttr::OptionValuePair(tag, *value));
88  }
89  return *this;
90 }

129 template <typename T>
130 static Optional<T>
131 getOption(ArrayRef<std::pair<LoopOptionCase, int64_t>> options,
132  LoopOptionCase option) {
133  auto it =
134  lower_bound(options, option, [](auto optionPair, LoopOptionCase option) {
135  return optionPair.first < option;
136  });
137  if (it == options.end())
138  return {};
139  return static_cast<T>(it->second);
140 }

170 void LoopOptionsAttr::print(AsmPrinter &printer) const {
171  printer << "<";
172  llvm::interleaveComma(getOptions(), printer, [&](auto option) {
173  printer << stringifyEnum(option.first) << " = ";
174  switch (option.first) {
175  case LoopOptionCase::disable_licm:
176  case LoopOptionCase::disable_unroll:
177  case LoopOptionCase::disable_pipeline:
178  printer << (option.second ? "true" : "false");
179  break;
180  case LoopOptionCase::interleave_count:
181  case LoopOptionCase::pipeline_initiation_interval:
182  printer << option.second;
183  break;
184  }
185  });
186  printer << ">";

189 Attribute LoopOptionsAttr::parse(AsmParser &parser, Type type) {
190  if (failed(parser.parseLess()))
191  return {};
192 
193  SmallVector<std::pair<LoopOptionCase, int64_t>> options;
194  llvm::SmallDenseSet<LoopOptionCase> seenOptions;
195  auto parseLoopOptions = [&]() -> ParseResult {
196  StringRef optionName;
197  if (parser.parseKeyword(&optionName))
198  return failure();
199 
200  auto option = symbolizeLoopOptionCase(optionName);
201  if (!option)
202  return parser.emitError(parser.getNameLoc(), "unknown loop option: ")
203  << optionName;
204  if (!seenOptions.insert(*option).second)
205  return parser.emitError(parser.getNameLoc(), "loop option present twice");
206  if (failed(parser.parseEqual()))
207  return failure();
208 
209  int64_t value;
210  switch (*option) {
211  case LoopOptionCase::disable_licm:
212  case LoopOptionCase::disable_unroll:
213  case LoopOptionCase::disable_pipeline:
214  if (succeeded(parser.parseOptionalKeyword("true")))
215  value = 1;
216  else if (succeeded(parser.parseOptionalKeyword("false")))
217  value = 0;
218  else {
219  return parser.emitError(parser.getNameLoc(),
220  "expected boolean value 'true' or 'false'");
221  }
222  break;
223  case LoopOptionCase::interleave_count:
224  case LoopOptionCase::pipeline_initiation_interval:
225  if (failed(parser.parseInteger(value)))
226  return parser.emitError(parser.getNameLoc(), "expected integer value");
227  break;
228  }
229  options.push_back(std::make_pair(*option, value));
230  return success();
231  };
232  if (parser.parseCommaSeparatedList(parseLoopOptions) || parser.parseGreater())
233  return {};
234 
235  llvm::sort(options, llvm::less_first());
236  return get(parser.getContext(), options);
237 }
