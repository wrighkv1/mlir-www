50 // Checks for a dynamic batch dim in any of the passed parameters of an op.
51 // The batch dimention must be #0 and the rest of the dimensions must be static.
52 template <typename Op>
53 Optional<SmallVector<Value>> checkHasDynamicBatchDims(PatternRewriter &rewriter,
54  Op op,
55  ArrayRef<Value> params) {
56  SmallVector<ShapedType> dynTypes;
57  SmallVector<Value> dynamicDims;
58  for (const Value &param : params) {
59  auto paramTy = param.getType().cast<ShapedType>();
60  if (!paramTy.hasStaticShape())
61  dynTypes.push_back(paramTy);
62  }
63 
64  if (dynTypes.empty())
65  return dynamicDims;
66 
67  for (const ShapedType &dynTy : dynTypes) {
68  if (llvm::any_of(dynTy.getShape().drop_front(), ShapedType::isDynamic)) {
69  (void)rewriter.notifyMatchFailure(
70  op, "input can only be dynamic for batch size");
71  return llvm::None;
72  }
73  }
74 
75  dynamicDims.push_back(
76  rewriter.create<tensor::DimOp>(op->getLoc(), params[0], 0));
77  return dynamicDims;
78 }
