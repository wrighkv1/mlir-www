67 /// Creates an attribute containing ABI and preferred alignment numbers parsed
68 /// a string. The string may be either "abi:preferred" or just "abi". In the
69 /// latter case, the prefrred alignment is considered equal to ABI alignment.
70 static DenseIntElementsAttr parseDataLayoutAlignment(MLIRContext &ctx,
71  StringRef spec) {
72  auto i32 = IntegerType::get(&ctx, 32);
73 
74  StringRef abiString, preferredString;
75  std::tie(abiString, preferredString) = spec.split(':');
76  int abi, preferred;
77  if (abiString.getAsInteger(/*Radix=*/10, abi))
78  return nullptr;
79 
80  if (preferredString.empty())
81  preferred = abi;
82  else if (preferredString.getAsInteger(/*Radix=*/10, preferred))
83  return nullptr;
84 
85  return DenseIntElementsAttr::get(VectorType::get({2}, i32), {abi, preferred});
86 }

247 DataLayoutSpecInterface
248 mlir::translateDataLayout(const llvm::DataLayout &dataLayout,
249  MLIRContext *context) {
250  assert(context && "expected MLIR context");
251  std::string layoutstr = dataLayout.getStringRepresentation();
252 
253  // Remaining unhandled default layout defaults
254  // e (little endian if not set)
255  // p[n]:64:64:64 (non zero address spaces have 64-bit properties)
256  std::string append =
257  "p:64:64:64-S0-i1:8:8-i8:8:8-i16:16:16-i32:32:32-i64:32:64-f16:16:16-f64:"
258  "64:64-f128:128:128-v64:64:64-v128:128:128-a:0:64";
259  if (layoutstr.empty())
260  layoutstr = append;
261  else
262  layoutstr = layoutstr + "-" + append;
263 
264  StringRef layout(layoutstr);
265 
266  SmallVector<DataLayoutEntryInterface> entries;
267  StringSet<> seen;
268  while (!layout.empty()) {
269  // Split at '-'.
270  std::pair<StringRef, StringRef> split = layout.split('-');
271  StringRef current;
272  std::tie(current, layout) = split;
273 
274  // Split at ':'.
275  StringRef kind, spec;
276  std::tie(kind, spec) = current.split(':');
277  if (seen.contains(kind))
278  continue;
279  seen.insert(kind);
280 
281  char symbol = kind.front();
282  StringRef parameter = kind.substr(1);
283 
284  if (symbol == 'i' || symbol == 'f') {
285  unsigned bitwidth;
286  if (parameter.getAsInteger(/*Radix=*/10, bitwidth))
287  return nullptr;
288  DenseIntElementsAttr params = parseDataLayoutAlignment(*context, spec);
289  if (!params)
290  return nullptr;
291  auto entry = DataLayoutEntryAttr::get(
292  symbol == 'i' ? static_cast<Type>(IntegerType::get(context, bitwidth))
293  : getDLFloatType(*context, bitwidth),
294  params);
295  entries.emplace_back(entry);
296  } else if (symbol == 'e' || symbol == 'E') {
297  auto value = StringAttr::get(
298  context, symbol == 'e' ? DLTIDialect::kDataLayoutEndiannessLittle
299  : DLTIDialect::kDataLayoutEndiannessBig);
300  auto entry = DataLayoutEntryAttr::get(
301  StringAttr::get(context, DLTIDialect::kDataLayoutEndiannessKey),
302  value);
303  entries.emplace_back(entry);
304  }
305  }
306 
307  return DataLayoutSpecAttr::get(context, entries);
308 }

310 /// Get a topologically sorted list of blocks for the given function.
311 static SetVector<llvm::BasicBlock *>
312 getTopologicallySortedBlocks(llvm::Function *func) {
313  SetVector<llvm::BasicBlock *> blocks;
314  for (llvm::BasicBlock &bb : *func) {
315  if (blocks.count(&bb) == 0) {
316  llvm::ReversePostOrderTraversal<llvm::BasicBlock *> traversal(&bb);
317  blocks.insert(traversal.begin(), traversal.end());
318  }
319  }
320  assert(blocks.size() == func->getBasicBlockList().size() &&
321  "some blocks are not sorted");
322 
323  return blocks;

500 // We only need integers, floats, doubles, and vectors and tensors thereof for
501 // attributes. Scalar and vector types are converted to the standard
502 // equivalents. Array types are converted to ranked tensors; nested array types
503 // are converted to multi-dimensional tensors or vectors, depending on the
504 // innermost type being a scalar or a vector.
505 Type Importer::getStdTypeForAttr(Type type) {
506  if (!type)
507  return nullptr;
508 
509  if (type.isa<IntegerType, FloatType>())
510  return type;
511 
512  // LLVM vectors can only contain scalars.
513  if (LLVM::isCompatibleVectorType(type)) {
514  llvm::ElementCount numElements = LLVM::getVectorNumElements(type);
515  if (numElements.isScalable()) {
516  emitError(UnknownLoc::get(context)) << "scalable vectors not supported";
517  return nullptr;
518  }
519  Type elementType = getStdTypeForAttr(LLVM::getVectorElementType(type));
520  if (!elementType)
521  return nullptr;
522  return VectorType::get(numElements.getKnownMinValue(), elementType);
523  }
524 
525  // LLVM arrays can contain other arrays or vectors.
526  if (auto arrayType = type.dyn_cast<LLVMArrayType>()) {
527  // Recover the nested array shape.
528  SmallVector<int64_t, 4> shape;
529  shape.push_back(arrayType.getNumElements());
530  while (arrayType.getElementType().isa<LLVMArrayType>()) {
531  arrayType = arrayType.getElementType().cast<LLVMArrayType>();
532  shape.push_back(arrayType.getNumElements());
533  }
534 
535  // If the innermost type is a vector, use the multi-dimensional vector as
536  // attribute type.
537  if (LLVM::isCompatibleVectorType(arrayType.getElementType())) {
538  llvm::ElementCount numElements =
539  LLVM::getVectorNumElements(arrayType.getElementType());
540  if (numElements.isScalable()) {
541  emitError(UnknownLoc::get(context)) << "scalable vectors not supported";
542  return nullptr;
543  }
544  shape.push_back(numElements.getKnownMinValue());
545 
546  Type elementType = getStdTypeForAttr(
547  LLVM::getVectorElementType(arrayType.getElementType()));
548  if (!elementType)
549  return nullptr;
550  return VectorType::get(shape, elementType);
551  }
552 
553  // Otherwise use a tensor.
554  Type elementType = getStdTypeForAttr(arrayType.getElementType());
555  if (!elementType)
556  return nullptr;
557  return RankedTensorType::get(shape, elementType);
558  }
559 
560  return nullptr;
561 }

563 // Get the given constant as an attribute. Not all constants can be represented
564 // as attributes.
565 Attribute Importer::getConstantAsAttr(llvm::Constant *value) {
566  if (auto *ci = dyn_cast<llvm::ConstantInt>(value))
567  return builder.getIntegerAttr(
568  IntegerType::get(context, ci->getType()->getBitWidth()),
569  ci->getValue());
570  if (auto *c = dyn_cast<llvm::ConstantDataArray>(value))
571  if (c->isString())
572  return builder.getStringAttr(c->getAsString());
573  if (auto *c = dyn_cast<llvm::ConstantFP>(value)) {
574  llvm::Type *type = c->getType();
575  FloatType floatTy;
576  if (type->isBFloatTy())
577  floatTy = FloatType::getBF16(context);
578  else
579  floatTy = getDLFloatType(*context, type->getScalarSizeInBits());
580  assert(floatTy && "unsupported floating point type");
581  return builder.getFloatAttr(floatTy, c->getValueAPF());
582  }
583  if (auto *f = dyn_cast<llvm::Function>(value))
584  return SymbolRefAttr::get(builder.getContext(), f->getName());
585 
586  // Convert constant data to a dense elements attribute.
587  if (auto *cd = dyn_cast<llvm::ConstantDataSequential>(value)) {
588  Type type = convertType(cd->getElementType());
589  auto attrType = getStdTypeForAttr(convertType(cd->getType()))
590  .dyn_cast_or_null<ShapedType>();
591  if (!attrType)
592  return nullptr;
593 
594  if (type.isa<IntegerType>()) {
595  SmallVector<APInt, 8> values;
596  values.reserve(cd->getNumElements());
597  for (unsigned i = 0, e = cd->getNumElements(); i < e; ++i)
598  values.push_back(cd->getElementAsAPInt(i));
599  return DenseElementsAttr::get(attrType, values);
600  }
601 
602  if (type.isa<Float32Type, Float64Type>()) {
603  SmallVector<APFloat, 8> values;
604  values.reserve(cd->getNumElements());
605  for (unsigned i = 0, e = cd->getNumElements(); i < e; ++i)
606  values.push_back(cd->getElementAsAPFloat(i));
607  return DenseElementsAttr::get(attrType, values);
608  }
609 
610  return nullptr;
611  }
612 
613  // Unpack constant aggregates to create dense elements attribute whenever
614  // possible. Return nullptr (failure) otherwise.
615  if (isa<llvm::ConstantAggregate>(value)) {
616  auto outerType = getStdTypeForAttr(convertType(value->getType()))
617  .dyn_cast_or_null<ShapedType>();
618  if (!outerType)
619  return nullptr;
620 
621  SmallVector<Attribute, 8> values;
622  SmallVector<int64_t, 8> shape;
623 
624  for (unsigned i = 0, e = value->getNumOperands(); i < e; ++i) {
625  auto nested = getConstantAsAttr(value->getAggregateElement(i))
626  .dyn_cast_or_null<DenseElementsAttr>();
627  if (!nested)
628  return nullptr;
629 
630  values.append(nested.value_begin<Attribute>(),
631  nested.value_end<Attribute>());
632  }
633 
634  return DenseElementsAttr::get(outerType, values);
635  }
636 
637  return nullptr;
638 }

686 Value Importer::convertConstantInPlace(llvm::Constant *constant) {
687  if (Attribute attr = getConstantAsAttr(constant)) {
688  // These constants can be represented as attributes.
689  Type type = convertType(constant->getType());
690  if (auto symbolRef = attr.dyn_cast<FlatSymbolRefAttr>())
691  return builder.create<AddressOfOp>(UnknownLoc::get(context), type,
692  symbolRef.getValue());
693  return builder.create<ConstantOp>(UnknownLoc::get(context), type, attr);
694  }
695  if (auto *cn = dyn_cast<llvm::ConstantPointerNull>(constant)) {
696  Type type = convertType(cn->getType());
697  return builder.create<NullOp>(UnknownLoc::get(context), type);
698  }
699  if (auto *gv = dyn_cast<llvm::GlobalVariable>(constant))
700  return builder.create<AddressOfOp>(UnknownLoc::get(context),
701  processGlobal(gv));
702 
703  if (auto *ce = dyn_cast<llvm::ConstantExpr>(constant)) {
704  llvm::Instruction *i = ce->getAsInstruction();
705  if (failed(processInstruction(i)))
706  return nullptr;
707  assert(valueMapping.count(i));
708 
709  // If we don't remove entry of `i` here, it's totally possible that the
710  // next time llvm::ConstantExpr::getAsInstruction is called again, which
711  // always allocates a new Instruction, memory address of the newly
712  // created Instruction might be the same as `i`. Making processInstruction
713  // falsely believe that the new Instruction has been processed before
714  // and raised an assertion error.
715  Value value = valueMapping[i];
716  valueMapping.erase(i);
717  // Remove this zombie LLVM instruction now, leaving us only with the MLIR
718  // op.
719  i->deleteValue();
720  return value;
721  }
722  if (auto *ue = dyn_cast<llvm::UndefValue>(constant)) {
723  Type type = convertType(ue->getType());
724  return builder.create<UndefOp>(UnknownLoc::get(context), type);
725  }
726 
727  if (isa<llvm::ConstantAggregate>(constant) ||
728  isa<llvm::ConstantAggregateZero>(constant)) {
729  unsigned numElements = constant->getNumOperands();
730  std::function<llvm::Constant *(unsigned)> getElement =
731  [&](unsigned index) -> llvm::Constant * {
732  return constant->getAggregateElement(index);
733  };
734  // llvm::ConstantAggregateZero doesn't take any operand
735  // so its getNumOperands is always zero.
736  if (auto *caz = dyn_cast<llvm::ConstantAggregateZero>(constant)) {
737  numElements = caz->getElementCount().getFixedValue();
738  // We want to capture the pointer rather than reference
739  // to the pointer since the latter will become dangling upon
740  // exiting the scope.
741  getElement = [=](unsigned index) -> llvm::Constant * {
742  return caz->getElementValue(index);
743  };
744  }
745 
746  // Generate a llvm.undef as the root value first.
747  Type rootType = convertType(constant->getType());
748  bool useInsertValue = rootType.isa<LLVMArrayType, LLVMStructType>();
749  assert((useInsertValue || LLVM::isCompatibleVectorType(rootType)) &&
750  "unrecognized aggregate type");
751  Value root = builder.create<UndefOp>(UnknownLoc::get(context), rootType);
752  for (unsigned i = 0; i < numElements; ++i) {
753  llvm::Constant *element = getElement(i);
754  Value elementValue = convertConstantInPlace(element);
755  if (!elementValue)
756  return nullptr;
757  if (useInsertValue) {
758  root = builder.create<InsertValueOp>(UnknownLoc::get(context), root,
759  elementValue, i);
760  } else {
761  Attribute indexAttr =
762  builder.getI32IntegerAttr(static_cast<int32_t>(i));
763  Value indexValue = builder.create<ConstantOp>(
764  UnknownLoc::get(context), builder.getI32Type(), indexAttr);
765  if (!indexValue)
766  return nullptr;
767  root = builder.create<InsertElementOp>(
768  UnknownLoc::get(context), rootType, root, elementValue, indexValue);
769  }
770  }
771  return root;
772  }
773 
774  return nullptr;
775 }

841 LogicalResult Importer::processInstruction(llvm::Instruction *inst) {
842  // FIXME: Support uses of SubtargetData.
843  // FIXME: Add support for inbounds GEPs.
844  // FIXME: Add support for fast-math flags and call / operand attributes.
845  // FIXME: Add support for the indirectbr, cleanupret, catchret, catchswitch,
846  // callbr, vaarg, landingpad, catchpad, cleanuppad instructions.
847 
848  // Convert all intrinsics that provide an MLIR builder.
849  if (auto *callInst = dyn_cast<llvm::CallInst>(inst))
850  if (succeeded(convertIntrinsic(builder, callInst)))
851  return success();
852 
853  // Convert all operations that provide an MLIR builder.
854  if (succeeded(convertOperation(builder, inst)))
855  return success();
856 
857  // Convert all special instructions that do not provide an MLIR builder.
858  Location loc = translateLoc(inst->getDebugLoc());
859  if (inst->getOpcode() == llvm::Instruction::Br) {
860  auto *brInst = cast<llvm::BranchInst>(inst);
861  OperationState state(loc,
862  brInst->isConditional() ? "llvm.cond_br" : "llvm.br");
863  if (brInst->isConditional()) {
864  Value condition = processValue(brInst->getCondition());
865  state.addOperands(condition);
866  }
867 
868  std::array<int32_t, 3> operandSegmentSizes = {1, 0, 0};
869  for (int i : llvm::seq<int>(0, brInst->getNumSuccessors())) {
870  llvm::BasicBlock *succ = brInst->getSuccessor(i);
871  SmallVector<Value, 4> blockArguments;
872  if (failed(processBranchArgs(brInst, succ, blockArguments)))
873  return failure();
874  state.addSuccessors(lookupBlock(succ));
875  state.addOperands(blockArguments);
876  operandSegmentSizes[i + 1] = blockArguments.size();
877  }
878 
879  if (brInst->isConditional()) {
880  state.addAttribute(LLVM::CondBrOp::getOperandSegmentSizeAttr(),
881  builder.getDenseI32ArrayAttr(operandSegmentSizes));
882  }
883 
884  builder.create(state);
885  return success();
886  }
887  if (inst->getOpcode() == llvm::Instruction::Switch) {
888  auto *swInst = cast<llvm::SwitchInst>(inst);
889  // Process the condition value.
890  Value condition = processValue(swInst->getCondition());
891  SmallVector<Value> defaultBlockArgs;
892  // Process the default case.
893  llvm::BasicBlock *defaultBB = swInst->getDefaultDest();
894  if (failed(processBranchArgs(swInst, defaultBB, defaultBlockArgs)))
895  return failure();
896 
897  // Process the cases.
898  unsigned numCases = swInst->getNumCases();
899  SmallVector<SmallVector<Value>> caseOperands(numCases);
900  SmallVector<ValueRange> caseOperandRefs(numCases);
901  SmallVector<int32_t> caseValues(numCases);
902  SmallVector<Block *> caseBlocks(numCases);
903  for (const auto &it : llvm::enumerate(swInst->cases())) {
904  const llvm::SwitchInst::CaseHandle &caseHandle = it.value();
905  llvm::BasicBlock *succBB = caseHandle.getCaseSuccessor();
906  if (failed(processBranchArgs(swInst, succBB, caseOperands[it.index()])))
907  return failure();
908  caseOperandRefs[it.index()] = caseOperands[it.index()];
909  caseValues[it.index()] = caseHandle.getCaseValue()->getSExtValue();
910  caseBlocks[it.index()] = lookupBlock(succBB);
911  }
912 
913  builder.create<SwitchOp>(loc, condition, lookupBlock(defaultBB),
914  defaultBlockArgs, caseValues, caseBlocks,
915  caseOperandRefs);
916  return success();
917  }
918  if (inst->getOpcode() == llvm::Instruction::PHI) {
919  Type type = convertType(inst->getType());
920  mapValue(inst, builder.getInsertionBlock()->addArgument(
921  type, translateLoc(inst->getDebugLoc())));
922  return success();
923  }
924  if (inst->getOpcode() == llvm::Instruction::Call) {
925  llvm::CallInst *ci = cast<llvm::CallInst>(inst);
926  SmallVector<llvm::Value *> args(ci->args());
927  SmallVector<Value> ops = processValues(args);
928  SmallVector<Type, 2> tys;
929  if (!ci->getType()->isVoidTy()) {
930  Type type = convertType(inst->getType());
931  tys.push_back(type);
932  }
933  Operation *op;
934  if (llvm::Function *callee = ci->getCalledFunction()) {
935  op = builder.create<CallOp>(
936  loc, tys, SymbolRefAttr::get(builder.getContext(), callee->getName()),
937  ops);
938  } else {
939  Value calledValue = processValue(ci->getCalledOperand());
940  ops.insert(ops.begin(), calledValue);
941  op = builder.create<CallOp>(loc, tys, ops);
942  }
943  if (!ci->getType()->isVoidTy())
944  mapValue(inst, op->getResult(0));
945  return success();
946  }
947  if (inst->getOpcode() == llvm::Instruction::LandingPad) {
948  llvm::LandingPadInst *lpi = cast<llvm::LandingPadInst>(inst);
949  SmallVector<Value, 4> ops;
950 
951  for (unsigned i = 0, ie = lpi->getNumClauses(); i < ie; i++)
952  ops.push_back(convertConstant(lpi->getClause(i)));
953 
954  Type ty = convertType(lpi->getType());
955  Value res = builder.create<LandingpadOp>(loc, ty, lpi->isCleanup(), ops);
956  mapValue(inst, res);
957  return success();
958  }
959  if (inst->getOpcode() == llvm::Instruction::Invoke) {
960  llvm::InvokeInst *ii = cast<llvm::InvokeInst>(inst);
961 
962  SmallVector<Type, 2> tys;
963  if (!ii->getType()->isVoidTy())
964  tys.push_back(convertType(inst->getType()));
965 
966  SmallVector<llvm::Value *> args(ii->args());
967  SmallVector<Value> ops = processValues(args);
968 
969  SmallVector<Value, 4> normalArgs, unwindArgs;
970  (void)processBranchArgs(ii, ii->getNormalDest(), normalArgs);
971  (void)processBranchArgs(ii, ii->getUnwindDest(), unwindArgs);
972 
973  Operation *op;
974  if (llvm::Function *callee = ii->getCalledFunction()) {
975  op = builder.create<InvokeOp>(
976  loc, tys, SymbolRefAttr::get(builder.getContext(), callee->getName()),
977  ops, lookupBlock(ii->getNormalDest()), normalArgs,
978  lookupBlock(ii->getUnwindDest()), unwindArgs);
979  } else {
980  ops.insert(ops.begin(), processValue(ii->getCalledOperand()));
981  op = builder.create<InvokeOp>(
982  loc, tys, ops, lookupBlock(ii->getNormalDest()), normalArgs,
983  lookupBlock(ii->getUnwindDest()), unwindArgs);
984  }
985 
986  if (!ii->getType()->isVoidTy())
987  mapValue(inst, op->getResult(0));
988  return success();
989  }
990  if (inst->getOpcode() == llvm::Instruction::GetElementPtr) {
991  // FIXME: Support inbounds GEPs.
992  llvm::GetElementPtrInst *gep = cast<llvm::GetElementPtrInst>(inst);
993  Value basePtr = processValue(gep->getOperand(0));
994  Type sourceElementType = convertType(gep->getSourceElementType());
995 
996  // Treat every indices as dynamic since GEPOp::build will refine those
997  // indices into static attributes later. One small downside of this
998  // approach is that many unused `llvm.mlir.constant` would be emitted
999  // at first place.
1000  SmallVector<GEPArg> indices;
1001  for (llvm::Value *operand : llvm::drop_begin(gep->operand_values())) {
1002  Value val = processValue(operand);
1003  indices.push_back(val);
1004  }
1005 
1006  Type type = convertType(inst->getType());
1007  Value res =
1008  builder.create<GEPOp>(loc, type, sourceElementType, basePtr, indices);
1009  mapValue(inst, res);
1010  return success();
1011  }
1012 
1013  return emitError(loc) << "unknown instruction: " << diag(*inst);
1014 }

1047 LogicalResult Importer::processFunction(llvm::Function *func) {
1048  blockMapping.clear();
1049  valueMapping.clear();
1050 
1051  auto functionType =
1052  convertType(func->getFunctionType()).dyn_cast<LLVMFunctionType>();
1053  if (func->isIntrinsic() && isConvertibleIntrinsic(func->getIntrinsicID()))
1054  return success();
1055 
1056  bool dsoLocal = func->hasLocalLinkage();
1057  CConv cconv = convertCConvFromLLVM(func->getCallingConv());
1058 
1059  // Insert the function at the end of the module.
1060  OpBuilder::InsertionGuard guard(builder);
1061  builder.setInsertionPoint(module.getBody(), module.getBody()->end());
1062 
1063  LLVMFuncOp funcOp = builder.create<LLVMFuncOp>(
1064  UnknownLoc::get(context), func->getName(), functionType,
1065  convertLinkageFromLLVM(func->getLinkage()), dsoLocal, cconv);
1066 
1067  for (const auto &it : llvm::enumerate(functionType.getParams())) {
1068  llvm::SmallVector<NamedAttribute, 1> argAttrs;
1069  if (auto *type = func->getParamByValType(it.index())) {
1070  Type mlirType = convertType(type);
1071  argAttrs.push_back(
1072  NamedAttribute(builder.getStringAttr(LLVMDialect::getByValAttrName()),
1073  TypeAttr::get(mlirType)));
1074  }
1075  if (auto *type = func->getParamByRefType(it.index())) {
1076  Type mlirType = convertType(type);
1077  argAttrs.push_back(
1078  NamedAttribute(builder.getStringAttr(LLVMDialect::getByRefAttrName()),
1079  TypeAttr::get(mlirType)));
1080  }
1081  if (auto *type = func->getParamStructRetType(it.index())) {
1082  Type mlirType = convertType(type);
1083  argAttrs.push_back(NamedAttribute(
1084  builder.getStringAttr(LLVMDialect::getStructRetAttrName()),
1085  TypeAttr::get(mlirType)));
1086  }
1087  if (auto *type = func->getParamInAllocaType(it.index())) {
1088  Type mlirType = convertType(type);
1089  argAttrs.push_back(NamedAttribute(
1090  builder.getStringAttr(LLVMDialect::getInAllocaAttrName()),
1091  TypeAttr::get(mlirType)));
1092  }
1093 
1094  funcOp.setArgAttrs(it.index(), argAttrs);
1095  }
1096 
1097  if (FlatSymbolRefAttr personality = getPersonalityAsAttr(func))
1098  funcOp.setPersonalityAttr(personality);
1099  else if (func->hasPersonalityFn())
1100  emitWarning(UnknownLoc::get(context),
1101  "could not deduce personality, skipping it");
1102 
1103  if (func->hasGC())
1104  funcOp.setGarbageCollector(StringRef(func->getGC()));
1105 
1106  // Handle Function attributes.
1107  processFunctionAttributes(func, funcOp);
1108 
1109  if (func->isDeclaration())
1110  return success();
1111 
1112  // Eagerly create all blocks.
1113  for (llvm::BasicBlock &bb : *func) {
1114  Block *block =
1115  builder.createBlock(&funcOp.getBody(), funcOp.getBody().end());
1116  mapBlock(&bb, block);
1117  }
1118 
1119  // Add function arguments to the entry block.
1120  for (const auto &it : llvm::enumerate(func->args())) {
1121  BlockArgument blockArg = funcOp.getFunctionBody().addArgument(
1122  functionType.getParamType(it.index()), funcOp.getLoc());
1123  mapValue(&it.value(), blockArg);
1124  }
1125 
1126  // Process the blocks in topological order. The ordered traversal ensures
1127  // operands defined in a dominating block have a valid mapping to an MLIR
1128  // value once a block is translated.
1129  SetVector<llvm::BasicBlock *> blocks = getTopologicallySortedBlocks(func);
1130  setConstantInsertionPointToStart(lookupBlock(blocks.front()));
1131  for (llvm::BasicBlock *bb : blocks) {
1132  if (failed(processBasicBlock(bb, lookupBlock(bb))))
1133  return failure();
1134  }
1135 
1136  return success();
1137 }
