25 /// Generate a unique label based on the current file name to prevent name
26 /// collisions if multiple generated files are included at once.
27 static std::string getUniqueOutputLabel(const llvm::RecordKeeper &records) {
28  // Use the input file name when generating a unique name.
29  std::string inputFilename = records.getInputFilename();
30 
31  // Drop all but the base filename.
32  StringRef nameRef = llvm::sys::path::filename(inputFilename);
33  nameRef.consume_back(".td");
34 
35  // Sanitize any invalid characters.
36  std::string uniqueName;
37  for (char c : nameRef) {
38  if (llvm::isAlnum(c) || c == '_')
39  uniqueName.push_back(c);
40  else
41  uniqueName.append(llvm::utohexstr((unsigned char)c));
42  }
43  return uniqueName;
44 }

107 /// Code templates for emitting type, attribute, successor, and region
108 /// constraints. Each of these templates require the following arguments:
109 ///
110 /// {0}: The unique constraint name.
111 /// {1}: The constraint code.
112 /// {2}: The constraint description.
113 
114 /// Code for a type constraint. These may be called on the type of either
115 /// operands or results.
116 static const char *const typeConstraintCode = R"(
117 static ::mlir::LogicalResult {0}(
118  ::mlir::Operation *op, ::mlir::Type type, ::llvm::StringRef valueKind,
119  unsigned valueIndex) {
120  if (!({1})) {
121  return op->emitOpError(valueKind) << " #" << valueIndex
122  << " must be {2}, but got " << type;
123  }
124  return ::mlir::success();
125 }
126 )";
127 
128 /// Code for an attribute constraint. These may be called from ops only.
129 /// Attribute constraints cannot reference anything other than `$_self` and
130 /// `$_op`.
131 ///
132 /// TODO: Unique constraints for adaptors. However, most Adaptor::verify
133 /// functions are stripped anyways.
134 static const char *const attrConstraintCode = R"(
135 static ::mlir::LogicalResult {0}(
136  ::mlir::Operation *op, ::mlir::Attribute attr, ::llvm::StringRef attrName) {
137  if (attr && !({1})) {
138  return op->emitOpError("attribute '") << attrName
139  << "' failed to satisfy constraint: {2}";
140  }
141  return ::mlir::success();
142 }
143 )";
144 
145 /// Code for a successor constraint.
146 static const char *const successorConstraintCode = R"(
147 static ::mlir::LogicalResult {0}(
148  ::mlir::Operation *op, ::mlir::Block *successor,
149  ::llvm::StringRef successorName, unsigned successorIndex) {
150  if (!({1})) {
151  return op->emitOpError("successor #") << successorIndex << " ('"
152  << successorName << ")' failed to verify constraint: {2}";
153  }
154  return ::mlir::success();
155 }
156 )";
157 
158 /// Code for a region constraint. Callers will need to pass in the region's name
159 /// for emitting an error message.
160 static const char *const regionConstraintCode = R"(
161 static ::mlir::LogicalResult {0}(
162  ::mlir::Operation *op, ::mlir::Region &region, ::llvm::StringRef regionName,
163  unsigned regionIndex) {
164  if (!({1})) {
165  return op->emitOpError("region #") << regionIndex
166  << (regionName.empty() ? " " : " ('" + regionName + "') ")
167  << "failed to verify constraint: {2}";
168  }
169  return ::mlir::success();
170 }
171 )";
172 
173 /// Code for a pattern type or attribute constraint.
174 ///
175 /// {3}: "Type type" or "Attribute attr".
176 static const char *const patternAttrOrTypeConstraintCode = R"(
177 static ::mlir::LogicalResult {0}(
178  ::mlir::PatternRewriter &rewriter, ::mlir::Operation *op, ::mlir::{3},
179  ::llvm::StringRef failureStr) {
180  if (!({1})) {
181  return rewriter.notifyMatchFailure(op, [&](::mlir::Diagnostic &diag) {
182  diag << failureStr << ": {2}";
183  });
184  }
185  return ::mlir::success();
186 }
187 )";

189 void StaticVerifierFunctionEmitter::emitConstraints(
190  const ConstraintMap &constraints, StringRef selfName,
191  const char *const codeTemplate) {
192  FmtContext ctx;
193  ctx.withOp("*op").withSelf(selfName);
194  for (auto &it : constraints) {
195  os << formatv(codeTemplate, it.second,
196  tgfmt(it.first.getConditionTemplate(), &ctx),
197  escapeString(it.first.getSummary()));
198  }

217 void StaticVerifierFunctionEmitter::emitPatternConstraints() {
218  FmtContext ctx;
219  ctx.withOp("*op").withBuilder("rewriter").withSelf("type");
220  for (auto &it : typeConstraints) {
221  os << formatv(patternAttrOrTypeConstraintCode, it.second,
222  tgfmt(it.first.getConditionTemplate(), &ctx),
223  escapeString(it.first.getSummary()), "Type type");
224  }
225  ctx.withSelf("attr");
226  for (auto &it : attrConstraints) {
227  os << formatv(patternAttrOrTypeConstraintCode, it.second,
228  tgfmt(it.first.getConditionTemplate(), &ctx),
229  escapeString(it.first.getSummary()), "Attribute attr");
230  }
231 }

236 /// An attribute constraint that references anything other than itself and the
237 /// current op cannot be generically extracted into a function. Most
238 /// prohibitive are operands and results, which require calls to
239 /// `getODSOperands` or `getODSResults`. Attribute references are tricky too
240 /// because ops use cached identifiers.
241 static bool canUniqueAttrConstraint(Attribute attr) {
242  FmtContext ctx;
243  auto test =
244  tgfmt(attr.getConditionTemplate(), &ctx.withSelf("attr").withOp("*op"))
245  .str();
246  return !StringRef(test).contains("<no-subst-found>");
247 }

264 void StaticVerifierFunctionEmitter::collectOpConstraints(
265  ArrayRef<Record *> opDefs) {
266  const auto collectTypeConstraints = [&](Operator::const_value_range values) {
267  for (const NamedTypeConstraint &value : values)
268  if (value.hasPredicate())
269  collectConstraint(typeConstraints, "type", value.constraint);
270  };
271 
272  for (Record *def : opDefs) {
273  Operator op(*def);
274  /// Collect type constraints.
275  collectTypeConstraints(op.getOperands());
276  collectTypeConstraints(op.getResults());
277  /// Collect attribute constraints.
278  for (const NamedAttribute &namedAttr : op.getAttributes()) {
279  if (!namedAttr.attr.getPredicate().isNull() &&
280  !namedAttr.attr.isDerivedAttr() &&
281  canUniqueAttrConstraint(namedAttr.attr))
282  collectConstraint(attrConstraints, "attr", namedAttr.attr);
283  }
284  /// Collect successor constraints.
285  for (const NamedSuccessor &successor : op.getSuccessors()) {
286  if (!successor.constraint.getPredicate().isNull()) {
287  collectConstraint(successorConstraints, "successor",
288  successor.constraint);
289  }
290  }
291  /// Collect region constraints.
292  for (const NamedRegion &region : op.getRegions())
293  if (!region.constraint.getPredicate().isNull())
294  collectConstraint(regionConstraints, "region", region.constraint);
295  }
296 }

298 void StaticVerifierFunctionEmitter::collectPatternConstraints(
299  const llvm::ArrayRef<DagLeaf> constraints) {
300  for (auto &leaf : constraints) {
301  assert(leaf.isOperandMatcher() || leaf.isAttrMatcher());
302  collectConstraint(
303  leaf.isOperandMatcher() ? typeConstraints : attrConstraints,
304  leaf.isOperandMatcher() ? "type" : "attr", leaf.getAsConstraint());
305  }
306 }
