29 /// Return true if all `neededValues` are in scope at the given
30 /// `insertionPoint`.
31 static bool
32 neededValuesDominateInsertionPoint(const DominanceInfo &domInfo,
33  Operation *insertionPoint,
34  const SmallVector<Value> &neededValues) {
35  for (Value val : neededValues) {
36  if (auto bbArg = val.dyn_cast<BlockArgument>()) {
37  Block *owner = bbArg.getOwner();
38  if (!owner->findAncestorOpInBlock(*insertionPoint))
39  return false;
40  } else {
41  auto opResult = val.cast<OpResult>();
42  if (!domInfo.dominates(opResult.getOwner(), insertionPoint))
43  return false;
44  }
45  }
46  return true;
47 }

49 /// Return true if the given `insertionPoint` dominates all uses of
50 /// `emptyTensorOp`.
51 static bool insertionPointDominatesUses(const DominanceInfo &domInfo,
52  Operation *insertionPoint,
53  Operation *emptyTensorOp) {
54  for (Operation *user : emptyTensorOp->getUsers())
55  if (!domInfo.dominates(insertionPoint, user))
56  return false;
57  return true;
58 }

60 /// Find a valid insertion point for a replacement of `emptyTensorOp`, assuming
61 /// that the replacement may use any value from `neededValues`.
62 static Operation *
63 findValidInsertionPoint(Operation *emptyTensorOp,
64  const SmallVector<Value> &neededValues) {
65  DominanceInfo domInfo;
66 
67  // Gather all possible insertion points: the location of `emptyTensorOp` and
68  // right after the definition of each value in `neededValues`.
69  SmallVector<Operation *> insertionPointCandidates;
70  insertionPointCandidates.push_back(emptyTensorOp);
71  for (Value val : neededValues) {
72  // Note: The anchor op is using all of `neededValues`, so:
73  // * in case of a block argument: There must be at least one op in the block
74  // (the anchor op or one of its parents).
75  // * in case of an OpResult: There must be at least one op right after the
76  // defining op (the anchor op or one of its
77  // parents).
78  if (auto bbArg = val.dyn_cast<BlockArgument>()) {
79  insertionPointCandidates.push_back(
80  &bbArg.getOwner()->getOperations().front());
81  } else {
82  insertionPointCandidates.push_back(val.getDefiningOp()->getNextNode());
83  }
84  }
85 
86  // Select first matching insertion point.
87  for (Operation *insertionPoint : insertionPointCandidates) {
88  // Check if all needed values are in scope.
89  if (!neededValuesDominateInsertionPoint(domInfo, insertionPoint,
90  neededValues))
91  continue;
92  // Check if the insertion point is before all uses.
93  if (!insertionPointDominatesUses(domInfo, insertionPoint, emptyTensorOp))
94  continue;
95  return insertionPoint;
96  }
97 
98  // No suitable insertion point was found.
99  return nullptr;
100 }

102 /// Try to eliminate tensor::EmptyOps inside `op`. A tensor::EmptyOp is replaced
103 /// with the result of `rewriteFunc` if it is anchored on a matching
104 /// OpOperand. "Anchored" means that there is a path on the reverse SSA use-def
105 /// chain, starting from the OpOperand and always following the aliasing
106 /// OpOperand, that eventually ends at a single tensor::EmptyOp.
107 LogicalResult mlir::bufferization::eliminateEmptyTensors(
108  RewriterBase &rewriter, Operation *op, AnalysisState &state,
109  AnchorMatchFn anchorMatchFunc, RewriteFn rewriteFunc) {
110  OpBuilder::InsertionGuard g(rewriter);
111 
112  WalkResult status = op->walk([&](Operation *op) {
113  for (OpOperand &operand : op->getOpOperands()) {
114  // Skip operands that do not bufferize inplace.
115  if (!state.isInPlace(operand))
116  continue;
117  // All values that are needed to create the replacement op.
118  SmallVector<Value> neededValues;
119  // Is this a matching OpOperand?
120  if (!anchorMatchFunc(operand, neededValues))
121  continue;
122  SetVector<Value> maybeEmptyTensor = state.findValueInReverseUseDefChain(
123  operand.get(), /*condition=*/[&](Value val) { return false; },
124  /*followEquivalentOnly=*/true);
125 
126  // Replace only if the reverse use-def chain ends at exactly one
127  // tensor::EmptyOp.
128  if (maybeEmptyTensor.size() != 1 ||
129  !maybeEmptyTensor.front().getDefiningOp<tensor::EmptyOp>())
130  return WalkResult::skip();
131  Value emptyTensor = maybeEmptyTensor.front();
132 
133  // Replace only if the types match.
134  // TODO: This could be extended to support IR such as:
135  // %0 = tensor.empty() : tensor<128xf32>
136  // %1 = "some_op"(%0) : (tensor<128xf32>) -> (tensor<128xf32>)
137  // %2 = tensor.expand_shape %1 ...
138  // %3 = tensor.insert_slice %2 into ...
139  if (emptyTensor.getType() != operand.get().getType())
140  return WalkResult::skip();
141 
142  // Find a suitable insertion point.
143  Operation *insertionPoint =
144  findValidInsertionPoint(emptyTensor.getDefiningOp(), neededValues);
145  if (!insertionPoint)
146  continue;
147 
148  // Create a replacement for the tensor::EmptyOp.
149  rewriter.setInsertionPoint(insertionPoint);
150  Value replacement = rewriteFunc(rewriter, emptyTensor.getLoc(), operand);
151  if (!replacement)
152  continue;
153 
154  // Replace the tensor::EmptyOp.
155  rewriter.replaceOp(emptyTensor.getDefiningOp(), replacement);
156  }
157 
158  // Advance to the next operation.
159  return WalkResult::advance();
160  });
161 
162  return failure(status.wasInterrupted());
163 }

165 /// Try to eliminate tensor::EmptyOps inside `op`. An tensor::EmptyOp can be
166 /// eliminated if it is eventually inserted into another tensor (and some other
167 /// conditions are met).
168 ///
169 /// E.g.:
170 /// %0 = tensor.empty()
171 /// %1 = linalg.fill(%cst, %0) {inplace = [true]}
172 /// %2 = tensor.insert_slice %1 into %t[10][20][1]
173 ///
174 /// tensor::EmptyOp elimination will try to fill %t inplace instead of filling a
175 /// new allocation %0 and inserting it into %t. This is done by replacing the
176 /// tensor::EmptyOp with:
177 ///
178 /// %0 = tensor.extract_slice %t[10][20][1]
179 ///
180 /// The analysis looks for matching ExtractSliceOp/InsertSliceOp pairs and lets
181 /// those bufferize inplace in the absence of other conflicts.
182 ///
183 /// Starting from an InsertSliceOp, an tensor::EmptyOp at the end of the insert
184 /// source's reverse use-def chain is eliminated if:
185 /// * On the reverse use-def chain path from the InsertSliceOp to the
186 /// tensor::EmptyOp, all ops were decided to bufferize inplace and the buffer
187 /// relation is "equivalent" (TODO: can be relaxed if needed).
188 /// * The reverse use-def chain has exactly one end, which is the
189 /// tensor::EmptyOp.
190 LogicalResult
191 mlir::bufferization::insertSliceAnchoredEmptyTensorEliminationStep(
192  RewriterBase &rewriter, Operation *op, AnalysisState &state) {
193  return eliminateEmptyTensors(
194  rewriter, op, state,
195  /*anchorMatchFunc=*/
196  [&](OpOperand &operand, SmallVector<Value> &neededValues) {
197  auto insertSliceOp =
198  dyn_cast<tensor::InsertSliceOp>(operand.getOwner());
199  if (!insertSliceOp)
200  return false;
201  if (&operand != &insertSliceOp->getOpOperand(0) /*source*/)
202  return false;
203 
204  // Collect all values that are needed to construct the replacement op.
205  neededValues.append(insertSliceOp.getOffsets().begin(),
206  insertSliceOp.getOffsets().end());
207  neededValues.append(insertSliceOp.getSizes().begin(),
208  insertSliceOp.getSizes().end());
209  neededValues.append(insertSliceOp.getStrides().begin(),
210  insertSliceOp.getStrides().end());
211  neededValues.push_back(insertSliceOp.getDest());
212 
213  return true;
214  },
215  /*rewriteFunc=*/
216  [](OpBuilder &b, Location loc, OpOperand &operand) {
217  auto insertOp = cast<tensor::InsertSliceOp>(operand.getOwner());
218  auto extractOp = b.create<tensor::ExtractSliceOp>(
219  loc, insertOp.getSourceType(), insertOp.getDest(),
220  insertOp.getMixedOffsets(), insertOp.getMixedSizes(),
221  insertOp.getMixedStrides());
222  return extractOp.getResult();
223  });
224 }

241 void EmptyTensorElimination::runOnOperation() {
242  Operation *op = getOperation();
243  OneShotBufferizationOptions options;
244  OneShotAnalysisState state(op, options);
245  if (failed(analyzeOp(op, state))) {
246  signalPassFailure();
247  return;
248  }
249 
250  IRRewriter rewriter(op->getContext());

252  rewriter, op, state)))
253  signalPassFailure();
