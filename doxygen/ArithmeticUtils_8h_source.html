35 //
36 // Safe comparison functions.
37 //
38 // Variants of the `==`, `!=`, `<`, `<=`, `>`, and `>=` operators which
39 // are careful to ensure that negatives are always considered strictly
40 // less than non-negatives regardless of the signedness of the types of
41 // the two arguments. They are "safe" in that they guarantee to *always*
42 // give an output and that that output is correct; in particular this means
43 // they never use assertions or other mechanisms for "returning an error".
44 //
45 // These functions are C++17-compatible backports of the safe comparison
46 // functions added in C++20, and the implementations are based on the
47 // sample implementations provided by the standard:
48 // <https://en.cppreference.com/w/cpp/utility/intcmp>.
49 //
50 //===----------------------------------------------------------------------===//
51 
52 template <typename T, typename U>
53 constexpr bool safelyEQ(T t, U u) noexcept {
54  using UT = std::make_unsigned_t<T>;
55  using UU = std::make_unsigned_t<U>;
56  if constexpr (std::is_signed_v<T> == std::is_signed_v<U>)
57  return t == u;
58  else if constexpr (std::is_signed_v<T>)
59  return t < 0 ? false : static_cast<UT>(t) == u;
60  else
61  return u < 0 ? false : t == static_cast<UU>(u);
62 }
63 
64 template <typename T, typename U>
65 constexpr bool safelyNE(T t, U u) noexcept {
66  return !safelyEQ(t, u);
67 }
68 
69 template <typename T, typename U>
70 constexpr bool safelyLT(T t, U u) noexcept {
71  using UT = std::make_unsigned_t<T>;
72  using UU = std::make_unsigned_t<U>;
73  if constexpr (std::is_signed_v<T> == std::is_signed_v<U>)
74  return t < u;
75  else if constexpr (std::is_signed_v<T>)
76  return t < 0 ? true : static_cast<UT>(t) < u;
77  else
78  return u < 0 ? false : t < static_cast<UU>(u);
79 }

81 template <typename T, typename U>
82 constexpr bool safelyGT(T t, U u) noexcept {
83  return safelyLT(u, t);
84 }

86 template <typename T, typename U>

92 constexpr bool safelyGE(T t, U u) noexcept {

97 //
98 // Overflow checking functions.
99 //
100 // These functions use assertions to ensure correctness with respect to
101 // overflow/underflow. Unlike the "safe" functions above, these "checked"
102 // functions only guarantee that *if* they return an answer then that answer
103 // is correct. When assertions are enabled, they do their best to remain
104 // as fast as possible (since MLIR keeps assertions enabled by default,
105 // even for optimized builds). When assertions are disabled, they use the
106 // standard unchecked implementations.
107 //
108 //===----------------------------------------------------------------------===//

116 /// A version of `static_cast<To>` which checks for overflow/underflow.
117 /// The implementation avoids performing runtime assertions whenever
118 /// the types alone are sufficient to statically prove that overflow
119 /// cannot happen.
120 template <typename To, typename From>
121 [[nodiscard]] inline To checkOverflowCast(From x) {
122  // Check the lower bound. (For when casting from signed types.)
123  constexpr To minTo = std::numeric_limits<To>::min();
124  constexpr From minFrom = std::numeric_limits<From>::min();
125  if constexpr (!safelyGE(minFrom, minTo))
126  assert(safelyGE(x, minTo) && "cast would underflow");
127  // Check the upper bound.
128  constexpr To maxTo = std::numeric_limits<To>::max();
129  constexpr From maxFrom = std::numeric_limits<From>::max();
130  if constexpr (!safelyLE(maxFrom, maxTo))
131  assert(safelyLE(x, maxTo) && "cast would overflow");
132  // Now do the cast itself.
133  return static_cast<To>(x);
134 }

136 /// A version of `operator*` on `uint64_t` which guards against overflows
137 /// (when assertions are enabled).
138 inline uint64_t checkedMul(uint64_t lhs, uint64_t rhs) {
139  // If assertions are enabled and we have the intrinsic, then use it to
140  // avoid the expensive division. If assertions are disabled, then don't
141  // bother with intrinsics (to avoid any possible slowdown vs `operator*`).
142 #if !defined(NDEBUG) && MLIR_SPARSETENSOR_HAS_BUILTIN(__builtin_mul_overflow)
143  uint64_t result;
144  bool overflowed = __builtin_mul_overflow(lhs, rhs, &result);
145  assert(!overflowed && "Integer overflow");
146  return result;
147 #else
148  assert((lhs == 0 || rhs <= std::numeric_limits<uint64_t>::max() / lhs) &&
149  "Integer overflow");
150  return lhs * rhs;
151 #endif
152 }
