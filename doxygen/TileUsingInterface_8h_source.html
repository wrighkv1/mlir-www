31 /// Options to use to control tiling.
32 struct SCFTilingOptions {
33  /// Computation function that returns the tile sizes for each operation.
34  /// Delayed construction of constant tile sizes should occur to interoperate
35  /// with folding.
36  SCFTileSizeComputationFunction tileSizeComputationFunction = nullptr;
37 
38  SCFTilingOptions &
39  setTileSizeComputationFunction(SCFTileSizeComputationFunction fun) {
40  tileSizeComputationFunction = std::move(fun);
41  return *this;
42  }
43  /// Set the `tileSizeComputationFunction` to return the values `ts`. The
44  /// values must not fold away when tiling. Otherwise, use a more robust
45  /// `tileSizeComputationFunction`.
46  SCFTilingOptions &setTileSizes(const SmallVector<Value, 4> &ts) {
47  tileSizeComputationFunction = [=](OpBuilder &, Operation *) { return ts; };
48  return *this;
49  }
50  /// Convenience function to set the `tileSizeComputationFunction` to a
51  /// function that computes tile sizes at the point they are needed. Allows
52  /// proper interaction with folding.
53  SCFTilingOptions &setTileSizes(ArrayRef<int64_t> ts);

55  /// The interchange vector to reorder the tiled loops.

57  SCFTilingOptions &setInterchange(ArrayRef<int64_t> interchange) {
58  interchangeVector = llvm::to_vector(interchange);

106 /// Method to tile and fuse a sequence of operations, by tiling the consumer
107 /// and fusing its producers. Note that this assumes that it is valid to
108 /// tile+fuse the producer into the innermost tiled loop. Its up to the caller
109 /// to ensure that the tile sizes provided make this fusion valid.
110 ///
111 /// For example, for the following sequence
112 ///
113 /// ```mlir
114 /// %0 =
115 /// %1 = linalg.fill ... outs(%0 : ... )
116 /// %2 = linalg.matmul ... outs(%1 : ...) ...
117 /// ```
118 ///
119 /// it is legal to fuse the fill with the matmul only if the matmul is tiled
120 /// along the parallel dimensions and not the reduction dimension, i.e. the tile
121 /// size for the reduction dimension should be 0. The resulting fused
122 /// transformation is
123 ///
124 /// ```mlir
125 /// %1 = scf.for ... iter_args(%arg0 = %0)
126 /// %2 = tensor.extract_slice %arg0
127 /// %3 = linalg.fill .. outs(%2 : ... )
128 /// %4 = linalg.matmul .. outs(%3 : ...)
129 /// }
130 /// ```
131 FailureOr<SCFTileAndFuseResult>
132 tileConsumerAndFuseProducerGreedilyUsingSCFForOp(
133  RewriterBase &rewriter, TilingInterface consumer,
134  const SCFTileAndFuseOptions &options);

153 /// Method to tile a reduction and generate a parallel op within a serial loop.
154 /// Each of the partial reductions are calculated in parallel. Then after the
155 /// loop all the partial reduction are merged into a final reduction.
156 /// For example for the following sequence
157 ///
158 /// ```mlir
159 /// %0 = linalg.generic %in ["parallel", "reduction"]
160 /// : tensor<7x9xf32> -> tensor<7xf32>
161 /// ```
162 ///
163 /// into:
164 ///
165 /// ```mlir
166 /// %0 = linalg.fill ... : tensor<7x4xf32>
167 /// %1 = scf.for ... iter_args(%arg0 = %0)
168 /// %2 = tensor.extract_slice %arg0 : tensor<7x4xf32> -> tensor<7x?xf32>
169 /// %3 = tensor.extract_slice %in : tensor<7x9xf32> -> tensor<7x?xf32>
170 /// %4 = linalg.generic %2, %3 ["parallel", "parallel"]
171 /// : tensor<7x?xf32> -> tensor<7x?xf32>
172 /// %5 = tensor.insert_slice %3, %0[0, 0] : tensor<7x4xf32>
173 /// }
174 /// %6 = linalg.generic %1 ["parallel", "reduction"]
175 /// : tensor<7x4xf32> -> tensor<7xf32>
176 /// ```
177 FailureOr<scf::SCFReductionTilingResult>
178 tileReductionUsingScf(PatternRewriter &b, PartialReductionOpInterface op,
179  ArrayRef<OpFoldResult> tileSize);
