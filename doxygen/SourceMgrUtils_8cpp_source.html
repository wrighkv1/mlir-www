19 /// Find the end of a string whose contents start at the given `curPtr`. Returns
20 /// the position at the end of the string, after a terminal or invalid character
21 /// (e.g. `"` or `\0`).
22 static const char *lexLocStringTok(const char *curPtr) {
23  while (char c = *curPtr++) {
24  // Check for various terminal characters.
25  if (StringRef("\"\n\v\f").contains(c))
26  return curPtr;
27 
28  // Check for escape sequences.
29  if (c == '\\') {
30  // Check a few known escapes and \xx hex digits.
31  if (*curPtr == '"' || *curPtr == '\\' || *curPtr == 'n' || *curPtr == 't')
32  ++curPtr;
33  else if (llvm::isHexDigit(*curPtr) && llvm::isHexDigit(curPtr[1]))
34  curPtr += 2;
35  else
36  return curPtr;
37  }
38  }
39 
40  // If we hit this point, we've reached the end of the buffer. Update the end
41  // pointer to not point past the buffer.
42  return curPtr - 1;
43 }

45 SMRange lsp::convertTokenLocToRange(SMLoc loc) {
46  if (!loc.isValid())
47  return SMRange();
48  const char *curPtr = loc.getPointer();
49 
50  // Check if this is a string token.
51  if (*curPtr == '"') {
52  curPtr = lexLocStringTok(curPtr + 1);
53 
54  // Otherwise, default to handling an identifier.
55  } else {
56  // Return if the given character is a valid identifier character.
57  auto isIdentifierChar = [](char c) {
58  return isalnum(c) || c == '$' || c == '.' || c == '_' || c == '-';
59  };
60 
61  while (*curPtr && isIdentifierChar(*(++curPtr)))
62  continue;
63  }
64 
65  return SMRange(loc, SMLoc::getFromPointer(curPtr));
66 }

68 Optional<std::string> lsp::extractSourceDocComment(llvm::SourceMgr &sourceMgr,
69  SMLoc loc) {
70  // This is a heuristic, and isn't intended to cover every case, but should
71  // cover the most common. We essentially look for a comment preceding the
72  // line, and if we find one, use that as the documentation.
73  if (!loc.isValid())
74  return llvm::None;
75  int bufferId = sourceMgr.FindBufferContainingLoc(loc);
76  if (bufferId == 0)
77  return llvm::None;
78  const char *bufferStart =
79  sourceMgr.getMemoryBuffer(bufferId)->getBufferStart();
80  StringRef buffer(bufferStart, loc.getPointer() - bufferStart);
81 
82  // Pop the last line from the buffer string.
83  auto popLastLine = [&]() -> Optional<StringRef> {
84  size_t newlineOffset = buffer.find_last_of("\n");
85  if (newlineOffset == StringRef::npos)
86  return llvm::None;
87  StringRef lastLine = buffer.drop_front(newlineOffset).trim();
88  buffer = buffer.take_front(newlineOffset);
89  return lastLine;
90  };
91 
92  // Try to pop the current line.
93  if (!popLastLine())
94  return llvm::None;
95 
96  // Try to parse a comment string from the source file.
97  SmallVector<StringRef> commentLines;
98  while (Optional<StringRef> line = popLastLine()) {
99  // Check for a comment at the beginning of the line.
100  if (!line->startswith("//"))
101  break;
102 
103  // Extract the document string from the comment.
104  commentLines.push_back(line->drop_while([](char c) { return c == '/'; }));
105  }
106 
107  if (commentLines.empty())
108  return llvm::None;
109  return llvm::join(llvm::reverse(commentLines), "\n");
110 }

131 void lsp::gatherIncludeFiles(llvm::SourceMgr &sourceMgr,
132  SmallVectorImpl<SourceMgrInclude> &includes) {
133  for (unsigned i = 1, e = sourceMgr.getNumBuffers(); i < e; ++i) {
134  // Check to see if this file was included by the main file.
135  SMLoc includeLoc = sourceMgr.getBufferInfo(i + 1).IncludeLoc;
136  if (!includeLoc.isValid() || sourceMgr.FindBufferContainingLoc(
137  includeLoc) != sourceMgr.getMainFileID())
138  continue;
139 
140  // Try to build a URI for this file path.
141  auto *buffer = sourceMgr.getMemoryBuffer(i + 1);
142  llvm::SmallString<256> path(buffer->getBufferIdentifier());
143  llvm::sys::path::remove_dots(path, /*remove_dot_dot=*/true);
144 
145  llvm::Expected<URIForFile> includedFileURI = URIForFile::fromFile(path);
146  if (!includedFileURI)
147  continue;
148 
149  // Find the end of the include token.
150  const char *includeStart = includeLoc.getPointer() - 2;
151  while (*(--includeStart) != '\"')
152  continue;
153 
154  // Push this include.
155  SMRange includeRange(SMLoc::getFromPointer(includeStart), includeLoc);
156  includes.emplace_back(*includedFileURI, Range(sourceMgr, includeRange));
157  }
158 }
