28 class OpFilter {
29 public:
30  /// An op filter entry. Filters can be used to specify which ops should be
31  /// processed by the bufferization.
32  struct Entry {
33  /// If the filter function evaluates to `true`, the filter matches.
34  using FilterFn = std::function<bool(Operation *)>;
35 
36  /// Filter type: A filter can either be a DENY filter or an ALLOW filter.
37  enum FilterType : int8_t { DENY = 0, ALLOW = 1 };
38 
39  FilterFn fn;
40  FilterType type;
41  };
42 
43  /// Return whether the op is allowed or not.
44  ///
45  /// If the filter does not have an ALLOW rule, ops are allowed by default,
46  /// unless they are explicitly marked as DENY. If the filter has at least one
47  /// ALLOW rule, ops are denied by default and only allowed if they match
48  /// an ALLOW rule and no DENY rule.
49  bool isOpAllowed(Operation *op) const;
50 
51  /// Allow the given dialects.
52  ///
53  /// This function adds one or multiple ALLOW entries.
54  template <typename... DialectTs>
55  void allowDialect() {
56  // The following expands a call to allowDialectImpl for each dialect
57  // in 'DialectTs'.
58  (allowDialectImpl<DialectTs>(), ...);
59  }
60 
61  /// Deny the given dialects.
62  ///
63  /// This function adds one or multiple DENY entries.
64  template <typename... DialectTs>
65  void denyDialect() {
66  (denyDialectImpl<DialectTs>(), ...);
67  }
68 
69  /// Allow the given dialect.
70  ///
71  /// This function adds an ALLOW entry.
72  void allowDialect(StringRef dialectNamespace) {
73  Entry::FilterFn filterFn = [=](Operation *op) {
74  return op->getDialect()->getNamespace() == dialectNamespace;
75  };
76  entries.push_back(Entry{filterFn, Entry::FilterType::ALLOW});
77  }
78 
79  /// Allow the given ops.
80  ///
81  /// This function adds one or multiple ALLOW entries.
82  template <typename... OpTys>
83  void allowOperation() {
84  (allowOperationImpl<OpTys>(), ...);
85  }
86 
87  /// Deny the given ops.
88  ///
89  /// This function adds one or multiple DENY entries.
90  template <typename... OpTys>
91  void denyOperation() {
92  (denyOperationImpl<OpTys>(), ...);
93  }
94 
95  /// Allow the given op.
96  ///
97  /// This function adds an ALLOW entry.
98  void allowOperation(StringRef opName) {
99  Entry::FilterFn filterFn = [=](Operation *op) {
100  return op->getName().getStringRef() == opName;
101  };
102  allowOperation(filterFn);
103  }
104 
105  /// Deny the given op.
106  ///
107  /// This function adds a DENY entry.
108  void denyOperation(StringRef opName) {
109  Entry::FilterFn filterFn = [=](Operation *op) {
110  return op->getName().getStringRef() == opName;
111  };
112  denyOperation(filterFn);
113  }
114 
115  /// Allow ops that are matched by `fn`.
116  ///
117  /// This function adds an ALLOW entry.
118  void allowOperation(Entry::FilterFn fn) {
119  entries.push_back(Entry{fn, Entry::FilterType::ALLOW});
120  }
121 
122  /// Deny ops that are matched by `fn`.
123  ///
124  /// This function adds a DENY entry.
125  void denyOperation(Entry::FilterFn fn) {
126  entries.push_back(Entry{fn, Entry::FilterType::DENY});
127  }
128 
129 private:
130  /// Return `true` if the filter has at least one ALLOW rule.
131  bool hasAllowRule() const {
132  for (const Entry &e : entries)
133  if (e.type == Entry::FilterType::ALLOW)
134  return true;
135  return false;
136  }
137 
138  /// Allow a dialect.
139  template <typename DialectT>
140  void allowDialectImpl() {
141  allowDialect(DialectT::getDialectNamespace());
142  }
143 
144  /// Deny a dialect.
145  template <typename DialectT>
146  void denyDialectImpl() {
147  denyDialect(DialectT::getDialectNamespace());
148  }
149 
150  /// Allow an op.
151  template <typename OpTy>
152  void allowOperationImpl() {
153  allowOperation(OpTy::getOperationName());
154  }
155 
156  /// Deny an op.
157  template <typename OpTy>
158  void denyOperationImpl() {
159  denyOperation(OpTy::getOperationName());
160  }
161 
162  /// A list of filter entries that determine whether an op should be allowed or
163  /// denied. If the filter has an ALLOW rule, only ops that are allowed and not
164  /// denied are allowed. If the filter does not have an ALLOW rule, only ops
165  /// that are not denied are allowed.
166  SmallVector<Entry> entries;
167 };

239  Optional<unsigned> defaultMemorySpace = 0;

241  /// Certain ops have aliasing OpOperand/OpResult invariants (e.g., scf.for).
242  /// If this flag is set to `false`, those invariants are no longer enforced
243  /// with buffer copies.
244  ///
245  /// Note: Deactivating this flag can lead to incorrect bufferization results
246  /// when used incorrectly. This flag is useful with
247  /// `AlwaysCopyAnalysisState` which bufferizes all writing tensor
248  /// OpOperands out-of-place.
249  bool enforceAliasingInvariants = true;
250 
251  /// This flag controls buffer types on function signatures.
252  ///
253  /// * InferLayoutMap: All function parameter types have a fully dynamic layout
254  /// map, but function result types are inferred from the body of the
255  /// function.
256  /// * FullyDynamicLayoutMap: All function parameter types and result types
257  /// have a fully dynamic layout map. This option is most efficient because
258  /// any layout map can be casted to a fully dynamic one.
259  /// * IdentityLayoutMap: All function parameter types and result types have a
260  /// static identity layout (i.e., no layout map). This option may introduce
261  /// additional buffer allocs and copies because layout maps cannot be casted
262  /// away.
263  ///
264  /// If `bufferizeFunctionBoundaries` is not set, this flag has no effect.
265  ///
266  /// Note: Inferred layout maps may not be desireable when interacting with
267  /// external functions, because the generated function signatures will be less
268  /// predictable.
269  LayoutMapOption functionBoundaryTypeConversion =
270  LayoutMapOption::InferLayoutMap;
271 
272  /// Type converter from tensors to memrefs. This type converter is used if no
273  /// memref type could be inferred during bufferization. By default, a type
274  /// converter that returns a memref type with a fully dynamic layout map is
275  /// used.
276  UnknownTypeConverterFn unknownTypeConverterFn = nullptr;
277 
278  /// Specifies whether dealloc ops should be generated along with alloc ops. If

280  bool createDeallocs = true;

294  /// If set to `true`, the IR is annotated with details about RaW conflicts.

296  bool printConflicts = false;

346  /// the op is not bufferizable.
347  bool bufferizesToMemoryRead(OpOperand &opOperand) const;
348 
349  /// Return true if `opOperand` bufferizes to a memory write. Return true` if
350  /// the op is not bufferizable.
351  bool bufferizesToMemoryWrite(OpOperand &opOperand) const;
352 
353  /// Return true if `opOperand` does neither read nor write but bufferizes to
354  /// an alias. Return false if the op is not bufferizable.
355  bool bufferizesToAliasOnly(OpOperand &opOperand) const;
356 
357  /// Return true if a copy can always be avoided when allocating a new tensor
358  /// for the given OpOperand.
359  bool canOmitTensorCopy(OpOperand &opOperand) const;
360 
361  /// Return true if the given value is read by an op that bufferizes to a
362  /// memory read. Also takes into account ops that create an alias but do not
363  /// read by themselves (e.g., ExtractSliceOp).
364  bool isValueRead(Value value) const;
365 
366  /// Starting from `value`, follow the use-def chain in reverse, always
367  /// selecting the aliasing OpOperands. Find and return Values for which
368  /// `condition` evaluates to true. OpOperands of such matching Values are not
369  /// traversed any further.
370  ///
371  /// When reaching the end of a chain (BlockArgument or Value without aliasing
372  /// OpOperands), also return the last Value of that chain.
373  ///
374  /// Example:
375  ///
376  /// 8
377  /// |
378  /// 6* 7* +-----+----+
379  /// | | | |
380  /// 2* 3 4* 5
381  /// | | | |
382  /// +----------+----------+----------+
383  /// |
384  /// 1
385  ///
386  /// In the above example, Values with a star satisfy the condition. When
387  /// starting the traversal from Value 1, the resulting SetVector is:
388  /// { 2, 7, 8, 5 }
389  ///
390  /// If `followEquivalentOnly` is set, only equivalent OpOperands are selected.
391  SetVector<Value>
392  findValueInReverseUseDefChain(Value value,
393  llvm::function_ref<bool(Value)> condition,
394  bool followEquivalentOnly = false) const;
395 
396  /// Find the Values of the last preceding write of a given Value.
397  ///
398  /// Note: Unknown ops are handled conservatively and assumed to be writes.
399  /// Furthermore, BlockArguments are also assumed to be writes. There is no
400  /// analysis across block boundaries.
401  ///
402  /// Note: When reaching an end of the reverse SSA use-def chain, that value
403  /// is returned regardless of whether it is a memory write or not.
404  SetVector<Value> findLastPrecedingWrite(Value value) const;
405 
406  /// Return `true` if the given OpResult has been decided to bufferize inplace.
407  virtual bool isInPlace(OpOperand &opOperand) const;
408 
409  /// Return true if `v1` and `v2` bufferize to equivalent buffers.
410  virtual bool areEquivalentBufferizedValues(Value v1, Value v2) const;
411 
412  /// Return true if `v1` and `v2` may bufferize to aliasing buffers.
413  virtual bool areAliasingBufferizedValues(Value v1, Value v2) const;
414 
415  /// Return `true` if the given tensor has undefined contents.
416  virtual bool hasUndefinedContents(OpOperand *opOperand) const;
417 
418  /// Return true if the given tensor (or an aliasing tensor) is yielded from
419  /// the containing block. Also include all aliasing tensors in the same block.
420  ///
421  /// Note: In the absence of an analysis, an implementation may return true for
422  /// any given tensor.
423  virtual bool isTensorYielded(Value tensor) const;
424 
425  /// Return `true` if the given dialect state exists.
426  bool hasDialectState(StringRef name) const {
427  auto it = dialectState.find(name);
428  return it != dialectState.end();
429  }
430 
431  /// Return dialect-specific bufferization state.
432  template <typename StateT>
433  Optional<const StateT *> getDialectState(StringRef name) const {
434  auto it = dialectState.find(name);
435  if (it == dialectState.end())
436  return None;
437  return static_cast<const StateT *>(it->getSecond().get());
438  }
439 
440  /// Return dialect-specific analysis state or create one if none exists.
441  template <typename StateT>
442  StateT &getOrCreateDialectState(StringRef name) {
443  // Create state if it does not exist yet.
444  if (!hasDialectState(name))
445  dialectState[name] = std::make_unique<StateT>();
446  return static_cast<StateT &>(*dialectState[name]);
447  }

452  dialectState[name] = std::move(state);
