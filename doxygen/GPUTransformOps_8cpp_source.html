37 /// Check if given mapping attributes are one of the desired attributes
38 static DiagnosedSilenceableFailure checkAttributeType(
39  const ArrayRef<DeviceMappingAttrInterface> &threadMappingAttributes,
40  const Optional<ArrayAttr> &foreachMapping,
41  llvm::Optional<TransformOpInterface> transformOp) {
42  if (!foreachMapping.has_value())
43  return transformOp->emitSilenceableError() << "mapping must be present";
44 
45  if (llvm::any_of(foreachMapping->getValue(),
46  [&](DeviceMappingAttrInterface map) {
47  return llvm::find(threadMappingAttributes, map) ==
48  threadMappingAttributes.end();
49  }))
50  return transformOp->emitDefiniteFailure()
51  << "mapping must be one of " << threadMappingAttributes;
52  return DiagnosedSilenceableFailure::success();
53 }

55 /// Determines if the size of the kernel configuration is supported by the GPU
56 /// architecture being used. It presently makes use of CUDA limitations, however
57 /// that aspect may be enhanced for other GPUs.
58 static DiagnosedSilenceableFailure
59 checkGpuLimits(TransformOpInterface transformOp, Optional<int64_t> gridDimX,
60  Optional<int64_t> gridDimY, Optional<int64_t> gridDimZ,
61  Optional<int64_t> blockDimX, Optional<int64_t> blockDimY,
62  Optional<int64_t> blockDimZ) {
63 
64  static constexpr int maxTotalBlockdim = 1024;
65  static constexpr int maxBlockdimx = 1024;
66  static constexpr int maxBlockdimy = 1024;
67  static constexpr int maxBlockdimz = 64;
68  static constexpr int maxTotalGriddim = 2147483647;
69  static constexpr int maxGriddimx = 2147483647;
70  static constexpr int maxGriddimy = 65535;
71  static constexpr int maxGriddimz = 65535;
72 
73  if ((blockDimX.value_or(1) * blockDimY.value_or(1) * blockDimZ.value_or(1)) >
74  maxTotalBlockdim ||
75  (gridDimX.value_or(1) * gridDimY.value_or(1) * gridDimZ.value_or(1)) >
76  maxTotalGriddim ||
77  blockDimX.value_or(1) > maxBlockdimx ||
78  blockDimY.value_or(1) > maxBlockdimy ||
79  blockDimZ.value_or(1) > maxBlockdimz ||
80  gridDimY.value_or(1) > maxGriddimy ||
81  gridDimZ.value_or(1) > maxGriddimz ||
82  gridDimX.value_or(1) > maxGriddimx) {
83  return transformOp.emitSilenceableError()
84  << "Trying to launch a GPU kernel with gridDim = ("
85  << gridDimX.value_or(1) << ", " << gridDimY.value_or(1) << ", "
86  << gridDimZ.value_or(1) << ") blockDim = (" << blockDimX.value_or(1)
87  << ", " << blockDimY.value_or(1) << ", " << blockDimZ.value_or(1)
88  << "). It is larger than the limits.";
89  }
90  return DiagnosedSilenceableFailure::success();
91 }

93 /// Creates an empty-body gpu::LaunchOp using the provided kernel settings and
94 /// put a terminator within.
95 static DiagnosedSilenceableFailure
96 createGpuLaunch(RewriterBase &rewriter, Location loc,
97  TransformOpInterface transformOp, LaunchOp &launchOp,
98  Optional<int64_t> gridDimX = llvm::None,
99  Optional<int64_t> gridDimY = llvm::None,
100  Optional<int64_t> gridDimZ = llvm::None,
101  Optional<int64_t> blockDimX = llvm::None,
102  Optional<int64_t> blockDimY = llvm::None,
103  Optional<int64_t> blockDimZ = llvm::None) {
104  DiagnosedSilenceableFailure diag =
105  checkGpuLimits(transformOp, gridDimX, gridDimY, gridDimZ, blockDimX,
106  blockDimY, blockDimZ);
107  if (!diag.succeeded())
108  return diag;
109 
110  auto createConst = [&](int dim) {
111  return rewriter.create<arith::ConstantIndexOp>(loc, dim);
112  };
113  OpBuilder::InsertionGuard guard(rewriter);
114  Value one = createConst(1);
115  Value gridSizeX = gridDimX.has_value() ? createConst(gridDimX.value()) : one;
116  Value gridSizeY = gridDimY.has_value() ? createConst(gridDimY.value()) : one;
117  Value gridSizeZ = gridDimZ.has_value() ? createConst(gridDimZ.value()) : one;
118  Value blkSizeX = blockDimX.has_value() ? createConst(blockDimX.value()) : one;
119  Value blkSizeY = blockDimY.has_value() ? createConst(blockDimY.value()) : one;
120  Value blkSizeZ = blockDimZ.has_value() ? createConst(blockDimZ.value()) : one;
121  launchOp = rewriter.create<LaunchOp>(loc, gridSizeX, gridSizeY, gridSizeZ,
122  blkSizeX, blkSizeY, blkSizeZ);
123  rewriter.setInsertionPointToEnd(&launchOp.getBody().front());
124  rewriter.create<TerminatorOp>(loc);
125  return DiagnosedSilenceableFailure(success());
126 }

128 /// Alter kernel configuration of the given kernel.
129 static DiagnosedSilenceableFailure
130 alterGpuLaunch(SimpleRewriter &rewriter, LaunchOp gpuLaunch,
131  TransformOpInterface transformOp,
132  Optional<int64_t> gridDimX = llvm::None,
133  Optional<int64_t> gridDimY = llvm::None,
134  Optional<int64_t> gridDimZ = llvm::None,
135  Optional<int64_t> blockDimX = llvm::None,
136  Optional<int64_t> blockDimY = llvm::None,
137  Optional<int64_t> blockDimZ = llvm::None) {
138  DiagnosedSilenceableFailure diag =
139  checkGpuLimits(transformOp, gridDimX, gridDimY, gridDimZ, blockDimX,
140  blockDimY, blockDimZ);
141  if (!diag.succeeded())
142  return diag;
143 
144  KernelDim3 currentBlockdim = gpuLaunch.getBlockSizeOperandValues();
145  OpBuilder::InsertionGuard guard(rewriter);
146  rewriter.setInsertionPointAfterValue(currentBlockdim.x);
147  auto createConstValue = [&](int dim) {
148  return rewriter.create<arith::ConstantIndexOp>(currentBlockdim.x.getLoc(),
149  dim);
150  };
151 
152  if (gridDimX.has_value())
153  gpuLaunch.getGridSizeXMutable().assign(createConstValue(gridDimX.value()));
154  if (gridDimY.has_value())
155  gpuLaunch.getGridSizeYMutable().assign(createConstValue(gridDimY.value()));
156  if (gridDimZ.has_value())
157  gpuLaunch.getGridSizeZMutable().assign(createConstValue(gridDimZ.value()));
158  if (blockDimX.has_value())
159  gpuLaunch.getBlockSizeXMutable().assign(
160  createConstValue(blockDimX.value()));
161  if (blockDimY.has_value())
162  gpuLaunch.getBlockSizeYMutable().assign(
163  createConstValue(blockDimY.value()));
164  if (blockDimZ.has_value())
165  gpuLaunch.getBlockSizeZMutable().assign(
166  createConstValue(blockDimZ.value()));
167  return DiagnosedSilenceableFailure::success();
168 }

174 DiagnosedSilenceableFailure mlir::transform::gpu::mapForeachToBlocksImpl(
175  RewriterBase &rewriter, scf::ForeachThreadOp foreachThreadOp,
176  function_ref<void(RewriterBase &, scf::ForeachThreadOp,
177  SmallVectorImpl<Value> &)>
178  blockIdGenerator,
179  SmallVectorImpl<int64_t> &gridDims, TransformOpInterface transformOp,
180  const ArrayRef<DeviceMappingAttrInterface> &mappingAttributes) {
181  // Step 0. Target-specific verifications. There is no good place to anchor
182  // those right now: the ForeachThreadOp is target-independent and the
183  // transform op does not apply to individual ForeachThreadOp.
184  Location loc = foreachThreadOp->getLoc();
185 
186  if (foreachThreadOp.getNumResults() > 0)
187  return transformOp.emitSilenceableError()
188  << "only bufferized scf.foreach_thread lowers to "
189  "gpu.block_id";
190  if (foreachThreadOp.getNumThreads().size() > 3)
191  return transformOp.emitSilenceableError()
192  << "scf.foreach_thread with rank > 3 does not lower to "
193  "gpu.block_id";
194  if (llvm::any_of(foreachThreadOp.getNumThreads(), [](Value v) {
195  return !v.getDefiningOp<arith::ConstantIndexOp>();
196  })) {
197  return transformOp.emitSilenceableError()
198  << "unsupported dynamic griddim size";
199  }
200  SmallVector<Attribute> blockMapping =
201  llvm::to_vector(foreachThreadOp.getMapping()->getValue());
202 
203  // Step 1. Complete the blockMapping to a full mapping (with 1s) if necessary.
204  SmallVector<Value> numBlocks =
205  llvm::to_vector(foreachThreadOp.getNumThreads());
206  // Ensure we have 3 block sizes, one for each id.
207  Value one;
208  for (auto attr : mappingAttributes) {
209  if (std::find(blockMapping.begin(), blockMapping.end(), attr) ==
210  blockMapping.end()) {
211  blockMapping.push_back(attr);
212  one = one ? one : rewriter.create<arith::ConstantIndexOp>(loc, 1);
213  numBlocks.push_back(one);
214  }
215  }
216 
217  // Step 2. sort the values by the corresponding DeviceMappingAttrInterface.
218  auto comparator = [&](DeviceMappingAttrInterface a,
219  DeviceMappingAttrInterface b) -> bool {
220  return a.getMappingId() < b.getMappingId();
221  };
222  SmallVector<Value> gridDimValues = scf::ForeachThreadOp::getValuesSortedByKey(
223  blockMapping, numBlocks, comparator);
224  for (Value v : gridDimValues)
225  gridDims.push_back(v.getDefiningOp<arith::ConstantIndexOp>().value());
226 
227  // Step 3. Generate the blockIds using the provided generator and map the
228  // induction variables to the newly created ops.
229  SmallVector<Value> blockOps;
230  blockIdGenerator(rewriter, foreachThreadOp, blockOps);
231  BlockAndValueMapping bvm;
232  for (auto [blockIdx, blockDim] :
233  llvm::zip(foreachThreadOp.getThreadIndices(), blockMapping)) {
234  bvm.map(blockIdx,
235  blockOps[static_cast<int64_t>(
236  blockDim.cast<DeviceMappingAttrInterface>().getMappingId())]);
237  }
238 
239  // Step 4. Move the body of foreachThreadOp.
240  // Erase the terminator first, it will not be used since we are on buffers.
241  rewriter.eraseOp(foreachThreadOp.getTerminator());
242  Block *targetBlock = foreachThreadOp->getBlock();
243  Block::iterator insertionPoint = Block::iterator(foreachThreadOp);
244  Block &sourceBlock = foreachThreadOp.getRegion().front();
245  targetBlock->getOperations().splice(insertionPoint,
246  sourceBlock.getOperations());
247 
248  // Step 5. RAUW thread indices to thread ops.
249  for (Value loopIndex : foreachThreadOp.getThreadIndices()) {
250  Value blockIdx = bvm.lookup(loopIndex);
251  rewriter.replaceAllUsesWith(loopIndex, blockIdx);
252  }
253 
254  // Step 6. Erase old op.
255  rewriter.eraseOp(foreachThreadOp);
256 
257  return DiagnosedSilenceableFailure::success();
258 }

260 DiagnosedSilenceableFailure mlir::transform::gpu::findTopLevelForeachThreadOp(
261  Operation *target, scf::ForeachThreadOp &topLevelForeachThreadOp,
262  TransformOpInterface transformOp) {
263  auto walkResult = target->walk([&](scf::ForeachThreadOp foreachThreadOp) {
264  if (foreachThreadOp->getParentOfType<scf::ForeachThreadOp>())
265  return WalkResult::advance();
266  if (topLevelForeachThreadOp)
267  // TODO: Handle multiple foreach if there is no dependences between them
268  return WalkResult::interrupt();
269  topLevelForeachThreadOp = foreachThreadOp;
270  return WalkResult::advance();
271  });
272 
273  if (walkResult.wasInterrupted())
274  return transformOp.emitSilenceableError()
275  << "could not find a unique topLevel scf.foreach_thread";
276  return DiagnosedSilenceableFailure::success();
277 }

295 DiagnosedSilenceableFailure
296 transform::MapForeachToBlocks::applyToOne(Operation *target,
297  SmallVectorImpl<Operation *> &results,
298  transform::TransformState &state) {
299  LaunchOp gpuLaunch = dyn_cast<LaunchOp>(target);
300  SimpleRewriter rewriter(getContext());
301  auto transformOp = cast<TransformOpInterface>(getOperation());
302 
303  if (!getGenerateGpuLaunch() && !gpuLaunch) {
304  results.assign({target});
305  DiagnosedSilenceableFailure diag =
306  emitSilenceableError()
307  << "Given target is not gpu.launch, set `generate_gpu_launch` "
308  "attribute";
309  diag.attachNote(target->getLoc()) << "when applied to this payload op";
310  return diag;
311  }
312 
313  scf::ForeachThreadOp topLevelForeachThreadOp;
314  DiagnosedSilenceableFailure diag =
315  mlir::transform::gpu::findTopLevelForeachThreadOp(
316  target, topLevelForeachThreadOp, transformOp);
317  if (!diag.succeeded()) {
318  results.assign({target});
319  diag.attachNote(target->getLoc()) << "when applied to this payload op";
320  return diag;
321  }
322 
323  OpBuilder::InsertionGuard guard(rewriter);
324  rewriter.setInsertionPoint(topLevelForeachThreadOp);
325 
326  // Generate gpu launch here and move the foreach_thread inside
327  if (getGenerateGpuLaunch()) {
328  DiagnosedSilenceableFailure diag =
329  createGpuLaunch(rewriter, target->getLoc(), transformOp, gpuLaunch);
330  if (!diag.succeeded()) {
331  results.assign({target});
332  return diag;
333  }
334  rewriter.setInsertionPointToStart(&gpuLaunch.getBody().front());
335  Operation *newForeachThreadOp = rewriter.clone(*topLevelForeachThreadOp);
336  rewriter.eraseOp(topLevelForeachThreadOp);
337  topLevelForeachThreadOp = cast<scf::ForeachThreadOp>(newForeachThreadOp);
338  }
339 
340  SmallVector<int64_t> gridDim = extractFromI64ArrayAttr(getGridDim());
341  SmallVector<DeviceMappingAttrInterface> blockMappingAttributes = {
342  GPUBlockMappingAttr::get(getContext(), Blocks::DimX),
343  GPUBlockMappingAttr::get(getContext(), Blocks::DimY),
344  GPUBlockMappingAttr::get(getContext(), Blocks::DimZ)};
345 
346  diag = checkAttributeType(blockMappingAttributes,
347  topLevelForeachThreadOp.getMapping(), transformOp);
348  if (diag.succeeded())
349  diag = mlir::transform::gpu::mapForeachToBlocksImpl(
350  rewriter, topLevelForeachThreadOp, generateGpuBlockIds, gridDim,
351  transformOp, blockMappingAttributes);
352  if (diag.succeeded()) {
353  diag = alterGpuLaunch(rewriter, gpuLaunch,
354  cast<TransformOpInterface>(getOperation()),
355  gridDim[0], gridDim[1], gridDim[2]);
356  }
357 
358  results.assign({gpuLaunch});
359  return diag;
360 }

366 /// Searches `scf.foreach_thread` ops nested under `target` and maps each such
367 /// op to GPU threads. Mapping is one-to-one and the induction variables of
368 /// `scf.foreach_thread` are rewritten to gpu.thread_id according to the
369 /// thread_dim_apping attribute. Sibling `scf.foreach_thread` are supported in
370 /// which case, the union of the number of threads is computed and may result
371 /// in predication. Dynamic, `scf.foreach_thread` trip counts are currently
372 /// not supported. Dynamic block dim sizes are currently not supported.
373 static DiagnosedSilenceableFailure rewriteOneForeachThreadToGpuThreads(
374  RewriterBase &rewriter, scf::ForeachThreadOp foreachThreadOp,
375  const SmallVectorImpl<int64_t> &globalBlockDims, bool syncAfterDistribute,
376  llvm::Optional<TransformOpInterface> transformOp,
377  const ArrayRef<DeviceMappingAttrInterface> &threadMappingAttributes) {
378  // Step 0. Target-specific verifications. There is no good place to anchor
379  // those right now: the ForeachThreadOp is target-independent and the
380  // transform op does not apply to individual ForeachThreadOp.
381  auto failureHelper =
382  [&](const Twine &message) -> DiagnosedSilenceableFailure {
383  if (transformOp.has_value()) {
384  return transformOp->emitSilenceableError() << message;
385  }
386  return emitDefiniteFailure(foreachThreadOp, message);
387  };
388  Location loc = foreachThreadOp->getLoc();
389  if (foreachThreadOp.getNumResults() > 0)
390  return failureHelper(
391  "only bufferized scf.foreach_thread lowers to gpu.thread_id");
392  if (foreachThreadOp.getNumThreads().size() > 3)
393  return failureHelper(
394  "scf.foreach_thread with rank > 3 does not lower to gpu.thread_id");
395  if (llvm::any_of(foreachThreadOp.getNumThreads(), [](Value v) {
396  return !v.getDefiningOp<arith::ConstantIndexOp>();
397  })) {
398  return failureHelper("unsupported dynamic blockdim size");
399  }
400  if (!foreachThreadOp.getMapping().has_value())
401  return failureHelper("mapping must be present");
402  SmallVector<Attribute> threadMapping =
403  llvm::to_vector(foreachThreadOp.getMapping()->getValue());
404 
405  // Step 1. Complete the threadMapping to a full mapping (with 1s) if
406  // necessary.
407  SmallVector<Value> numThreads =
408  llvm::to_vector(foreachThreadOp.getNumThreads());
409  // Ensure we have 3 block sizes, one for each id.
410  Value one;
411  for (auto attr : threadMappingAttributes) {
412  if (std::find(threadMapping.begin(), threadMapping.end(), attr) ==
413  threadMapping.end()) {
414  threadMapping.push_back(attr);
415  one = one ? one : rewriter.create<arith::ConstantIndexOp>(loc, 1);
416  numThreads.push_back(one);
417  }
418  }
419 
420  // Step 2. sort the values by the corresponding DeviceMappingAttrInterface.
421  auto comparator = [&](DeviceMappingAttrInterface a,
422  DeviceMappingAttrInterface b) -> bool {
423  return a.getMappingId() < b.getMappingId();
424  };
425  SmallVector<Value> blockDimValues =
426  scf::ForeachThreadOp::getValuesSortedByKey(threadMapping, numThreads,
427  comparator);
428  SmallVector<int64_t> blockDims =
429  llvm::to_vector(llvm::map_range(blockDimValues, [](Value v) {
430  return v.getDefiningOp<arith::ConstantIndexOp>().value();
431  }));
432 
433  // Step 3. Create the gpu.thread ops and map the induction variables to the
434  // newly created ops.
435  IndexType indexType = rewriter.getIndexType();
436  SmallVector<Value> threadOps{
437  rewriter.create<ThreadIdOp>(loc, indexType, Dimension::x),
438  rewriter.create<ThreadIdOp>(loc, indexType, Dimension::y),
439  rewriter.create<ThreadIdOp>(loc, indexType, Dimension::z)};
440  BlockAndValueMapping bvm;
441  for (auto [blockIdx, blockDim] :
442  llvm::zip(foreachThreadOp.getThreadIndices(), threadMapping)) {
443  bvm.map(
444  blockIdx,
445  threadOps[blockDim.cast<DeviceMappingAttrInterface>().getMappingId()]);
446  }
447 
448  // Step 4. Maybe create conditionals to predicate the region.
449  Value predicate;
450  for (auto [threadId, blockDim, globalBlockDim] :
451  llvm::zip(threadOps, blockDims, globalBlockDims)) {
452  if (blockDim > globalBlockDim) {
453  return failureHelper(
454  "The requested GPU threads are fewer than the number of loop trip "
455  "counts. Try to tile scf.foreach_thread before mapping or set "
456  "small blockDim.");
457  }
458  if (blockDim == globalBlockDim)
459  continue;
460  Value blockIdx = rewriter.create<arith::ConstantIndexOp>(loc, blockDim);
461  Value tmpPredicate = rewriter.create<arith::CmpIOp>(
462  loc, arith::CmpIPredicate::ult, threadId, blockIdx);
463  predicate =
464  predicate ? rewriter.create<arith::AndIOp>(loc, predicate, tmpPredicate)
465  : tmpPredicate;
466  }
467 
468  // Step 5. Move the body of foreachThreadOp.
469  // Erase the terminator first, it will not be used.
470  rewriter.eraseOp(foreachThreadOp.getTerminator());
471  Block *targetBlock;
472  Block::iterator insertionPoint;
473  if (predicate) {
474  // Step 5.a. If predicated, move at the beginning.
475  auto ifOp =
476  rewriter.create<scf::IfOp>(loc, predicate, /*withElseRegion=*/false);
477  targetBlock = ifOp.thenBlock();
478  insertionPoint = ifOp.thenBlock()->begin();
479  } else {
480  // Step 5.b. Otherwise, move inline just before foreachThreadOp.
481  targetBlock = foreachThreadOp->getBlock();
482  insertionPoint = Block::iterator(foreachThreadOp);
483  }
484  Block &sourceBlock = foreachThreadOp.getRegion().front();
485  targetBlock->getOperations().splice(insertionPoint,
486  sourceBlock.getOperations());
487 
488  // Step 6. RAUW thread indices to thread ops.
489  for (Value loopIndex : foreachThreadOp.getThreadIndices()) {
490  Value threadIdx = bvm.lookup(loopIndex);
491  rewriter.replaceAllUsesWith(loopIndex, threadIdx);
492  }
493 
494  // Step 7. syncthreads.
495  // TODO: Need warpsync
496  if (syncAfterDistribute)
497  rewriter.create<BarrierOp>(loc);
498 
499  // Step 8. Erase old op.
500  rewriter.eraseOp(foreachThreadOp);
501 
502  return DiagnosedSilenceableFailure::success();
503 }

505 DiagnosedSilenceableFailure mlir::transform::gpu::mapNestedForeachToThreadsImpl(
506  RewriterBase &rewriter, Operation *target,
507  const SmallVectorImpl<int64_t> &blockDim, bool syncAfterDistribute,
508  llvm::Optional<TransformOpInterface> transformOp,
509  const ArrayRef<DeviceMappingAttrInterface> &threadMappingAttributes) {
510  DiagnosedSilenceableFailure diag = DiagnosedSilenceableFailure::success();
511  target->walk([&](scf::ForeachThreadOp foreachThreadOp) {
512  diag = checkAttributeType(threadMappingAttributes,
513  foreachThreadOp.getMapping(), transformOp);
514  if (diag.succeeded()) {
515  rewriter.setInsertionPoint(foreachThreadOp);
516  diag = rewriteOneForeachThreadToGpuThreads(
517  rewriter, foreachThreadOp, blockDim, syncAfterDistribute, transformOp,
518  threadMappingAttributes);
519  }
520  return diag.succeeded() ? WalkResult::advance() : WalkResult::interrupt();
521  });
522  return diag;
523 }

525 DiagnosedSilenceableFailure transform::MapNestedForeachToThreads::applyToOne(
526  ::mlir::Operation *target,
527  ::llvm::SmallVectorImpl<::mlir::Operation *> &results,
528  ::mlir::transform::TransformState &state) {
529  LaunchOp gpuLaunch = dyn_cast<LaunchOp>(target);
530  auto transformOp = cast<TransformOpInterface>(getOperation());
531 
532  if (!gpuLaunch) {
533  results.assign({target});
534  return emitSilenceableError() << "Given target is not gpu.launch";
535  }
536 
537  SmallVector<int64_t> blockDim = extractFromI64ArrayAttr(getBlockDim());
538  blockDim.resize(/*size=*/3, /*value=*/1);
539 
540  DiagnosedSilenceableFailure diag =
541  checkGpuLimits(transformOp, llvm::None, llvm::None, llvm::None,
542  blockDim[0], blockDim[1], blockDim[2]);
543  if (diag.isSilenceableFailure()) {
544  results.assign({target});
545  diag.attachNote(getLoc()) << getBlockDimAttrName() << " is very large";
546  return diag;
547  }
548 
549  MLIRContext *ctx = getContext();
550  SimpleRewriter rewriter(ctx);
551  rewriter.setInsertionPoint(target);
