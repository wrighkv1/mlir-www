23 template <typename InterfaceT>
24 static void walkSubElementsImpl(InterfaceT interface,
25  function_ref<void(Attribute)> walkAttrsFn,
26  function_ref<void(Type)> walkTypesFn,
27  DenseSet<Attribute> &visitedAttrs,
28  DenseSet<Type> &visitedTypes) {
29  interface.walkImmediateSubElements(
30  [&](Attribute attr) {
31  // Guard against potentially null inputs. This removes the need for the
32  // derived attribute/type to do it.
33  if (!attr)
34  return;
35 
36  // Avoid infinite recursion when visiting sub attributes later, if this
37  // is a mutable attribute.
38  if (LLVM_UNLIKELY(attr.hasTrait<AttributeTrait::IsMutable>())) {
39  if (!visitedAttrs.insert(attr).second)
40  return;
41  }
42 
43  // Walk any sub elements first.
44  if (auto interface = attr.dyn_cast<SubElementAttrInterface>())
45  walkSubElementsImpl(interface, walkAttrsFn, walkTypesFn, visitedAttrs,
46  visitedTypes);
47 
48  // Walk this attribute.
49  walkAttrsFn(attr);
50  },
51  [&](Type type) {
52  // Guard against potentially null inputs. This removes the need for the
53  // derived attribute/type to do it.
54  if (!type)
55  return;
56 
57  // Avoid infinite recursion when visiting sub types later, if this
58  // is a mutable type.
59  if (LLVM_UNLIKELY(type.hasTrait<TypeTrait::IsMutable>())) {
60  if (!visitedTypes.insert(type).second)
61  return;
62  }
63 
64  // Walk any sub elements first.
65  if (auto interface = type.dyn_cast<SubElementTypeInterface>())
66  walkSubElementsImpl(interface, walkAttrsFn, walkTypesFn, visitedAttrs,
67  visitedTypes);
68 
69  // Walk this type.
70  walkTypesFn(type);
71  });
72 }

98 void AttrTypeReplacer::replaceElementsIn(Operation *op, bool replaceAttrs,
99  bool replaceLocs, bool replaceTypes) {
100  // Functor that replaces the given element if the new value is different,
101  // otherwise returns nullptr.
102  auto replaceIfDifferent = [&](auto element) {
103  auto replacement = replace(element);
104  return (replacement && replacement != element) ? replacement : nullptr;
105  };
106 
107  // Update the attribute dictionary.
108  if (replaceAttrs) {
109  if (auto newAttrs = replaceIfDifferent(op->getAttrDictionary()))
110  op->setAttrs(cast<DictionaryAttr>(newAttrs));
111  }
112 
113  // If we aren't updating locations or types, we're done.
114  if (!replaceTypes && !replaceLocs)
115  return;
116 
117  // Update the location.
118  if (replaceLocs) {
119  if (Attribute newLoc = replaceIfDifferent(op->getLoc()))
120  op->setLoc(cast<LocationAttr>(newLoc));
121  }
122 
123  // Update the result types.
124  if (replaceTypes) {
125  for (OpResult result : op->getResults())
126  if (Type newType = replaceIfDifferent(result.getType()))
127  result.setType(newType);
128  }
129 
130  // Update any nested block arguments.
131  for (Region &region : op->getRegions()) {
132  for (Block &block : region) {
133  for (BlockArgument &arg : block.getArguments()) {
134  if (replaceLocs) {
135  if (Attribute newLoc = replaceIfDifferent(arg.getLoc()))
136  arg.setLoc(cast<LocationAttr>(newLoc));
137  }
138 
139  if (replaceTypes) {
140  if (Type newType = replaceIfDifferent(arg.getType()))
141  arg.setType(newType);
142  }
143  }
144  }
145  }
146 }

148 template <typename T>
149 static void updateSubElementImpl(T element, AttrTypeReplacer &replacer,
150  DenseMap<T, T> &elementMap,
151  SmallVectorImpl<T> &newElements,
152  FailureOr<bool> &changed) {
153  // Bail early if we failed at any point.
154  if (failed(changed))
155  return;
156 
157  // Guard against potentially null inputs. We always map null to null.
158  if (!element) {
159  newElements.push_back(nullptr);
160  return;
161  }
162 
163  // Replace the element.
164  if (T result = replacer.replace(element)) {
165  newElements.push_back(result);
166  if (result != element)
167  changed = true;
168  } else {
169  changed = failure();
170  }
171 }

173 template <typename InterfaceT, typename T>
174 T AttrTypeReplacer::replaceSubElements(InterfaceT interface,
175  DenseMap<T, T> &interfaceMap) {
176  // Walk the current sub-elements, replacing them as necessary.
177  SmallVector<Attribute, 16> newAttrs;
178  SmallVector<Type, 16> newTypes;
179  FailureOr<bool> changed = false;
180  interface.walkImmediateSubElements(
181  [&](Attribute element) {
182  updateSubElementImpl(element, *this, attrMap, newAttrs, changed);
183  },
184  [&](Type element) {
185  updateSubElementImpl(element, *this, typeMap, newTypes, changed);
186  });
187  if (failed(changed))
188  return nullptr;
189 
190  // If any sub-elements changed, use the new elements during the replacement.
191  T result = interface;
192  if (*changed)
193  result = interface.replaceImmediateSubElements(newAttrs, newTypes);
194  return result;
195 }

197 /// Shared implementation of replacing a given attribute or type element.
198 template <typename InterfaceT, typename ReplaceFns, typename T>
199 T AttrTypeReplacer::replaceImpl(T element, ReplaceFns &replaceFns,
200  DenseMap<T, T> &map) {
201  auto [it, inserted] = map.try_emplace(element, element);
202  if (!inserted)
203  return it->second;
204 
205  T result = element;
206  WalkResult walkResult = WalkResult::advance();
207  for (auto &replaceFn : llvm::reverse(replaceFns)) {
208  if (Optional<std::pair<T, WalkResult>> newRes = replaceFn(element)) {
209  std::tie(result, walkResult) = *newRes;
210  break;
211  }
212  }
213 
214  // If an error occurred, return nullptr to indicate failure.
215  if (walkResult.wasInterrupted() || !result)
216  return map[element] = nullptr;
217 
218  // Handle replacing sub-elements if this element is also a container.
219  if (!walkResult.wasSkipped()) {
220  if (auto interface = dyn_cast<InterfaceT>(result)) {
221  // Replace the sub elements of this element, bailing if we fail.
222  if (!(result = replaceSubElements(interface, map)))
223  return map[element] = nullptr;
224  }
225  }
226 
227  return map[element] = result;
228 }
