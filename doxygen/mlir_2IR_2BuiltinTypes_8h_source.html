212 /// This is a builder type that keeps local references to arguments. Arguments
213 /// that are passed into the builder must outlive the builder.
214 class RankedTensorType::Builder {
215 public:
216  /// Build from another RankedTensorType.
217  explicit Builder(RankedTensorType other)
218  : shape(other.getShape()), elementType(other.getElementType()),
219  encoding(other.getEncoding()) {}
220 
221  /// Build from scratch.
222  Builder(ArrayRef<int64_t> shape, Type elementType, Attribute encoding)
223  : shape(shape), elementType(elementType), encoding(encoding) {}
224 
225  Builder &setShape(ArrayRef<int64_t> newShape) {
226  shape = newShape;
227  return *this;
228  }
229 
230  Builder &setElementType(Type newElementType) {
231  elementType = newElementType;
232  return *this;
233  }
234 
235  Builder &setEncoding(Attribute newEncoding) {
236  encoding = newEncoding;
237  return *this;
238  }
239 
240  /// Erase a dim from shape @pos.
241  Builder &dropDim(unsigned pos) {
242  assert(pos < shape.size() && "overflow");
243  if (storage.empty())
244  storage.append(shape.begin(), shape.end());
245  storage.erase(storage.begin() + pos);
246  shape = {storage.data(), storage.size()};
247  return *this;
248  }
249 
250  /// Insert a val into shape @pos.
251  Builder &insertDim(int64_t val, unsigned pos) {
252  assert(pos <= shape.size() && "overflow");
253  if (storage.empty())
254  storage.append(shape.begin(), shape.end());
255  storage.insert(storage.begin() + pos, val);
256  shape = {storage.data(), storage.size()};
257  return *this;
258  }
259 
260  operator RankedTensorType() {
261  return RankedTensorType::get(shape, elementType, encoding);
262  }
263 
264 private:
265  ArrayRef<int64_t> shape;
266  // Owning shape data for copy-on-write operations.
267  SmallVector<int64_t> storage;
268  Type elementType;
269  Attribute encoding;
270 };

276 /// This is a builder type that keeps local references to arguments. Arguments
277 /// that are passed into the builder must outlive the builder.
278 class VectorType::Builder {
279 public:
280  /// Build from another VectorType.
281  explicit Builder(VectorType other)
282  : shape(other.getShape()), elementType(other.getElementType()),
283  numScalableDims(other.getNumScalableDims()) {}
284 
285  /// Build from scratch.
286  Builder(ArrayRef<int64_t> shape, Type elementType,
287  unsigned numScalableDims = 0)
288  : shape(shape), elementType(elementType),
289  numScalableDims(numScalableDims) {}
290 
291  Builder &setShape(ArrayRef<int64_t> newShape,
292  unsigned newNumScalableDims = 0) {
293  numScalableDims = newNumScalableDims;
294  shape = newShape;
295  return *this;
296  }
297 
298  Builder &setElementType(Type newElementType) {
299  elementType = newElementType;
300  return *this;
301  }
302 
303  /// Erase a dim from shape @pos.
304  Builder &dropDim(unsigned pos) {
305  assert(pos < shape.size() && "overflow");
306  if (pos >= shape.size() - numScalableDims)
307  numScalableDims--;
308  if (storage.empty())
309  storage.append(shape.begin(), shape.end());
310  storage.erase(storage.begin() + pos);
311  shape = {storage.data(), storage.size()};
312  return *this;
313  }
314 
315  /// In the particular case where the vector has a single dimension that we
316  /// drop, return the scalar element type.
317  // TODO: unify once we have a VectorType that supports 0-D.
318  operator Type() {
319  if (shape.empty())
320  return elementType;
321  return VectorType::get(shape, elementType, numScalableDims);
322  }
323 
324 private:
325  ArrayRef<int64_t> shape;
326  // Owning shape data for copy-on-write operations.
327  SmallVector<int64_t> storage;
328  Type elementType;
329  unsigned numScalableDims;
330 };

422 /// Returns the strides of the MemRef if the layout map is in strided form.
423 /// MemRefs with a layout map in strided form include:
424 /// 1. empty or identity layout map, in which case the stride information is
425 /// the canonical form computed from sizes;
426 /// 2. a StridedLayoutAttr layout;
427 /// 3. any other layout that be converted into a single affine map layout of
428 /// the form `K + k0 * d0 + ... kn * dn`, where K and ki's are constants or
429 /// symbols.
430 ///
431 /// A stride specification is a list of integer values that are either static
432 /// or dynamic (encoded with ShapedType::kDynamicStrideOrOffset). Strides encode
433 /// the distance in the number of elements between successive entries along a
434 /// particular dimension.
435 LogicalResult getStridesAndOffset(MemRefType t,
436  SmallVectorImpl<int64_t> &strides,
437  int64_t &offset);
438 
439 /// Return a version of `t` with identity layout if it can be determined
440 /// statically that the layout is the canonical contiguous strided layout.
441 /// Otherwise pass `t`'s layout into `simplifyAffineMap` and return a copy of
442 /// `t` with simplified layout.
443 MemRefType canonicalizeStridedLayout(MemRefType t);
444 
445 /// Given MemRef `sizes` that are either static or dynamic, returns the
446 /// canonical "contiguous" strides AffineExpr. Strides are multiplicative and
447 /// once a dynamic dimension is encountered, all canonical strides become
448 /// dynamic and need to be encoded with a different symbol.
449 /// For canonical strides expressions, the offset is always 0 and and fastest
450 /// varying stride is always `1`.
451 ///
452 /// Examples:
453 /// - memref<3x4x5xf32> has canonical stride expression
454 /// `20*exprs[0] + 5*exprs[1] + exprs[2]`.
455 /// - memref<3x?x5xf32> has canonical stride expression
456 /// `s0*exprs[0] + 5*exprs[1] + exprs[2]`.
457 /// - memref<3x4x?xf32> has canonical stride expression
458 /// `s1*exprs[0] + s0*exprs[1] + exprs[2]`.
459 AffineExpr makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
460  ArrayRef<AffineExpr> exprs,
461  MLIRContext *context);
462 
463 /// Return the result of makeCanonicalStrudedLayoutExpr for the common case
464 /// where `exprs` is {d0, d1, .., d_(sizes.size()-1)}
465 AffineExpr makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
466  MLIRContext *context);
