29 /// Return the SSA values that represent the data point accessed using a given
30 /// `indexingMap` for a given point in the iteration space represented by `ivs`.
31 static SmallVector<Value> getIndicesForAccess(OpBuilder &b, Location loc,
32  AffineMap indexingMap,
33  ValueRange ivs) {
34  SmallVector<Value> indices;
35  indices.reserve(indexingMap.getNumResults());
36  for (auto result : indexingMap.getResults()) {
37  AffineMap m = AffineMap::get(indexingMap.getNumDims(),
38  indexingMap.getNumSymbols(), result);
39  Value v = b.create<AffineApplyOp>(loc, m, ivs);
40  indices.push_back(v);
41  }
42  return indices;
43 }

45 /// Method to inline the payload of a `linalgOp` given the iteration space
46 /// point and values for the arguments of the payload.
47 static LogicalResult inlinePayload(OpBuilder &b, LinalgOp linalgOp,
48  ValueRange ivs, ValueRange argValues) {
49  Block *body = linalgOp.getBlock();
50  BlockAndValueMapping map;
51  map.map(body->getArguments(), argValues);
52  for (auto &op : body->without_terminator()) {
53  if (auto indexOp = dyn_cast<IndexOp>(&op)) {
54  map.map(indexOp.getResult(), ivs[indexOp.getDim()]);
55  continue;
56  }
57  b.clone(op, map);
58  }
59 
60  Operation *terminator = body->getTerminator();
61  Location loc = terminator->getLoc();
62  for (const auto &operand : llvm::enumerate(terminator->getOperands())) {
63  Value toStore = map.lookupOrDefault(operand.value());
64  OpOperand *storeInto = linalgOp.getDpsInitOperand(operand.index());
65  auto indices = getIndicesForAccess(
66  b, loc, linalgOp.getMatchingIndexingMap(storeInto), ivs);
67  b.create<memref::StoreOp>(
68  loc, toStore, linalgOp.getDpsInitOperand(operand.index())->get(),
69  indices);
70  }
71  return success();
72 }

79 /// External model implementation of TilingInterface for LinalgOps. An external
80 /// model implementation is used for now till the use of `TilingInterface` is

94  /// Return the iteration domain range.
95  SmallVector<Range> getIterationDomain(Operation *op, OpBuilder &b) const {
96  OpBuilder::InsertionGuard g(b);
97  b.setInsertionPoint(op);
98  Location loc = op->getLoc();
99  LinalgOp linalgOp = cast<LinalgOp>(op);
100  SmallVector<OpFoldResult> allShapesSizes =
101  linalgOp.createFlatListOfOperandDims(b, loc);
102  AffineMap map = linalgOp.getShapesToLoopsMap();
103 
104  return llvm::to_vector(
105  llvm::map_range(map.getResults(), [&](AffineExpr loopExpr) {
106  OpFoldResult ofr =
107  makeComposedFoldedAffineApply(b, loc, loopExpr, allShapesSizes);
108  return Range{b.getIndexAttr(0), ofr, b.getIndexAttr(1)};
109  }));
110  }

112  // Instantiate the tiled implementation of the operation.
113  SmallVector<Operation *>
114  getTiledImplementation(Operation *op, OpBuilder &b,
115  ArrayRef<OpFoldResult> offsets,
116  ArrayRef<OpFoldResult> sizes) const {
117  // Leave the `sizeBounds` value empty. That is only needed when the `sizes`
118  // specified could lead to out of bounds accesses.
119  Location loc = op->getLoc();
120  LinalgOp linalgOp = cast<LinalgOp>(op);
121  SmallVector<Value> valuesToTile = linalgOp->getOperands();
122  SmallVector<Value, 4> tiledOperands = makeTiledShapes(
123  b, loc, linalgOp, valuesToTile, offsets, sizes, {}, true);
124 
125  SmallVector<Type> resultTensorTypes =
126  getTensorOutputTypes(linalgOp, tiledOperands);
127 
128  Operation *tiledOp =
129  linalgOp.clone(b, loc, resultTensorTypes, tiledOperands);
130  offsetIndices(b, cast<LinalgOp>(tiledOp), offsets);
131 
132  return {tiledOp};
133  }

135  // Return the details of the output tile generated by the tiled
136  // implementation.
137  LogicalResult
138  getResultTilePosition(Operation *op, OpBuilder &b, unsigned resultNumber,
139  ArrayRef<OpFoldResult> offsets,
140  ArrayRef<OpFoldResult> sizes,
141  SmallVector<OpFoldResult> &resultOffsets,
142  SmallVector<OpFoldResult> &resultSizes) const {
143  Location loc = op->getLoc();
144  LinalgOp linalgOp = cast<LinalgOp>(op);
145 
146  AffineExpr d0;
147  bindDims(b.getContext(), d0);
148  SmallVector<OpFoldResult> subShapeSizes =
149  llvm::to_vector(llvm::map_range(sizes, [&](OpFoldResult ofr) {
150  return makeComposedFoldedAffineApply(b, loc, d0 - 1, ofr);
151  }));
152 
153  OpOperand *outOperand = linalgOp.getDpsInitOperand(resultNumber);
154  SliceParameters sliceParams = computeSliceParameters(
155  b, loc, outOperand->get(), sizes,
156  linalgOp.getMatchingIndexingMap(outOperand), offsets,
157  /*ubs*/ {}, subShapeSizes, true);
158  resultOffsets = sliceParams.offsets;
159  resultSizes = sliceParams.sizes;
160  return success();
161  }

163  FailureOr<Value> generateResultTileValue(Operation *op, OpBuilder &b,
164  unsigned resultNumber,
165  ArrayRef<OpFoldResult> offsets,
166  ArrayRef<OpFoldResult> sizes) const {
167  auto linalgOp = cast<LinalgOp>(op);
168 
169  // Check that the indexing map used for the output is a projected
170  // permutation. This could be relaxed with a more general approach that can
171  // map the offsets and sizes from the result to iteration space tiles
172  // (filling in full extent for dimensions not used to access the result).
173  AffineMap indexingMap =
174  linalgOp.getIndexingMapMatchingResult(op->getResult(resultNumber));
175  if (!indexingMap.isProjectedPermutation()) {
176  return op->emitOpError(
177  "unhandled tiled implementation generation when result is not "
178  "accessed using a permuted projection");
179  }
180 
181  auto numLoops = linalgOp.getNumLoops();
182  auto tilingInterfaceOp = cast<TilingInterface>(op);
183  SmallVector<OpFoldResult> iterationTileOffsets(numLoops),
184  iterationTileSizes(numLoops);
185  if (!indexingMap.isPermutation()) {
186  SmallVector<Range> iterationDomain =
187  tilingInterfaceOp.getIterationDomain(b);
188  for (const auto &range : llvm::enumerate(iterationDomain)) {
189  iterationTileOffsets[range.index()] = range.value().offset;
190  iterationTileSizes[range.index()] = range.value().size;
191  }
192  }
193  for (const auto &resultExpr : llvm::enumerate(indexingMap.getResults())) {
194  unsigned dimPosition =
195  resultExpr.value().cast<AffineDimExpr>().getPosition();
196  iterationTileOffsets[dimPosition] = offsets[resultExpr.index()];
197  iterationTileSizes[dimPosition] = sizes[resultExpr.index()];
198  }
199 
200  SmallVector<Operation *> tiledOp = tilingInterfaceOp.getTiledImplementation(
201  b, iterationTileOffsets, iterationTileSizes);
202  if (tiledOp.size() != 1)
203  return op->emitOpError("failed to generate tiled implementation");
204 
205  return tiledOp[0]->getResult(resultNumber);
206  }

208  LogicalResult generateScalarImplementation(Operation *op, OpBuilder &builder,
209  Location loc,
210  ValueRange ivs) const {
211  auto linalgOp = cast<LinalgOp>(op);
212  if (!linalgOp.hasBufferSemantics())
213  return op->emitOpError("expected operation to have buffer semantics");
214 
215  SmallVector<Value> indexedValues;
216  indexedValues.reserve(linalgOp->getNumOperands());
217  Location linalgOpLoc = op->getLoc();
218  /// Load the data corresponding to the block arguments that
219  /// represent input operands.
220  for (OpOperand &operand : linalgOp->getOpOperands()) {
221  if (!linalgOp.payloadUsesValueFromOperand(&operand)) {
222  indexedValues.push_back(nullptr);
223  continue;
224  }
225  if (linalgOp.isScalar(&operand)) {
226  indexedValues.push_back(operand.get());
227  continue;
228  }
229  SmallVector<Value> indices = getIndicesForAccess(
230  builder, linalgOpLoc, linalgOp.getMatchingIndexingMap(&operand), ivs);
231  Value load =
232  builder.create<memref::LoadOp>(linalgOpLoc, operand.get(), indices);
233  indexedValues.push_back(load);
234  }
235 
236  /// Inline the op payload and store the result.
237  return inlinePayload(builder, linalgOp, ivs, indexedValues);
238  }

245 /// External model implementation of PartialReductionInterface for LinalgOps.
246 template <typename LinalgOpTy>
247 struct LinalgOpPartialReductionInterface
248  : public PartialReductionOpInterface::ExternalModel<
249  LinalgOpPartialReductionInterface<LinalgOpTy>, LinalgOpTy> {
250  FailureOr<Operation *> generateInitialTensorForPartialReduction(
251  Operation *op, OpBuilder &b, Location loc, ArrayRef<OpFoldResult> sizes,
252  ArrayRef<int> reductionDims) const {
253  auto linalgOp = cast<LinalgOp>(op);
254  OpBuilder::InsertionGuard guard(b);
255  assert(reductionDims.size() == 1 &&
256  "only support single reduction right now.");
257  if (linalgOp.hasBufferSemantics())
258  return op->emitOpError("expected operation to have tensor semantics");
259  // Insert the new parallel dimension based on the index of the reduction
260  // loop. This could be controlled by user for more flexibility.
261  int64_t insertSplitDimension = reductionDims[0];
262 
263  SmallVector<Operation *, 4> combinerOps;
264  if (!matchReduction(linalgOp.getRegionOutputArgs(), 0, combinerOps) ||
265  combinerOps.size() != 1)
266  return op->emitOpError("Failed to anaysis the reduction operation.");
267 
268  Operation *reductionOp = combinerOps[0];
269  Optional<Attribute> identity = getNeutralElement(reductionOp);
270  if (!identity.has_value())
271  return op->emitOpError(
272  "Failed to get an identity value for the reduction operation.");
273 
274  // Calculate the new shape, we insert the new dimension based on the index
275  // of the reduction dimension.
276  SmallVector<int64_t> newOutputShape;
277  ArrayRef<int64_t> oldShape =
278  linalgOp.getShape(linalgOp.getDpsInitOperand(0));
279  SmallVector<Value> dynamicDims;
280  for (int64_t idx : llvm::seq<int64_t>(0, oldShape.size() + 1)) {
281  if (idx == insertSplitDimension) {
282  dispatchIndexOpFoldResults(sizes[idx], dynamicDims, newOutputShape,
283  ShapedType::kDynamicStrideOrOffset);
284  continue;
285  }
286  int64_t oldIdx = idx < insertSplitDimension ? idx : idx - 1;
287  int64_t dim = oldShape[oldIdx];
288  newOutputShape.push_back(dim);
289  if (ShapedType::isDynamic(dim))
290  dynamicDims.push_back(b.createOrFold<tensor::DimOp>(
291  loc, linalgOp.getDpsInitOperand(0)->get(), oldIdx));
292  }
293  Value emptyTensor = b.create<tensor::EmptyOp>(
294  loc, newOutputShape, linalgOp.getRegionOutputArgs()[0].getType(),
295  dynamicDims);
296  Value constantOp = b.create<arith::ConstantOp>(loc, *identity);
297  auto identityTensor =
298  b.create<linalg::FillOp>(loc, constantOp, emptyTensor);
299  return identityTensor.getOperation();
300  }

302  Operation *tileToPartialReduction(Operation *op, OpBuilder &b, Location loc,
303  ValueRange init,
304  ArrayRef<OpFoldResult> offsets,
305  ArrayRef<OpFoldResult> sizes,
306  ArrayRef<int> reductionDims) const {
307  OpBuilder::InsertionGuard guard(b);
308  auto linalgOp = cast<LinalgOp>(op);
309  assert(reductionDims.size() == 1 &&
310  "only support single reduction right now.");
311  int64_t insertSplitDimension = reductionDims[0];
312 
313  AffineMap oldOutputMap =
314  linalgOp.getMatchingIndexingMap(linalgOp.getDpsInitOperand(0));
315  SmallVector<AffineExpr> outputExpr;
316  for (auto &[idx, expr] : llvm::enumerate(oldOutputMap.getResults())) {
317  if (static_cast<int64_t>(idx) == insertSplitDimension) {
318  outputExpr.push_back(b.getAffineDimExpr(reductionDims[0]));
319  }
320  outputExpr.push_back(expr);
321  }
322  if (insertSplitDimension == oldOutputMap.getNumResults())
323  outputExpr.push_back(b.getAffineDimExpr(reductionDims[0]));
324 
325  // Step 1: Extract a slice of the input operands.
326  SmallVector<Value> valuesToTile = linalgOp.getDpsInputOperands();
327  SmallVector<Value, 4> tiledOperands =
328  makeTiledShapes(b, loc, op, valuesToTile, offsets, sizes, {}, true);
329 
330  // Step 2: Extract the accumulator operands
331  SmallVector<OpFoldResult> strides(offsets.size(), b.getIndexAttr(1));
332  SmallVector<OpFoldResult> outOffsets(offsets.size(), b.getIndexAttr(0));
333  // TODO: use SubsetExtractOpInterface once it is available.
334  Value out = b.create<tensor::ExtractSliceOp>(loc, init[0], outOffsets,
335  sizes, strides);
336 
337  // Step3. create a generic op where the reduction dimension is replaced by a
338  // parallel dimension of the size of reduction.
339  SmallVector<utils::IteratorType> newIteratorTypes =
340  linalgOp.getIteratorTypesArray();
341  newIteratorTypes[reductionDims[0]] = utils::IteratorType::parallel;
342  SmallVector<AffineMap> newMaps = linalgOp.getIndexingMapsArray();
343  newMaps.back() = AffineMap::get(newMaps.back().getNumDims(), 0, outputExpr,
344  linalgOp.getContext());
345  auto genericOp =
346  b.create<GenericOp>(loc, TypeRange({out.getType()}), tiledOperands,
347  ValueRange({out}), newMaps, newIteratorTypes);
348  BlockAndValueMapping mapping;
349  op->getRegion(0).cloneInto(&genericOp.getRegion(),
350  genericOp.getRegion().begin(), mapping);
351  return genericOp.getOperation();
352  }

354  Operation *mergeReductions(Operation *op, OpBuilder &b, Location loc,
355  ValueRange partialReduce,
356  ArrayRef<int> reductionDims) const {
357  auto linalgOp = cast<LinalgOp>(op);
358  assert(reductionDims.size() == 1 &&
359  "only support single reduction right now.");
360  int64_t dimToMerge = reductionDims[0];
361 
362  // Then create a new reduction that only reduce the newly added dimension
363  // from the previous op.
364  int64_t intermRank =
365  partialReduce[0].getType().cast<ShapedType>().getRank();
366  AffineMap inputMap = b.getMultiDimIdentityMap(intermRank);
367  SmallVector<utils::IteratorType> reductionIteratorTypes;
368  SmallVector<AffineExpr> exprs;
369  for (int64_t i : llvm::seq<int64_t>(0, intermRank)) {
370  if (dimToMerge == i) {
371  reductionIteratorTypes.push_back(utils::IteratorType::reduction);
372  } else {
373  exprs.push_back(b.getAffineDimExpr(i));
374  reductionIteratorTypes.push_back(utils::IteratorType::parallel);
375  }
376  }
377  AffineMap outputMap =
378  AffineMap::get(intermRank, 0, exprs, op->getContext());
379  SmallVector<AffineMap> reductionMaps = {inputMap, outputMap};
380 
381  SmallVector<Operation *, 4> combinerOps;
382  matchReduction(linalgOp.getRegionOutputArgs(), 0, combinerOps);
383  Operation *reductionOp = combinerOps[0];
384 
385  auto reduction = b.create<GenericOp>(
386  loc, op->getResultTypes(), ValueRange({partialReduce[0]}),
387  SmallVector<Value>{linalgOp.getDpsInitOperands()}, reductionMaps,
388  reductionIteratorTypes,
389  [reductionOp](OpBuilder &b, Location loc, ValueRange inputs) {
390  Operation *clonedReductionOp = b.clone(*reductionOp);
391  clonedReductionOp->setOperand(0, inputs[0]);
392  clonedReductionOp->setOperand(1, inputs[1]);
393  b.create<linalg::YieldOp>(loc, clonedReductionOp->getResult(0));
394  });
395  return reduction.getOperation();
396  }
