26 /// Return configuration options for One-Shot Bufferize.
27 static bufferization::OneShotBufferizationOptions
28 getBufferizationOptions(bool analysisOnly) {
29  using namespace bufferization;
30  OneShotBufferizationOptions options;
31  options.bufferizeFunctionBoundaries = true;
32  // TODO(springerm): To spot memory leaks more easily, returning dense allocs
33  // should be disallowed.
34  options.allowReturnAllocs = true;
35  options.functionBoundaryTypeConversion = LayoutMapOption::IdentityLayoutMap;
36  options.unknownTypeConverterFn = [](Value value, unsigned memorySpace,
37  const BufferizationOptions &options) {
38  return getMemRefTypeWithStaticIdentityLayout(
39  value.getType().cast<TensorType>(), memorySpace);
40  };
41  if (analysisOnly) {
42  options.testAnalysisOnly = true;
43  options.printConflicts = true;
44  }
45  return options;
46 }

52 void mlir::sparse_tensor::buildSparseCompiler(
53  OpPassManager &pm, const SparseCompilerOptions &options) {
54  pm.addNestedPass<func::FuncOp>(createLinalgGeneralizationPass());
55  pm.addPass(
56  bufferization::createTensorCopyInsertionPass(getBufferizationOptions(
57  /*analysisOnly=*/options.testBufferizationAnalysisOnly)));
58  if (options.testBufferizationAnalysisOnly)
59  return;
60  pm.addPass(createPreSparsificationRewritePass());
61  pm.addPass(createSparsificationPass(options.sparsificationOptions()));
62  pm.addPass(createPostSparsificationRewritePass(options.enableRuntimeLibrary));
63  if (options.enableRuntimeLibrary) {
64  pm.addPass(createSparseTensorConversionPass(
65  options.sparseTensorConversionOptions()));
66  } else {
67  pm.addPass(
68  createSparseTensorCodegenPass(options.enableBufferInitialization));
69  pm.addPass(
70  createSparseBufferRewritePass(options.enableBufferInitialization));
71  }
72  pm.addPass(createDenseBufferizationPass(
73  getBufferizationOptions(/*analysisOnly=*/false)));
74  pm.addNestedPass<func::FuncOp>(createCanonicalizerPass());
75  pm.addNestedPass<func::FuncOp>(
76  mlir::bufferization::createFinalizingBufferizePass());
77  // TODO(springerm): Add sparse support to the BufferDeallocation pass and add
78  // it to this pipeline.
79  pm.addNestedPass<func::FuncOp>(createConvertLinalgToLoopsPass());
80  pm.addNestedPass<func::FuncOp>(createConvertVectorToSCFPass());
81  pm.addNestedPass<func::FuncOp>(createConvertSCFToCFPass());
82  pm.addPass(createLowerAffinePass());
83  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
84  pm.addPass(createMemRefToLLVMConversionPass());
85  pm.addNestedPass<func::FuncOp>(createConvertComplexToStandardPass());
86  pm.addNestedPass<mlir::func::FuncOp>(mlir::arith::createArithExpandOpsPass());
87  pm.addNestedPass<func::FuncOp>(createConvertMathToLLVMPass());
88  pm.addPass(createConvertMathToLibmPass());
89  pm.addPass(createConvertComplexToLibmPass());
90  pm.addPass(createConvertComplexToLLVMPass());
91  pm.addPass(createConvertFuncToLLVMPass());
92  pm.addPass(createReconcileUnrealizedCastsPass());
93 }
