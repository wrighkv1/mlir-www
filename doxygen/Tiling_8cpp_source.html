43 static bool isZero(OpFoldResult v) {
44  if (!v)
45  return false;
46  if (auto attr = v.dyn_cast<Attribute>()) {
47  IntegerAttr intAttr = attr.dyn_cast<IntegerAttr>();
48  return intAttr && intAttr.getValue().isZero();
49  }
50  if (auto cst = v.get<Value>().getDefiningOp<arith::ConstantIndexOp>())
51  return cst.value() == 0;
52  return false;
53 }

55 std::tuple<SmallVector<Range, 4>, LoopIndexToRangeIndexMap>
56 mlir::linalg::makeTiledLoopRanges(RewriterBase &b, Location loc, AffineMap map,
57  ArrayRef<OpFoldResult> allShapeSizes,
58  ArrayRef<OpFoldResult> allTileSizes) {
59  assert(allTileSizes.size() == map.getNumResults());
60  // Apply `map` to get shape sizes in loop order.
61  SmallVector<OpFoldResult> shapeSizes =
62  makeComposedFoldedMultiResultAffineApply(b, loc, map, allShapeSizes);
63  SmallVector<OpFoldResult> tileSizes(allTileSizes.begin(), allTileSizes.end());
64 
65  // Traverse the tile sizes, which are in loop order, erase zeros everywhere.
66  LoopIndexToRangeIndexMap loopIndexToRangeIndex;
67  for (int idx = 0, e = tileSizes.size(), zerosCount = 0; idx < e; ++idx) {
68  if (isZero(tileSizes[idx - zerosCount])) {
69  shapeSizes.erase(shapeSizes.begin() + idx - zerosCount);
70  tileSizes.erase(tileSizes.begin() + idx - zerosCount);
71  ++zerosCount;
72  continue;
73  }
74  loopIndexToRangeIndex[idx] = idx - zerosCount;
75  }
76 
77  // Create a new range with the applied tile sizes.
78  SmallVector<Range, 4> res;
79  for (unsigned idx = 0, e = tileSizes.size(); idx < e; ++idx)
80  res.push_back(Range{b.getIndexAttr(0), shapeSizes[idx], tileSizes[idx]});
81  return std::make_tuple(res, loopIndexToRangeIndex);
82 }

116 FailureOr<MultiSizeSpecification>
117 mlir::linalg::computeMultiTileSizes(OpBuilder &builder, LinalgOp op,
118  unsigned dimension, OpFoldResult targetSize,
119  OpFoldResult divisor, bool emitAssertions) {
120  // Bail out on dimension overflow.
121  if (dimension >= op.getNumLoops())
122  return failure();
123 
124  // The code below works only on values.
125  Location loc = op.getLoc();
126  ImplicitLocOpBuilder b(loc, builder);
127  if (emitAssertions) {
128  emitIsPositiveIndexAssertion(b, targetSize);
129  emitIsPositiveIndexAssertion(b, divisor);
130  }
131  Value targetSizeValue =
132  getValueOrCreateConstantIndexOp(builder, loc, targetSize);
133  Value divisorValue = getValueOrCreateConstantIndexOp(builder, loc, divisor);
134 
135  // Find the trip count of the iteration space dimension for which the tile
136  // sizes are computed.
137  SmallVector<OpFoldResult> allShapes =
138  op.createFlatListOfOperandDims(b, b.getLoc());
139  AffineMap shapesToLoops = op.getShapesToLoopsMap();
140  SmallVector<OpFoldResult> loopRanges =
141  makeComposedFoldedMultiResultAffineApply(b, op.getLoc(), shapesToLoops,
142  allShapes);
143  Value tripCount =
144  getValueOrCreateConstantIndexOp(b, op.getLoc(), loopRanges[dimension]);
145 
146  // Compute the tile sizes and the respective numbers of tiles.
147  AffineExpr s0 = b.getAffineSymbolExpr(0);
148  AffineExpr s1 = b.getAffineSymbolExpr(1);
149  AffineExpr s2 = b.getAffineSymbolExpr(2);
150  auto apply = [&](AffineExpr expr, ValueRange values) -> Value {
151  return makeComposedAffineApply(b, b.getLoc(), expr, values);
152  };
153  Value a = apply(s0.floorDiv(s1), {tripCount, divisorValue});
154  Value t = apply((s0 + s1 - 1).floorDiv(s1), {targetSizeValue, divisorValue});
155  Value d = apply((s0 + s1 - 1).floorDiv(s1), {a, t});
156  Value s = apply(s0.floorDiv(s1) * s2, {a, d, divisorValue});
157  Value v = apply(s0 % s1, {a, d});
158  Value u = apply(s0 - s1, {d, v});
159 
160  MultiSizeSpecification spec;
161  spec.lowTileSize = s;
162  spec.highTileSize = apply(s0 + s1, {s, divisorValue});
163  spec.lowTripCount = u;
164  spec.highTripCount = v;
165 
166  // If requested, emit the check that the tile sizes are computed correctly.
167  // For example, for iteration dimension size of 15 and the target size 8 it is
168  // impossible to find two tile sizes both divisible by 8 that fully cover the
169  // original space dimension.
170  if (emitAssertions) {
171  AffineExpr s3 = builder.getAffineSymbolExpr(3);
172  Value coveredSize =
173  apply(s0 * s1 + s2 * s3, {spec.lowTileSize, spec.lowTripCount,
174  spec.highTileSize, spec.highTripCount});
175  Value equals = b.create<arith::CmpIOp>(arith::CmpIPredicate::eq,
176  coveredSize, tripCount);
177  b.create<cf::AssertOp>(
178  equals, builder.getStringAttr(
179  "could not compute dynamic multi-size tile shapes"));
180  }
181 
182  return spec;
183 }

185 /// Returns true if the maximum tile offset `tileSize * numThreads-1` is less
186 /// than `iterationSize`.
187 static bool canOmitTileOffsetInBoundsCheck(OpFoldResult tileSize,
188  OpFoldResult numThreads,
189  OpFoldResult iterationSize) {
190  Optional<int64_t> tileSizeConst = getConstantIntValue(tileSize);
191  Optional<int64_t> numThreadsConst = getConstantIntValue(numThreads);
192  Optional<int64_t> iterSizeConst = getConstantIntValue(iterationSize);
193  if (!tileSizeConst || !numThreadsConst || !iterSizeConst)
194  return false;
195  return *tileSizeConst * (*numThreadsConst - 1) < *iterSizeConst;
196 }

214 /// Fill out the `tiledOffsets` and `tiledSizes` to be used to tile to a given
215 /// number of threads.
216 static void calculateTileOffsetsAndSizes(
217  RewriterBase &b, Location loc, scf::ForeachThreadOp foreachThreadOp,
218  ArrayRef<OpFoldResult> numThreads, SmallVector<Range> loopRanges,
219  bool omitTileOffsetBoundsCheck,
220  Optional<ArrayRef<OpFoldResult>> nominalTileSizes,
221  SmallVector<OpFoldResult> &tiledOffsets,
222  SmallVector<OpFoldResult> &tiledSizes) {
223  ValueRange threadIds = foreachThreadOp.getThreadIndices();
224  SmallVector<OpFoldResult> nonZeroNumThreads =
225  llvm::to_vector(llvm::make_filter_range(numThreads, [](OpFoldResult ofr) {
226  return !isConstantIntValue(ofr, 0);
227  }));
228  int64_t nLoops = loopRanges.size();
229  tiledOffsets.reserve(nLoops);
230  tiledSizes.reserve(nLoops);
231  for (unsigned loopIdx = 0, threadIdIdx = 0; loopIdx < nLoops; ++loopIdx) {
232  bool overflow = loopIdx >= numThreads.size();
233  bool isZero = !overflow && isConstantIntValue(numThreads[loopIdx], 0);
234  // Degenerate case: take the whole domain.
235  if (overflow || isZero) {
236  tiledOffsets.push_back(loopRanges[loopIdx].offset);
237  tiledSizes.push_back(loopRanges[loopIdx].size);
238  continue;
239  }
240 
241  // Tiled case: compute the offset and size.
242  AffineExpr i, j, m, n, o;
243  bindDims(b.getContext(), i, j);
244  bindSymbols(b.getContext(), m, n, o);
245  OpFoldResult size = loopRanges[loopIdx].size;
246  OpFoldResult offset = loopRanges[loopIdx].offset;
247  OpFoldResult threadId = threadIds[threadIdIdx];
248  // Symbolic fixed max size per thread.
249  // TODO: floor + 0/1 depending on case for better load-balancing.
250  OpFoldResult tileSizePerThread =
251  nominalTileSizes.has_value()
252  ? (*nominalTileSizes)[loopIdx]
253  : makeComposedFoldedAffineApply(
254  b, loc, m.ceilDiv(n),
255  ArrayRef<OpFoldResult>{size, nonZeroNumThreads[threadIdIdx]});
256 
257  // Dynamic offset shifted by threadId * maxSizePerThread.
258  OpFoldResult offsetPerThread = makeComposedFoldedAffineApply(
259  b, loc, i + j * m, {offset, threadId, tileSizePerThread});
260  // Dynamic upper-bound depending on the threadId.
261  OpFoldResult residualTileSize = makeComposedFoldedAffineApply(
262  b, loc, i + j * m - n,
263  {offset, nonZeroNumThreads[threadIdIdx], tileSizePerThread, size});
264  if (!isConstantIntValue(residualTileSize, 0)) {
265  OpFoldResult sizeMinusOffsetPerThread = makeComposedFoldedAffineApply(
266  b, loc, -i + m, {offsetPerThread, size});
267  tileSizePerThread =
268  buildMin(b, loc, {sizeMinusOffsetPerThread, tileSizePerThread});
269  }
270 
271  tiledOffsets.push_back(offsetPerThread);
272  // TODO: if tileSizePerThread <= 0 early exit.
273  if (!omitTileOffsetBoundsCheck &&
274  !canOmitTileOffsetInBoundsCheck(tileSizePerThread,
275  nonZeroNumThreads[threadIdIdx], size))
276  tileSizePerThread =
277  buildMax(b, loc, {b.getIndexAttr(0), tileSizePerThread});
278 
279  tiledSizes.push_back(tileSizePerThread);
280  ++threadIdIdx;
281  }
282 }

284 /// Rewrite a TilingInterface `op` to a tiled `scf.foreach_thread`. The
285 /// tiling is specified by the number of tiles/threads `numThreads` and the
286 /// optional nominal tile size `nominalTileSizes`. If `nominalTilSizes` is
287 /// not specified, then it is derived from `numThreads` as `ceilDiv(dimSize[i],
288 /// numThreads[i])`. If non-empty, the `mapping` is added as an
289 /// attribute to the resulting `scf.foreach_thread`. A zero tile sizes indicate
290 /// that the dimension is not tiled, and can be thought of as tiling by the full
291 /// size of data.
292 /// It is the user's responsibility to ensure that `numThreads` is a valid
293 /// tiling specification (i.e. that only tiles parallel dimensions, e.g. in the
294 /// Linalg case). If `omitTileOffsetBoundsCheck` is true, then the function will
295 /// assume that `tileSize[i] * (numThread[i] -1) <= dimSize[i]` holds.
296 static FailureOr<ForeachThreadTilingResult> tileToForeachThreadOpImpl(
297  RewriterBase &b, TilingInterface op, ArrayRef<OpFoldResult> numThreads,
298  Optional<ArrayRef<OpFoldResult>> nominalTileSizes,
299  Optional<ArrayAttr> mapping, bool omitTileOffsetBoundsCheck) {
300  Location loc = op->getLoc();
301  OpBuilder::InsertionGuard g(b);
302  SmallVector<Range> loopRanges = op.getIterationDomain(b);
303  if (loopRanges.empty())
304  return op->emitOpError("expected non-empty loop ranges");
305  auto hasStrideOne = [](Range r) { return !isConstantIntValue(r.stride, 1); };
306  if (llvm::any_of(loopRanges, hasStrideOne))
307  return op->emitOpError("only stride-1 supported atm");
308 
309  // Gather destination tensors.
310  SmallVector<Value> dest;
311  if (failed(tensor::getOrCreateDestinations(b, loc, op, dest)))
312  return op->emitOpError("failed to get destination tensors");
313 
314  SmallVector<OpFoldResult> nonZeroNumThreads =
315  llvm::to_vector(llvm::make_filter_range(numThreads, [](OpFoldResult ofr) {
316  return !isConstantIntValue(ofr, 0);
317  }));
318  SmallVector<Value> materializedNonZeroNumThreads =
319  llvm::to_vector(llvm::map_range(nonZeroNumThreads, [&](OpFoldResult ofr) {
320  return getValueOrCreateConstantIndexOp(b, loc, ofr);
321  }));
322 
323  Operation *tiledOp = nullptr;
324 
325  // Create the ForeachThreadOp. We don't use the lambda body-builder
326  // version because we require the use of RewriterBase in the body, so we
327  // manually move the insertion point to the body below.
328  scf::ForeachThreadOp foreachThreadOp = b.create<scf::ForeachThreadOp>(
329  loc, dest, ValueRange(materializedNonZeroNumThreads), mapping);
330 
331  // Fill out the ForeachThreadOp body.
332  b.setInsertionPointToStart(foreachThreadOp.getBody(0));
333  SmallVector<OpFoldResult> tiledOffsets, tiledSizes;
334  calculateTileOffsetsAndSizes(b, loc, foreachThreadOp, numThreads, loopRanges,
335  omitTileOffsetBoundsCheck, nominalTileSizes,
336  tiledOffsets, tiledSizes);
337 
338  // Clone the tileable op and update its destination operands to use the output
339  // bbArgs of the ForeachThreadOp.
340  ArrayRef<BlockArgument> destBbArgs =
341  foreachThreadOp.getOutputBlockArguments();
342  Operation *clonedOp = b.clone(*op.getOperation());
343  auto destinationStyleOp = dyn_cast<DestinationStyleOpInterface>(clonedOp);
344  if (destinationStyleOp) {
345  for (OpOperand *outOperand : destinationStyleOp.getDpsInitOperands()) {
346  auto *it = llvm::find(dest, outOperand->get());
347  assert(it != dest.end() && "dest operand not found in dest");
348  unsigned destNum = std::distance(dest.begin(), it);
349  outOperand->set(destBbArgs[destNum]);
350  }
351  }
352 
353  // Tile the cloned op and delete the clone.
354  SmallVector<Operation *> tiledOps =
355  cast<TilingInterface>(clonedOp).getTiledImplementation(b, tiledOffsets,
356  tiledSizes);
357  b.eraseOp(clonedOp);
358  assert(tiledOps.size() == 1 && "expected a single produced tiled op");
359  tiledOp = tiledOps.front();
360 
361  auto tilingInterfaceOp = dyn_cast<TilingInterface>(tiledOp);
362  assert(tilingInterfaceOp && "Tiled op does not implement TilingInterface");
363  OpBuilder::InsertPoint insertPt = b.saveInsertionPoint();
364  for (auto it : llvm::zip(llvm::seq(unsigned(0), unsigned(dest.size())),
365  tilingInterfaceOp->getResults(), destBbArgs)) {
366  b.setInsertionPoint(insertPt.getBlock(), insertPt.getPoint());
367  SmallVector<OpFoldResult> resultOffsets, resultSizes;
368  if (failed(op.getResultTilePosition(b, std::get<0>(it), tiledOffsets,
369  tiledSizes, resultOffsets,
370  resultSizes)))
371  return op->emitOpError("output offsets couldn't be calculated");
372  SmallVector<OpFoldResult> strides(resultSizes.size(), b.getIndexAttr(1));
373  b.setInsertionPointToEnd(foreachThreadOp.getTerminator().getBody());
374  b.create<tensor::ParallelInsertSliceOp>(loc, std::get<1>(it),
375  std::get<2>(it), resultOffsets,
376  resultSizes, strides);
377  }
378  return ForeachThreadTilingResult{foreachThreadOp, tiledOp};
379 }

406  numThreads.push_back(numTiles);
407  }
408  return tileToForeachThreadOpImpl(b, op, numThreads,
409  /*nominalTileSizes=*/tileSizes, mapping,
410  /*omitTileOffsetBoundsCheck=*/true);

413 FailureOr<linalg::ForeachThreadReductionTilingResult>
414 linalg::tileReductionUsingForeachThread(RewriterBase &b,
415  PartialReductionOpInterface op,
416  ArrayRef<OpFoldResult> numThreads,
417  Optional<ArrayAttr> mapping) {
418  Location loc = op.getLoc();
419  OpBuilder::InsertionGuard g(b);
420  // Ops implementing PartialReductionOpInterface are expected to implement
421  // TilingInterface.
422  auto tilingInterfaceOp = cast<TilingInterface>(op.getOperation());
423  SmallVector<Range> iterationDomain = tilingInterfaceOp.getIterationDomain(b);
424  if (op->getNumResults() != 1)
425  return b.notifyMatchFailure(
426  op, "don't support ops with multiple results for now");
427  SmallVector<utils::IteratorType> iterators =
428  tilingInterfaceOp.getLoopIteratorTypes();
429  SmallVector<unsigned> redDims;
430  cast<linalg::LinalgOp>(op.getOperation()).getReductionDims(redDims);
431  if (redDims.size() != 1)
432  return b.notifyMatchFailure(
433  op, "only support ops with one reduction dimension.");
434  int reductionDim = static_cast<int>(redDims.front());
435  // 1. create the inital tensor value.
436  FailureOr<Operation *> identityTensor =
437  op.generateInitialTensorForPartialReduction(b, loc, numThreads,
438  reductionDim);
439  if (failed(identityTensor))
440  return b.notifyMatchFailure(op,
441  "cannot create a tensor of identity value.");
442 
443  // Gather destination tensors.
444  SmallVector<Value> dest;
445  if (failed(tensor::getOrCreateDestinations(b, loc, op, dest)))
446  return b.notifyMatchFailure(op, "failed to get destination tensors");
447 
448  Operation *tiledOp = nullptr;
449 
450  SmallVector<OpFoldResult> nonZeroNumThreads =
451  llvm::to_vector(llvm::make_filter_range(numThreads, [](OpFoldResult ofr) {
452  return !isConstantIntValue(ofr, 0);
453  }));
454  SmallVector<Value> materializedNonZeroNumThreads =
455  llvm::to_vector(llvm::map_range(nonZeroNumThreads, [&](OpFoldResult ofr) {
456  return getValueOrCreateConstantIndexOp(b, loc, ofr);
457  }));
458 
459  // 2. Create the ForeachThreadOp with an empty region.
460  scf::ForeachThreadOp foreachThreadOp = b.create<scf::ForeachThreadOp>(
461  loc, identityTensor.value()->getResults(),
462  ValueRange(materializedNonZeroNumThreads), mapping);
463 
464  // 3. calculate the tile offsets and sizes.
465  b.setInsertionPointToStart(foreachThreadOp.getBody(0));
466  SmallVector<OpFoldResult> tiledOffsets, tiledSizes;
467  calculateTileOffsetsAndSizes(
468  b, loc, foreachThreadOp, numThreads, iterationDomain,
469  /*omitTileOffsetBoundsCheck =*/false,
470  /*nominalTileSizes=*/llvm::None, tiledOffsets, tiledSizes);
471 
472  // 4. Clone the tileable op and update its destination operands to use the
473  // output bbArgs of the ForeachThreadOp.
474  ArrayRef<BlockArgument> destBbArgs =
475  foreachThreadOp.getOutputBlockArguments();
476  Operation *clonedOp = b.clone(*op.getOperation());
477  auto destinationStyleOp = cast<DestinationStyleOpInterface>(clonedOp);
478  for (OpOperand *outOperand : destinationStyleOp.getDpsInitOperands()) {
479  auto *it = llvm::find(dest, outOperand->get());
480  assert(it != dest.end() && "dest operand not found in dest");
481  unsigned destNum = std::distance(dest.begin(), it);
482  SmallVector<OpFoldResult> strides(numThreads.size(), b.getIndexAttr(1));
483  SmallVector<OpFoldResult> outOffsets(numThreads.size(), b.getIndexAttr(0));
484  SmallVector<OpFoldResult> sizes = tiledSizes;
485  sizes[reductionDim] = b.getIndexAttr(1);
486  outOffsets[reductionDim] = foreachThreadOp.getThreadIndices().front();
487  // TODO: use SubsetExtractOpInterface once it is available.
488  Value patial = b.create<tensor::ExtractSliceOp>(
489  loc, outOperand->get().getType().cast<RankedTensorType>(),
490  destBbArgs[destNum], outOffsets, sizes, strides);
491  outOperand->set(patial);
492  }
493 
494  // 5. Tile the cloned op and delete the clone.
495  SmallVector<Operation *> tiledOps =
496  cast<TilingInterface>(clonedOp).getTiledImplementation(b, tiledOffsets,
497  tiledSizes);
498  b.eraseOp(clonedOp);
499  assert(tiledOps.size() == 1 && "expected a single produced tiled op");
500  tiledOp = tiledOps.front();
501 
502  // 6. Insert the partial reductions back into a new tensor.
503  auto tiledInterfaceOp = dyn_cast<TilingInterface>(tiledOp);
504  assert(tiledInterfaceOp && "Tiled op does not implement TilingInterface");
505  OpBuilder::InsertPoint insertPt = b.saveInsertionPoint();
506  for (auto it : llvm::zip(llvm::seq(unsigned(0), unsigned(dest.size())),
507  tiledInterfaceOp->getResults(), destBbArgs)) {
508  b.setInsertionPoint(insertPt.getBlock(), insertPt.getPoint());

538  b.replaceOp(op, mergeOp->getResults());
539  ForeachThreadReductionTilingResult results;
540  results.initialOp = identityTensor.value();
541  results.loops = foreachThreadOp;
542  results.parallelTiledOp = tiledOp;
543  results.mergeOp = mergeOp;

559 template <typename LoopTy>
560 static FailureOr<TiledLinalgOp>
561 tileLinalgOpImpl(RewriterBase &b, LinalgOp op, ArrayRef<OpFoldResult> tileSizes,
562  const LinalgTilingOptions &options) {
563  auto nLoops = op.getNumLoops();
564  // Initial tile sizes may be too big, only take the first nLoops.
565  tileSizes = tileSizes.take_front(nLoops);
566 
567  if (llvm::all_of(tileSizes, isZero)) {
568  TiledLinalgOp tiledOp;
569  tiledOp.op = cast<LinalgOp>(b.clone(*op.getOperation()));
570  tiledOp.tensorResults.assign(tiledOp.op->result_begin(),
571  tiledOp.op->result_end());
572  return tiledOp;
573  }
574 
575  // 1. Build the tiled loop ranges.
576  SmallVector<OpFoldResult> allShapeSizes =
577  op.createFlatListOfOperandDims(b, op.getLoc());
578  AffineMap shapeSizesToLoopsMap = op.getShapesToLoopsMap();
579  if (!shapeSizesToLoopsMap)
580  return failure();
581 
582  auto [loopRanges, loopIndexToRangeIndex] = makeTiledLoopRanges(
583  b, op.getLoc(), shapeSizesToLoopsMap, allShapeSizes, tileSizes);
584 
585  SmallVector<utils::IteratorType, 4> iteratorTypes;
586  for (const auto &attr : enumerate(op.getIteratorTypesArray())) {
587  if (loopIndexToRangeIndex.count(attr.index()))
588  iteratorTypes.push_back(attr.value());
589  }
590  // If interchangeVector is empty, use the identity. Build the permutation map
591  // otherwise.
592  auto invPermutationMap =
593  AffineMap::getMultiDimIdentityMap(tileSizes.size(), b.getContext());
594  if (!options.interchangeVector.empty()) {
595  // Based on the pruned iterations (due to zero tile size), recompute the
596  // interchange vector.
597  SmallVector<unsigned, 4> interchangeVector;
598  interchangeVector.reserve(options.interchangeVector.size());
599  for (auto pos : options.interchangeVector) {
600  auto it = loopIndexToRangeIndex.find(pos);
601  if (it == loopIndexToRangeIndex.end())
602  continue;
603  interchangeVector.push_back(it->second);
604  }
605  // Interchange vector is guaranteed to be a permutation,
606  // `inversePermutation` must succeed.
607  invPermutationMap = inversePermutation(
608  AffineMap::getPermutationMap(interchangeVector, b.getContext()));
609  assert(invPermutationMap);
610  SmallVector<int64_t> permutation(interchangeVector.begin(),
611  interchangeVector.end());
612  applyPermutationToVector(loopRanges, permutation);
613  applyPermutationToVector(iteratorTypes, permutation);
614  }
615 
616  // Handle distribution. Create a vector of the same size of loops that are to
617  // be tiled.
618  SmallVector<linalg::ProcInfo> procInfo;
619  if (options.distribution) {
620  procInfo.resize(
621  iteratorTypes.size(),
622  linalg::ProcInfo{nullptr, nullptr, linalg::DistributionMethod::None});
623  // Collect loop ranges of tiled loopss, loops that are parallel.
624  SmallVector<Range> parallelLoopRanges;
625  for (const auto &iteratorType : llvm::enumerate(iteratorTypes)) {
626  if (!isParallelIterator(iteratorType.value()))
627  break;
628  parallelLoopRanges.push_back(loopRanges[iteratorType.index()]);
629  }
630  auto returnedProcInfo =
631  options.distribution->procInfo(b, op.getLoc(), parallelLoopRanges);
632  unsigned procIdIdx = 0;
633  // Update the distribution information for the loops.
634  for (const auto &iteratorType : llvm::enumerate(iteratorTypes)) {
635  if (!isParallelIterator(iteratorType.value()))
636  break;
637  procInfo[iteratorType.index()] = returnedProcInfo[procIdIdx++];
638  }
639  }
640 
641  // 2. Create the tiled loops.
642  LinalgOp res = op;
643  SmallVector<Value, 4> ivs, tensorResults;
644  auto tiledLoopBodyBuilder =
645  [&](OpBuilder &builder, Location loc, ValueRange localIvs,
646  ValueRange operandValuesToUse) -> scf::ValueVector {
647  ivs.assign(localIvs.begin(), localIvs.end());
648 
649  // When an `interchangeVector` is present, it has been applied to the
650  // loop ranges and the iterator types. Apply its inverse to the
651  // resulting loop `ivs` to match the op definition.
652  SmallVector<Value, 4> interchangedIvs;
653  if (!options.interchangeVector.empty())
654  interchangedIvs = applyMapToValues(b, loc, invPermutationMap, ivs);
655  else
656  interchangedIvs.assign(ivs.begin(), ivs.end());
657 
658  // Tile the `operandValuesToUse` that either match the `op` operands
659  // themselves or the tile loop arguments forwarding them.
660  assert(operandValuesToUse.size() ==
661  static_cast<size_t>(op->getNumOperands()) &&
662  "expect the number of operands and inputs and outputs to match");
663  SmallVector<Value> valuesToTile = operandValuesToUse;
664  SmallVector<OpFoldResult> sizeBounds =
665  makeComposedFoldedMultiResultAffineApply(b, loc, shapeSizesToLoopsMap,
666  allShapeSizes);
667  SmallVector<Value> tiledOperands = makeTiledShapes(
668  b, loc, op, valuesToTile, getAsOpFoldResult(interchangedIvs), tileSizes,
669  sizeBounds,
670  /*omitPartialTileCheck=*/false);
671 
672  SmallVector<Type> resultTensorTypes =
673  getTensorOutputTypes(op, tiledOperands);
674  res = op.clone(b, loc, resultTensorTypes, tiledOperands);
675  tensorResults =
676  insertSlicesBack(builder, loc, op, tiledOperands, res->getResults());
677  return scf::ValueVector(tensorResults.begin(), tensorResults.end());
678  };
679  GenerateLoopNest<LoopTy>::doit(b, op.getLoc(), loopRanges, op, iteratorTypes,
680  tiledLoopBodyBuilder, procInfo);
681 
682  // 3. Transform IndexOp results w.r.t. the tiling.
683  transformIndexOps(b, res, ivs, loopIndexToRangeIndex);
684 
685  // 4. Gather the newly created loops and return them with the new op.
686  SmallVector<Operation *, 8> loops;
687  loops.reserve(ivs.size());
688  for (auto iv : ivs) {
689  if (iv.isa<BlockArgument>()) {
690  loops.push_back(iv.cast<BlockArgument>().getOwner()->getParentOp());
691  assert(loops.back() && "no owner found for induction variable!");
692  } else {
693  // TODO: Instead of doing this, try to recover the ops used instead of the
694  // loop.
695  loops.push_back(nullptr);
696  }
697  }
698 
699  // 5. Get the tensor results from the outermost loop if available. Otherwise
700  // use the previously captured `tensorResults`.
701  Operation *outermostLoop = nullptr;
702  for (Operation *loop : loops)
703  if ((outermostLoop = loop))
704  break;
705 
706  return TiledLinalgOp{
707  res, loops, outermostLoop ? outermostLoop->getResults() : tensorResults};
708 }

745 /// Generate a loop nest around a given tensor::PadOp (for tiling). `newPadOp`
746 /// and `loopNest` are output parameters that return the new (tiled)
747 /// tensor::PadOp and the loop nest.
748 static LogicalResult tilePadOp(RewriterBase &builder, tensor::PadOp op,
749  tensor::PadOp &newPadOp, LoopNest &loopNest,
750  const LinalgTilingOptions &options) {
751  Location loc = op.getLoc();
752  OpBuilder::InsertionGuard g(builder);
753  builder.setInsertionPoint(op);
754 
755  // Clone tensor::PadOp so that the existing op can be replaced more easily.
756  newPadOp = cast<tensor::PadOp>(builder.clone(*op.getOperation()));
757  // Get rank and tile sizes.
758  int64_t rank = op.getResultType().getRank();
759  SmallVector<OpFoldResult> tileSizes =
760  getAsOpFoldResult(options.tileSizeComputationFunction(builder, op));
761  // Normalize untiled padding dimensions to 0.
762  tileSizes.append(rank - tileSizes.size(), builder.getIndexAttr(0));
763  // Compute lower and upper bounds of the loop nest.
764  TilingInterface tilingInterface =
765  dyn_cast<TilingInterface>(op.getOperation());
766  SmallVector<Range> ranges = tilingInterface.getIterationDomain(builder);
767  SmallVector<Value> lbs, dims, steps;
768  SmallVector<OpFoldResult> allDims;
769  for (int64_t i = 0; i < rank; ++i) {
770  allDims.push_back(ranges[i].size);
771  if (!isZero(tileSizes[i])) {
772  lbs.push_back(
773  getValueOrCreateConstantIndexOp(builder, loc, ranges[i].offset));
774  dims.push_back(
775  getValueOrCreateConstantIndexOp(builder, loc, ranges[i].size));
776  steps.push_back(
777  getValueOrCreateConstantIndexOp(builder, loc, tileSizes[i]));
778  }
779  }
780  SmallVector<Value> destinationTensors;
781  if (failed(tensor::getOrCreateDestinations(builder, loc, tilingInterface,
782  destinationTensors)))
783  return failure();
784 
785  loopNest = mlir::scf::buildLoopNest(
786  builder, loc, lbs, /*ubs=*/dims, steps, ValueRange(destinationTensors),
787  [&](OpBuilder &b, Location loc, ValueRange localIvs,
788  ValueRange iterArgs) -> scf::ValueVector {
789  // Compute offsets and sizes of ExtractSliceOp.
790  SmallVector<Value> localIVVector = llvm::to_vector(localIvs);
791  SmallVector<OpFoldResult> offsets = computeTileOffsets(
792  b, loc, getAsOpFoldResult(localIVVector), tileSizes);
793  SmallVector<OpFoldResult> sizes =
794  computeTileSizes(b, loc, tileSizes, allDims);
795  // Create ExtractSliceOp: Extract a tile from the tensor::PadOp.
796  // Note: The tensor::PadOp is located outside of the loop nest. It is
797  // later moved inside by ExtractSliceOfPadTensorSwapPattern.
798  auto map = AffineMap::getMultiDimIdentityMap(rank, b.getContext());
799  Value tiledOutput = makeTiledShape(
800  b, loc, newPadOp->getResult(0), tileSizes, map, offsets, allDims,
801  sizes, /*omitPartialTileCheck=*/false);
802  auto sliceOp = tiledOutput.getDefiningOp<tensor::ExtractSliceOp>();
803  assert(sliceOp && "expected ExtractSliceOp");
804  // Insert the tile into the output tensor.
805  Value yieldValue =
806  insertSliceIntoTensor(b, loc, sliceOp, sliceOp, iterArgs[0]);
807  return scf::ValueVector({yieldValue});
808  });
809  return success();
810 }
