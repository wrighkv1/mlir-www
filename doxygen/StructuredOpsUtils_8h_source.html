60 /// Helper StructuredGenerator class to manipulate and rewrite ops with
61 /// `StructuredOpInterface`. This is templated for now because VectorOps do not
62 /// yet implement the StructuredOpInterface itself.
63 template <typename StructuredOpInterface, typename IteratorTypeT>
64 class StructuredGenerator {
65 public:
66  using MapList = ArrayRef<ArrayRef<AffineExpr>>;
67 
68  struct IteratorType {
69  IteratorType(IteratorTypeT iter) : iter(iter) {}
70  bool isOfType(IteratorTypeT expectedIter) const {
71  return expectedIter == iter;
72  }
73  IteratorTypeT iter;
74  };
75  struct Par : public IteratorType {
76  Par() : IteratorType(IteratorTypeT::parallel) {}
77  };
78  struct Red : public IteratorType {
79  Red() : IteratorType(IteratorTypeT::reduction) {}
80  };
81 
82  StructuredGenerator(OpBuilder &builder, StructuredOpInterface op)
83  : builder(builder), ctx(op.getContext()), loc(op.getLoc()),
84  iterators(op.getIteratorTypesArray()), maps(op.getIndexingMapsArray()),
85  op(op) {}
86 
87  bool iters(ArrayRef<IteratorType> its) {
88  if (its.size() != iterators.size())
89  return false;
90  for (int i = 0, e = its.size(); i != e; ++i) {
91  if (!its[i].isOfType(iterators[i]))
92  return false;
93  }
94  return true;
95  }
96 
97  bool layout(MapList l) {
98  auto infer = [](MapList m) { return AffineMap::inferFromExprList(m); };
99  return maps == infer(l);
100  }
101 
102 protected:
103  OpBuilder &builder;
104  MLIRContext *ctx;
105  Location loc;
106  SmallVector<IteratorTypeT> iterators;
107  SmallVector<AffineMap, 4> maps;
108  Operation *op;
109 };
