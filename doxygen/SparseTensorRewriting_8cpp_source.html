63 // Helper to detect sampling operation.
64 static bool isSampling(GenericOp op) {
65  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
66  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
67  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def)) {
68  // Both scalar input arguments used exactly once.
69  Value s1 = op.getBlock()->getArgument(0);
70  Value s2 = op.getBlock()->getArgument(1);
71  return (def->getOperand(0) == s1 && def->getOperand(1) == s2) ||
72  (def->getOperand(1) == s1 && def->getOperand(0) == s2);
73  }
74  }
75  return false;
76 }

78 // Helper to detect chain of multiplications that do not involve x.
79 static bool isMulChain(Value val, Value x) {
80  if (auto arg = val.dyn_cast<BlockArgument>())
81  return arg != x;
82  if (auto *def = val.getDefiningOp()) {
83  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def))
84  return isMulChain(def->getOperand(0), x) &&
85  isMulChain(def->getOperand(1), x);
86  }
87  return false;
88 }

90 // Helper to detect x = x + <multiplications>.
91 static bool isSumOfMul(GenericOp op) {
92  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
93  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
94  if (isa<arith::AddFOp>(def) || isa<arith::AddIOp>(def)) {
95  Value x = op.getBlock()->getArguments().back();
96  return (def->getOperand(0) == x && isMulChain(def->getOperand(1), x)) ||
97  (def->getOperand(1) == x && isMulChain(def->getOperand(0), x));
98  }
99  }
100  return false;
101 }

103 // Helper to detect direct yield of a zero value.
104 static bool isZeroYield(GenericOp op) {
105  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
106  if (auto arg = yieldOp.getOperand(0).dyn_cast<BlockArgument>()) {
107  if (arg.getOwner()->getParentOp() == op) {

129 // shape relies on the output tensor

131 static RankedTensorType getUnorderedCOOFromType(RankedTensorType src) {
132  auto *ctx = src.getContext();
133  auto rank = src.getRank();
134  SmallVector<DimLevelType> dims;
135 
136  // An unordered and non-unique compressed dim at beginning.
137  dims.push_back(DimLevelType::CompressedNuNo);
138 
139  if (rank > 1) {
140  // TODO: it is actually ordered at the level for ordered input.
141  // Followed by unordered non-unique n-2 singleton levels.
142  std::fill_n(std::back_inserter(dims), rank - 2,
143  DimLevelType::SingletonNuNo);
144  // TODO: only if all the inputs (for concatentate) are unique at the last
145  // level should the COO has a unique level at the end. Ends by a unordered
146  // unique singleton level unless the tensor rank is 1.
147  dims.push_back(DimLevelType::SingletonNo);
148  }
149  SparseTensorEncodingAttr encSrc = getSparseTensorEncoding(src);
150  // TODO: Maybe pick the bitwidth based on input/output tensors (probably the
151  // largest one among them) in the original operation instead of using the
152  // default value.
153  unsigned pointerBitWidth = encSrc ? encSrc.getPointerBitWidth() : 0;
154  unsigned indexBitWidth = encSrc ? encSrc.getIndexBitWidth() : 0;
155  auto enc = SparseTensorEncodingAttr::get(
156  ctx, dims, AffineMap::getMultiDimIdentityMap(rank, ctx), AffineMap(),
157  pointerBitWidth, indexBitWidth);
158  return RankedTensorType::get(src.getShape(), src.getElementType(), enc);
159 }

173 static LogicalResult genForeachOnSparseConstant(ForeachOp op,
174  RewriterBase &rewriter,
175  SparseElementsAttr attr) {
176  auto loc = op.getLoc();
177  SmallVector<Value> reduc = op.getInitArgs();
178 
179  // Foreach on constant.
180  foreachInSparseConstant(
181  loc, rewriter, attr,
182  [&reduc, &rewriter, op](ArrayRef<Value> coords, Value v) mutable {
183  SmallVector<Value> args;
184  args.append(coords.begin(), coords.end());
185  args.push_back(v);
186  args.append(reduc);
187  // Clones the foreach op to get a copy of the loop body.
188  auto cloned = cast<ForeachOp>(rewriter.clone(*op.getOperation()));
189  assert(args.size() == cloned.getBody()->getNumArguments());
190  Operation *yield = cloned.getBody()->getTerminator();
191  rewriter.mergeBlockBefore(cloned.getBody(), op, args);
192  // clean up
193  rewriter.eraseOp(cloned);
194  reduc = yield->getOperands();
195  rewriter.eraseOp(yield);
196  });
197 
198  rewriter.replaceOp(op, reduc);
199  return success();
200 }

208 /// Rewriting rule that converts direct yield of zero with initial allocation.
209 struct FoldInvariantYield : public OpRewritePattern<GenericOp> {
210 public:
211  using OpRewritePattern<GenericOp>::OpRewritePattern;
212 
213  LogicalResult matchAndRewrite(GenericOp op,
214  PatternRewriter &rewriter) const override {
215  if (!op.hasTensorSemantics() || op.getNumResults() != 1 ||
216  !isAlloc(op.getDpsInitOperand(0), /*isZero=*/false) ||
217  !isZeroYield(op) || !op.getDpsInitOperand(0)->get().hasOneUse())
218  return failure();
219  auto outputType = op.getResult(0).getType().cast<RankedTensorType>();
220  // Yielding zero on newly allocated (all-zero) sparse tensors can be
221  // optimized out directly (regardless of dynamic or static size).
222  if (getSparseTensorEncoding(outputType)) {
223  rewriter.replaceOp(op, op.getDpsInitOperand(0)->get());
224  return success();
225  }
226  // Incorporate zero value into allocation copy.
227  if (!outputType.hasStaticShape())
228  return failure();
229  Value zero = constantZero(rewriter, op.getLoc(), op.getResult(0).getType());
230  AllocTensorOp a =
231  op.getDpsInitOperand(0)->get().getDefiningOp<AllocTensorOp>();
232  rewriter.updateRootInPlace(a, [&]() { a.getCopyMutable().assign(zero); });
233  rewriter.replaceOp(op, op.getDpsInitOperand(0)->get());
234  return success();
235  }
236 };

238 /// Rewriting rule that converts two kernels:
239 ///
240 /// T(i,j) = SUM(k, A(i,j,k) * B(i,j,k) * ... )
241 /// X(i,j) = S(i,j) * T(i,j)
242 ///
243 /// into a single kernel, using distributive law:
244 ///
245 /// X(i,j) = SUM(k, S(i,j) * A(i,j,k) * B(i,j,k) * ... )
246 ///
247 /// This kind of fusion (merging two ops into one but using arithmetic
248 /// equalities that may not hold for floating-point computations) would
249 /// be undesirable in the dense case, since we distribute the multiplication
250 /// into the reduction loop. However, for sparse sampling tensor S, such
251 /// a fusion may actually reduce the asymptotic complexity of the kernel,
252 /// since intermediate results may be nullified.
253 struct FuseSparseMultiplyOverAdd : public OpRewritePattern<GenericOp> {
254 public:
255  using OpRewritePattern<GenericOp>::OpRewritePattern;
256 
257  LogicalResult matchAndRewrite(GenericOp op,
258  PatternRewriter &rewriter) const override {
259  // Check consumer.
260  if (!op.hasTensorSemantics() || op.getNumDpsInputs() != 2 ||
261  op.getNumResults() != 1 ||
262  op.getNumParallelLoops() != op.getNumLoops() ||
263  !op.getMatchingIndexingMap(op.getDpsInitOperand(0)).isIdentity() ||
264  !op.getMatchingIndexingMap(op.getDpsInputOperand(0)).isIdentity() ||
265  !op.getMatchingIndexingMap(op.getDpsInputOperand(1)).isIdentity())
266  return failure();
267  // Find consuming OP2(sparse, other) or OP2(other, sparse). The other
268  // operand can be sparse or dense, since the point of this rewriting rule
269  // is detecting a situation in which *more* sparsity is introduced into
270  // a computation, be it already sparse or still dense.
271  unsigned other = 0;
272  if (isSparseTensor(op.getDpsInputOperand(0)))
273  other = 1;
274  else if (!isSparseTensor(op.getDpsInputOperand(1)))
275  return failure();
276  // Check producer.
277  auto prod = dyn_cast_or_null<GenericOp>(
278  op.getDpsInputOperand(other)->get().getDefiningOp());
279  if (!prod || !prod.hasTensorSemantics() || prod.getNumResults() != 1 ||
280  !prod.getResult(0).hasOneUse())
281  return failure();
282  // Sampling consumer and sum of multiplication chain producer.
283  if (!isAlloc(op.getDpsInitOperand(0), /*isZero=*/false) ||
284  !isAlloc(prod.getDpsInitOperand(0), /*isZero=*/true) ||
285  !isSampling(op) || !isSumOfMul(prod))
286  return failure();
287  // Modify operand structure of producer and consumer.
288  Location loc = prod.getLoc();
289  SmallVector<Value> inputOps = prod.getInputs();
290  SmallVector<Value> outputOps = op.getOutputs();
291  SmallVector<AffineMap> fusedIndexMaps = prod.getIndexingMapsArray();
292  inputOps.push_back(op.getDpsInputOperand(1 - other)->get());
293  fusedIndexMaps.push_back(fusedIndexMaps.back()); // mimic other
294  // Fuse producer and consumer into a new generic op.
295  auto fusedOp = rewriter.create<GenericOp>(
296  loc, op.getResult(0).getType(), inputOps, outputOps,
297  rewriter.getAffineMapArrayAttr(fusedIndexMaps), prod.getIteratorTypes(),
298  /*doc=*/nullptr, /*library_call=*/nullptr);
299  Block &prodBlock = prod.getRegion().front();
300  Block &consBlock = op.getRegion().front();
301  BlockAndValueMapping mapper;
302  Block *fusedBlock = new Block();
303  fusedOp.getRegion().push_back(fusedBlock);
304  unsigned num = prodBlock.getNumArguments();
305  for (unsigned i = 0; i < num - 1; i++)
306  addArg(mapper, fusedBlock, prodBlock.getArgument(i));
307  addArg(mapper, fusedBlock, consBlock.getArgument(1 - other));
308  addArg(mapper, fusedBlock, prodBlock.getArgument(num - 1));
309  // Clone bodies of the producer and consumer in new evaluation order.
310  auto *acc = prodBlock.getTerminator()->getOperand(0).getDefiningOp();
311  auto *sampler = consBlock.getTerminator()->getOperand(0).getDefiningOp();
312  rewriter.setInsertionPointToStart(fusedBlock);
313  Value last;
314  for (auto &op : prodBlock.without_terminator())
315  if (&op != acc) {
316  last = op.getResult(0);
317  rewriter.clone(op, mapper);
318  }
319  mapper.map(consBlock.getArgument(other), fusedBlock->back().getResult(0));
320  mapper.map(last, rewriter.clone(*sampler, mapper)->getResult(0));
321  last = rewriter.clone(*acc, mapper)->getResult(0);
322  rewriter.create<linalg::YieldOp>(loc, last);
323  // Force initial value on merged allocation for dense outputs.
324  if (!getSparseTensorEncoding(op.getResult(0).getType())) {
325  Value init = prod.getDpsInitOperand(0)
326  ->get()
327  .getDefiningOp<AllocTensorOp>()
328  .getCopy();
329  AllocTensorOp a =
330  op.getDpsInitOperand(0)->get().getDefiningOp<AllocTensorOp>();
331  rewriter.updateRootInPlace(a, [&]() { a.getCopyMutable().assign(init); });
332  }
333  // Replace consumer with fused operation. Old producer
334  // and consumer ops will be removed by DCE.
335  rewriter.replaceOp(op, fusedOp->getResults());
336  return success();
337  }
338 
339 private:
340  // Helper to add argument and record the mapping.
341  static void addArg(BlockAndValueMapping &mapper, Block *b, BlockArgument a) {
342  mapper.map(a, b->addArgument(a.getType(), a.getLoc()));
343  }
344 };

346 /// Sparse rewriting rule for sparse-to-sparse reshape operator.
347 template <typename ReshapeOp>
348 struct Sparse2SparseReshapeRewriter : public OpRewritePattern<ReshapeOp> {
349 public:
350  using OpRewritePattern<ReshapeOp>::OpRewritePattern;
351 
352  LogicalResult matchAndRewrite(ReshapeOp op,
353  PatternRewriter &rewriter) const override {
354  Location loc = op.getLoc();
355  Value srcTensor = op.getSrc();
356  auto srcTp = srcTensor.getType().template cast<RankedTensorType>();
357  auto dstTp = op.getResult().getType().template cast<RankedTensorType>();
358  SparseTensorEncodingAttr encSrc = getSparseTensorEncoding(srcTp);
359  SparseTensorEncodingAttr encDst = getSparseTensorEncoding(dstTp);
360  if (!encDst || !encSrc) {
361  return failure();
362  }
363 
364  // Generate code to represent the static dimension constants or compute
365  // the dynamic dimension values.
366  SmallVector<Value> srcSizes;
367  sizesForTensor(rewriter, srcSizes, loc, srcTp, srcTensor);
368  SmallVector<Value> dstSizes;
369  SmallVector<Value> dstDynSizes;
370  if (dstTp.hasStaticShape()) {
371  for (auto d : dstTp.getShape())
372  dstSizes.push_back(constantIndex(rewriter, loc, d));
373  } else {
374  ArrayRef<int64_t> dstShape = dstTp.getShape();
375  genReshapeDstShape(loc, rewriter, dstSizes, srcSizes, dstShape,
376  op.getReassociationIndices());
377  for (auto &d : llvm::enumerate(dstShape)) {
378  if (d.value() == ShapedType::kDynamicSize)
379  dstDynSizes.push_back(dstSizes[d.index()]);
380  }
381  }
382 
383  // Implement the sparse2sparse reshape as follows:
384  // %tmp = bufferization.alloc_tensor : unordered COO
385  // foreach srcCoords %srcTensor
386  // insert translateIndicesArray(srcCoords), %tmp
387  // %t = sparse_tensor.cast %tmp
388  RankedTensorType cooTp = getUnorderedCOOFromType(dstTp);
389  auto cooBuffer =
390  rewriter.create<AllocTensorOp>(loc, cooTp, dstDynSizes).getResult();
391  ForeachOp foreachOp = rewriter.create<ForeachOp>(
392  loc, srcTensor, cooBuffer,
393  [&](OpBuilder &builder, Location loc, ValueRange args, Value v,
394  ValueRange reduc) {
395  SmallVector<Value> srcIndices;
396  SmallVector<Value> dstIndices;
397  for (int64_t i = 0, e = srcTp.getRank(); i < e; i++) {
398  uint64_t dim = toStoredDim(encSrc, i);
399  srcIndices.push_back(args[dim]);
400  }
401  translateIndicesArray(builder, loc, op.getReassociationIndices(),
402  srcIndices, srcSizes, dstSizes, dstIndices);
403  auto t = builder.create<InsertOp>(loc, v, reduc.front(), dstIndices);
404  builder.create<sparse_tensor::YieldOp>(loc, t);
405  });
406  auto t = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
407  auto converted = rewriter.create<ConvertOp>(loc, dstTp, t).getResult();
408  rewriter.create<DeallocTensorOp>(loc, t);
409  rewriter.replaceOp(op, converted);
410  return success();
411  }
412 };

414 /// Sparse rewriting rule for sparse-to-dense and dense-to-sparse reshape
415 /// operator.
416 template <typename ReshapeOp>
417 struct ReshapeRewriter : public OpRewritePattern<ReshapeOp> {
418 public:
419  using OpRewritePattern<ReshapeOp>::OpRewritePattern;
420 
421  LogicalResult matchAndRewrite(ReshapeOp op,
422  PatternRewriter &rewriter) const override {
423  Location loc = op->getLoc();
424  auto encDst = getSparseTensorEncoding(op.getResult().getType());
425  auto encSrc = getSparseTensorEncoding(op.getSrc().getType());
426  // Since a pure dense expansion is very cheap (change of view), for
427  // a sparse2dense or dense2sparse, we can simply unfuse a sparse
428  // conversion from the reshape operation itself.
429  // All other cases are handled elsewhere.
430  if (encDst && encSrc) {
431  return failure();
432  }
433  if (encSrc) {
434  RankedTensorType rtp =
435  op.getSrc().getType().template cast<RankedTensorType>();
436  auto denseTp =
437  RankedTensorType::get(rtp.getShape(), rtp.getElementType());
438  auto convert = rewriter.create<ConvertOp>(loc, denseTp, op.getSrc());
439  op->setOperand(0, convert);
440  return success();
441  }
442  if (encDst) {
443  RankedTensorType rtp =
444  op.getResult().getType().template cast<RankedTensorType>();
445  auto denseTp =
446  RankedTensorType::get(rtp.getShape(), rtp.getElementType());
447  auto reshape = rewriter.create<ReshapeOp>(loc, denseTp, op.getSrc(),
448  op.getReassociation());
449  Value convert = rewriter.create<ConvertOp>(loc, rtp, reshape);
450  rewriter.replaceOp(op, convert);
451  return success();
452  }
453  return failure();
454  }

457 struct ConcatenateRewriter : public OpRewritePattern<ConcatenateOp> {
458  using OpRewritePattern::OpRewritePattern;
459  LogicalResult matchAndRewrite(ConcatenateOp op,
460  PatternRewriter &rewriter) const override {
461  auto loc = op.getLoc();
462  auto rtp = op.getType().cast<RankedTensorType>();
463  size_t conDim = op.getDimension().getZExtValue();
464  SmallVector<Value> dynSizes;
465  if (!rtp.hasStaticShape()) {
466  ArrayRef<int64_t> rShape = rtp.getShape();
467  for (const auto &d : llvm::enumerate(rShape)) {
468  if (d.value() == ShapedType::kDynamicSize) {
469  Value v =
470  createOrFoldDimOp(rewriter, loc, op.getOperand(0), d.index());
471  rewriter.create<tensor::DimOp>(loc, op.getOperand(0), d.index());
472  for (const auto &opnd : op.getOperands().drop_front()) {
473  Value t = createOrFoldDimOp(rewriter, loc, opnd, d.index());
474  v = rewriter.create<arith::AddIOp>(loc, v, t);
475  }
476  dynSizes.push_back(v);
477  }
478  }
479  }
480 
481  // %t = concatenate %s1, %s2, %s3 {dim = 1}
482  // ==>
483  // %tmp = bufferization.alloc_tensor : unordered COO
484  // foreach in %s1 : insert d0, d1, %tmp
485  // foreach in %s2 : insert d0, d1 + size(s1), %tmp
486  // foreach in %s3 : insert d0, d1 + size(s1) + size(s2), %tmp
487  // %t = sparse_tensor.cast %tmp
488  auto cooTp = getUnorderedCOOFromType(rtp);
489  auto cooBuffer =
490  rewriter.create<AllocTensorOp>(loc, cooTp, dynSizes).getResult();
491  auto rank = rtp.getRank();
492  Value offset = constantIndex(rewriter, loc, 0);
493  ForeachOp foreachOp;
494  for (Value input : op.getInputs()) {
495  // Build a for op for each input tensor to append new values into the
496  // output tensor.
497  foreachOp = rewriter.create<ForeachOp>(
498  loc, input, cooBuffer,
499  [&](OpBuilder &builder, Location loc, ValueRange args, Value v,
500  ValueRange reduc) {
501  SmallVector<Value> indices;
502  for (int64_t i = 0; i < rank; i++) {
503  Value idx = args[i];
504  if (i == static_cast<int64_t>(conDim))
505  // transform coordinates on matching dim
506  idx = builder.create<arith::AddIOp>(loc, idx, offset);
507  indices.push_back(idx);
508  }
509  Value cond = genIsNonzero(rewriter, loc, v);
510  scf::IfOp ifOp = builder.create<scf::IfOp>(
511  loc, TypeRange(reduc.front().getType()), cond, /*else*/ true);
512  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
513  Value t = builder.create<InsertOp>(loc, v, reduc.front(), indices);
514  rewriter.create<scf::YieldOp>(loc, t);
515  rewriter.setInsertionPointToStart(&ifOp.getElseRegion().front());
516  rewriter.create<scf::YieldOp>(loc, reduc.front());
517  rewriter.setInsertionPointAfter(ifOp);
518  rewriter.create<sparse_tensor::YieldOp>(loc, ifOp.getResult(0));
519  });
520  // Accumulates the offset. Note that only static-shaped inputs are allowed
521  // by concatenate op verifier, which saves us from computing the offset
522  // dynamically.
523  auto d = input.getType().cast<RankedTensorType>().getShape()[conDim];
524  assert(!ShapedType::isDynamic(d));
525  offset = rewriter.create<arith::AddIOp>(loc, offset,
526  constantIndex(rewriter, loc, d));
527  cooBuffer = foreachOp.getResult(0);
528  }
529 
530  cooBuffer = rewriter.create<LoadOp>(loc, cooBuffer, true);

563  // conversion as follows:
564  // t = new sparse COO tensor
565  // fill t using src
566  // dst = convert t

568  // To fill the COO tensor from a dense tensor:
569  // for i1 in dim1
570  // ..
571  // for ik in dimk

573  // if val != 0
574  // t->add(val, [i1,..,ik], [p1,..,pk])

577  // for i in range(NNZ)
578  // val = values[i]

580  // t->add(val, [i1,..,ik], [p1,..,pk])

582  PatternRewriter &rewriter) const {

588  SmallVector<Value> dynSizes;

593  if (constOp.getValue().dyn_cast<SparseElementsAttr>()) {
594  fromSparseConst = true;
595  }
596  }
597

600  rewriter.create<AllocTensorOp>(loc, cooTp, dynSizes).getResult();
601  auto foreachOp = rewriter.create<ForeachOp>(
602  loc, src, cooBuffer,
603  [&](OpBuilder &builder, Location loc, ValueRange indices, Value v,
604  ValueRange reduc) {
605  Value input = reduc.front();

607  input = builder.create<InsertOp>(loc, v, input, indices);
608  } else {
609  Value cond = genIsNonzero(builder, loc, v);
610  auto ifOp = builder.create<scf::IfOp>(
611  loc, TypeRange(input.getType()), cond, /*else*/ true);
612  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
613  Value insert = builder.create<InsertOp>(loc, v, input, indices);
614  builder.create<scf::YieldOp>(loc, insert);
615  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
616  builder.create<scf::YieldOp>(loc, input);
617  builder.setInsertionPointAfter(ifOp);
618  input = ifOp.getResult(0);
619  }
620  builder.create<sparse_tensor::YieldOp>(loc, input);
621  });
622  rewriter.setInsertionPointAfter(op);
623  src = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
624  rewriter.replaceOpWithNewOp<ConvertOp>(op, dstTp, src);
625  rewriter.create<DeallocTensorOp>(loc, src);

666  // Handles sparse tensor to sparse tensor conversion as follows:
667  // if src is not COO
668  // construct a COO to represent the src
669  // sort the src COO
670  // foreach elemment in the sorted src COO
671  // insert element to dst
672  LogicalResult sparse2SparseRewrite(ConvertOp op,
673  PatternRewriter &rewriter) const {
674  Location loc = op->getLoc();
675  Value src = op.getSource();
676  RankedTensorType srcTp = src.getType().cast<RankedTensorType>();
677  RankedTensorType dstTp = op.getType().cast<RankedTensorType>();
678  SparseTensorEncodingAttr encDst = getSparseTensorEncoding(dstTp);
679 
680  SmallVector<Value> srcSizes;
681  sizesForTensor(rewriter, srcSizes, loc, srcTp, src);
682  Value tmpCoo = Value();
683  if (!isUniqueCOOType(srcTp)) {
684  // Construct a COO tensor from the src tensor.
685  // TODO: there may be cases for which more efficiently without
686  // going through an intermediate COO, such as cases that only change
687  // the overhead types.
688  SmallVector<Value> dynSrcSizes;
689  getDynamicSizes(srcTp, srcSizes, dynSrcSizes);
690  srcTp = getUnorderedCOOFromType(srcTp);
691  tmpCoo =
692  rewriter.create<AllocTensorOp>(loc, srcTp, dynSrcSizes).getResult();
693  auto foreachOp = rewriter.create<ForeachOp>(
694  loc, src, tmpCoo,
695  [&](OpBuilder &builder, Location loc, ValueRange args, Value v,
696  ValueRange reduc) {
697  // The resulting COO tensor has identity ordering.
698  auto t = builder.create<InsertOp>(loc, v, reduc.front(),
699  args.slice(0, srcTp.getRank()));
700  builder.create<sparse_tensor::YieldOp>(loc, t);
701  });
702  src = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
703  }
704 
705  // Sort the COO tensor so that its elements are ordered via increasing
706  // indices for the storage ordering of the dst tensor.
707  SparseTensorEncodingAttr encSrc = getSparseTensorEncoding(srcTp);
708  auto dynShape = {ShapedType::kDynamicSize};
709  auto indTp =
710  MemRefType::get(dynShape, getIndexOverheadType(rewriter, encSrc));
711  uint64_t rank = dstTp.getRank();
712  // Gather the indices-arrays in the dst tensor storage order.
713  SmallVector<Value> xs(rank, Value());
714  for (uint64_t i = 0; i < rank; i++) {
715  uint64_t orgDim = toOrigDim(encSrc, i);
716  xs[toStoredDim(encDst, orgDim)] = rewriter.create<ToIndicesOp>(
717  loc, indTp, src, rewriter.getIndexAttr(i));
718  }
719 
720  // Retrieve NNZ.
721  Value nnz = rewriter.create<NumberOfEntriesOp>(loc, src);
722  nnz =
723  rewriter.create<arith::IndexCastOp>(loc, rewriter.getIndexType(), nnz);
724 
725  // Retrieve the values-array.
726  auto valTp = MemRefType::get(dynShape, srcTp.getElementType());
727  Value y = rewriter.create<ToValuesOp>(loc, valTp, src);
728 
729  // Sort the COO tensor.
730  rewriter.create<SortOp>(loc, nnz, xs, ValueRange{y});
731 
732  // For each element in the COO tensor, insert the element to the dst tensor.
733  SmallVector<Value> dynDstSizes;
734  getDynamicSizes(dstTp, srcSizes, dynDstSizes);
735  Value dst =
736  rewriter.create<AllocTensorOp>(loc, dstTp, dynDstSizes).getResult();
737  SmallVector<Value> indices(srcTp.getRank(), Value());
738  auto foreachOp = rewriter.create<ForeachOp>(
739  loc, src, dst,
740  [&](OpBuilder &builder, Location loc, ValueRange args, Value v,
741  ValueRange reduc) {
742  for (int64_t i = 0, e = srcTp.getRank(); i < e; i++) {
743  uint64_t dim = toStoredDim(encDst, i);
744  indices[dim] = args[i];
745  }
746  auto t = builder.create<InsertOp>(loc, v, reduc.front(), indices);
747  builder.create<sparse_tensor::YieldOp>(loc, t);
748  });
749 
750  // Release the temporary COO if it is created. Note that tmpCoo is
751  // invalidated due to foreach and updated to src.
752  if (tmpCoo)
753  rewriter.create<DeallocTensorOp>(loc, src);
754 
755  // Directly replace op with dst results in bufferization error message
756  // "sparse tensor allocation should not escape function".
757  // As such, we insert a trivial tensor convert which will be removed by
758  // codegen.
759  rewriter.setInsertionPointAfter(op);
760  auto t = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
761  rewriter.replaceOpWithNewOp<ConvertOp>(op, dstTp, t);
762  return success();
763  }

766 /// Sparse rewriting rule for the foreach operator.
767 struct ForeachRewriter : public OpRewritePattern<ForeachOp> {
768 public:
769  using OpRewritePattern::OpRewritePattern;
770 
771  LogicalResult matchAndRewrite(ForeachOp op,
772  PatternRewriter &rewriter) const override {
773 
774  auto loc = op.getLoc();
775  Value input = op.getTensor();
776  SmallVector<Value> reduc = op.getInitArgs();
777  auto rtp = input.getType().cast<RankedTensorType>();
778  int64_t rank = rtp.getRank();
779 
780  // Special-case: for each over a sparse constant uses its own rewriting
781  // rule.
782  if (auto constOp = input.getDefiningOp<arith::ConstantOp>()) {
783  if (auto attr = constOp.getValue().dyn_cast<SparseElementsAttr>()) {
784  return genForeachOnSparseConstant(op, rewriter, attr);
785  }
786  }
787 
788  // Otherwise, use loop emitter to generate loops.
789  auto enc = getSparseTensorEncoding(rtp);
790 
791  // 1. Generates loop for the sparse input.
792  SparseTensorLoopEmitter loopEmitter(
793  ValueRange{input},
794  StringAttr::get(getContext(), ForeachOp::getOperationName()));
795  loopEmitter.initializeLoopEmit(rewriter, loc);
796  for (int64_t i = 0; i < rank; i++) {
797  // TODO: provide utility function for loop sequences that only contains
798  // one for loop?
799  loopEmitter.enterNewLoopSeq(rewriter, loc, 0, static_cast<size_t>(i));
800  // Note that reduc will be taken care of by loop emitter and get updated
801  // in place.
802  loopEmitter.enterLoopOverTensorAtDim(rewriter, loc, 0, i, reduc);
803  }
804 
805  SmallVector<Value> coords;
806  coords.reserve(rank);
807  loopEmitter.getCoordinateArray(coords);
808 
809  Value vals = loopEmitter.getValBuffer()[0];
810  Value pidx = loopEmitter.getPidxs()[0].back();
811  // Loads the value from sparse tensor using pointer index;
812  // loads the value from dense tensor using coordinate array.
813  Value val = enc ? rewriter.create<memref::LoadOp>(loc, vals, pidx)
814  : rewriter.create<memref::LoadOp>(loc, vals, coords);
815 
816  // 2. Inline the block in the foreach operator.
817  Block *srcBlock = op.getBody();
818 
819  // Remap coordinates.
820  SmallVector<Value> args;
821  for (int64_t i = 0; i < rank; i++) {
822  Value actual = coords[toStoredDim(enc, i)];
823  args.push_back(actual);
824  }
825  // Remap value.
826  args.push_back(val);
827  // Remap reduction variables.
828  args.append(reduc);
829 
830  // Remove sparse_tensor.yield.
831  SmallVector<Value> reducValue = srcBlock->getTerminator()->getOperands();
832  rewriter.eraseOp(srcBlock->getTerminator());
833 
834  // Inline body.
835  if (!reducValue.empty()) {
836  rewriter.mergeBlocks(srcBlock, rewriter.getBlock(), args);
837  } else {
838  // This is annoying, since scf.for inserts a implicit yield op when
839  // there is no reduction variable upon creation, in this case we need to
840  // merge the block *before* the yield op.
841  rewriter.mergeBlockBefore(srcBlock, &*rewriter.getInsertionPoint(), args);
842  }
843 
844  for (int64_t i = 0; i < rank; i++) {
845  // Link the reduction chain. Note that loop emitter update the reducValue
846  // in place.
847  loopEmitter.exitCurrentLoop(rewriter, loc, reducValue);
848  loopEmitter.exitCurrentLoopSeq();
849  }
850 
851  // Replace the foreach operator with the value returned by the outtermost
852  // for loop.
853  rewriter.replaceOp(op, reducValue);
854  return success();
855  }

910  Value c1 = constantIndex(rewriter, loc, 1);
911  Value nnz = createFuncCall(rewriter, loc, "getSparseTensorReaderNNZ",
912  {indexTp}, {reader}, EmitCInterface::Off)
913  .getResult(0);

915  // We assume only rank 2 tensors may have the isSymmetric flag set.
916  if (rank == 2) {
917  symmetric =

919  {rewriter.getI1Type()}, {reader}, EmitCInterface::Off)
920  .getResult(0);

926  scf::ForOp forOp = rewriter.create<scf::ForOp>(loc, c0, nnz, c1,

931  primaryTypeFunctionSuffix(eltTp)};

940  }

942  Value t = rewriter.create<InsertOp>(loc, v, forOp.getRegionIterArg(0),
943  indicesArray);

960  rewriter.setInsertionPointAfter(forOp);

965  createFuncCall(rewriter, loc, "delSparseTensorReader", {}, {reader},
966  EmitCInterface::Off);
967  cooBuffer = rewriter.create<LoadOp>(loc, cooBuffer, true);
968  Value newOp = rewriter.replaceOpWithNewOp<ConvertOp>(op, dstTp, cooBuffer);
969 
970  // Release the unordered COO tensor buffer.
971  rewriter.setInsertionPointAfterValue(newOp);
972  rewriter.create<DeallocTensorOp>(loc, cooBuffer);
973 
974  return success();
975  }
