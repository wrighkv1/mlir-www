50 /// Registers conversions into BufferizeTypeConverter
51 BufferizeTypeConverter::BufferizeTypeConverter() {
52  // Keep all types unchanged.
53  addConversion([](Type type) { return type; });
54  // Convert RankedTensorType to MemRefType.
55  addConversion([](RankedTensorType type) -> Type {
56  return MemRefType::get(type.getShape(), type.getElementType());
57  });
58  // Convert UnrankedTensorType to UnrankedMemRefType.
59  addConversion([](UnrankedTensorType type) -> Type {
60  return UnrankedMemRefType::get(type.getElementType(), 0);
61  });
62  addArgumentMaterialization(materializeToTensor);
63  addSourceMaterialization(materializeToTensor);
64  addTargetMaterialization([](OpBuilder &builder, BaseMemRefType type,
65  ValueRange inputs, Location loc) -> Value {
66  assert(inputs.size() == 1 && "expected exactly one input");
67 
68  if (auto inputType = inputs[0].getType().dyn_cast<MemRefType>()) {
69  // MemRef to MemRef cast.
70  assert(inputType != type && "expected different types");
71  // Unranked to ranked and ranked to unranked casts must be explicit.
72  auto rankedDestType = type.dyn_cast<MemRefType>();
73  if (!rankedDestType)
74  return nullptr;
75  FailureOr<Value> replacement =
76  castOrReallocMemRefValue(builder, inputs[0], rankedDestType);
77  if (failed(replacement))
78  return nullptr;
79  return *replacement;
80  }
81 
82  if (inputs[0].getType().isa<TensorType>()) {
83  // Tensor to MemRef cast.
84  return builder.create<bufferization::ToMemrefOp>(loc, type, inputs[0]);
85  }
86 
87  llvm_unreachable("only tensor/memref input types supported");
88  });
89 }

137  FinalizingBufferizePass> {

141  void runOnOperation() override {
142  auto func = getOperation();
143  auto *context = &getContext();
144 
145  BufferizeTypeConverter typeConverter;
146  RewritePatternSet patterns(context);
147  ConversionTarget target(*context);
148 
149  populateEliminateBufferizeMaterializationsPatterns(typeConverter, patterns);
150 
151  // If all result types are legal, and all block arguments are legal (ensured
152  // by func conversion above), then all types in the program are legal.
153  //
154  // We also check that the operand types are legal to avoid creating invalid
155  // IR. For example, this prevents
156  // populateEliminateBufferizeMaterializationsPatterns from updating the
157  // types of the operands to a return op without updating the enclosing
158  // function.
159  target.markUnknownOpDynamicallyLegal(
160  [&](Operation *op) { return typeConverter.isLegal(op); });
161 
162  if (failed(applyFullConversion(func, target, std::move(patterns))))
163  signalPassFailure();
164  }

205  opt.allowUnknownOps = allowUnknownOps;
206  opt.analysisFuzzerSeed = analysisFuzzerSeed;
207  opt.analysisHeuristic = parseHeuristicOption(analysisHeuristic);

236  // Filter may be specified via options.
237  if (this->dialectFilter.hasValue())
238  return llvm::is_contained(this->dialectFilter,

257  return;
258  }
259  }
260 
261  if (opt.testAnalysisOnly)
262  return;
263 
264  OpPassManager cleanupPipeline("builtin.module");
265  cleanupPipeline.addPass(createCanonicalizerPass());
266  cleanupPipeline.addPass(createCSEPass());

319 /// Return true if the given op has a tensor result or a tensor operand.
320 static bool hasTensorSemantics(Operation *op) {
321  if (auto funcOp = dyn_cast<FunctionOpInterface>(op)) {
322  bool hasTensorArg = any_of(funcOp.getArgumentTypes(), isaTensor);
323  bool hasTensorResult = any_of(funcOp.getResultTypes(), isaTensor);
324  return hasTensorArg || hasTensorResult;
325  }
326 
327  bool hasTensorResult = any_of(op->getResultTypes(), isaTensor);
328  bool hasTensorOperand = any_of(op->getOperandTypes(), isaTensor);
329  return hasTensorResult || hasTensorOperand;
330 }

333 /// A rewriter that keeps track of extra information during bufferization.
334 class BufferizationRewriter : public IRRewriter {
335 public:
336  BufferizationRewriter(MLIRContext *ctx, DenseSet<Operation *> &erasedOps,
337  DenseSet<Operation *> &toMemrefOps,
338  SmallVector<Operation *> &worklist,
339  const BufferizationOptions &options,
340  const OpFilter *opFilter)
341  : IRRewriter(ctx), erasedOps(erasedOps), toMemrefOps(toMemrefOps),
342  worklist(worklist), analysisState(options), opFilter(opFilter) {}
343 
344 protected:
345  void notifyOperationRemoved(Operation *op) override {
346  IRRewriter::notifyOperationRemoved(op);
347  erasedOps.insert(op);
348  // Erase if present.
349  toMemrefOps.erase(op);
350  }
351 
352  void notifyOperationInserted(Operation *op) override {
353  IRRewriter::notifyOperationInserted(op);
354  erasedOps.erase(op);
355 
356  // Keep track of to_memref ops.
357  if (isa<ToMemrefOp>(op)) {
358  toMemrefOps.insert(op);
359  return;
360  }
361 
362  // Skip to_tensor ops.
363  if (isa<ToTensorOp>(op))
364  return;
365 
366  // Skip non-tensor ops.
367  if (!hasTensorSemantics(op))
368  return;
369 
370  // Skip ops that are not allowed to be bufferized.
371  auto const &options = analysisState.getOptions();
372  if (!options.isOpAllowed(op) || (opFilter && !opFilter->isOpAllowed(op)))
373  return;
374 
375  // Add op to worklist.
376  worklist.push_back(op);
377  }
378 
379 private:
380  /// A set of all erased ops.
381  DenseSet<Operation *> &erasedOps;
382 
383  /// A set of all to_memref ops.
384  DenseSet<Operation *> &toMemrefOps;
385 
386  /// The worklist of ops to be bufferized.
387  SmallVector<Operation *> &worklist;
388 
389  /// The analysis state. Used for debug assertions and access to the
390  /// bufferization options.
391  const AnalysisState analysisState;
392 
393  /// An extra op filter for bufferization.
394  const OpFilter *opFilter;
395 };
396 } // namespace

398 LogicalResult bufferization::bufferizeOp(Operation *op,
399  const BufferizationOptions &options,
400  bool copyBeforeWrite,
401  const OpFilter *opFilter) {
402  if (copyBeforeWrite) {
403  AnalysisState state(options);
404  if (failed(insertTensorCopies(op, state)))
405  return failure();
406  }
407 
408  // Keep track of to_memref ops.
409  DenseSet<Operation *> toMemrefOps;
410  op->walk([&](ToMemrefOp toMemrefOp) { toMemrefOps.insert(toMemrefOp); });
411 
412  // Gather all bufferizable ops in top-to-bottom order.
413  //
414  // We should ideally know the exact memref type of all operands when
415  // bufferizing an op. (This is the case when bufferizing top-to-bottom.)
416  // Otherwise, we have to use a memref type with a fully dynamic layout map to
417  // avoid copies. We are currently missing patterns for layout maps to
418  // canonicalize away (or canonicalize to more precise layouts).
419  //
420  // FuncOps must be bufferized before their bodies, so add them to the worklist
421  // first.
422  SmallVector<Operation *> worklist;
423  op->walk([&](func::FuncOp funcOp) {
424  if (hasTensorSemantics(funcOp))
425  worklist.push_back(funcOp);
426  });
427  op->walk<WalkOrder::PostOrder>([&](Operation *op) {
428  if (hasTensorSemantics(op) && !isa<func::FuncOp>(op))
429  worklist.push_back(op);
430  });
431 
432  // Keep track of all erased ops.
433  DenseSet<Operation *> erasedOps;
434 
435  // Bufferize all ops.
436  BufferizationRewriter rewriter(op->getContext(), erasedOps, toMemrefOps,
437  worklist, options, opFilter);
438  for (unsigned i = 0; i < worklist.size(); ++i) {
439  Operation *nextOp = worklist[i];
440  // Skip ops that were erased.
441  if (erasedOps.contains(nextOp))
442  continue;
443  // Skip ops that are not bufferizable or not allowed.
444  auto bufferizableOp = options.dynCastBufferizableOp(nextOp);
445  if (!bufferizableOp)
446  continue;
447  if (opFilter && !opFilter->isOpAllowed(nextOp))
448  continue;
449  // Skip ops that no longer have tensor semantics.
450  if (!hasTensorSemantics(nextOp))
451  continue;
452  // Bufferize the op.
453  LLVM_DEBUG(llvm::dbgs()
454  << "//===-------------------------------------------===//\n"
455  << "IR after bufferizing: " << nextOp->getName() << "\n");
456  rewriter.setInsertionPoint(nextOp);
457  if (failed(bufferizableOp.bufferize(rewriter, options))) {
458  LLVM_DEBUG(llvm::dbgs()
459  << "failed to bufferize\n"
460  << "//===-------------------------------------------===//\n");
461  return nextOp->emitError("failed to bufferize op");
462  }
463  LLVM_DEBUG(llvm::dbgs()
464  << *op
465  << "\n//===-------------------------------------------===//\n");
466  }
467 
468  // Fold all to_memref(to_tensor(x)) pairs.
469  for (Operation *op : toMemrefOps) {
470  rewriter.setInsertionPoint(op);
471  (void)bufferization::foldToMemrefToTensorPair(rewriter,
472  cast<ToMemrefOp>(op));
473  }
474 
475  /// Check the result of bufferization. Return an error if an op was not
476  /// bufferized, unless partial bufferization is allowed.
477  if (options.allowUnknownOps)
478  return success();
479 
480  for (Operation *op : worklist) {
481  // Skip ops that are entirely gone.
482  if (erasedOps.contains(op))
483  continue;
484  // Ops that no longer have tensor semantics (because they were updated
485  // in-place) are allowed.
486  if (!hasTensorSemantics(op))
487  continue;
488  // Continue ops that are not allowed.
489  if (!options.isOpAllowed(op))
490  continue;
491  if (opFilter && !opFilter->isOpAllowed(op))
492  continue;
493  // Ops without any uses and no side effects will fold away.
494  if (op->getUses().empty() && isMemoryEffectFree(op))
495  continue;
496  // ToTensorOps/ToMemrefOps are allowed in the output.
497  if (isa<ToTensorOp, ToMemrefOp>(op))
498  continue;
499  return op->emitError("op was not bufferized");
500  }
501 
502  return success();
503 }
