9 // This file implements a light-weight runtime support library for
10 // manipulating sparse tensors from MLIR. More specifically, it provides
11 // C-API wrappers so that MLIR-generated code can call into the C++ runtime
12 // support library. The functionality provided in this library is meant
13 // to simplify benchmarking, testing, and debugging of MLIR code operating
14 // on sparse tensors. However, the provided functionality is **not**
15 // part of core MLIR itself.
16 //
17 // The following memory-resident sparse storage schemes are supported:
18 //
19 // (a) A coordinate scheme for temporarily storing and lexicographically
20 // sorting a sparse tensor by index (SparseTensorCOO).
21 //
22 // (b) A "one-size-fits-all" sparse tensor storage scheme defined by
23 // per-dimension sparse/dense annnotations together with a dimension
24 // ordering used by MLIR compiler-generated code (SparseTensorStorage).
25 //
26 // The following external formats are supported:
27 //
28 // (1) Matrix Market Exchange (MME): *.mtx
29 // https://math.nist.gov/MatrixMarket/formats.html
30 //
31 // (2) Formidable Repository of Open Sparse Tensors and Tools (FROSTT): *.tns
32 // http://frostt.io/tensors/file-formats.html
33 //
34 // Two public APIs are supported:
35 //
36 // (I) Methods operating on MLIR buffers (memrefs) to interact with sparse
37 // tensors. These methods should be used exclusively by MLIR
38 // compiler-generated code.
39 //
40 // (II) Methods that accept C-style data structures to interact with sparse
41 // tensors. These methods can be used by any external runtime that wants
42 // to interact with MLIR compiler-generated code.
43 //
44 // In both cases (I) and (II), the SparseTensorStorage format is externally
45 // only visible as an opaque pointer.
46 //
47 //===----------------------------------------------------------------------===//

74 /// Wrapper class to avoid memory leakage issues. The `SparseTensorCOO<V>`
75 /// class provides a standard C++ iterator interface, where the iterator
76 /// is implemented as per `std::vector`'s iterator. However, for MLIR's
77 /// usage we need to have an iterator which also holds onto the underlying
78 /// `SparseTensorCOO<V>` so that it can be freed whenever the iterator
79 /// is freed.
80 //
81 // We name this `SparseTensorIterator` rather than `SparseTensorCOOIterator`
82 // for future-proofing, since the use of `SparseTensorCOO` is an
83 // implementation detail that we eventually want to change (e.g., to
84 // use `SparseTensorEnumerator` directly, rather than constructing the
85 // intermediate `SparseTensorCOO` at all).
86 template <typename V>
87 class SparseTensorIterator final {
88 public:
89  /// This ctor requires `coo` to be a non-null pointer to a dynamically
90  /// allocated object, and takes ownership of that object. Therefore,
91  /// callers must not free the underlying COO object, since the iterator's
92  /// dtor will do so.
93  explicit SparseTensorIterator(const SparseTensorCOO<V> *coo)
94  : coo(coo), it(coo->begin()), end(coo->end()) {}
95 
96  ~SparseTensorIterator() { delete coo; }
97 
98  // Disable copy-ctor and copy-assignment, to prevent double-free.
99  SparseTensorIterator(const SparseTensorIterator<V> &) = delete;
100  SparseTensorIterator<V> &operator=(const SparseTensorIterator<V> &) = delete;
101 
102  /// Gets the next element. If there are no remaining elements, then
103  /// returns nullptr.
104  const Element<V> *getNext() { return it < end ? &*it++ : nullptr; }
105 
106 private:
107  const SparseTensorCOO<V> *const coo; // Owning pointer.
108  typename SparseTensorCOO<V>::const_iterator it;
109  const typename SparseTensorCOO<V>::const_iterator end;
110 };

124 /// Initializes sparse tensor from an external COO-flavored format.
125 /// The `rank` argument is both dimension-rank and level-rank, and the
126 /// `dim2lvl` argument must be a permutation.
127 /// Used by `IMPL_CONVERTTOMLIRSPARSETENSOR`.
128 //
129 // TODO: generalize beyond 64-bit indices.
130 template <typename V>
131 static SparseTensorStorage<uint64_t, uint64_t, V> *
132 toMLIRSparseTensor(uint64_t rank, uint64_t nse, const uint64_t *dimSizes,
133  const V *values, const uint64_t *dimIndices,
134  const uint64_t *dim2lvl, const DimLevelType *lvlTypes) {
135 #ifndef NDEBUG
136  // Verify that the sparsity values are supported.
137  // TODO: update this check to match what we actually support.
138  for (uint64_t i = 0; i < rank; ++i)
139  if (lvlTypes[i] != DimLevelType::Dense &&
140  lvlTypes[i] != DimLevelType::Compressed)
141  MLIR_SPARSETENSOR_FATAL("unsupported level type: %d\n",
142  static_cast<uint8_t>(lvlTypes[i]));
143 #endif
144  // Verify that `dim2lvl` is a permutation of `[0..(rank-1)]`.
145  // NOTE: The construction of `lvlSizes` and `lvl2dim` don't generalize
146  // to arbitrary `dim2lvl` mappings. Whereas constructing `lvlInd` from
147  // `dimInd` does (though the details would have to be updated, just
148  // like for `IMPL_ADDELT`).
149  detail::PermutationRef d2l(rank, dim2lvl);
150  // Convert external format to internal COO.
151  auto lvlSizes = d2l.pushforward(rank, dimSizes);
152  auto *lvlCOO = new SparseTensorCOO<V>(lvlSizes, nse);
153  std::vector<uint64_t> lvlInd(rank);
154  const uint64_t *dimInd = dimIndices;
155  for (uint64_t i = 0; i < nse; ++i) {
156  d2l.pushforward(rank, dimInd, lvlInd.data());
157  lvlCOO->add(lvlInd, values[i]);
158  dimInd += rank;
159  }
160  // Return sparse tensor storage format as opaque pointer.
161  auto lvl2dim = d2l.inverse();
162  auto *tensor = SparseTensorStorage<uint64_t, uint64_t, V>::newFromCOO(
163  rank, dimSizes, rank, lvlTypes, lvl2dim.data(), *lvlCOO);
164  delete lvlCOO;
165  return tensor;
166 }

168 /// Converts a sparse tensor to an external COO-flavored format.
169 /// Used by `IMPL_CONVERTFROMMLIRSPARSETENSOR`.
170 //
171 // TODO: Currently, values are copied from SparseTensorStorage to
172 // SparseTensorCOO, then to the output. We may want to reduce the number
173 // of copies.
174 //
175 // TODO: generalize beyond 64-bit indices, no dim ordering, all dimensions
176 // compressed
177 template <typename V>
178 static void
179 fromMLIRSparseTensor(const SparseTensorStorage<uint64_t, uint64_t, V> *tensor,
180  uint64_t *pRank, uint64_t *pNse, uint64_t **pShape,
181  V **pValues, uint64_t **pIndices) {
182  assert(tensor && "Received nullptr for tensor");
183  uint64_t dimRank = tensor->getDimRank();
184  const auto &dimSizes = tensor->getDimSizes();
185  std::vector<uint64_t> identityPerm(dimRank);
186  std::iota(identityPerm.begin(), identityPerm.end(), 0);
187  SparseTensorCOO<V> *coo =
188  tensor->toCOO(dimRank, dimSizes.data(), dimRank, identityPerm.data());
189 
190  const std::vector<Element<V>> &elements = coo->getElements();
191  uint64_t nse = elements.size();
192 
193  const auto &cooSizes = coo->getDimSizes();
194  assert(cooSizes.size() == dimRank && "Rank mismatch");
195  uint64_t *shape = new uint64_t[dimRank];
196  std::memcpy((void *)shape, (const void *)cooSizes.data(),
197  sizeof(uint64_t) * dimRank);
198 
199  V *values = new V[nse];
200  uint64_t *indices = new uint64_t[dimRank * nse];
201 
202  for (uint64_t i = 0, base = 0; i < nse; ++i) {
203  values[i] = elements[i].value;
204  for (uint64_t d = 0; d < dimRank; ++d)
205  indices[base + d] = elements[i].indices[d];
206  base += dimRank;
207  }
208 
209  delete coo;
210  *pRank = dimRank;
211  *pNse = nse;
212  *pShape = shape;
213  *pValues = values;
214  *pIndices = indices;
215 }

223 // We shouldn't need to use `detail::safelyEQ` here since the `1` is a literal.
224 #define ASSERT_NO_STRIDE(MEMREF) \
225  do { \
226  assert((MEMREF) && "Memref is nullptr"); \
227  assert(((MEMREF)->strides[0] == 1) && "Memref has non-trivial stride"); \
228  } while (false)
229 
230 // All our functions use `uint64_t` for ranks, but `StridedMemRefType::sizes`
231 // uses `int64_t` on some platforms. So we explicitly cast this lookup to
232 // ensure we get a consistent type, and we use `checkOverflowCast` rather
233 // than `static_cast` just to be extremely sure that the casting can't
234 // go awry. (The cast should aways be safe since (1) sizes should never
235 // be negative, and (2) the maximum `int64_t` is smaller than the maximum
236 // `uint64_t`. But it's better to be safe than sorry.)
237 #define MEMREF_GET_USIZE(MEMREF) \
238  detail::checkOverflowCast<uint64_t>((MEMREF)->sizes[0])
239 
240 #define ASSERT_USIZE_EQ(MEMREF, SZ) \
241  assert(detail::safelyEQ(MEMREF_GET_USIZE(MEMREF), (SZ)) && \
242  "Memref size mismatch")
243 
244 #define MEMREF_GET_PAYLOAD(MEMREF) ((MEMREF)->data + (MEMREF)->offset)
245 
246 // We make this a function rather than a macro mainly for type safety
247 // reasons. This function does not modify the vector, but it cannot
248 // be marked `const` because it is stored into the non-`const` memref.
249 template <typename T>
250 static void vectorToMemref(std::vector<T> &v, StridedMemRefType<T, 1> &ref) {
251  ref.basePtr = ref.data = v.data();
252  ref.offset = 0;
253  using SizeT = typename std::remove_reference_t<decltype(ref.sizes[0])>;
254  ref.sizes[0] = detail::checkOverflowCast<SizeT>(v.size());
255  ref.strides[0] = 1;
256 }

269 #define CASE(p, i, v, P, I, V) \
270  if (ptrTp == (p) && indTp == (i) && valTp == (v)) { \
271  switch (action) { \
272  case Action::kEmpty: \
273  return SparseTensorStorage<P, I, V>::newEmpty( \
274  dimRank, dimSizes, lvlRank, lvlSizes, lvlTypes, lvl2dim); \
275  case Action::kFromFile: { \
276  char *filename = static_cast<char *>(ptr); \
277  return openSparseTensor<P, I, V>(dimRank, dimSizes, lvlRank, lvlTypes, \
278  lvl2dim, dim2lvl, filename, v); \
279  } \
280  case Action::kFromCOO: { \
281  assert(ptr && "Received nullptr for SparseTensorCOO object"); \
282  auto &coo = *static_cast<SparseTensorCOO<V> *>(ptr); \
283  return SparseTensorStorage<P, I, V>::newFromCOO( \
284  dimRank, dimSizes, lvlRank, lvlTypes, lvl2dim, coo); \
285  } \
286  case Action::kSparseToSparse: { \
287  assert(ptr && "Received nullptr for SparseTensorStorage object"); \
288  auto &tensor = *static_cast<SparseTensorStorageBase *>(ptr); \
289  return SparseTensorStorage<P, I, V>::newFromSparseTensor( \
290  dimRank, dimSizes, lvlRank, lvlSizes, lvlTypes, lvl2dim, dimRank, \
291  dim2lvl, tensor); \
292  } \
293  case Action::kEmptyCOO: \
294  return new SparseTensorCOO<V>(lvlRank, lvlSizes); \
295  case Action::kToCOO: { \
296  assert(ptr && "Received nullptr for SparseTensorStorage object"); \
297  auto &tensor = *static_cast<SparseTensorStorage<P, I, V> *>(ptr); \
298  return tensor.toCOO(lvlRank, lvlSizes, dimRank, dim2lvl); \
299  } \
300  case Action::kToIterator: { \
301  assert(ptr && "Received nullptr for SparseTensorStorage object"); \
302  auto &tensor = *static_cast<SparseTensorStorage<P, I, V> *>(ptr); \
303  auto *coo = tensor.toCOO(lvlRank, lvlSizes, dimRank, dim2lvl); \
304  return new SparseTensorIterator<V>(coo); \
305  } \
306  } \
307  MLIR_SPARSETENSOR_FATAL("unknown action: %d\n", \
308  static_cast<uint32_t>(action)); \
309  }
