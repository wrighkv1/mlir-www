26 /// This class provides a utility for replacing attributes/types, and their sub
27 /// elements. Multiple replacement functions may be registered.
28 class AttrTypeReplacer {
29 public:

31  // Application
32  //===--------------------------------------------------------------------===//

34  /// Replace the elements within the given operation. If `replaceAttrs` is
35  /// true, this updates the attribute dictionary of the operation. If
36  /// `replaceLocs` is true, this also updates its location, and the locations
37  /// of any nested block arguments. If `replaceTypes` is true, this also
38  /// updates the result types of the operation, and the types of any nested
39  /// block arguments.
40  void replaceElementsIn(Operation *op, bool replaceAttrs = true,
41  bool replaceLocs = false, bool replaceTypes = false);
42 
43  /// Replace the given attribute/type, and recursively replace any sub
44  /// elements. Returns either the new attribute/type, or nullptr in the case of
45  /// failure.
46  Attribute replace(Attribute attr);
47  Type replace(Type type);
48 
49  //===--------------------------------------------------------------------===//
50  // Registration
51  //===--------------------------------------------------------------------===//
52 
53  /// A replacement mapping function, which returns either None (to signal the
54  /// element wasn't handled), or a pair of the replacement element and a
55  /// WalkResult.
56  template <typename T>
57  using ReplaceFnResult = Optional<std::pair<T, WalkResult>>;
58  template <typename T>
59  using ReplaceFn = std::function<ReplaceFnResult<T>(T)>;
60 
61  /// Register a replacement function for mapping a given attribute or type. A
62  /// replacement function must be convertible to any of the following
63  /// forms(where `T` is a class derived from `Type` or `Attribute`, and `BaseT`
64  /// is either `Type` or `Attribute` respectively):
65  ///
66  /// * Optional<BaseT>(T)
67  /// - This either returns a valid Attribute/Type in the case of success,
68  /// nullptr in the case of failure, or `llvm::None` to signify that
69  /// additional replacement functions may be applied (i.e. this function
70  /// doesn't handle that instance).
71  ///
72  /// * Optional<std::pair<BaseT, WalkResult>>(T)
73  /// - Similar to the above, but also allows specifying a WalkResult to
74  /// control the replacement of sub elements of a given attribute or
75  /// type. Returning a `skip` result, for example, will not recursively
76  /// process the resultant attribute or type value.
77  ///
78  /// Note: When replacing, the mostly recently added replacement functions will
79  /// be invoked first.
80  void addReplacement(ReplaceFn<Attribute> fn) {
81  attrReplacementFns.emplace_back(std::move(fn));
82  }
83  void addReplacement(ReplaceFn<Type> fn) {
84  typeReplacementFns.push_back(std::move(fn));
85  }
86 
87  /// Register a replacement function that doesn't match the default signature,
88  /// either because it uses a derived parameter type, or it uses a simplified
89  /// result type.
90  template <typename FnT,
91  typename T = typename llvm::function_traits<
92  std::decay_t<FnT>>::template arg_t<0>,
93  typename BaseT = std::conditional_t<std::is_base_of_v<Attribute, T>,
94  Attribute, Type>,
95  typename ResultT = std::invoke_result_t<FnT, T>>
96  std::enable_if_t<!std::is_same_v<T, BaseT> ||
97  !std::is_convertible_v<ResultT, ReplaceFnResult<BaseT>>>
98  addReplacement(FnT &&callback) {
99  addReplacement([callback = std::forward<FnT>(callback)](
100  BaseT base) -> ReplaceFnResult<BaseT> {
101  if (auto derived = dyn_cast<T>(base)) {
102  if constexpr (std::is_convertible_v<ResultT, Optional<BaseT>>) {
103  Optional<BaseT> result = callback(derived);
104  return result ? std::make_pair(*result, WalkResult::advance())
105  : ReplaceFnResult<BaseT>();
106  } else {
107  return callback(derived);
108  }
109  }
110  return ReplaceFnResult<BaseT>();
111  });
112  }
113 
114 private:
115  /// Internal implementation of the `replace` methods above.
116  template <typename InterfaceT, typename ReplaceFns, typename T>
117  T replaceImpl(T element, ReplaceFns &replaceFns, DenseMap<T, T> &map);
118 
119  /// Replace the sub elements of the given interface.
120  template <typename InterfaceT, typename T = typename InterfaceT::ValueType>
121  T replaceSubElements(InterfaceT interface, DenseMap<T, T> &interfaceMap);
122 
123  /// The set of replacement functions that map sub elements.
124  std::vector<ReplaceFn<Attribute>> attrReplacementFns;
125  std::vector<ReplaceFn<Type>> typeReplacementFns;
126 
127  /// The set of cached mappings for attributes/types.
128  DenseMap<Attribute, Attribute> attrMap;
129  DenseMap<Type, Type> typeMap;
130 };

136 /// This class is used by AttrTypeSubElementHandler instances to walking sub
137 /// attributes and types.
138 class AttrTypeSubElementWalker {
139 public:
140  AttrTypeSubElementWalker(function_ref<void(Attribute)> walkAttrsFn,
141  function_ref<void(Type)> walkTypesFn)
142  : walkAttrsFn(walkAttrsFn), walkTypesFn(walkTypesFn) {}
143 
144  /// Walk an attribute.
145  void walk(Attribute element) {
146  if (element)
147  walkAttrsFn(element);
148  }
149  /// Walk a type.
150  void walk(Type element) {
151  if (element)
152  walkTypesFn(element);
153  }
154  /// Walk a range of attributes or types.
155  template <typename RangeT>
156  void walkRange(RangeT &&elements) {
157  for (auto element : elements)
158  walk(element);
159  }
160 
161 private:
162  function_ref<void(Attribute)> walkAttrsFn;
163  function_ref<void(Type)> walkTypesFn;
164 };

166 /// This class is used by AttrTypeSubElementHandler instances to process sub
167 /// element replacements.
168 template <typename T>
169 class AttrTypeSubElementReplacements {
170 public:
171  AttrTypeSubElementReplacements(ArrayRef<T> repls) : repls(repls) {}
172 
173  /// Take the first N replacements as an ArrayRef, dropping them from
174  /// this replacement list.
175  ArrayRef<T> take_front(unsigned n) {
176  ArrayRef<T> elements = repls.take_front(n);
177  repls = repls.drop_front(n);
178  return elements;
179  }
180 
181 private:
182  /// The current set of replacements.
183  ArrayRef<T> repls;
184 };

188 /// This class provides support for interacting with the
189 /// SubElementInterfaces for different types of parameters. An
190 /// implementation of this class should be provided for any parameter class
191 /// that may contain an attribute or type. There are two main methods of
192 /// this class that need to be implemented:
193 ///
194 /// - walk
195 ///
196 /// This method should traverse into any sub elements of the parameter
197 /// using the provided walker, or by invoking handlers for sub-types.
198 ///
199 /// - replace
200 ///
201 /// This method should extract any necessary sub elements using the
202 /// provided replacer, or by invoking handlers for sub-types. The new
203 /// post-replacement parameter value should be returned.
204 ///
205 template <typename T, typename Enable = void>
206 struct AttrTypeSubElementHandler {
207  /// Default walk implementation that does nothing.
208  static inline void walk(const T &param, AttrTypeSubElementWalker &walker) {}
209 
210  /// Default replace implementation just forwards the parameter.
211  template <typename ParamT>
212  static inline decltype(auto) replace(ParamT &&param,
213  AttrSubElementReplacements &attrRepls,
214  TypeSubElementReplacements &typeRepls) {
215  return std::forward<ParamT>(param);
216  }
217 
218  /// Tag indicating that this handler does not support sub-elements.
219  using DefaultHandlerTag = void;
220 };

223 namespace detail {

225 using has_default_sub_element_handler_t = decltype(T::DefaultHandlerTag);
226 } // namespace detail
227 template <typename... Ts>
228 inline constexpr bool has_sub_attr_or_type_v =
229  (!llvm::is_detected<detail::has_default_sub_element_handler_t, Ts>::value ||
230  ...);

232 /// Implementation for derived Attributes and Types.
233 template <typename T>
234 struct AttrTypeSubElementHandler<
235  T, std::enable_if_t<std::is_base_of_v<Attribute, T> ||
236  std::is_base_of_v<Type, T>>> {
237  static void walk(T param, AttrTypeSubElementWalker &walker) {
238  walker.walk(param);
239  }
240  static T replace(T param, AttrSubElementReplacements &attrRepls,
241  TypeSubElementReplacements &typeRepls) {
242  if (!param)
243  return T();
244  if constexpr (std::is_base_of_v<Attribute, T>) {
245  return cast<T>(attrRepls.take_front(1)[0]);
246  } else {
247  return cast<T>(typeRepls.take_front(1)[0]);
248  }
249  }
250 };
251 template <>
252 struct AttrTypeSubElementHandler<NamedAttribute> {
253  template <typename T>
254  static void walk(T param, AttrTypeSubElementWalker &walker) {
255  walker.walk(param.getName());
256  walker.walk(param.getValue());
257  }
258  template <typename T>
259  static T replace(T param, AttrSubElementReplacements &attrRepls,
260  TypeSubElementReplacements &typeRepls) {
261  ArrayRef<Attribute> paramRepls = attrRepls.take_front(2);
262  return T(cast<decltype(param.getName())>(paramRepls[0]), paramRepls[1]);
263  }
264 };
265 /// Implementation for derived ArrayRef.
266 template <typename T>
267 struct AttrTypeSubElementHandler<ArrayRef<T>,
268  std::enable_if_t<has_sub_attr_or_type_v<T>>> {
269  using EltHandler = AttrTypeSubElementHandler<T>;
270 
271  static void walk(ArrayRef<T> param, AttrTypeSubElementWalker &walker) {
272  for (const T &subElement : param)
273  EltHandler::walk(subElement, walker);
274  }
275  static auto replace(ArrayRef<T> param, AttrSubElementReplacements &attrRepls,
276  TypeSubElementReplacements &typeRepls) {
277  // Normal attributes/types can extract using the replacer directly.
278  if constexpr (std::is_base_of_v<Attribute, T> &&
279  sizeof(T) == sizeof(Attribute)) {
280  ArrayRef<Attribute> attrs = attrRepls.take_front(param.size());
281  return ArrayRef<T>((const T *)attrs.data(), attrs.size());
282  } else if constexpr (std::is_base_of_v<Type, T> &&
283  sizeof(T) == sizeof(Type)) {
284  ArrayRef<Type> types = typeRepls.take_front(param.size());
285  return ArrayRef<T>((const T *)types.data(), types.size());
286  } else {
287  // Otherwise, we need to allocate storage for the new elements.
288  SmallVector<T> newElements;
289  for (const T &element : param)
290  newElements.emplace_back(
291  EltHandler::replace(element, attrRepls, typeRepls));
292  return newElements;
293  }
294  }
295 };
296 /// Implementation for Tuple.
297 template <typename... Ts>
298 struct AttrTypeSubElementHandler<
299  std::tuple<Ts...>, std::enable_if_t<has_sub_attr_or_type_v<Ts...>>> {
300  static void walk(const std::tuple<Ts...> &param,
301  AttrTypeSubElementWalker &walker) {
302  std::apply(
303  [&](const Ts &...params) {

309  AttrSubElementReplacements &attrRepls,

316  typeRepls)...};
317  },
318  param);
319  }

330 /// This function provides the underlying implementation for the
331 /// SubElementInterface walk method, using the key type of the derived
332 /// attribute/type to interact with the individual parameters.
333 template <typename T>
334 void walkImmediateSubElementsImpl(T derived,
335  function_ref<void(Attribute)> walkAttrsFn,
336  function_ref<void(Type)> walkTypesFn) {
337  auto key = static_cast<typename T::ImplType *>(derived.getImpl())->getAsKey();
338 
339  // If we don't have any sub-elements, there is nothing to do.
340  if constexpr (!has_sub_attr_or_type_v<decltype(key)>) {
341  return;
342  } else {
343  AttrTypeSubElementWalker walker(walkAttrsFn, walkTypesFn);
344  AttrTypeSubElementHandler<decltype(key)>::walk(key, walker);
345  }
346 }

348 /// This function invokes the proper `get` method for a type `T` with the given
349 /// values.
350 template <typename T, typename... Ts>
351 T constructSubElementReplacement(MLIRContext *ctx, Ts &&...params) {
352  // Prefer a direct `get` method if one exists.
353  if constexpr (llvm::is_detected<has_get_method, T, Ts...>::value) {
354  (void)ctx;
355  return T::get(std::forward<Ts>(params)...);
356  } else if constexpr (llvm::is_detected<has_get_method, T, MLIRContext *,
357  Ts...>::value) {
358  return T::get(ctx, std::forward<Ts>(params)...);
359  } else {
360  // Otherwise, pass to the base get.
361  return T::Base::get(ctx, std::forward<Ts>(params)...);
362  }
363 }

365 /// This function provides the underlying implementation for the
366 /// SubElementInterface replace method, using the key type of the derived
367 /// attribute/type to interact with the individual parameters.
368 template <typename T>
369 T replaceImmediateSubElementsImpl(T derived, ArrayRef<Attribute> &replAttrs,
370  ArrayRef<Type> &replTypes) {
371  auto key = static_cast<typename T::ImplType *>(derived.getImpl())->getAsKey();
372 
373  // If we don't have any sub-elements, we can just return the original.
374  if constexpr (!has_sub_attr_or_type_v<decltype(key)>) {
375  return derived;
376 
377  // Otherwise, we need to replace any necessary sub-elements.
378  } else {
379  AttrSubElementReplacements attrRepls(replAttrs);
380  TypeSubElementReplacements typeRepls(replTypes);
381  auto newKey = AttrTypeSubElementHandler<decltype(key)>::replace(
382  key, attrRepls, typeRepls);
383  if constexpr (is_tuple<decltype(key)>::value) {
384  return std::apply(
385  [&](auto &&...params) {
386  return constructSubElementReplacement<T>(
387  derived.getContext(),
388  std::forward<decltype(params)>(params)...);
389  },
390  newKey);
391  } else {
392  return constructSubElementReplacement<T>(derived.getContext(), newKey);
393  }
394  }
