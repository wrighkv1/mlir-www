143 DataLayoutSpecInterface ModuleOp::getDataLayoutSpec() {
144  // Take the first and only (if present) attribute that implements the
145  // interface. This needs a linear search, but is called only once per data
146  // layout object construction that is used for repeated queries.
147  for (NamedAttribute attr : getOperation()->getAttrs())
148  if (auto spec = attr.getValue().dyn_cast<DataLayoutSpecInterface>())
149  return spec;
150  return {};

153 LogicalResult ModuleOp::verify() {
154  // Check that none of the attributes are non-dialect attributes, except for
155  // the symbol related attributes.
156  for (auto attr : (*this)->getAttrs()) {
157  if (!attr.getName().strref().contains('.') &&
158  !llvm::is_contained(
159  ArrayRef<StringRef>{mlir::SymbolTable::getSymbolAttrName(),
160  mlir::SymbolTable::getVisibilityAttrName()},
161  attr.getName().strref()))
162  return emitOpError() << "can only contain attributes with "
163  "dialect-prefixed names, found: '"
164  << attr.getName().getValue() << "'";
165  }
166 
167  // Check that there is at most one data layout spec attribute.
168  StringRef layoutSpecAttrName;
169  DataLayoutSpecInterface layoutSpec;
170  for (const NamedAttribute &na : (*this)->getAttrs()) {
171  if (auto spec = na.getValue().dyn_cast<DataLayoutSpecInterface>()) {
172  if (layoutSpec) {
173  InFlightDiagnostic diag =
174  emitOpError() << "expects at most one data layout attribute";
175  diag.attachNote() << "'" << layoutSpecAttrName
176  << "' is a data layout attribute";
177  diag.attachNote() << "'" << na.getName().getValue()
178  << "' is a data layout attribute";
179  }
180  layoutSpecAttrName = na.getName().strref();
181  layoutSpec = spec;
182  }
183  }
184 
185  return success();
186 }

192 LogicalResult
193 UnrealizedConversionCastOp::fold(ArrayRef<Attribute> attrOperands,
194  SmallVectorImpl<OpFoldResult> &foldResults) {
195  OperandRange operands = getInputs();
196  ResultRange results = getOutputs();
197 
198  if (operands.getType() == results.getType()) {
199  foldResults.append(operands.begin(), operands.end());
200  return success();
201  }
202 
203  if (operands.empty())
204  return failure();
205 
206  // Check that the input is a cast with results that all feed into this
207  // operation, and operand types that directly match the result types of this
208  // operation.
209  Value firstInput = operands.front();
210  auto inputOp = firstInput.getDefiningOp<UnrealizedConversionCastOp>();
211  if (!inputOp || inputOp.getResults() != operands ||
212  inputOp.getOperandTypes() != results.getTypes())
213  return failure();
214 
215  // If everything matches up, we can fold the passthrough.
216  foldResults.append(inputOp->operand_begin(), inputOp->operand_end());
217  return success();
218 }
