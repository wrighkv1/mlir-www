121 static OwningOpRef<Operation *> parseMLIRInput(StringRef inputFilename,
122  bool insertImplicitModule,
123  MLIRContext *context) {
124  // Set up the input file.
125  std::string errorMessage;
126  auto file = openInputFile(inputFilename, &errorMessage);
127  if (!file) {
128  llvm::errs() << errorMessage << "\n";
129  return nullptr;
130  }
131 
132  llvm::SourceMgr sourceMgr;
133  sourceMgr.AddNewSourceBuffer(std::move(file), SMLoc());
134  OwningOpRef<Operation *> module =
135  parseSourceFileForTool(sourceMgr, context, insertImplicitModule);
136  if (!module)
137  return nullptr;
138  if (!module.get()->hasTrait<OpTrait::SymbolTable>()) {
139  llvm::errs() << "Error: top-level op must be a symbol table.\n";
140  return nullptr;
141  }
142  return module;
143 }

150 static Optional<unsigned> getCommandLineOptLevel(Options &options) {
151  Optional<unsigned> optLevel;
152  SmallVector<std::reference_wrapper<llvm::cl::opt<bool>>, 4> optFlags{
153  options.optO0, options.optO1, options.optO2, options.optO3};
154 
155  // Determine if there is an optimization flag present.
156  for (unsigned j = 0; j < 4; ++j) {
157  auto &flag = optFlags[j].get();
158  if (flag) {
159  optLevel = j;
160  break;
161  }
162  }
163  return optLevel;
164 }

166 // JIT-compile the given module and run "entryPoint" with "args" as arguments.
167 static Error compileAndExecute(Options &options, Operation *module,
168  StringRef entryPoint,
169  CompileAndExecuteConfig config, void **args) {
170  Optional<llvm::CodeGenOpt::Level> jitCodeGenOptLevel;
171  if (auto clOptLevel = getCommandLineOptLevel(options))
172  jitCodeGenOptLevel = static_cast<llvm::CodeGenOpt::Level>(*clOptLevel);
173 
174  // If shared library implements custom mlir-runner library init and destroy
175  // functions, we'll use them to register the library with the execution
176  // engine. Otherwise we'll pass library directly to the execution engine.
177  SmallVector<SmallString<256>, 4> libPaths;
178 
179  // Use absolute library path so that gdb can find the symbol table.
180  transform(
181  options.clSharedLibs, std::back_inserter(libPaths),
182  [](std::string libPath) {
183  SmallString<256> absPath(libPath.begin(), libPath.end());
184  cantFail(llvm::errorCodeToError(llvm::sys::fs::make_absolute(absPath)));
185  return absPath;
186  });
187 
188  // Libraries that we'll pass to the ExecutionEngine for loading.
189  SmallVector<StringRef, 4> executionEngineLibs;
190 
191  using MlirRunnerInitFn = void (*)(llvm::StringMap<void *> &);
192  using MlirRunnerDestroyFn = void (*)();
193 
194  llvm::StringMap<void *> exportSymbols;
195  SmallVector<MlirRunnerDestroyFn> destroyFns;
196 
197  // Handle libraries that do support mlir-runner init/destroy callbacks.
198  for (auto &libPath : libPaths) {
199  auto lib = llvm::sys::DynamicLibrary::getPermanentLibrary(libPath.c_str());
200  void *initSym = lib.getAddressOfSymbol("__mlir_runner_init");
201  void *destroySim = lib.getAddressOfSymbol("__mlir_runner_destroy");
202 
203  // Library does not support mlir runner, load it with ExecutionEngine.
204  if (!initSym || !destroySim) {
205  executionEngineLibs.push_back(libPath);
206  continue;
207  }
208 
209  auto initFn = reinterpret_cast<MlirRunnerInitFn>(initSym);
210  initFn(exportSymbols);
211 
212  auto destroyFn = reinterpret_cast<MlirRunnerDestroyFn>(destroySim);
213  destroyFns.push_back(destroyFn);
214  }
215 
216  // Build a runtime symbol map from the config and exported symbols.
217  auto runtimeSymbolMap = [&](llvm::orc::MangleAndInterner interner) {
218  auto symbolMap = config.runtimeSymbolMap ? config.runtimeSymbolMap(interner)
219  : llvm::orc::SymbolMap();
220  for (auto &exportSymbol : exportSymbols)
221  symbolMap[interner(exportSymbol.getKey())] =
222  llvm::JITEvaluatedSymbol::fromPointer(exportSymbol.getValue());
223  return symbolMap;
224  };
225 
226  mlir::ExecutionEngineOptions engineOptions;
227  engineOptions.llvmModuleBuilder = config.llvmModuleBuilder;
228  if (config.transformer)
229  engineOptions.transformer = config.transformer;
230  engineOptions.jitCodeGenOptLevel = jitCodeGenOptLevel;
231  engineOptions.sharedLibPaths = executionEngineLibs;
232  engineOptions.enableObjectDump = true;
233  auto expectedEngine = mlir::ExecutionEngine::create(module, engineOptions);
234  if (!expectedEngine)
235  return expectedEngine.takeError();
236 
237  auto engine = std::move(*expectedEngine);
238  engine->registerSymbols(runtimeSymbolMap);
239 
240  auto expectedFPtr = engine->lookupPacked(entryPoint);
241  if (!expectedFPtr)
242  return expectedFPtr.takeError();
243 
244  if (options.dumpObjectFile)
245  engine->dumpToObjectFile(options.objectFilename.empty()
246  ? options.inputFilename + ".o"
247  : options.objectFilename);
248 
249  void (*fptr)(void **) = *expectedFPtr;
250  (*fptr)(args);
251 
252  // Run all dynamic library destroy callbacks to prepare for the shutdown.
253  for (MlirRunnerDestroyFn destroy : destroyFns)
254  destroy();
255 
256  return Error::success();
257 }

259 static Error compileAndExecuteVoidFunction(Options &options, Operation *module,
260  StringRef entryPoint,
261  CompileAndExecuteConfig config) {
262  auto mainFunction = dyn_cast_or_null<LLVM::LLVMFuncOp>(
263  SymbolTable::lookupSymbolIn(module, entryPoint));
264  if (!mainFunction || mainFunction.empty())
265  return makeStringError("entry point not found");
266  void *empty = nullptr;
267  return compileAndExecute(options, module, entryPoint, std::move(config),
268  &empty);
269 }

302 template <typename Type>
303 Error compileAndExecuteSingleReturnFunction(Options &options, Operation *module,
304  StringRef entryPoint,
305  CompileAndExecuteConfig config) {
306  auto mainFunction = dyn_cast_or_null<LLVM::LLVMFuncOp>(
307  SymbolTable::lookupSymbolIn(module, entryPoint));
308  if (!mainFunction || mainFunction.isExternal())
309  return makeStringError("entry point not found");
310 
311  if (mainFunction.getFunctionType()
312  .cast<LLVM::LLVMFunctionType>()
313  .getNumParams() != 0)
314  return makeStringError("function inputs not supported");
315 
316  if (Error error = checkCompatibleReturnType<Type>(mainFunction))
317  return error;
318 
319  Type res;
320  struct {
321  void *data;
322  } data;
323  data.data = &res;
324  if (auto error = compileAndExecute(options, module, entryPoint,
325  std::move(config), (void **)&data))
326  return error;
327 
328  // Intentional printing of the output so we can test.
329  llvm::outs() << res << '\n';
330 
331  return Error::success();
332 }

334 /// Entry point for all CPU runners. Expects the common argc/argv arguments for
335 /// standard C++ main functions.
336 int mlir::JitRunnerMain(int argc, char **argv, const DialectRegistry &registry,
337  JitRunnerConfig config) {
338  // Create the options struct containing the command line options for the
339  // runner. This must come before the command line options are parsed.
340  Options options;
341  llvm::cl::ParseCommandLineOptions(argc, argv, "MLIR CPU execution driver\n");
342 
343  if (options.hostSupportsJit) {
344  auto j = llvm::orc::LLJITBuilder().create();
345  if (j)
346  llvm::outs() << "true\n";
347  else {
348  llvm::consumeError(j.takeError());
349  llvm::outs() << "false\n";
350  }
351  return 0;
352  }
353 
354  Optional<unsigned> optLevel = getCommandLineOptLevel(options);
355  SmallVector<std::reference_wrapper<llvm::cl::opt<bool>>, 4> optFlags{
356  options.optO0, options.optO1, options.optO2, options.optO3};
357 
358  MLIRContext context(registry);
359 
360  auto m = parseMLIRInput(options.inputFilename, !options.noImplicitModule,
361  &context);
362  if (!m) {
363  llvm::errs() << "could not parse the input IR\n";
364  return 1;
365  }
366 
367  JitRunnerOptions runnerOptions{options.mainFuncName, options.mainFuncType};
368  if (config.mlirTransformer)
369  if (failed(config.mlirTransformer(m.get(), runnerOptions)))
370  return EXIT_FAILURE;
371 
372  auto tmBuilderOrError = llvm::orc::JITTargetMachineBuilder::detectHost();
373  if (!tmBuilderOrError) {
374  llvm::errs() << "Failed to create a JITTargetMachineBuilder for the host\n";
375  return EXIT_FAILURE;
376  }
377  auto tmOrError = tmBuilderOrError->createTargetMachine();
378  if (!tmOrError) {
379  llvm::errs() << "Failed to create a TargetMachine for the host\n";
380  return EXIT_FAILURE;
381  }
382 
383  CompileAndExecuteConfig compileAndExecuteConfig;
384  if (optLevel) {
385  compileAndExecuteConfig.transformer = mlir::makeOptimizingTransformer(
386  *optLevel, /*sizeLevel=*/0, /*targetMachine=*/tmOrError->get());
387  }
388  compileAndExecuteConfig.llvmModuleBuilder = config.llvmModuleBuilder;
389  compileAndExecuteConfig.runtimeSymbolMap = config.runtimesymbolMap;
390 
391  // Get the function used to compile and execute the module.
392  using CompileAndExecuteFnT =
393  Error (*)(Options &, Operation *, StringRef, CompileAndExecuteConfig);
394  auto compileAndExecuteFn =
395  StringSwitch<CompileAndExecuteFnT>(options.mainFuncType.getValue())
396  .Case("i32", compileAndExecuteSingleReturnFunction<int32_t>)
397  .Case("i64", compileAndExecuteSingleReturnFunction<int64_t>)
398  .Case("f32", compileAndExecuteSingleReturnFunction<float>)
399  .Case("void", compileAndExecuteVoidFunction)
400  .Default(nullptr);
401 
402  Error error = compileAndExecuteFn
403  ? compileAndExecuteFn(options, m.get(),
404  options.mainFuncName.getValue(),
405  compileAndExecuteConfig)
406  : makeStringError("unsupported function type");
407 
408  int exitCode = EXIT_SUCCESS;
409  llvm::handleAllErrors(std::move(error),
410  [&exitCode](const llvm::ErrorInfoBase &info) {
411  llvm::errs() << "Error: ";
412  info.log(llvm::errs());
413  llvm::errs() << '\n';
414  exitCode = EXIT_FAILURE;
415  });
416 
417  return exitCode;
418 }
