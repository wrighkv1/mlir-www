26 FailureOr<Value>
27 mlir::bufferization::castOrReallocMemRefValue(OpBuilder &b, Value value,
28  MemRefType destType) {
29  auto srcType = value.getType().cast<MemRefType>();
30 
31  // Element type, rank and memory space must match.
32  if (srcType.getElementType() != destType.getElementType())
33  return failure();
34  if (srcType.getMemorySpaceAsInt() != destType.getMemorySpaceAsInt())
35  return failure();
36  if (srcType.getRank() != destType.getRank())
37  return failure();
38 
39  // In case the affine maps are different, we may need to use a copy if we go
40  // from dynamic to static offset or stride (the canonicalization cannot know
41  // at this point that it is really cast compatible).
42  auto isGuaranteedCastCompatible = [](MemRefType source, MemRefType target) {
43  int64_t sourceOffset, targetOffset;
44  SmallVector<int64_t, 4> sourceStrides, targetStrides;
45  if (failed(getStridesAndOffset(source, sourceStrides, sourceOffset)) ||
46  failed(getStridesAndOffset(target, targetStrides, targetOffset)))
47  return false;
48  auto dynamicToStatic = [](int64_t a, int64_t b) {
49  return a == MemRefType::getDynamicStrideOrOffset() &&
50  b != MemRefType::getDynamicStrideOrOffset();
51  };
52  if (dynamicToStatic(sourceOffset, targetOffset))
53  return false;
54  for (auto it : zip(sourceStrides, targetStrides))
55  if (dynamicToStatic(std::get<0>(it), std::get<1>(it)))
56  return false;
57  return true;
58  };
59 
60  // Note: If `areCastCompatible`, a cast is valid, but may fail at runtime. To
61  // ensure that we only generate casts that always succeed at runtime, we check
62  // a fix extra conditions in `isGuaranteedCastCompatible`.
63  if (memref::CastOp::areCastCompatible(srcType, destType) &&
64  isGuaranteedCastCompatible(srcType, destType)) {
65  Value casted = b.create<memref::CastOp>(value.getLoc(), destType, value);
66  return casted;
67  }
68 
69  auto loc = value.getLoc();
70  SmallVector<Value, 4> dynamicOperands;
71  for (int i = 0; i < destType.getRank(); ++i) {
72  if (destType.getShape()[i] != ShapedType::kDynamicSize)
73  continue;
74  auto index = b.createOrFold<arith::ConstantIndexOp>(loc, i);
75  Value size = b.create<memref::DimOp>(loc, value, index);
76  dynamicOperands.push_back(size);
77  }
78  // TODO: Use alloc/memcpy callback from BufferizationOptions if called via
79  // BufferizableOpInterface impl of ToMemrefOp.
80  Value copy = b.create<memref::AllocOp>(loc, destType, dynamicOperands);
81  b.create<memref::CopyOp>(loc, value, copy);
82  return copy;
83 }

85 /// Try to fold to_memref(to_tensor(x)). If x's type and the result type of the
86 /// to_memref op are different, a memref.cast is needed.
87 LogicalResult
88 mlir::bufferization::foldToMemrefToTensorPair(RewriterBase &rewriter,
89  ToMemrefOp toMemref) {
90  auto memrefToTensor = toMemref.getTensor().getDefiningOp<ToTensorOp>();
91  if (!memrefToTensor)
92  return failure();
93 
94  Type srcType = memrefToTensor.getMemref().getType();
95  Type destType = toMemref.getType();
96 
97  // Directly rewrite if the type did not change.
98  if (srcType == destType) {
99  rewriter.replaceOp(toMemref, memrefToTensor.getMemref());
100  return success();
101  }
102 
103  auto rankedSrcType = srcType.dyn_cast<MemRefType>();
104  auto rankedDestType = destType.dyn_cast<MemRefType>();
105  auto unrankedSrcType = srcType.dyn_cast<UnrankedMemRefType>();
106 
107  // Ranked memref -> Ranked memref cast.
108  if (rankedSrcType && rankedDestType) {
109  FailureOr<Value> replacement = castOrReallocMemRefValue(
110  rewriter, memrefToTensor.getMemref(), rankedDestType);
111  if (failed(replacement))
112  return failure();
113 
114  rewriter.replaceOp(toMemref, *replacement);
115  return success();
116  }
117 
118  // Unranked memref -> Ranked memref cast: May require a copy.
119  // TODO: Not implemented at the moment.
120  if (unrankedSrcType && rankedDestType)
121  return failure();
122 
123  // Unranked memref -> unranked memref cast
124  // Ranked memref -> unranked memref cast: No copy needed.
125  assert(memref::CastOp::areCastCompatible(srcType, destType) &&
126  "expected that types are cast compatible");
127  rewriter.replaceOpWithNewOp<memref::CastOp>(toMemref, destType,
128  memrefToTensor.getMemref());
129  return success();
130 }

152 LogicalResult AllocTensorOp::bufferize(RewriterBase &rewriter,
153  const BufferizationOptions &options) {
154  OpBuilder::InsertionGuard g(rewriter);
155  Location loc = getLoc();
156 
157  // Nothing to do for dead AllocTensorOps.
158  if (getOperation()->getUses().empty()) {
159  rewriter.eraseOp(getOperation());
160  return success();
161  }
162 
163  // Get "copy" buffer.
164  Value copyBuffer;
165  if (getCopy()) {
166  FailureOr<Value> maybeCopyBuffer = getBuffer(rewriter, getCopy(), options);
167  if (failed(maybeCopyBuffer))
168  return failure();
169  copyBuffer = *maybeCopyBuffer;
170  }
171 
172  // Create memory allocation.
173  auto allocType = bufferization::getBufferType(getResult(), options);
174  if (failed(allocType))
175  return failure();
176  SmallVector<Value> dynamicDims = getDynamicSizes();
177  if (getCopy()) {
178  assert(dynamicDims.empty() && "expected either `copy` or `dynamicDims`");
179  populateDynamicDimSizes(rewriter, loc, copyBuffer, dynamicDims);
180  }
181  FailureOr<Value> alloc = options.createAlloc(
182  rewriter, loc, allocType->cast<MemRefType>(), dynamicDims);
183  if (failed(alloc))
184  return failure();
185 
186  // Create memory copy (if any).
187  if (getCopy()) {
188  if (failed(options.createMemCpy(rewriter, loc, copyBuffer, *alloc)))
189  return failure();
190  }
191 
192  // Should the buffer be deallocated?
193  bool dealloc =
194  shouldDeallocateOpResult(getResult().cast<OpResult>(), options);
195 
196  // Replace op.
197  replaceOpWithBufferizedValues(rewriter, getOperation(), *alloc);
198 
199  // Create buffer deallocation (if requested).
200  if (!dealloc)
201  return success();
202 
203  rewriter.setInsertionPoint(rewriter.getInsertionBlock()->getTerminator());
204  if (failed(options.createDealloc(rewriter, loc, *alloc)))
205  return failure();
206  return success();
207 }

236 FailureOr<BaseMemRefType> AllocTensorOp::getBufferType(
237  Value value, const BufferizationOptions &options,
238  const DenseMap<Value, BaseMemRefType> &fixedTypes) {
239  assert(value == getResult() && "invalid value");
240 
241  // Compute memory space of this allocation.
242  unsigned memorySpace;
243  if (getMemorySpace().has_value()) {
244  memorySpace = *getMemorySpace();
245  } else if (getCopy()) {
246  auto copyBufferType =
247  bufferization::getBufferType(getCopy(), options, fixedTypes);
248  if (failed(copyBufferType))
249  return failure();
250  memorySpace = copyBufferType->getMemorySpaceAsInt();
251  } else if (options.defaultMemorySpace.has_value()) {
252  memorySpace = *options.defaultMemorySpace;
253  } else {
254  return getOperation()->emitError("could not infer memory space");
255  }
256 
257  return getMemRefTypeWithStaticIdentityLayout(getType(), memorySpace);
258 }

260 LogicalResult AllocTensorOp::verify() {
261  if (getCopy() && !getDynamicSizes().empty())
262  return emitError("dynamic sizes not needed when copying a tensor");
263  if (!getCopy() && getType().getNumDynamicDims() !=
264  static_cast<int64_t>(getDynamicSizes().size()))
265  return emitError("expected ")
266  << getType().getNumDynamicDims() << " dynamic sizes";
267  if (getCopy() && getCopy().getType() != getType())
268  return emitError("expected that `copy` and return type match");
269 
270  // For sparse tensor allocation, we require that none of its
271  // uses escapes the function boundary directly.
272  if (sparse_tensor::getSparseTensorEncoding(getType())) {
273  for (auto &use : getOperation()->getUses())
274  if (isa<func::ReturnOp, func::CallOp, func::CallIndirectOp>(
275  use.getOwner()))
276  return emitError("sparse tensor allocation should not escape function");
277  }
278 
279  return success();
280 }

304 /// Change the type of the result of a `bufferization.alloc_tensor` by making
305 /// the result type statically sized along dimension that in the original
306 /// operation where defined as dynamic, but the size was defined using a
307 /// `constant` op. For example:
308 ///
309 /// %c5 = arith.constant 5: index
310 /// %0 = bufferization.alloc_tensor(%arg0, %c5) : tensor<?x?xf32>
311 ///
312 /// to
313 ///
314 /// %0 = bufferization.alloc_tensor(%arg0) : tensor<?x5xf32>
315 struct ReplaceStaticShapeDims : OpRewritePattern<AllocTensorOp> {
316  using OpRewritePattern<AllocTensorOp>::OpRewritePattern;
317 
318  LogicalResult matchAndRewrite(AllocTensorOp op,
319  PatternRewriter &rewriter) const override {
320  if (op.getCopy())
321  return failure();
322  SmallVector<int64_t> newShape = llvm::to_vector(op.getType().getShape());
323  SmallVector<Value> newDynamicSizes;
324  unsigned int dynValCounter = 0;
325  for (int64_t i = 0; i < op.getType().getRank(); ++i) {
326  if (!op.isDynamicDim(i))
327  continue;
328  Value value = op.getDynamicSizes()[dynValCounter++];
329  APInt intVal;
330  if (matchPattern(value, m_ConstantInt(&intVal))) {
331  newShape[i] = intVal.getSExtValue();
332  } else {
333  newDynamicSizes.push_back(value);
334  }
335  }
336  RankedTensorType newType = RankedTensorType::get(
337  newShape, op.getType().getElementType(), op.getType().getEncoding());
338  if (newType == op.getType())
339  return failure();
340  auto newOp = rewriter.create<AllocTensorOp>(
341  op.getLoc(), newType, newDynamicSizes, /*copy=*/Value());
342  rewriter.replaceOpWithNewOp<tensor::CastOp>(op, op.getType(), newOp);
343  return success();
344  }
345 };

383 ParseResult AllocTensorOp::parse(OpAsmParser &parser, OperationState &result) {
384  SmallVector<OpAsmParser::UnresolvedOperand> dynamicSizesOperands;
385  if (parser.parseLParen() || parser.parseOperandList(dynamicSizesOperands) ||
386  parser.parseRParen())
387  return failure();
388  ParseResult copyKeyword = parser.parseOptionalKeyword("copy");
389  OpAsmParser::UnresolvedOperand copyOperand;
390  if (copyKeyword.succeeded())
391  if (parser.parseLParen() || parser.parseOperand(copyOperand) ||
392  parser.parseRParen())
393  return failure();
394  ParseResult sizeHintKeyword = parser.parseOptionalKeyword("size_hint");
395  OpAsmParser::UnresolvedOperand sizeHintOperand;
396  if (sizeHintKeyword.succeeded())
397  if (parser.parseEqual() || parser.parseOperand(sizeHintOperand))
398  return failure();
399  if (parser.parseOptionalAttrDict(result.attributes) || parser.parseColon())
400  return failure();
401 
402  TensorType type;
403  if (parser.parseCustomTypeWithFallback(type))
404  return failure();
405  result.addTypes(type);
406 
407  Type indexType = parser.getBuilder().getIndexType();
408  if (parser.resolveOperands(dynamicSizesOperands, indexType, result.operands))
409  return failure();
410  if (copyKeyword.succeeded())
411  if (parser.resolveOperand(copyOperand, type, result.operands))
412  return failure();
413  if (sizeHintKeyword.succeeded())
414  if (parser.resolveOperand(sizeHintOperand, indexType, result.operands))
415  return failure();
416  result.addAttribute(AllocTensorOp::getOperandSegmentSizeAttr(),
417  parser.getBuilder().getDenseI32ArrayAttr(
418  {static_cast<int32_t>(dynamicSizesOperands.size()),
419  static_cast<int32_t>(copyKeyword.succeeded()),
420  static_cast<int32_t>(sizeHintKeyword.succeeded())}));
421  return success();
422 }

468 /// Merge the clone and its source (by converting the clone to a cast) when
469 /// possible.
470 struct SimplifyClones : public OpRewritePattern<CloneOp> {
471  using OpRewritePattern<CloneOp>::OpRewritePattern;
472 
473  LogicalResult matchAndRewrite(CloneOp cloneOp,
474  PatternRewriter &rewriter) const override {
475  if (cloneOp.use_empty()) {
476  rewriter.eraseOp(cloneOp);
477  return success();
478  }
479 
480  Value source = cloneOp.getInput();
481  // Aims to find the dealloc op for the canonical source
482  // which otherwise could prevent removal of unnecessary allocs.
483  Value canonicalSource = source;
484  while (auto iface = dyn_cast_or_null<ViewLikeOpInterface>(
485  canonicalSource.getDefiningOp()))
486  canonicalSource = iface.getViewSource();
487 
488  llvm::Optional<Operation *> maybeCloneDeallocOp =
489  memref::findDealloc(cloneOp.getOutput());
490  // Skip if either of them has > 1 deallocate operations.
491  if (!maybeCloneDeallocOp.has_value())
492  return failure();
493  llvm::Optional<Operation *> maybeSourceDeallocOp =
494  memref::findDealloc(canonicalSource);
495  if (!maybeSourceDeallocOp.has_value())
496  return failure();
497  Operation *cloneDeallocOp = *maybeCloneDeallocOp;
498  Operation *sourceDeallocOp = *maybeSourceDeallocOp;
499 
500  // If both are deallocated in the same block, their in-block lifetimes
501  // might not fully overlap, so we cannot decide which one to drop.
502  if (cloneDeallocOp && sourceDeallocOp &&
503  cloneDeallocOp->getBlock() == sourceDeallocOp->getBlock())
504  return failure();
505 
506  Block *currentBlock = cloneOp->getBlock();
507  Operation *redundantDealloc = nullptr;
508  if (cloneDeallocOp && cloneDeallocOp->getBlock() == currentBlock) {
509  redundantDealloc = cloneDeallocOp;
510  } else if (sourceDeallocOp && sourceDeallocOp->getBlock() == currentBlock) {
511  redundantDealloc = sourceDeallocOp;
512  }
513 
514  if (!redundantDealloc)
515  return failure();
516 
517  // Safety check that there are no other deallocations inbetween
518  // cloneOp and redundantDealloc, as otherwise we might deallocate an alias
519  // of source before the uses of the clone. With alias information, we could
520  // restrict this to only fail of the dealloc's operand is an alias
521  // of the source.
522  for (Operation *pos = cloneOp->getNextNode(); pos != redundantDealloc;
523  pos = pos->getNextNode()) {
524  auto effectInterface = dyn_cast<MemoryEffectOpInterface>(pos);
525  if (!effectInterface)
526  continue;
527  if (effectInterface.hasEffect<MemoryEffects::Free>())
528  return failure();
529  }
530 
531  rewriter.replaceOpWithNewOp<memref::CastOp>(cloneOp, cloneOp.getType(),
532  source);
533  rewriter.eraseOp(redundantDealloc);
534  return success();
535  }
536 };

609 /// Replace tensor.cast + to_memref by to_memref + memref.cast.
610 struct ToMemrefOfCast : public OpRewritePattern<ToMemrefOp> {
611  using OpRewritePattern<ToMemrefOp>::OpRewritePattern;
612 
613  LogicalResult matchAndRewrite(ToMemrefOp toMemref,
614  PatternRewriter &rewriter) const final {
615  auto tensorCastOperand =
616  toMemref.getOperand().getDefiningOp<tensor::CastOp>();
617  if (!tensorCastOperand)
618  return failure();
619  auto srcTensorType =
620  tensorCastOperand.getOperand().getType().dyn_cast<RankedTensorType>();
621  if (!srcTensorType)

629  return success();
630  }

669  return failure();

672  dimOp.getIndex());
673  return success();
